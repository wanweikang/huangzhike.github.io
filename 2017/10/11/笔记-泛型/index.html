参数化类型不考虑类型参数的继承关系

// 指定上限，只能是Number类型或其子类，并实现List接口
class Info<T extends Number & java.util.List> {}

变量被声明为泛型时，只能被实例方法调用（还有内嵌类型），不能被静态方法调用。

静态成员是被所参数化的类所共享的，不应该有类级别的类型参数。

public class Apple<T> {

    // static T var; 错误，静态属性是共享的
    // static void method (T arg) {} 错误，静态方法声明不能使用类型形参
    // static <T> void method (T arg) {} ok
    // xx instanceof Apple<String>; 错误，instanceof后不能加

    // 泛型类的成员方法
    public T getInfo() {}
    // 泛型方法，这里的T是不同的
    public static <T> T genericFunc(T a) {
        return null;
    }
}

// 错误，必须指明T
class A extends Apple<T> {}

class A extends Apple<String> {
    // ok
    public String getInfo() {}

    // 错误，重写父类方法时返回值类型不一致
    public Object getInfo() {}
}
// 泛型接口
interface Info<T> {}

class InfoImpl1<T> implements Info<T> {};

class InfoImpl2 implements Info<String> {};
class Info<T> {
    private T var; // 定义泛型变量

    Info(T var) {
        this.var = var;
    }
};

public class Test {
    public static void main(String args[]) {

        Info<Float> i1 = new Info<Float>(0.1f); // 自动装箱

        fun1(i1);

        Info<String> i2 = new Info<String>("hello");
        fun2(i2);
    }

    // 泛型上限，只能接收Number及其Number的子类
    public static void fun1(Info<? extends Number> arg) {}

    // 泛型下限，只能接收String或Object类型的泛型
    public static void fun2(Info<? super String> arg) {}
};
// 也可以在类声明中定义
class Info<T extends Number> {
    private T var;

    public void setVar(T var) {
        this.var = var;
    }

    public T getVar() {
        return this.var;
    }
};
public class Test {
    public static void main(String args[]) {
        Info<Integer> i = fun(30);
    }

    // 泛型方法，返回一个泛型类的实例对象
    public static <T extends Number> Info<T> fun(T args) {
        Info<T> temp = new Info<T>(); // 根据传入的数据类型实例化Info
        temp.setVar(args); // 将传递的内容设置到Info对象的var属性之中
        return temp; // 返回实例化对象
    }
};

public class Test {
    static <T> void fromArrayToCollection(T[] a, Collection<T> c) {
        for (T o : a) {
            c.add(o);
        }
    }

    public static void main(String[] args) {
        Object[] oa = new Object[100];
        Collection<Object> co = new ArrayList<>();
        fromArrayToCollection(oa, co); // T代表Object

        String[] sa = new String[100];
        fromArrayToCollection(sa, co); // T代表Object

        Integer[] ia = new Integer[100];
        Collection<Number> cn = new ArrayList<>();
        fromArrayToCollection(ia, cn); // T代表Number类型

        Number[] na = new Number[100];
        fromArrayToCollection(na, co); // T代表String类型

    }
}

public class GenericTest {
    public static void main(String[] args) {
        new GenericTest().testType();
    }

    public void testType(){
        ArrayList<Integer> c1 = new ArrayList<Integer>();
        ArrayList<String> c2 = new ArrayList<String>();

        System.out.println(c1.getClass() == c2.getClass());
        // 两者class类型一样，即字节码一致

        System.out.println(c2.getClass().getName());
        // class均为java.util.ArrayList，无实际类型参数信息
    }
