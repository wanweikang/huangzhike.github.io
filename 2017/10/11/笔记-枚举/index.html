enum Color {
    // 枚举对象，实例可按静态字方式引用： Color.RED
    RED, BLUE; // 实例列表末分号可选
    private String name;

    public void setName(String name) {
        switch (this) { // 判断枚举对象
            // 如果是加了类名的如Color.RED则编译会报错
            case RED: {
                if ("红色".equals(name)) {
                    this.name = name;
                }
                break;
            }
            case BLUE: {
                if ("蓝色".equals(name)) {
                    this.name = name;
                }
                break;
            }
        }
        this.name = name;
    }

    public String getName() {
        return this.name;
    }
};

        // Color c = Color.valueOf(Color.class, "BLUE");

        c.setName("蓝色");
        System.out.println(c.getName());
    }
};
public enum RegularPolygon {
    // 已经声明了有参构造方法，必须调用
    TRIANGLE(3), SQUARE(4), PENTAGON(5); // 有参数的枚举必须使用分号
    private Shape shape;
    public Shape getShape() {
        return shape;
    }
    // 枚举实例运行时创建，外部不能实例化，构造方法默认private，只能private
    private RegularPolygon(int sides) {
        switch (sides) {
        case 3:
            shape = new Triangle(1, 1, 1, 60, 60, 60);
            break;
        case 4:
            shape = new Rectangle(1, 1);
            break;
        case 5:
            shape = new Pentagon(1, 1, 1, 1, 1, 108, 108, 108, 108, 108);
            break;
        }
    }
};
interface Print {
    public String getColor();

    RED {
        public String getColor() {
            return "红色";
        }
    },
    GREEN {
        public String getColor() {
            return "绿色";
        }
    };
};
