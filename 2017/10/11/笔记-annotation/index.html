自定义注解

public @interface X {
    public String[] value(); // 数组
}

// @X(value = { "cnm", "mdzz" })
enum Name { // 定义枚举类型
    CNM, MDZZ, HHD;
};

public @interface E {
    public Name name() default Name.CNM;
};

// 只能从固定的枚举中取内容
// @E(name = Name.CNM)

@Retention：生命周期

RetentionPolicy.SOURCE // 注解在Java源文件
RetentionPolicy.CLASS // 注解在Class类文件
RetentionPolicy.RUNTIME // 运行期也保留注解，可以通过反射机制读取注解

@Target：注解位置

ElemenetType.CONSTRUCTOR // 构造器
ElemenetType.FIELD // 域（包括 enum 实例）
ElemenetType.LOCAL_VARIABLE // 局部变量声
ElemenetType.METHOD // 方法
ElemenetType.PACKAGE // 包
ElemenetType.PARAMETER // 参数
ElemenetType.TYPE // 类，接口（包括注解类型）或enum
// 元注解：注解其他注解
@Retention(value = RetentionPolicy.RUNTIME) // 此Annotation在运行时有效
@Target({ElementType.TYPE, ElementType.METHOD}) // 此注释只能用在类及方法上，@Target({ METHOD, TYPE })
@Documented // 生成java doc时就可以加入说明
@Inherited // 使用该注释的类的子类也可以继承该注释使用
public @interface JDBCConfig {
    String ip(); // 接收设置的内容，注意括号

    int port() default 3306;

    String database();

    String encoding();

    String loginName();

    String password();
}
// 如果定义了不使用会报错，除非有default
@JDBCConfig(ip = "127.0.0.1", database = "test", encoding = "UTF-8", loginName = "root", password = "admin")
public class DBUtil {
    static {
        try {
            Class.forName("com.mysql.jdbc.Driver");
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }
    }

    public static Connection getConnection() throws SQLException, NoSuchMethodException, SecurityException {
        // 通过反射，获取这个DBUtil这个类上的注解对象
        JDBCConfig config = DBUtil.class.getAnnotation(JDBCConfig.class);
        // 拿到注解对象之后，通过其方法，获取各个注解元素的值
        String ip = config.ip();
        int port = config.port();
        String database = config.database();
        String encoding = config.encoding();
        String loginName = config.loginName();
        String password = config.password();

        String url = String.format("jdbc:mysql://%s:%d/%s?characterEncoding=%s", ip, port, database, encoding);
        return DriverManager.getConnection(url, loginName, password);
    }

    public static void main(String[] args) throws NoSuchMethodException, SecurityException, SQLException {
        Connection c = getConnection();
    }
};

Annotation[] annotations = aClass.getAnnotations();
