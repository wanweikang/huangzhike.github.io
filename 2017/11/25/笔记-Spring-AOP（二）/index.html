注解方式

package mmp;

@Aspect
public class Logging {
    // 切入点，声明哪种方法将被拦截，签名就是下面的方法名
    @Pointcut("execution(* mmp.*.*(..))")
    private void selectAll() {}

    @Before("selectAll()")
    public void beforeAdvice() {
        System.out.println("BEFORE");
    }

    @After("selectAll()")
    public void afterAdvice() {
        System.out.println("AFTER");
    }

    @AfterReturning(pointcut = "selectAll()", returning = "retVal")
    public void afterReturningAdvice(Object retVal) {
        System.out.println(retVal.toString());
    }

    @AfterThrowing(pointcut = "selectAll()", throwing = "ex")
    public void AfterThrowingAdvice(IllegalArgumentException ex) {
        System.out.println(ex.toString());
    }
}

package mmp;

public class Customer {

    public void addCustomer() {}

    public String addCustomerReturnValue() {
        return "abc";
    }

    public void addCustomerThrowException() throws Exception {
        throw new Exception("Generic Error");
    }

    public void addCustomerAround(String name) {}
}
package mmp.aspect;

@Aspect
public class LoggingAspect {

    // ...

    @Before("execution(* mmp.Customer.addCustomer(..))")
    public void logBefore() {}

    @After("execution(* mmp.Customer.addCustomer(..))")
    public void logAfter() {}

    @AfterReturning(pointcut = "execution(* mmp.Customer.addCustomerReturnValue(..))", returning = "result")
    public void logAfterReturning(Object result) {}

    @AfterThrowing(pointcut = "execution(* mmp.Customer.addCustomerThrowException(..))", throwing = "error")
    public void logAfterThrowing(Throwable error) {}

    /*
     * addCustomerAround()方法执行之前执行
     * ProceedingJoinPoint.proceed();控制何时拦截器返回控制到原来的addCustomerAround()方法 
     * start stopwatch:
     * Object retVal = pjp.proceed();
     * stop stopwatch:
     * return retVal;
     */

    @Around("execution(* mmp.Customer.addCustomerAround(..))")
    public void logAround(ProceedingJoinPoint pjp) throws Throwable {

        int numAttempts = 0;
        PessimisticLockingFailureException lockFailureException;
        do {
            numAttempts++;
            try {
                return pjp.proceed();
            } catch (PessimisticLockingFailureException ex) {
                lockFailureException = ex;
            }
        } while (numAttempts <= this.maxRetries);
        throw lockFailureException;

    }

}

<aop:before 
    pointcut="切入点表达式" 
    pointcut-ref="切入点引用" 
    method="前置通知实现方法名，如果是多态需要加上参数类型" 
    arg-names="前置通知实现方法参数列表参数名字"/>
<!-- pointcut和pointcut-ref：二选一，指定切入点 -->
<aop:before 
    pointcut="execution(* mmp..*.sayBefore(..)) and args(param)" 
    method="beforeAdvice(java.lang.String)" 
    arg-names="param"/>
<!-- arg-names="param"指定通知实现方法参数名为“param”，切入点中用“args(param)”匹配的目标方法参数将传递给通知实现方法同名参数 -->
<aop:before 
    pointcut="execution(* test(*)) and args(param)" 
    method="beforeAdvice(java.lang.String)" 
    arg-names="param"/>

execution(* test(*)) && args(param);

  • execution(* test(*))匹配任何方法名为test，且有一个任何类型的参数；
  • args(param)匹配只有一个参数且传入的参数类型为通知实现方法中同名的参数类型；
  • args(param)将首先查找通知方法上同名的参数，并在方法执行时匹配传入的参数是使用该同名参数类型，即java.lang.String，匹配将把该被通知参数传递给通知方法上同名参数。

java.*.String 匹配java包下的任何“一级子包”下的String类型；
java..* 匹配java包及任何子包下的任何类型;如匹配java.lang.String、java.lang.annotation.Annotation
java.lang.*ing 匹配任何java.lang包下的以ing结尾的类型；
java.lang.Number+ 匹配java.lang包下的任何Number的自类型；

<aop:after-returning 
    pointcut="execution(* mmp..*.sayAfterReturning(..))" 
    method="afterReturningAdvice" 
    arg-names="retVal" 
    returning="retVal"/>
<!-- arg-names="retVal"指定通知实现方法afterReturningAdvice参数名为"retVal"  -->
<!-- returning="retVal"将目标方法返回值赋值给通知实现方法afterReturningAdvice参数名为“retVal”的参数 -->

<!-- arg-names="exception"指定通知实现方法afterThrowingAdvice参数名为“exception” -->
<!-- returning="exception"将目标方法抛出的异常赋值给通知实现方法afterThrowingAdvice参数名为“exception”的参数 -->

execution(public * *(..))

任何方法以”set”开头
execution(* set*(..))

任何被AccountService接口定义的方法
execution(* com.xyz.service.AccountService.*(..))

任何定义在service包的方法
execution(* com.xyz.service.*.*(..))

任何定义在service包及子包的方法
execution(* com.xyz.service..*.*(..))

mmp包及所有子包下IPointcutService接口中的任何无参方法
execution(* mmp..IPointcutService.*())

mmp包及所有子包下任何类的任何方法
execution(* mmp..*.*(..))

mmp包及所有子包下IPointcutService接口的任何只有一个参数方法
execution(* mmp..IPointcutService.*(*))

非“mmp包及所有子包下IPointcutService接口及子类型”的任何方法
execution(* (!mmp..IPointcutService+).*(..))

mmp包及所有子包下IPointcutService接口及子类型的的任何无参方法
execution(* mmp..IPointcutService+.*())

mmp包及所有子包下IPointcut前缀类型的的以test开头的只有一个参数类型为java.util.Date的方法
execution(* mmp..IPointcut*.test*(java.util.Date))

mmp包及所有子包下IPointcut前缀类型的的任何方法，且抛出IllegalArgumentException和ArrayIndexOutOfBoundsException异常
execution(* mmp..IPointcut*.test*(..) throws IllegalArgumentException, ArrayIndexOutOfBoundsException)

任何实现了mmp包及所有子包下IPointcutService接口和java.io.Serializable接口的类型的任何方法
execution(* (mmp..IPointcutService+ && java.io.Serializable+).*(..))

任何持有@java.lang.Deprecated注解的方法
execution(@java.lang.Deprecated * *(..))

<aop:before/>：在切入点选择的方法执行之前执行通知

<aop:after/>：在切入点选择的方法返回时执行通知

<aop:after-returning/>：在切入点选择的方法正常返回后执行通知

<aop:after-throwing/>：在切入点选择的方法抛出异常时执行

<aop:around/>：环绕着在切入点选择的连接点处的方法执行通知，可以决定目标方法是否执行，什么时候执行，执行时是否需要替换方法参数，执行完毕是否需要替换返回值。在建议方法调用之前和之后，执行通知
