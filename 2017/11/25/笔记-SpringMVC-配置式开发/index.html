[笔记]-SpringMVC-配置式开发

2017-11-25

12-27重新整理更新Spring笔记

web.xml

<web-app>
    <!-- 注册ContextLoaderListener，创建 Spring容器并放入到ServletContext 域 -->
    <listener>
        <listener-class>org.springframework.web.context.ContextLoaderListener</listener-class>
    </listener>
    <!-- 加载Spring容器 -->
    <context-param>
        <!-- 读取其他XML配置文件，Spring在加载配置文件后，把它放在ServletContext中 -->
        <param-name>contextConfigLocation</param-name>
        <param-value>/WEB-INF/root-context.xml, /WEB-INF/app-context.xml</param-value>
    </context-param>
    <servlet>
        <servlet-name>app</servlet-name>
        <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
        <init-param>
            <!-- SpringMVC加载的配置文件，配置处理器映射器、适配器等 -->
            <!-- 如果不配置contextConfigLocation，默认加载的是/WEB-INF/[servlet-name]-serlvet.xml -->
            <param-name>contextConfigLocation</param-name>
            <param-value>/WEB-INF/springMVC.xml</param-value>
        </init-param>
        <load-on-startup>1</load-on-startup>
    </servlet>
    <servlet-mapping>
        <servlet-name>app</servlet-name>
        <!-- 在web.xml中配置拦截*.action后，访问的时候xxxx.action，注解路由可以@RequestMapping("xxxx") -->
        <url-pattern>/app/*</url-pattern>
    </servlet-mapping>
</web-app>

MVC中完整的C（控制逻辑+功能处理）由DispatcherServlet + Controller组成。

配置式开发：处理器类是手工定义的、实现了特定接口的类，再在SpringMVC配置文件中显式注册该类。

Spring2.5前，通过实现Controller接口或其实现定义处理器类，已过时。

package mmp;

public class LoginController implements Controller {
    public ModelAndView handleRequest() throws Exception {
        return new ModelAndView("welcome"); // 返回welcome.jsp
    }
}
<!-- HandlerMapping将请求的URL和Bean名字映射，如URL为 上下文/login ，必须有名为 /login 的Bean -->
<bean class="org.springframework.web.servlet.handler.BeanNameUrlHandlerMapping"/>
<!-- HandlerAdapter：实现了Controller接口的Bean可作为处理器，其他处理器可以通过实现HadlerAdapter解决 -->
<bean class="org.springframework.web.servlet.mvc.SimpleControllerHandlerAdapter"/>
<!-- 注册Controller -->
<bean name="/login" class="mmp.LoginController"></bean>
<!-- ViewResolver视图解析器 -->
<bean class="org.springframework.web.servlet.view.InternalResourceViewResolver">
    <!-- JSP模板页面需要使用JSTL标签库，classpath中必须包含jstl的相关jar包 -->
    <property name="viewClass" value="org.springframework.web.servlet.view.JstlView"/>
    <!-- 默认在WebRoot下 -->
    <property name="prefix" value="/WEB-INF/jsp/"/>
    <property name="suffix" value=".jsp"/>
</bean>

处理器映射HandlerMapping常用的实现类有两种：

  • BeanNameUrlHandlerMapping
  • SimpleUrlHandlerMapping
<bean class="org.springframework.web.servlet.handler.SimpleUrlHandlerMapping">
    <property name="mappings">
        <props>
            <prop key="/welcome.html">welcomeController</prop> 
            <prop key="userAction.do">userController</prop>
        </props>
    </property>
</bean>
<!-- 处理器 -->
<bean id="welcomeController" class="mmp.WelcomeController"/>
<!-- 处理器 -->
<bean id="userController" class="mmp.UserController">
    <property name="methodNameResolver">
        <bean class="org.springframework.web.servlet.mvc.multiaction.ParameterMethodNameResolver">
            <!-- 指定参数名为action -->
            <property name="paramName" value="action"/>
            <!-- http://localhost:8080/xxx/userAction.do?action=add -->
            <!-- http://localhost:8080/xxx/userAction.do?action=delete -->
        </bean>
    </property>
</bean>

处理器适配器HandlerAdapter实现类有两种：

  • SimpleControllerHandlerAdapter
    • 所有实现了Controller接口的处理器Bean，通过此适配器适配执行。
  • HttpRequestHandlerAdapter
    • 所有实现了HttpRequestHandler接口的处理器Bean，通过此适配器适配执行。

视图解析器ViewResolver接口将处理结果生成View视图。

常用实现类：InternalResourceViewResolver

<!-- URL映射 -->
<bean class="org.springframework.web.servlet.view.InternalResourceViewResolver">
    <property name="prefix" value="/WEB-INF/jsp/"/>
    <property name="suffix" value=".jsp"/>
</bean>
<!-- 对于/login请求，DispatcherServlet将请求转发到prefix+view-name+suffix=/WEB-INF/jsp/login.jsp -->

XmlViewResolver：需要定义的视图对象很多时，可以将视图对象抽取出来，单独定义为一个xml文件。

<bean class="org.springframework.web.servlet.view.XmlViewResolver">
    <property name="location">
        <value>/WEB-INF/views.xml</value>
    </property>
</bean>

views.xml

<!-- 对于/login请求，转发到由view.xml中对应的login.jsp -->
<bean id="login" class="org.springframework.web.servlet.view.JstlView">
    <property name="url" value="/WEB-INF/jsp/login.jsp"/>
</bean>

ResourceBundleViewResolver：在properties文件中注册视图解析器。

<bean class="org.springframework.web.servlet.view.ResourceBundleViewResolver">
    <!-- basename：视图的资源束名，views.properties -->
    <property name="basename" value="views"/>
</bean>

src/views.properties

# 对于/login请求，转发到login.jsp
login.(class)=org.springframework.web.servlet.view.JstlView
login.url=/WEB-INF/jsp/login.jsp

使用多个视图解析器，order属性设置优先级顺序。

<bean class="org.springframework.web.servlet.view.ResourceBundleViewResolver">
    <property name="basename" value="views"/>
    <!-- 0作为第一解析器，1作为下一解析器 -->
    <property name="order" value="0"/>
</bean>
<bean class="org.springframework.web.servlet.view.InternalResourceViewResolver">
    <property name="prefix" value="/WEB-INF/jsp/"/>
    <property name="suffix" value=".jsp"/>
    <property name="order" value="1"/>
</bean>

处理器除了实现Controller接口外，还可以继承自其它类。

继承AbstractController类

package mmp;

public class HelloController extends AbstractController {
    @Override
    protected ModelAndView handleRequestInternal() throws Exception {
        ModelAndView model = new ModelAndView("hello");
        model.addObject("message", "Hello World!");
        return model;
    }
}
<bean class="org.springframework.web.servlet.mvc.support.ControllerClassNameHandlerMapping"/>
<!-- ControllerClassNameHandlerMapping接受Controller名称并转换为带有/的小写形式 -->
<!-- HelloController映射到/hello*，对于/helloWorld.html或/hello{任何字母}.html，请求转发到HelloController -->
<!-- 对于/hello1.html，不会找到控制器 -->
<bean class="mmp.HelloController"/>

继承MultiActionController类

package mmp;

public class UserController extends MultiActionController {
    // MultiActionController在单个控制器中分别映射多个URL到对应的方法

    public ModelAndView home() throws Exception {}

    public ModelAndView add() throws Exception {}
}
<bean class="org.springframework.web.servlet.handler.BeanNameUrlHandlerMapping"/>

<!-- 对于/home.html，请求转发到UserController类的home()方法 -->
<bean name="/home.html" class="mmp.UserController"/>
<!-- 对于user/add.html，请求转发到UserController类的add()方法 -->
<bean name="/user/*.html" class="mmp.UserController"/>
