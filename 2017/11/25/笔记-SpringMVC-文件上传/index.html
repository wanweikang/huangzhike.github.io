使用SpringMVC的CommonsMultipartFile类读取文件，需要用到两个jar包commons-logging.jar、commons-io-xxx.jar。

<!-- CommonsMultipartResolver为MultipartFile 接口实现类，注册文件上传解析器，id是固定的 -->
<bean id="multipartResolver" class="org.springframework.web.multipart.commons.CommonsMultipartResolver">
    <property name="defaultEncoding" value="utf-8"></property>
    <!-- 单位字节B，不设置，或指定为-1，表示不限制上传文件大小 -->
    <!-- 该大小为上传文件的总大小，若上传多个文件，大小之和不能大于该设定值 -->
    <!-- 最大上传文件大小 -->
    <property name="maxUploadSize" value="10485760000"></property>
    <property name="maxInMemorySize" value="10960"></property>
    <!-- 设置属性 maxUploadSizePerFile，设置每个上传文件的大小 -->
    <!-- 当上传文件超出指定大小时，抛出MaxUploadSizeExceededException异常，设置SimpleMappingExceptionResolver处理该异常 -->
</bean>
@Controller
@RequestMapping("/test")
public class MyController {

    @RequestMapping(value = "/upload.do")
    public ModelAndView doUpload(@RequestParam MultipartFile[] photos, HttpSession session) throws Exception {
        // 未选择上传文件就提交，MultipateFile形参接的实参值并非null，而是empty文件
        // 上传多个文件，即使没有选择要上传的文件，MultipartFile数组也不为null，length值也大于0
        String path = session.getServletContext().getRealPath("images");

        for (int i = 0; 1 < photos.length; i++) {
            // MultipartFile接口方法：判断文件是否为空
            if (!photos[i].isEmpty()) {
                // 获取文件的原始名
                String fileName = photos[i].getOriginalFilename();
                // 文件名后辍判断文件类型
                if (fileName.endsWith(".jpg") || fileName.endsWith(".png")) {
                    File file = new File(path, fileName);
                    // 文件上传
                    photos[i].transferTo(file);
                } else {
                    return new ModelAndView("/fail.jsp");
                }
            }
        }
        return new ModelAndView("/success.jsp");
    }
}
