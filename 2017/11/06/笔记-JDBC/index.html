public class Test {
    private String driver, url, user, pass;
    Connection conn;

    public void initParam(String paramFile) throws Exception {
        Properties props = new Properties(); // 使用Properties类加载属性文件
        props.load(new FileInputStream(paramFile));

        driver = props.getProperty("driver");
        url = props.getProperty("url");
        user = props.getProperty("user");
        pass = props.getProperty("pass");
    }
    // Statement
    public void Statement() throws Exception {
        String sql = "SELECT id, name, age, birthday FROM user";
        try (java.sql.Statement stmt = conn.createStatement();) {
            // Statement 3种执行SQL方法
            // 1 execute 执行任何SQL语句 检索到ResultSet对象，返回true，否则false
            // 2 executeQuery 返回查询到的 ResultSet
            // 3 executeUpdate 返回被影响的记录数 int
            if (stmt.execute(sql)) {
                try (ResultSet rs = stmt.getResultSet();) {
                    // ResultSet有getXxx系列
                    while (rs.next()) { // 指针移动
                        int id = rs.getInt("id"); // int id = rs.getInt(1) ;
                        String name = rs.getString("name"); // String name = rs.getString(2) ;
                        int age = rs.getInt("age"); // int age = rs.getInt(3) ;
                        java.util.Date d = rs.getDate("birthday"); // java.util.Date d = rs.getDate(4);
                    }
                } catch (Exception e) {
                }
            } else {
                System.out.println("该SQL语句影响" + stmt.getUpdateCount());
            }
        } catch (Exception e) {
        }
    }
    // PreparedStatement
    public void PreparedStatement() throws Exception {
        long start = System.currentTimeMillis();
        try (PreparedStatement pstmt = conn.prepareStatement("insert into user values(null, ?, 1)");) {
            // 插入100条记录
            for (int i = 0; i < 100; i++) {
                // 参数标记由1开始，与索引不同(它不从0开始)
                pstmt.setString(1, "姓名" + i);
                pstmt.executeUpdate();
            }
            System.out.println("费时:" + (System.currentTimeMillis() - start));
        }
    }

    public void getConn() throws Exception {
        if (conn == null) {
            Class.forName(driver);
            conn = DriverManager.getConnection(url, user, pass);
        }
    }

    public void closeConn() throws Exception {
        if (conn != null)
            conn.close();
    }

    public static void main(String[] args) throws Exception {
        Test test = null;
        try {
            test = new Test();
            test.initParam("mysql.ini");
            test.getConn();
            test.Statement();
            test.PreparedStatement();
        } finally {
            test.closeConn();
        }
    }
}
public class Test {
    public static final String DRIVER = "com.mysql.jdbc.Driver";
    public static final String URL = "jdbc:mysql://127.0.0.1:3306/user";
    public static final String USER = "root";
    public static final String PASS = "123456";

    public static void main(String args[]) throws Exception {

        String sql = "SELECT id, name, age, birthday FROM user";
        Class.forName(DRIVER);
        Connection conn = DriverManager.getConnection(URL, USER, PASS);
        PreparedStatement pstmt = conn.prepareStatement(sql, 
            ResultSet.TYPE_SCROLL_SENSITIVE, // ResultSet可以滚动，更新内容
            ResultSet.CONCUR_READ_ONLY); // 以只读打开
        ResultSet rs = pstmt.executeQuery();

        rs.absolute(1); // 移到第一行
        print(rs, 1); // 第2条

        rs.beforeFirst(); // 移到首行前
        print(rs, 1); // 第1条

        rs.afterLast(); // 移到最后一条后
        print(rs, -1); // 最后一条

        rs.close();
        pstmt.close();
        conn.close();
    }

    public static void print(ResultSet rs, int re) throws Exception {
        if (re > 0) {
            rs.next(); // 由前向后
        } else {
            rs.previous(); // 由后向前
        }
        int id = rs.getInt(1);
        String name = rs.getString(2);
        int age = rs.getInt(3);
        java.util.Date d = rs.getDate(4);
        // ...
    }
};

批处理：一次向数据库发送多个SQL语句时，可以减少通信开销，提高性能。

使用Statement对象进行批处理

Statement stmt = conn.createStatement();

conn.setAutoCommit(false);

String[] sqls = {
    "INSERT INTO user(name, age) VALUES ('Java', 99)",
    "INSERT INTO user(name, age) VALUES ('Java', 99)",
    "INSERT INTO user(name, age) VALUES ('Java', 99)",
}

for (String sql : sqls) {
    stmt.addBatch(sql); // 加入批处理
}

int[] count = stmt.executeBatch();

conn.commit();

使用PrepareStatement对象进行批处理

PreparedStatement pstmt = conn.prepareStatement(sql);

conn.setAutoCommit(false);

String sql = "INSERT INTO user(name, age, birthday) VALUES (?,?,?)";

for (int i = 0; i < 10; i++) {
    pstmt.setString(1, "cnm - " + i);
    pstmt.setInt(2, 18 + i);
    pstmt.setDate(3, new java.sql.Date(new java.util.Date().getTime()));
    pstmt.addBatch(); // 加入批处理
}

int count[] = pstmt.executeBatch();

conn.commit();

如果处于自动提交模式，默认每个SQL语句完成后都会提交。

关闭自动提交并管理事务原因：

  • 提高性能
  • 保持业务流程的完整性
  • 使用分布式事务

事务将单个SQL语句或一组SQL语句视为一个逻辑单元，如果任何语句失败，整个事务将失败。

JDBC默认自动提交，调用Connection对象的setAutoCommit()方法启用手动事务。

try {
    // 假设下面一条不正确，则自动回滚
    String sql = "insert into user values(null , 'bbb' ,1)";

    conn.setAutoCommit(false); // 关闭自动提交，开启事务

    PreparedStatement stmt = conn.createStatement();
    // Savepoint savepoint = conn.setSavepoint("SavepointXxx");
    stmt.executeUpdate(sql); // SQL执行了，但未生效
    // INSERT语句不会成功执行，因为所有操作都被回滚了
    conn.commit(); // 提交事务

} catch (SQLException se) {
    conn.rollback();
    // conn.rollback(savepoint);
}

String sql1 = "INSERT INTO userclob(name,note) VALUES (?,?) ";
PreparedStatement pstmt = conn.prepareStatement(sql1);
File f = new File("d:" + File.separator + "cnm.txt");
InputStream input1 = new FileInputStream(f); // 通过输入流读取文件
pstmt.setString(1, "cnm"); // 设置第一个?
pstmt.setAsciiStream(2, input1, (int) f.length());
pstmt.executeUpdate();

String sql2 = "SELECT name,note FROM userclob WHERE id=? ";
pstmt = conn.prepareStatement(sql2);
pstmt.setInt(1, 1); // 读取的编号

ResultSet rs = pstmt.executeQuery();
if (rs.next()) {
    // 方法一
    StringBuffer note1 = new StringBuffer();
    InputStream input2 = rs.getAsciiStream(2);
    Scanner scan = new Scanner(input2); // 使用Scanner类读取内容
    scan.useDelimiter("\r\n"); // 将文件换行作为分割符
    while (scan.hasNext()) {
        note1.append(scan.next()).append("\n");
    }
    System.out.println("内容：" + note1);
    input2.close();
    // 方法二
    Clob c = rs.getClob(2);
    String note2 = c.getSubString(1, (int) c.length());
    System.out.println("内容：" + note2);
}

String sql1 = "INSERT INTO userblob (name, photo) VALUES (?, ?)";
PreparedStatement pstmt = conn.prepareStatement(sql1);
File f = new File("d:" + File.separator + "cnm.jpg"); // 图片
InputStream input1 = new FileInputStream(f);
pstmt.setString(1, "cnm");
pstmt.setBinaryStream(2, input1, (int) f.length()); // 设置输入流
pstmt.executeUpdate(); // 更新数据库

String sql2 = "SELECT name,photo FROM userblob WHERE id=?";
pstmt = conn.prepareStatement(sql2);
pstmt.setInt(1, 1);

ResultSet rs = pstmt.executeQuery();
if (rs.next()) {
    OutputStream out = new FileOutputStream(f);
    // 方法一
    InputStream input2 = rs.getBinaryStream(2);
    int temp = 0;
    while ((temp = input2.read()) != -1) { // 边读边写
        out.write(temp);
    }
    input2.close();
    // 方法二
    Blob b = rs.getBlob(2);
    out.write(b.getBytes(1, (int) b.length()));

    out.close();
}

SQL使用NULL值和Java使用null是不同的概念，在Java中处理SQL NULL值：

  • 避免使用返回原始类型的getXXX()方法。
  • 对原始数据类型使用包装类，并使用ResultSet对象的wasNull()方法来测试接收getXXX()方法的返回值的包装器类变量是否应设置为null。
  • 使用原始数据类型和ResultSet对象的wasNull()方法来测试接收到由getXXX()方法返回的值的原始变量是否应设置为表示NULL的可接受值。
Statement stmt = conn.createStatement();
String sql = "SELECT id, first, last, age FROM Employees";
ResultSet rs = stmt.executeQuery(sql);

int id = rs.getInt(1);
if(rs.wasNull()) {
    id = 0;
}
