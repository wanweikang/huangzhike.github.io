[笔记]-JavaBean

2017-11-01

12-31重新整理

JavaBean是一个遵循特定写法的Java类：

  • 必须有无参构造
  • 属性必须私有化
  • 私有属性必须通过public方法暴露，方法命名必须遵守规范
public class Person {
    // 封装私有属性
    private String name;

    private boolean married;

    // 无参构造
    public Person() {
        // ...
    }

    // 对外提供访问私有属性的public方法
    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public boolean isMarried() {
        return married;
    }

    public void setMarried(boolean married) {
        this.married = married;
    }

    // ...
}

每个属性通常要具有相应的setter、getter方法。

属性修改器以小写的set前缀开始，后跟属性名，属性名的第一个字母大写。

属性访问器以小写的get前缀开始，后跟属性名，属性名的第一个字母大写。

属性也可以只有set方法或get方法，也称只写、只读属性。

<jsp:useBean>：查找或实例化JavaBean。
<jsp:setProperty>：设置JavaBean属性。
<jsp:getProperty>：获取JavaBean属性。

<%@ page language="java" import="java.util.*" pageEncoding="UTF-8"%>
<!-- 在指定的域范围查找，存在返回引用，不存在则实例化 -->
<jsp:useBean id="person" class="xxx.Person" scope="page"/>
<%
    // person对象使用jsp:useBean标签实例化了，直接使用
    person.setName("cnm");
%>
<!-- 自动把字符串转换成8种基本数据类型，但复合数据类型无法自动转换 -->
<jsp:setProperty property="married" name="person" value="false"/>
<%--
    birthday属性是一个Date类型，无法将字符串自动转换成Date，下面写法报错
    <jsp:setProperty property="birthday" name="person" value="1999-09-09"/>
--%>
<jsp:setProperty property="birthday" name="person" value="<%=new Date()%>"/>
<!-- param="param_name"接收参数名为param_name的参数值，将接收到的值赋给name属性 -->
<jsp:setProperty property="name" name="person" param="param_name"/>
<!-- 用所有的请求参数为bean的属性赋值，property="*"代表bean的所有属性 -->
<jsp:setProperty property="*" name="person"/>

<body>
    <h2>姓名：<%=person.getName()%></h2>
    <h2>已婚：<jsp:getProperty property="married" name="person"/></h2>
    <!-- 如果JavaBean实例对象的属性的值为null，<jsp:getProperty>输出“null”字符串 -->
</body>
