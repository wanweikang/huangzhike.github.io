单例模式

// 饿汉式单例：在单例类被加载时候，就实例化一个对象交给自己的引用
class Singleton {
    // 指向自己实例的私有静态引用
    private static Singleton singleton = new Singleton();
    // 私有构造方法
    private Singleton() {}
    // 以自己实例为返回值的静态公有方法
    public static Singleton getInstance() {
        return singleton;
    }
}

// 懒汉式单例：在调用取得实例方法的时候才会实例化对象
class Singleton {
    private static Singleton singleton;
    private Singleton() {}
    public static Singleton getInstance() {
        if (instance == null) {
            synchronized (Singleton.class) {
                if (instance == null){
                    instance = new Singleton();
                }
            }
        }
        return singleton;
    }
}

Class c = Class.forName(Singleton.class.getName());
Constructor ct = c.getDeclaredConstructor();
ct.setAccessible(true);
// 每次运行都会产生新的单例对象
Singleton singleton = (Singleton)ct.newInstance();

  • 只能使用单例类提供的方法得到单例对象，不要使用反射，否则将会实例化一个新对象。

  • 不要做断开单例类对象与类中静态引用的危险操作。

  • 多线程使用单例使用共享资源时，注意线程安全问题。

工厂模式：简单工厂模式（静态工厂模式）、工厂方法模式、抽象工厂模式。

抽象工厂模式

// 产品接口
interface IProduct1 {
    public void show();
}

interface IProduct2 {
    public void show();
}
// 产品实现
class Product1 implements IProduct1 {
    public void show() {}
}

class Product2 implements IProduct2 {
    public void show() {}
}
// 工厂接口
interface IFactory {
    public IProduct1 createProduct1();
    public IProduct2 createProduct2();
}
// 工厂实现
class Factory implements IFactory {
    public IProduct1 createProduct1() {
        return new Product1();
    }

    public IProduct2 createProduct2() {
        return new Product2();
    }
}

public class Client {
    public static void main(String[] args) {
        IFactory factory = new Factory();
        factory.createProduct1().show();
        factory.createProduct2().show();
    }
}

  • 工厂方法模式提供的所有产品都是衍生自同一个接口或抽象类，而抽象工厂模式所提供的产品则是衍生自不同的接口或抽象类。

  • 对于抽象工厂模式，当减少一个方法使的提供的产品不再构成产品族之后，它就变成了工厂方法模式。

  • 扩展只需要再增加一个工厂类的实现就可以。

  • 简单工厂只有三个要素，没有工厂接口，并且得到产品的方法一般是静态的。

  • 工厂模式最终目的是解耦。

模版方法模式

购物一般需要三个步骤：登录、挑选商品、付款，对于登录与付款，每个人操作相同，但每个人挑选的商品是不同的。

某类的某个方法的实现，需要几个固定步骤，对于该类的不同对象，有些步骤是固定的，有些步骤是不同的，有些步骤是可变可不变的。

这种情况适合模板方法：定义一个操作中某种算法的框架，而将一些步骤延迟到子类中。

模板方法中：由一个抽象类和一个（或一组）实现类通过继承结构组成：

  • 模板方法：实现某种算法的方法步骤。
  • 步骤方法：完成模板方法的每个阶段性方法：
    • 抽象方法：子类必须实现。
    • 最终方法：子类不能重写的方法，所有子类执行都一样的步骤。
    • 钩子方法：父类给出了默认实现，但子类也可以重写。
abstract class ShoppingService {
    // 模版方法：抽象类声明并实现，大多会定义为final
    public void shopping() {
        login();
        choose();
        pay();
    }

    // 子类不能重写
    public final void login() {}

    // 子类必须实现
    public abstract void choose();

    // 钩子方法：子类可以重写
    public void pay() {}
}

class ShoseShoppingService extends ShoppingService {
    @Override
    public void choose() {}
}

class DressShoppingService extends ShoppingService {
    @Override
    public void choose() {}

    @Override
    public void pay() {}
}

观察者模式：如果一个对象的状态发生改变，与它相关的对象也要随之做出相应的变化。

// 被观察者：接口或抽象类
abstract class Subject {
    // 存放观察者对象，Vector线程安全，List不安全
    private Vector obs = new Vector();
    // 添加观察者对象
    public void addObserver(Observer obs) {
        this.obs.add(obs);
    }
    // 移除观察者对象
    public void delObserver(Observer obs) {
        this.obs.remove(obs);
    }
    // 依次调用观察者对象的对应方法
    protected void notifyObserver() {
        for (Observer o : obs) {
            o.update();
        }
    }

    public abstract void doSomething();
}
// 具体的观察者：定义被观察者对象状态发生变化时要处理的逻辑
class ConcreteSubject extends Subject {
    public void doSomething() {
        System.out.println("被观察者事件反生");
        this.notifyObserver();
    }
}
// 观察者
interface Observer {
    // update()：被观察者状态发生变化时，被触发
    public void update();
}
// 具体的被观察者：便于扩展，定义具体的业务逻辑
class ConcreteObserver1 implements Observer {
    public void update() {
        System.out.println("观察者1收到信息，并进行处理。");
    }
}

class ConcreteObserver2 implements Observer {
    public void update() {
        System.out.println("观察者2收到信息，并进行处理。");
    }
}

public class Client {
    public static void main(String[] args) {
        Subject sub = new ConcreteSubject();

        sub.addObserver(new ConcreteObserver1()); // 添加观察者1
        sub.addObserver(new ConcreteObserver2()); // 添加观察者2
        sub.doSomething();
    }
}

就是在Subject类中关联一下Observer类，并且在doSomething方法中遍历一下Observer的update方法就行了。

Java有Observer接口，以及实现类Observable，对观察者角色常进行了实现。

回调就是，你不要主动问我，只需要把事情注册给我（放到一个集合里），到时我会主动通知你（遍历集合干活）。

责任链模式

public void test(int i, Request request) {
    if (i == 1) {
        Handler1.response(request);
    } else if (i == 2) {
        Handler2.response(request);
    } else if (i == 3) {
        Handler3.response(request);
    } else {
        Handler4.response(request);
    }
}
// 模拟判定条件
class Level {
    private int level = 0;

    public Level(int level) {
        this.level = level;
    };

    public boolean above(Level level) {
        if (this.level >= level.level) {
            return true;
        }
        return false;
    }
}
// 请求
class Request {
    Level level;

    public Request(Level level) {
        this.level = level;
    }

    public Level getLevel() {
        return level;
    }
}
// 响应
class Response {
}
// 抽象处理类
abstract class Handler {
    // 指向下一处理类
    private Handler nextHandler;
    // 处理请求的方法，把判定条件的部分放到处理类中
    public final Response handleRequest(Request request) {
        Response response = null;
        // 如果满足条件，则本处理类处理，否则nextHandler处理
        // 只有处理类的处理等级高于Request的等级才处理，否则交给下一个
        if (this.getHandlerLevel().above(request.getLevel())) {
            response = this.response(request);
        } else {
            if (this.nextHandler != null) {
                this.nextHandler.handleRequest(request);
            } else {
                System.out.println("-----没有合适的处理器-----");
            }
        }
        return response;
    }

    public void setNextHandler(Handler handler) {
        this.nextHandler = handler;
    }

    protected abstract Level getHandlerLevel();

    public abstract Response response(Request request);
}
// 具体处理类
class ConcreteHandler1 extends Handler {
    protected Level getHandlerLevel() {
        return new Level(1);
    }

    public Response response(Request request) {
        System.out.println("-----请求由处理器1进行处理-----");
        return null;
    }
}

class ConcreteHandler2 extends Handler {
    protected Level getHandlerLevel() {
        return new Level(3);
    }

    public Response response(Request request) {
        System.out.println("-----请求由处理器2进行处理-----");
        return null;
    }
}

class ConcreteHandler3 extends Handler {
    protected Level getHandlerLevel() {
        return new Level(5);
    }

    public Response response(Request request) {
        System.out.println("-----请求由处理器3进行处理-----");
        return null;
    }
}

public class Client {
    public static void main(String[] args) {
        Handler handler1 = new ConcreteHandler1();
        Handler handler2 = new ConcreteHandler2();
        Handler handler3 = new ConcreteHandler3();

        handler1.setNextHandler(handler2);
        handler2.setNextHandler(handler3);
        // 执行时将请求交给第一个处理类
        Response response = handler1.handleRequest(new Request(new Level(4)));
    }
}

策略模式

// 抽象策略
interface IStrategy {
    // 重复的逻辑，公共代码
    public void doSomething();
}
// 具体策略
class ConcreteStrategy1 implements IStrategy {
    public void doSomething() {
        System.out.println("具体策略1");
    }
}

class ConcreteStrategy2 implements IStrategy {
    public void doSomething() {
        System.out.println("具体策略2");
    }
}
// 上下文
class Context {
    private IStrategy strategy;

    public Context(IStrategy strategy) {
        // 具体策略可以根据需要替换
        this.strategy = strategy;
    }
    // 对策略二次封装，避免高层模块对策略的直接调用
    public void execute() {
        strategy.doSomething();
    }
}

public class Client {
    public static void main(String[] args) {
        Context context;

        // 执行策略1
        context = new Context(new ConcreteStrategy1());
        context.execute();

        // 执行策略2
        context = new Context(new ConcreteStrategy2());
        context.execute();
    }
}
