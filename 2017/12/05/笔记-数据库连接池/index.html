// 实现java.sql.DateSource接口
public class JdbcPool implements DataSource {
    // LinkedList集合存放数据库链接
    private static LinkedList<Connection> list = new LinkedList<>();
    static {
        // 在静态代码块中加载db.properties配置文件
        InputStream in; 
        Properties prop;
        try {
            in = JdbcPool.class.getClassLoader().getResourceAsStream("db.properties");
            prop = new Properties();
            prop.load(in); // 加载配置文件

            String driver = prop.getProperty("driver");
            String url = prop.getProperty("url");
            String username = prop.getProperty("username");
            String password = prop.getProperty("password");

            // 数据库连接池的初始化连接数大小
            int jdbcPoolInitSize = Integer.parseInt(prop.getProperty("jdbcPoolInitSize"));

            Class.forName(driver); // 加载驱动

            // 连接池中创建Connection
            for (int i = 0; i < jdbcPoolInitSize; i++) {
                Connection conn = DriverManager.getConnection(url, username, password);
                System.out.println("获取到了连接" + conn);
                // 在DateSource构造函数或静态代码块中批量创建与数据库的连接
                // 并把创建的连接加入LinkedList中
                // LinkedList集合此时就是一个存放了数据库连接的连接池
                list.add(conn);
            }
        } catch (Exception e) {
            e.printStackTrace();
            throw new ExceptionInInitializerError(e);
        }
    }

    // getConnection方法每次调用时，从LinkedList中取一个Connection返回给用户
    @Override
    public synchronized Connection getConnection() throws SQLException {
        // 如果数据库连接池中的连接对象的个数大于0
        if (list.size() > 0) {
            // 从list集合中获取一个数据库连接
            final Connection conn = list.removeFirst(); // 删掉并返回给conn
            // 不能直接 return conn; 因为conn.close()没有把这个conn返回给连接池中
            System.out.println("池大小是" + list.size());

            // 动态代理：返回Connection对象的代理对象，拦截close方法的调用，对close进行增强
            // 第一个参数this.getClass().getClassLoader()
            // 第二个参数原来为conn.getClass.getInterface()，不过会出错
            return (Connection) Proxy.newProxyInstance(JdbcPool.class.getClassLoader(),
                    new Class[] { Connection.class }, new InvocationHandler() {
                        // 此处为内部类，当close方法被调用时将conn还回池中，其它方法直接执行
                        @Override
                        public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
                            // 调用Connection.close()方法时，Connection对象将自己返回到LinkedList中(动态代理实现)，而不要把conn还给数据库
                            if (method.getName().equals("close")) {
                                // 如果调用的是Connection对象的close方法，就把conn还给数据库连接池
                                list.add(conn);
                                System.out.println(conn + "被还到池中");
                                System.out.println("池大小为" + list.size());
                                return null;
                            }
                            // 不是调用close方法，不管就是了
                            return method.invoke(conn, args);
                        }
                    });
        } else {
            throw new RuntimeException("对不起，数据库忙");
        }
    }

}
public class JdbcUtils {

    private static JdbcPool pool = new JdbcPool();

    public static Connection getConnection() throws SQLException {
        return pool.getConnection(); // 直接从连接池中获取一个Connection
    }

    public static void release(Connection conn, Statement st, ResultSet rs) {
        if (rs != null) {
            try {
                rs.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
            rs = null;
        }
        if (st != null) {
            try {
                st.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
            st = null;
        }

        if (conn != null) {
            try {
                conn.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
            conn = null;
        }
    }

    public static void update(String sql, Object params[]) throws SQLException {
        Connection conn = null;
        PreparedStatement st = null;
        ResultSet rs = null;
        try {
            conn = getConnection();
            st = conn.prepareStatement(sql);
            for (int i = 0; i < params.length; i++) {
                st.setObject(i + 1, params[i]);
            }
            st.executeUpdate();
        } finally {
            release(conn, st, rs);
        }
    }

    public static Object query(String sql, Object params[], ResultSetHandler rsh) throws SQLException {

        Connection conn = null;
        PreparedStatement st = null;
        ResultSet rs = null;
        try {
            conn = getConnection();
            st = conn.prepareStatement(sql);
            for (int i = 0; i < params.length; i++) {
                st.setObject(i + 1, params[i]);
            }
            rs = st.executeQuery();
            // 策略模式，query方法事先不知如何处理查询结果集，只要实现了ResultSetHandler接口
            return rsh.handler(rs);
        } finally {
            release(conn, st, rs);
        }
    }
}
public interface ResultSetHandler {
    public Object handler(ResultSet rs);
}
// 将结果集转换成bean对象的处理器
public class BeanHandler implements ResultSetHandler {
    private Class<?> clazz;

    public BeanHandler(Class<?> clazz) {
        this.clazz = clazz;
    }

    public Object handler(ResultSet rs) {
        try {
            if (!rs.next()) {
                return null;
            }
            Object bean = clazz.newInstance();
            // 得到结果集元数据
            ResultSetMetaData metadata = rs.getMetaData();
            int columnCount = metadata.getColumnCount(); // 得到结果集中有几列数据
            for (int i = 0; i < columnCount; i++) {
                String coulmnName = metadata.getColumnName(i + 1); // 得到每列的列名
                Object coulmnData = rs.getObject(i + 1);
                Field f = clazz.getDeclaredField(coulmnName); // 反射出类上列名对应的属性
                f.setAccessible(true);
                f.set(bean, coulmnData);
            }
            return bean;
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}
// 将结果集转换成bean对象的list集合的处理器
public class BeanListHandler implements ResultSetHandler {
    private Class<?> clazz;

    public BeanListHandler(Class<?> clazz) {
        this.clazz = clazz;
    }

    public Object handler(ResultSet rs) {
        try {
            List<Object> list = new ArrayList<Object>();
            while (rs.next()) {
                Object bean = clazz.newInstance();
                ResultSetMetaData metadata = rs.getMetaData();
                int count = metadata.getColumnCount();
                for (int i = 0; i < count; i++) {
                    String name = metadata.getColumnName(i + 1);
                    Object value = rs.getObject(name);
                    Field f = bean.getClass().getDeclaredField(name);
                    f.setAccessible(true);
                    f.set(bean, value);
                }
                list.add(bean);
            }
            return list.size() > 0 ? list : null;
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }
}
public class AccountDao {

    public void add(Account account) throws SQLException {
        String sql = "insert into account(name, money) values(?, ?)";
        Object params[] = { account.getName(), account.getMoney() };
        JdbcUtils.update(sql, params);
    }

    public void delete(int id) throws SQLException {
        String sql = "delete from account where id = ?";
        Object params[] = { id };
        JdbcUtils.update(sql, params);
    }

    public void update(Account account) throws SQLException {
        String sql = "update account set name = ?,money = ? where id = ?";
        Object params[] = { account.getName(), account.getMoney(), account.getId() };
        JdbcUtils.update(sql, params);
    }

    public Account find(int id) throws SQLException {
        String sql = "select * from account where id = ?";
        Object params[] = { id };
        return (Account) JdbcUtils.query(sql, params, new BeanHandler(Account.class));
    }

    public List<Account> getAll() throws SQLException {
        String sql = "select * from account";
        Object params[] = {};
        return (List<Account>) JdbcUtils.query(sql, params, new BeanListHandler(Account.class));
    }
}

数据库连接池负责分配，管理和释放数据库连接，允许重复使用一个现有的数据库连接，而不是重新建立一个。

在初始化时创建一定数量的数据库连接放到连接池中，由最小数据库连接数设定，无论连接是否被使用。

当应用程序向连接池请求的连接数超过最大连接数量时，这些请求将被加入到等待队列。

DataSource的实现，称之为数据源，数据源中都包含了数据库连接池的实现。
