2017-12-13

12-28重新整理更新Hibernate笔记

集合映射：映射持久类的集合元素。

从以下类型之一声明持久类中的集合接口类型：

  • java.util.List：和<list>匹配并用java.util.ArrayList初始化。
  • java.util.Set：和<set>匹配并用java.util.HashSet初始化。
  • java.util.SortedSet：和<set>匹配并用java.util.TreeSet初始化。
  • java.util.Map：和<map>匹配并用java.util.HashMap初始化。
  • java.util.SortedMap：和<map>匹配并用java.util.TreeMap初始化。
  • java.util.Collection：和<bag>匹配并用java.util.ArrayList初始化。

集合元素可以具有值或实体引用(另一个对象)：

  • 普通值，如String
    • element
    • component-element
  • 实体引用
    • one-to-many
    • many-to-many

如果持久类有List对象，通过<list>或<bag>元素映射。

public class Question {
    private int id;
    private List<String> ans;
    private List<Answer> answers;
    // getters and setters
}

public class Answer {
    private int id;
    // ...
}

使用<list>

<class name="mmp.Question" table="q100">
    <id name="id">
        <generator class="increment"/>
    </id>
    <!-- 集合存储字符串对象 -->
    <list name="ans" table="ans100">
        <!-- key：集合外键的列名，被连接表中定义了一个外键引用原表的主键 -->
        <!-- 外键元素默认为空，对于不可空的外键，需要指定 not-null="true" -->
        <key column="qid"/>
        <!-- 集合可分为两种：索引，非索引 -->
        <!-- List和Map都可被索引，需要额外的<index> -->
        <!-- 用于存放索引的列 -->
        <index column="type"/>
        <!-- 集合的元素 -->
        <element column="answer" type="string"/>
    </list>
    <!-- 集合存储实体对象 -->
    <list name="answers" cascade="all">
        <key column="qid"/>
        <index column="type"/>
        <!-- 一对多关系映射，一个问题有很多答案 -->
        <one-to-many class="mmp.Answer"/>
    </list>
</class>

<class name="mmp.Answer" table="ans501">
    <id name="id">
        <generator class="increment"/>
    </id>
</class>

使用<bag>

<class name="mmp.Question" table="q101">
    <id name="id">
        <generator class="increment"/>
    </id>
    <bag name="ans" table="ans101">
        <key column="qid"/>
        <!-- bag不需要<index> -->
        <element column="answer" type="string"/>
    </bag>
    <!-- 略 -->
</class>

如果持久类有数组对象，通过<array>元素映射，基本是把<list>标签改成<array>就行了。

如果持久类具有Set对象，用<set>元素映射Set集合。

  • <set>元素不需要索引元素。
  • Set只存储唯一的值。
public class Question {
    private int id;
    private String qname;
    private Set<Answer> answers;
    // ...
}
public class Answer {
    private int id;
    private String answer;
    private int qid;
    private Question question;
    // ...
}
<class name="mmp.Question" table="q1002">
    <id name="id">
        <generator class="increment"/>
    </id>
    <property name="qname"/>
    <set name="answers" table="a1002" inverse="true" lazy="true" fetch="select">
        <key column="qid" not-null="true"/>
        <one-to-many class="mmp.Answer"/>
    </set>
</class>

<class name="mmp.Answer" table="a1002">
    <id name="id">
        <generator class="increment"/>
    </id>
    <property name="answer"/>
    <many-to-one name="question" class="mmp.Question" fetch="select" column="qid" not-null="true"/>
</class>
Question question1 = new Question();
question1.setQname("What is Java?");
session.save(question1);

Answer ans1 = new Answer();
ans1.setAnswer("java is a programming language");
ans1.setQuestion(question1);
// question1.getAnswers().add(ans1);
session.save(ans1);

Answer ans2 = new Answer();
ans2.setAnswer("java is a platform");
ans2.setQuestion(question1);
session.save(ans2);
List<Question> list = session.createQuery("from Question").list();

Iterator<Question> itr = list.iterator();

while (itr.hasNext()) {
    Question q = itr.next();

    Set<Answer> set = q.getAnswers();
    Iterator<Answer> itr2 = set.iterator();

    while (itr2.hasNext()) {
        System.out.println(itr2.next());
    }
}

List和Map是基于索引的集合。Map索引列作键，元素列作值。

public class Question {
    private int id;
    private String name;
    private Map<String, String> answers;
    //
}
<hibernate-mapping>
    <class name="mmp.Question" table="question_736">
        <id name="id">
            <generator class="native"/>
        </id>
        <property name="name"/>
        <map name="answers" table="answer736" cascade="all">
            <key column="questionid"/>
            <index column="answer" type="string"/>
            <element column="username" type="string"/>
        </map>
    </class>
</hibernate-mapping>
HashMap<String, String> map1 = new HashMap<>();
map1.put("java is a programming language", "John Lee");
map1.put("java is a platform", "Ashok Su");

HashMap<String, String> map2 = new HashMap<>();
map2.put("Servlet is an Interface", "Ashok Lee");
map2.put("Servlet is a package", "Rahul Su");

Question question1 = new Question("What is java?", map1);
Question question2 = new Question("What is servlet?", map2);

session.persist(question1);
session.persist(question2);
List<Question> list = session.createQuery("from Question").list();

Iterator<Question> iterator = list.iterator();

while (iterator.hasNext()) {
    Question question = iterator.next();

    Map<String, String> map = question.getAnswers();

    Set<Map.Entry<String, String>> set = map.entrySet();

    Iterator<Map.Entry<String, String>> iteratoranswer = set.iterator();

    while (iteratoranswer.hasNext()) {
        Map.Entry<String, String> entry = (Map.Entry<String, String>) iteratoranswer.next();
        System.out.println("answer:" + entry.getKey());
        System.out.println("answer posted by: " + entry.getValue());
    }
}

}

<class name="mmp.Question" table="question_m2m">

    <property name="name"/>

        <index column="answer" type="string"/>

组件映射Component Mappings

  • 作为变量的一员实体类很可能和其它类有相关关系。
  • 组件是存储为值而不是实体引用的对象。
  • 如果从属对象没有主键，则要使用此方法。
  • 它用于组合(HAS-A关系)的情况下。
public class Employee {
    private int id;
    private String name;
    private Address address;
    //
}
public class Address {
    private String country;
    //
}
<hibernate-mapping>
    <class name="mmp.Employee" table="emp_cpmap">
        <id name="id">
            <generator class="increment"/>
        </id>
        <property name="name"/>
        <component name="address" class="mmp.Address">
            <property name="country"/>
            <!-- EMPLOYEE表中将包括country字段 -->
        </component>
    </class>
</hibernate-mapping>
Employee e = new Employee("Mina Sun", new Address("China"));
