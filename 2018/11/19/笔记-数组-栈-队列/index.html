栈：

  • 后进先出（LIFO）的有序集合。

  • 新添加的或待删除的元素在栈的末尾，称作栈顶，另一端叫栈底。

  • 新元素都靠近栈顶，旧元素都接近栈底，如一堆盘子。

class Stack {
    constructor() {
        this.count = 0;
        this.items = {};
    }
    push(element) {
        this.items[this.count] = element;
        this.count++;
    }
    pop() {
        if (this.isEmpty()) {
            return undefined;
        }
        this.count--;
        const result = this.items[this.count];
        delete this.items[this.count];
        return result;
    }
    peek() {
        if (this.isEmpty()) {
            return undefined;
        }
        return this.items[this.count - 1];
    }
    isEmpty() {
        return this.count === 0;
    }
    size() {
        return this.count;
    }
    clear() {
        // while (!this.isEmpty()) {
        //     this.pop();
        // }
        this.items = {};
        this.count = 0;
    }
    toString() {
        if (this.isEmpty()) {
            return '';
        }
        let objString = `$ {
            this.items[0]
        }`;
        for (let i = 1; i < this.count; i++) {
            objString = `$ {
                objString
            },
            $ {
                this.items[i]
            }`;
        }
        return objString;
    }
}

把十进制转换成任何进制

/*
    The folow algorithm converts from base 10 to any base
*/
function baseConverter(decNumber, base) {

    let remStack = new Stack(), rem, baseString = '', digits = '0123456789ABCDEF';

    while (decNumber > 0) {
        rem = Math.floor(decNumber % base); // 获取余数
        remStack.push(rem); // 添加进去
        decNumber = Math.floor(decNumber / base); // 除后的数
    }

    while (!remStack.isEmpty()) {
        baseString += digits[remStack.pop()];
        // 十进制内可以reverse再join
    }

    return baseString;
}

    if (n > 0) {

}

虽然代码简单，但过程还是非常繁琐的，以下是我的理解：

  • 步骤一：

    • C是目标，将1~n-1号盘子放在中转站B上，方便n号底盘转移到目标柱C。

    • 如何将1n-1号盘子放在中转站B上？B成为了目标，将1n-1-1号盘子放到中转站C上，方便n-1号底盘转移到目标柱B。

    • 如何将1~n-1-1号盘子放在中转站C上？然后是不断递归。

  • 步骤二：

    • n号底盘转移到目标柱C，不再移动。

  • 步骤三：

    • n号底盘不再移动，以后都可忽略不管，n-1变成最大号盘子，此时B变为起始柱，A为中转站，C为目标柱。

    • 然后是不断递归。

步骤1和步骤3均包含2^{n-1}-1次挪动，步骤2包含1次挪动。总体是2^{n} -1次挪动

F(n) = F(n-1) + 1 + F(n-1) = 2 * F(n-1) + 1， F(1) = 1

function Fib(n) {
    if (n == 0) return 0;
    if (n == 1) return 1;
    if (n > 1) return Fib(n-1) + Fib(n-2);
}

function parenthesesChecker(symbols) {

    let stack = new Stack(), balanced = true, index = 0, symbol, top, opens = "([{", closers = ")]}";

    while (index < symbols.length && balanced) {
        symbol = symbols.charAt(index);

        if (opens.indexOf(symbol) >= 0) {
            // 获得左侧的符号
            stack.push(symbol);
        } else {
            // 轮到右侧的符号了
            if (stack.isEmpty()) {
                // 左侧没有符号
                balanced = false;
            } else {
                top = stack.pop(); // 获得左侧最右边的符号，循环下来就是跟右侧对应位置比
                // 由内到外开始
                if (!(opens.indexOf(top) === closers.indexOf(symbol))) {
                    balanced = false;
                }
            }
        }
        index++;
    }
    return balanced && stack.isEmpty();
}

队列：

  • 遵循FIFO（First In First Out，先进先出）原则的一组有序的项。

  • 队列在尾部添加新元素，并从顶部移除元素。最新添加的元素必须排在末尾，如排队。

class Queue {
    constructor() {
        this.count = 0;
        this.lowestCount = 0;
        this.items = {};
    }
    // 向队列尾部添加一个或多个新的项
    enqueue(element) {
        this.items[this.count] = element;
        this.count++;
    }
    // 移除队列第一项，并返回被移除的元素
    dequeue() {
        if (this.isEmpty()) {
            return undefined;
        }
        const result = this.items[this.lowestCount];
        delete this.items[this.lowestCount];
        this.lowestCount++;
        return result;
    }
    // 返回队列第一个元素，最先被添加，最先被移除
    peek() {
        if (this.isEmpty()) {
            return undefined;
        }
        return this.items[this.lowestCount];
    }

    isEmpty() {
        return this.size() === 0;
    }

    clear() {
        this.items = {};
        this.count = 0;
        this.lowestCount = 0;
    }

    size() {
        return this.count - this.lowestCount;
    }

    toString() {
        if (this.isEmpty()) {
            return '';
        }
        let objString = `$ {
            this.items[this.lowestCount]
        }`;
        for (let i = this.lowestCount + 1; i < this.count; i++) {
            objString = `$ {
                objString
            },
            $ {
                this.items[i]
            }`;
        }
        return objString;
    }
}

优先队列

function PriorityQueue() {

    let items = [];

    // 设置元素及其优先级，0为最高，以此类推
    function QueueElement(element, priority) {
        this.element = element;
        this.priority = priority;
    }

    this.enqueue = function(element, priority) {
        let queueElement = new QueueElement(element, priority);

        let added = false;
        // 添加前循环所有已有项
        for (let i = 0; i < items.length; i++) {
            // 比添加项优先高
            if (queueElement.priority < items[i].priority) {
                items.splice(i, 0, queueElement); // 添加到它后面
                added = true;
                break;
            }
        }
        // 没有找到更低优先级的项，直接push末尾
        if (!added) {
            items.push(queueElement);
        }
    };

    this.dequeue = function() {};

    this.front = function() {};

    this.isEmpty = function() {};

    this.size = function() {};

    this.print = function() {
        for (let i = 0; i < items.length; i++) {
            console.log(`$ {items[i].element} - $ {items[i].priority}`);
        }
    };
}

function hotPotato(elementsList, num) {
    const queue = new Queue();
    const elimitatedList = [];

    for (let i = 0; i < elementsList.length; i++) {
        queue.enqueue(elementsList[i]);
    }

    while (queue.size() > 1) {
        // 进行num次传递
        for (let i = 0; i < num; i++) {
            queue.enqueue(queue.dequeue()); // 从队列开头移除一项，再将其添加到队列末尾
        }
        elimitatedList.push(queue.dequeue());
    }

    return {
        eliminated: elimitatedList,
        winner: queue.dequeue()
    };
}

let names = ['John', 'Jack', 'Camila', 'Ingrid', 'Carl'];
let winner = hotPotato(names, 7);
console.log('The winner is: ' + winner);
