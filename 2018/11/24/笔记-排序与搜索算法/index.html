冒泡排序：比较任何两个相邻的项，如果第一个比第二个大，则交换它们。元素项向上移动像气泡一样。

export function bubbleSort(array) {
    const {
        length
    } = array;
    // 外循环从数组的第一位迭代至最后一位，数组中每项都经过一轮
    // 第一轮确保最大（小）的被排序，第二轮确保第二大（小）的被排序，以此类推
    // 每一轮排序，该次比较的最后一个位置上，正确的数已确定
    for (let i = 0; i < length; i++) {
        // 内循环从0迭代至length - 2，比较当前项和下一项
        for (let j = 0; j < length - 1; j++) {
            array[j] > array[j + 1] && swap(array, j, j + 1);
        }
    }
    return array;
}
export function modifiedBubbleSort(array) {
    const {
        length
    } = array;
    for (let i = 0; i < length; i++) {
        // 从内循环减去外循环中已跑过的轮数
        for (let j = 0; j < length - 1 - i; j++) {
            array[j] > array[j + 1] && swap(array, j, j + 1);
        }
    }
    return array;
}

选择排序：原址比较排序，找到数据结构中的最小值并将其放置在第一位，接着找到第二小的值并将其放在第二位，以此类推。

export const selectionSort = (array) => {
    const {length} = array;
    let indexMin;
    for (let i = 0; i < length - 1; i++) {
        // 假设本迭代轮次的第一个值为数组最小值
        indexMin = i;
        for (let j = i; j < length; j++) {
            if (array[indexMin] > array[j]) {
                indexMin = j;
            }
        }
        // 得出数组第n小的值
        // 如果该最小值和原最小值不同，则交换其值
        i !== indexMin && swap(array, i, indexMin);
    }
    return array;
};

插入排序：假定第一项已经排序了，接着它和第二项进行比较，第二项是应该待在原位还是插到第一项之前呢？这样，头两项就已正确排序，接着和第三项比较（它是该插入到第一、第二还是第三的位置呢？），以此类推。

export const insertionSort = (array) => {
    const {length} = array;
    let temp;
    // 从第二个位置（ 1 ）而不是 0 开始（默认第一项已排序了）
    for (let i = 1; i < length; i++) {
        let j = i;
        temp = array[i]; // 中间变量
        // 找到正确的位置来插入项目
        // 只要变量 j 比 0 大（没有负值的索引）并且前一个比temp大
        while (j > 0 && array[j - 1] > temp) {
            // 把这个值后移覆盖
            array[j] = array[j - 1];
            j--;
        }
        array[j] = temp;
    }
    return array;
};

归并排序：分治算法，将原始数组切分成小数组，直到每个小数组只有一个元素，接着将小数组归并成较大的数组，归并过程完成排序，直到最后只有一个排序完毕的大数组。由于是分治法，归并排序也是递归的。

function merge(left, right) {
    let l = 0;
    let r = 0;
    const result = []; // 归并创建的新数组

    while (l < left.length && r < right.length) {
        // 如果 left 数组的项比 right 数组的项小，将该项从 left 数组添加至数组，并递增
        // 否则，从 right 数组添加项并递增
        result.push(left[l] < right[r] ? left[l++] : right[r++]);
    }
    // 将 left 或 right 数组所有剩余的项添加到归并数组中
    return result.concat(l < left.length ? left.slice(l) : right.slice(r));
}

export function mergeSort(array) {
    // 如果数组长度比 1 大，将其分成小数组
    if (array.length > 1) {
        const {length} = array;
        // 找到数组的中间位
        const middle = Math.floor(length / 2);
        // 将数组分成两个小数组
        const left = mergeSort(array.slice(0, middle));
        const right = mergeSort(array.slice(middle, length));
        // 合并和排序小数组来产生大数组
        array = merge(left, right);
    }
    return array;
}

快速排序：使用分治的方法，将原始数组分为较小的数组（但没有像归并排序那样将它们分割开）。

  • 首先，从数组中选择中间一项作为主元。

  • 创建两个指针，左边一个指向数组第一个项，右边一个指向数组最后一个项。

    • 移动左指针直到找到一个比主元大的元素，接着，移动右指针直到找到一个比主元小的元素，然后交换它们，重复这个过程，直到左指针超过了右指针。

    • 这个过程将使得比主元小的值都排在主元之前，而比主元大的值都排在主元之后。

  • 接着，对划分后的小数组（较主元小的值组成的子数组，以及较主元大的值组成的子数组）重复之前的两个步骤，直至数组已完全排序。

function partition(array, left, right) {
    // 选择主元，其实随便选一个就行
    const pivot = array[Math.floor((right + left) / 2)];
    let i = left;
    let j = right;
    // 只要 left 和 right 指针没有相互交错，就执行划分操作
    while (i <= j) {
        // 移动 left 指针直到找到一个元素比主元大
        while (array[i] < pivot) {
            i++;
        }
        // 移动 right 指针直到找到一个元素比主元小
        while (array[j] > pivot) {
            j--;
        }
        // 当左指针指向的元素比主元大且右指针指向的元素比主元小，并且左指针<=右指针
        if (i <= j) {
            // 左项比右项大，交换它们
            swap(array, i, j);
            // 移动两个指针，并重复此过程
            i++;
            j--;
        }
    }
    // 划分操作结束，返回左指针的索引
    return i;
}

function quick(array, left, right) {
    let index;
    // 如果数组的长度比1大（只有一个元素的数组必然是已排序了的）
    // 对给定子数组执行 partition 操作（第一次调用是针对整个数组）以得到 index
    if (array.length > 1) {
        index = partition(array, left, right);
        // 如果子数组存在较小值，重复
        left < index - 1 && quick(array, left, index - 1);
        // 如果子数组存在较大值，重复
        index < right && quick(array, index, right);
    }
    return array;
}

export function quickSort(array) {
    // 传递待排序数组，索引0及其最末的位置（因为要排整个数组，而不是子数组）
    return quick(array, 0, array.length - 1);
}

堆排序：把数组当作二叉树来排序。

function heapify(array, index, heapSize) {
    let largest = index;
    const left = (2 * index) + 1;
    const right = (2 * index) + 2;
    if (left < heapSize && array[left] > array[index]) {
        largest = left;
    }
    if (right < heapSize && array[right] > array[largest]) {
        largest = right;
    }
    if (largest !== index) {
        swap(array, index, largest);
        heapify(array, largest, heapSize);
    }
}

function buildMaxHeap(array) {
    for (let i = Math.floor(array.length / 2); i >= 0; i -= 1) {
        heapify(array, i, array.length);
    }
    return array;
}

export default function heapSort(array) {
    let heapSize = array.length;
    // 构造一个满足 array[parent(i)] ≥ array[i] 的堆结构
    buildMaxHeap(array);
    while (heapSize > 1) {
        // 交换堆里第一个元素（数组中较大的值）和最后一个元素的位置
        // 最大的值就会出现在它已排序的位置
        swap(array, 0, --heapSize);
        // 再次将数组转换成堆，找到当前堆的根节点（较小的值），重新放到树的底部
        heapify(array, 0, heapSize);
    }
    return array;
}

大顶堆：每个节点的值都大于或等于其子节点的值，用于升序排列。

小顶堆：每个节点的值都小于或等于其子节点的值，用于降序排列。

[3, 5, 1, 6, 4, 7, 2]创建堆：

  • 索引0是树的根节点；

  • 除根节点外，任意节点N的父节点是N/2；

  • 节点L的左子节点是2*L；

  • 节点R的右子节点是2*R+1。

export function defaultEquals(a, b) {
    return a === b;
}

顺序搜索：将每一个数据结构中的元素和要找的元素做比较。

export function sequentialSearch(array, value, equalsFn = defaultEquals) {
    for (let i = 0; i < array.length; i++) {
        if (equalsFn(value, array[i])) {
            return i;
        }
    }
    return -1;
}

二分搜索：要求被搜索的数据结构已排序。

  • 选择数组的中间值。

  • 如果选中值是待搜索值，完毕（值找到了）。

  • 如果待搜索值比选中值要小，则返回第一步并在选中值左边的子数组中寻找。

  • 如果待搜索值比选中值要大，则返回第一步并在选种值右边的子数组中寻找。

export function binarySearch(array, value) {
    const sortedArray = quickSort(array);
    let low = 0;
    let high = sortedArray.length - 1;
    while (low <= high) {
        const mid = Math.floor((low + high) / 2);
        const element = sortedArray[mid];
        if (element < value) {
            low = mid + 1;
        } else if (element > value) {
            high = mid - 1;
        } else {
            return mid;
        }
    }
    return -1;
}
function binarySearchRecursive(array, value, low, high) {
    if (low <= high) {
        const mid = Math.floor((low + high) / 2);
        const element = array[mid];
        if (element < value) {
            return binarySearchRecursive(array, value, mid + 1, high);
        }
        if (element > value) {
            return binarySearchRecursive(array, value, low, mid - 1);
        }
        return mid;
    }
    return -1;
}

export function binarySearch(array, value) {
    const sortedArray = quickSort(array);

    return binarySearchRecursive(array, value, 0, sortedArray.length - 1);
}

如果没有停止函数递归调用的边界条件，递归并不会无限地执行下去，浏览器会抛出栈溢出错误（stack overflow error）。

ECMAScript6有尾调用优化（tail call optimization）。

如果函数内最后一个操作是调用函数，会通过“跳转指令”（jump）而不是“子程序调用”（subroutine call）来控制。

在ES6中，代码可以一直执行下去。

尾调用：函数的返回值为纯函数调用，即函数最终的返回值与它返回的调用函数的返回值完全一致，要求函数最后一句话是返回调用某个函数值，且该函数调用不出现在表达式中。

尾递归：尾调用函数最后调用的是它自身，特殊的尾调用。

// 普通线性递归写法
function fib(n) {
    if (n <= 2) {
        return 1;
    }
    return fib(n - 1) + fib(n - 2);
}
console.log(fib(3));
console.log(fib(50));

// 尾递归的写法，此时当递归层级深时执行速度比上面的快很多
function fibtail(n, acc1, acc2) {
    if (n < 2) {
        return acc1;
    }
    return fibtail(n - 1, acc2, acc1 + acc2);
}
console.log(fibtail(3, 1, 1));
console.log(fibtail(50, 1, 1));
