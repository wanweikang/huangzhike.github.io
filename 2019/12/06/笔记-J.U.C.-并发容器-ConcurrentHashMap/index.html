ConcurrentHashMap

public class ConcurrentHashMap<K, V> extends AbstractMap<K, V> implements ConcurrentMap<K, V>, Serializable {

    /**
     * 数组
     */
    transient volatile ConcurrentHashMap.Node<K, V>[] table;

    /**
     * 暂存扩容中的新数组，扩容完成后用该数组替代
     */
    private transient volatile ConcurrentHashMap.Node<K, V>[] nextTable;

    private transient volatile long baseCount;
    /**
     * volatile修饰，修改立即可见，为什么不使用Atomic类，还要Unsafe手动CAS，可能是性能更好？
     * 用于数组初始化和扩容的控制，低16位代表正在扩容的线程数加一
     * 0为初始值，表示数组未初始化
     * 负数的情况：
     * 1（低16位）表示正在初始化，只有一个线程可以执行初始化，其它Thread#yield
     * n（低16位）表示有n-1个线程正在扩容
     * 正数的情况：
     * 如果数组未初始化，表示数组初始化后的大小
     * 如果数组初始化完成，表示下次扩容的阈值，Map的元素数量超过后就会扩容
     */
    private transient volatile int sizeCtl;

    /**
     * 并发扩容时，数组已被分配的最新索引
     */
    private transient volatile int transferIndex;

    private transient volatile int cellsBusy;

    private transient volatile ConcurrentHashMap.CounterCell[] counterCells;
}

ConcurrentHashMap#putVal

    final V putVal(K key, V value, boolean onlyIfAbsent) {
        // 键值都不允许为空
        // 百度了一下，对于value不允许为null，是因为如果get返回了null，那么无法判断是key不存在还是value就是null，在并发的情况下，如果再次调用containsKey判断，可能在这期间Map已经发生了改变
        // 对于key不允许为null，还没找到合理的解释，无法判断key为null或不存在？
        if (key == null || value == null)
            throw new NullPointerException();
        // 计算哈希值，会保证哈希值为正
        int hash = spread(key.hashCode());
        int binCount = 0;
        // 重试直到成功
        for (ConcurrentHashMap.Node<K, V>[] tab = table; ; ) {
            ConcurrentHashMap.Node<K, V> f;
            int n, i, fh;
            // 第一次put，初始化数组，所以数组是延迟初始化的
            if (tab == null || (n = tab.length) == 0)
                tab = initTable();
                // 如果数组对应位置无节点，CAS尝试插入，成功就退出，否则下个循环继续
                // tabAt实际是调用的Unsafe#getObjectVolatile直接获取指定位置的数据
                // 因为虽然table是volatile的，但是table中的元素不是volatile的，不能保证线程每次都拿到最新元素
            else if ((f = tabAt(tab, i = (n - 1) & hash)) == null) {
                // 使用Unsafe#compareAndSwapObject，原子操作table[i]位置，如果为null，则插入，不需要锁
                if (casTabAt(tab, i, null, new ConcurrentHashMap.Node<K, V>(hash, key, value, null)))
                    break;
            }
            // ForwardingNode节点，是一个标记，说明有其它线程在扩容，而原本的节点被搬到了扩容后的新数组
            else if ((fh = f.hash) == MOVED)
                // 协助扩容，直到扩容完成才能插入新数组
                tab = helpTransfer(tab, f);
                // 有哈希冲突的情况
            else {
                V oldVal = null;
                // 加锁，没有用ReentrantLock是因为JDK8下synchronized性能已经足够好，推荐哦
                // 这里只锁住了一条链表或树，缩小了锁的范围
                synchronized (f) {
                    // 再次检查一次该位置加锁的节点，DCL
                    if (tabAt(tab, i) == f) {
                        // 哈希值大于零，说明是链表节点
                        if (fh >= 0) {
                            binCount = 1;
                            // 遍历链表
                            for (ConcurrentHashMap.Node<K, V> e = f; ; ++binCount) {
                                K ek;
                                // 找到了key对应的节点，然后决定是否覆盖更新
                                if (e.hash == hash && ((ek = e.key) == key || (ek != null && key.equals(ek)))) {
                                    oldVal = e.val;
                                    if (!onlyIfAbsent)
                                        e.val = value;
                                    break;
                                }
                                ConcurrentHashMap.Node<K, V> pred = e;
                                // 到了链表尾还没找到
                                if ((e = e.next) == null) {
                                    // 追加到链表尾
                                    pred.next = new ConcurrentHashMap.Node<K, V>(hash, key, value, null);
                                    break;
                                }
                            }
                        }
                        // 红黑树以TreeBin形式存放，哈希值为TREEBIN，即-2
                        else if (f instanceof ConcurrentHashMap.TreeBin) {
                            ConcurrentHashMap.Node<K, V> p;
                            binCount = 2;
                            // 红黑树中找到节点就返回，没有找到就直接追加，这里插入新节点后红黑树重平衡还会另外加锁
                            if ((p = ((ConcurrentHashMap.TreeBin<K, V>) f).putTreeVal(hash, key, value)) != null) {
                                oldVal = p.val;
                                if (!onlyIfAbsent)
                                    p.val = value;
                            }
                        }
                    }
                }
                if (binCount != 0) {
                    // 链表过长，就转换为红黑树
                    if (binCount >= TREEIFY_THRESHOLD)
                        // 如果数组容量太小，则不转化，优先扩容，否则DCL对头节点加锁转化为红黑树
                        treeifyBin(tab, i);
                    if (oldVal != null)
                        return oldVal;
                    break;
                }
            }
        }
        // 插入完成后，更新Map大小，检查数组是否需要扩容
        addCount(1L, binCount);
        return null;
    }

ConcurrentHashMap#initTable

    private final ConcurrentHashMap.Node<K, V>[] initTable() {
        ConcurrentHashMap.Node<K, V>[] tab;
        int sc;
        // 数组还没初始化
        while ((tab = table) == null || tab.length == 0) {
            // 小于0说明有别的线程在初始化或扩容数组，但还没完成，那么当前线程就让一让，避免并发修改
            if ((sc = sizeCtl) < 0)
                Thread.yield();
                // CAS将sizeCtl改为-1成功才能初始化，类似获得了锁，避免多次初始化
            else if (U.compareAndSwapInt(this, SIZECTL, sc, -1)) {
                try {
                    // 再次检查数组有没有初始化，类似DCL，如两个线程同时进入了while循环的代码，然后线程1在执行CAS前被挂起，线程2完成了数组初始化，然后线程1唤醒，执行CAS，需要再次判断
                    if ((tab = table) == null || tab.length == 0) {
                        // 数组的初始容量，有设置就使用设置的，否则使用默认
                        int n = (sc > 0) ? sc : DEFAULT_CAPACITY;
                        @SuppressWarnings("unchecked") ConcurrentHashMap.Node<K, V>[] nt = (ConcurrentHashMap.Node<K, V>[]) new ConcurrentHashMap.Node<?, ?>[n];
                        // 用新的数组替换
                        table = tab = nt;
                        // sc == n - (n / 2^2) == 0.75 * n
                        sc = n - (n >>> 2);
                    }
                } finally {
                    // 扩容完成后将sizeCtl更新为下次扩容的阈值
                    // 这里没有竞争，因为数组只会初始化一次，不需要CAS
                    sizeCtl = sc;
                }
                break;
            }
        }
        return tab;
    }

ConcurrentHashMap$TreeBin#lockRoot

        private final void lockRoot() {
            // CAS将红黑树的锁状态由空闲改为写才能执行重平衡，否则说明有线程在读或写这棵红黑树，自旋重试
            if (!U.compareAndSwapInt(this, LOCKSTATE, 0, WRITER))
                contendedLock();
        }

ConcurrentHashMap$TreeBin#contendedLock

        private final void contendedLock() {
            boolean waiting = false;
            // 自旋重试直到成功
            for (int s; ; ) {
                // 如果当前树仅处于等待状态，再次CAS尝试，失败则继续，成功则返回
                if (((s = lockState) & ~WAITER) == 0) {
                    // 尝试CAS修改为写状态，成功则表明获取写锁成功，否则说明状态发生了改变或者有竞争，下次自旋继续
                    if (U.compareAndSwapInt(this, LOCKSTATE, s, WRITER)) {
                        // 如果上次自旋后要进入等待状态，现在已经不需要了，置空等待线程并返回
                        if (waiting)
                            waiter = null;
                        return;
                    }
                }
                // 如果当前树不是等待状态，即没有等待线程，尝试修改为等待状态，准备把当前线程挂起
                else if ((s & WAITER) == 0) {
                    // CAS添加等待状态成功
                    if (U.compareAndSwapInt(this, LOCKSTATE, s, s | WAITER)) {
                        // 准备下个循环把线程挂起，当然也可能下次自旋CAS成功
                        waiting = true;
                        // 将当前线程设为等待线程，其它的就CAS竞争去吧，只能有一个被park
                        waiter = Thread.currentThread();
                    }
                }
                // 挂起当前写线程，等待读线程完成后将它唤醒
                else if (waiting)
                    LockSupport.park(this);
            }
        }

ConcurrentHashMap#treeifyBin

    private final void treeifyBin(ConcurrentHashMap.Node<K, V>[] tab, int index) {
        ConcurrentHashMap.Node<K, V> b;
        int n, sc;
        if (tab != null) {
            // 数组容量过小，则扩容
            if ((n = tab.length) < MIN_TREEIFY_CAPACITY)
                // 扩容，容量翻倍
                tryPresize(n << 1);
                // 否则，如果该位置有元素，并且是个链表节点
            else if ((b = tabAt(tab, index)) != null && b.hash >= 0) {
                // 锁定该头节点
                synchronized (b) {
                    // DCL
                    if (tabAt(tab, index) == b) {
                        ConcurrentHashMap.TreeNode<K, V> hd = null, tl = null;
                        // 遍历链表，转化为红黑树节点组成的链表
                        for (ConcurrentHashMap.Node<K, V> e = b; e != null; e = e.next) {
                            ConcurrentHashMap.TreeNode<K, V> p = new ConcurrentHashMap.TreeNode<K, V>(e.hash, e.key, e.val, null, null);
                            if ((p.prev = tl) == null)
                                hd = p;
                            else
                                tl.next = p;
                            tl = p;
                        }
                        // 和HashMap不一样，插入的是TreeBin，代表了这棵红黑树，不是TreeNode，TreeBin的构造函数内包含了红黑树的转化过程
                        // 应该是为了方便加锁，因为红黑树重平衡后根节点可能会变的，不好直接对根节点加锁
                        setTabAt(tab, index, new ConcurrentHashMap.TreeBin<K, V>(hd));
                    }
                }
            }
        }
    }

ConcurrentHashMap#tryPresize

    private final void tryPresize(int size) {
        // 计算扩容后的数组大小，最小的2的幂
        int c = (size >= (MAXIMUM_CAPACITY >>> 1)) ? MAXIMUM_CAPACITY : tableSizeFor(size + (size >>> 1) + 1);
        int sc;
        while ((sc = sizeCtl) >= 0) {
            ConcurrentHashMap.Node<K, V>[] tab = table;
            int n;
            // 数组还没初始化，那就先初始化数组，下个循环再检测是否需要扩容
            if (tab == null || (n = tab.length) == 0) {
                // 新数组大小
                n = (sc > c) ? sc : c;
                // CAS尝试获取锁，尝试更新sizeCtl为-1，数组正在初始化
                if (U.compareAndSwapInt(this, SIZECTL, sc, -1)) {
                    try {
                        // DCL，再次检查数组是否已经初始化过了
                        if (table == tab) {
                            @SuppressWarnings("unchecked") ConcurrentHashMap.Node<K, V>[] nt = (ConcurrentHashMap.Node<K, V>[]) new ConcurrentHashMap.Node<?, ?>[n];
                            // 更新数组
                            table = nt;
                            // 扩容阈值为数组大小的0.75倍
                            sc = n - (n >>> 2);
                        }
                    } finally {
                        sizeCtl = sc;
                    }
                }
            }
            // 数组还没达到扩容的阈值或者超过了数组的最大容量阈值那就跳出
            else if (c <= sc || n >= MAXIMUM_CAPACITY)
                break;
                // 扩容前要再次检查当前要扩容的数组是否还是原本的数组，因为扩容是并发的，可能此时扩容已经完成了，大概是线程挂起再唤醒的情况
            else if (tab == table) {
                int rs = resizeStamp(n);
                // 数组正在扩容
                if (sc < 0) {
                    ConcurrentHashMap.Node<K, V>[] nt;
                    // 参考addCount的注释，总之就是不需要新的线程参与扩容了
                    if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 || sc == rs + MAX_RESIZERS || (nt = nextTable) == null || transferIndex <= 0)
                        break;
                    // 否则，CAS尝试更新sizeCtl，协助扩容的线程数加1，成功才能参与扩容
                    if (U.compareAndSwapInt(this, SIZECTL, sc, sc + 1))
                        transfer(tab, nt);
                }
                // 数组还没扩容，那当前线程就是发起扩容的线程，CAS更新sizeCtl成功才能扩容
                else if (U.compareAndSwapInt(this, SIZECTL, sc, (rs << RESIZE_STAMP_SHIFT) + 2))
                    transfer(tab, null);
            }
        }
    }

ConcurrentHashMap#addCount

    private final void addCount(long x, int check) {
        ConcurrentHashMap.CounterCell[] as;
        long b, s;
        // 如果counterCells不为null，说明存在并发修改 || 如果CAS更新baseCount失败，说明存在并发修改
        // 即是说，优先CAS尝试修改baseCount，在存在并发修改的情况下，避免集中的CAS操作，而是将修改分散地暂存在counterCells
        if ((as = counterCells) != null || !U.compareAndSwapLong(this, BASECOUNT, b = baseCount, s = b + x)) {
            ConcurrentHashMap.CounterCell a;
            long v;
            int m;
            boolean uncontended = true;
            // 如果CounterCell数组不存在 || 如果CounterCell数组长度为0 || 该线程的哈希定位到CounterCell数组的位置为空 || 该位置不为空，并且CAS更新该CounterCell失败
            if (as == null || (m = as.length - 1) < 0 || (a = as[ThreadLocalRandom.getProbe() & m]) == null || !(uncontended = U.compareAndSwapLong(a, CELLVALUE, v = a.value, v + x))) {
                // 尝试自旋更新CounterCell，uncontended为false
                fullAddCount(x, uncontended);
                // 成功就返回，也不继续检查要不要扩容了
                return;
            }
            // 说明是被删除方法调用，直接返回
            if (check <= 1)
                return;
            // 否则说明是被添加方法调用，需要计算元素个数，检查是否需要扩容，另外Map没有自动缩容的功能
            s = sumCount();
        }
        // 如果需要检查是否需要扩容
        if (check >= 0) {
            ConcurrentHashMap.Node<K, V>[] tab, nt;
            int n, sc;
            // 元素数量超出扩容阈值 && 数组不为空 && 数组容量小于最大限制
            // 那么就扩容
            while (s >= (long) (sc = sizeCtl) && (tab = table) != null && (n = tab.length) < MAXIMUM_CAPACITY) {
                // 当前数组的长度的二进制值中开头0的数量
                int rs = resizeStamp(n);
                // 说明正在扩容
                if (sc < 0) {
                    /*
                     * 扩容已经结束的5种情况，退出循环：
                     * sizeCtl无符号右移16位，即高16位不等于rs，说明扩容已经结束，sizeCtl已经被更新
                     * 第一个开始扩容的线程会将sizeCtl无符号左移16位再加2，即所有线程退出时sc == rs + 1
                     * 大概是扩容的线程数已经到了65535，不能再增加扩容的线程了，因为低16位已经溢出了
                     * nextTable为空说明扩容已经完成，并且新数组已经更新
                     * transferIndex <= 0说明原本数组全部位置已经分配好线程了，不需要新的线程参与
                     */
                    if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 || sc == rs + MAX_RESIZERS || (nt = nextTable) == null || transferIndex <= 0)
                        break;
                    // 否则说明扩容未完成，需要线程协助，如果CAS将sizeCtl加1成功，则帮忙进行扩容，加1是代表参与扩容的线程增加
                    if (U.compareAndSwapInt(this, SIZECTL, sc, sc + 1))
                        // 帮助扩容，nextTable是扩容后的新表
                        transfer(tab, nt);
                }
                /*
                 * 当前没有扩容，说明是第一个扩容的线程，那么CAS尝试更新sizeCtl，成功了就扩容
                 * 这里把rs有符号左移16位，那么最高位就是符号位1，是个负数，高位剩下的15位则保存了当前数组的长度的二进制值中开头0的数量，因为数组长度是2的指数幂，所以二进制值肯定是0...010...0这样，所以只要知道开头几个0就行了
                 * 然后加2，所以最开始扩容时是加2的，扩容完成后是减1，即sizeCtl == rs + 1
                 * 假设当前数组容量为16，那么resizeStamp方法返回rs = 0000 0000 0000 0000 1000 0000 0001 0000
                 * resizeStamp(16) << RESIZE_STAMP_SHIFT) + 2 == 10000000000100000000000000000010，表示有2-1=1个线程正在扩容
                 */
                else if (U.compareAndSwapInt(this, SIZECTL, sc, (rs << RESIZE_STAMP_SHIFT) + 2))
                    // 发起扩容，迁移数据
                    transfer(tab, null);
                // 再次统计元素个数，是否还要继续扩容，可能扩容期间有新的元素添加进来
                s = sumCount();
            }
        }
    }

    private final void fullAddCount(long x, boolean wasUncontended) {
        int h;
        // 说明该线程第一次进入
        if ((h = ThreadLocalRandom.getProbe()) == 0) {
            // 先初始化该线程的ThreadLocalRandom
            ThreadLocalRandom.localInit();
            // 线程的随机数，可以理解为本次的哈希值
            h = ThreadLocalRandom.getProbe();
            wasUncontended = true;
        }
        boolean collide = false;
        // 自旋重试
        for (; ; ) {
            ConcurrentHashMap.CounterCell[] as;
            ConcurrentHashMap.CounterCell a;
            int n;
            long v;
            // 如果CounterCell数组已初始化
            if ((as = counterCells) != null && (n = as.length) > 0) {
                // 如果哈希定位到的位置没有CounterCell
                if ((a = as[(n - 1) & h]) == null) {
                    // 如果当前的CounterCell数组不存在竞争
                    if (cellsBusy == 0) {
                        // 创建一个CounterCell
                        ConcurrentHashMap.CounterCell r = new ConcurrentHashMap.CounterCell(x);
                        // DCL && CAS将cellsBusy由空闲修改为忙碌状态成功，类似获取锁
                        if (cellsBusy == 0 && U.compareAndSwapInt(this, CELLSBUSY, 0, 1)) {
                            boolean created = false;
                            try {
                                ConcurrentHashMap.CounterCell[] rs;
                                int m, j;
                                // DCL，再次检查
                                if ((rs = counterCells) != null && (m = rs.length) > 0 && rs[j = (m - 1) & h] == null) {
                                    // 把创建的CounterCell加入数组
                                    rs[j] = r;
                                    // 标记操作为成功
                                    created = true;
                                }
                            } finally {
                                // 重置为空闲状态
                                cellsBusy = 0;
                            }
                            // 如果操作成功，不再继续，否则下个循环继续尝试
                            if (created)
                                break;
                            continue;
                        }
                    }
                    collide = false;
                }
                /*
                 * 第一次进来时，如果发生了哈希冲突，首先进来这里，然后到下面换一个哈希值尝试别的位置，而不是尝试CAS
                 * 如果第二次循环还是有哈希冲突，才尝试CAS更新CounterCell，失败则检查是否扩容数组，不能扩容则换一个哈希下个循环继续
                 */
                // 到这里说明发生了哈希冲突
                // contended表示竞争，双重否定就是肯定
                else if (!wasUncontended)
                    // 标志为无竞争，然后不会尝试CAS更新，而是会跳到最下面，换一个哈希值，重新执行循环
                    wasUncontended = true;
                    // 尝试CAS更新该CounterCell的值，如果成功则不需继续
                else if (U.compareAndSwapLong(a, CELLVALUE, v = a.value, v + x))
                    break;
                    // CAS更新失败，再次检查CounterCell数组是否已经扩容或容量达到最大值，即CPU核心数
                else if (counterCells != as || n >= NCPU)
                    // 不能扩容数组，那么就标记冲突为false，继续匹配下个分支，不再匹配扩容分支，然后到下面换一个哈希值
                    collide = false;
                    // 如果冲突标记为false，那么设置为true，下面就换一个哈希，继续下一轮循环判断，只有为true才会尝试扩容
                else if (!collide)
                    collide = true;
                    // CAS尝试加锁，然后才能扩容CounterCell数组
                else if (cellsBusy == 0 && U.compareAndSwapInt(this, CELLSBUSY, 0, 1)) {
                    try {
                        // 检查是否已被扩容
                        if (counterCells == as) {
                            // CounterCell数组扩容，翻倍
                            ConcurrentHashMap.CounterCell[] rs = new ConcurrentHashMap.CounterCell[n << 1];
                            // 数据迁移
                            for (int i = 0; i < n; ++i)
                                rs[i] = as[i];
                            counterCells = rs;
                        }
                    } finally {
                        cellsBusy = 0;
                    }
                    collide = false;
                    continue;
                }
                // 如果上面的条件都不满足，那么更新当前线程的哈希值，下个循环尝试别的位置
                h = ThreadLocalRandom.advanceProbe(h);
            }
            // 如果CounterCell数组未初始化，尝试CAS将cellsBusy由0更新为1，成功才能初始化
            else if (cellsBusy == 0 && counterCells == as && U.compareAndSwapInt(this, CELLSBUSY, 0, 1)) {
                boolean init = false;
                try {
                    // 类似DCL的双重检查，反复出现的老套路了
                    if (counterCells == as) {
                        // 初始化CounterCell数组，初始容量为2
                        ConcurrentHashMap.CounterCell[] rs = new ConcurrentHashMap.CounterCell[2];
                        // 根据哈希插入CounterCell
                        rs[h & 1] = new ConcurrentHashMap.CounterCell(x);
                        counterCells = rs;
                        init = true;
                    }
                } finally {
                    cellsBusy = 0;
                }
                if (init)
                    break;
            }
            // 说明当前CounterCell数组存在竞争初始化，那么CAS尝试更新baseCount
            else if (U.compareAndSwapLong(this, BASECOUNT, v = baseCount, v + x))
                break;
        }
    }

    final ConcurrentHashMap.Node<K, V>[] helpTransfer(ConcurrentHashMap.Node<K, V>[] tab, ConcurrentHashMap.Node<K, V> f) {
        ConcurrentHashMap.Node<K, V>[] nextTab;
        int sc;
        // 数组不为空 && 该位置第一个节点为ForwardingNode && ForwardingNode指向的当前在扩容的新数组不为空
        // 说明原数组该位置的节点已经完成迁移了，并且迁移还没完成，那么就帮忙迁移别的位置
        if (tab != null && (f instanceof ConcurrentHashMap.ForwardingNode) && (nextTab = ((ConcurrentHashMap.ForwardingNode<K, V>) f).nextTable) != null) {

                // 参考addCount的注释，总之就是不需要新的线程参与扩容了
                if ((sc >>> RESIZE_STAMP_SHIFT) != rs || sc == rs + 1 || sc == rs + MAX_RESIZERS || transferIndex <= 0)
                    // 跳出循环
                    break;
                // 否则，CAS尝试更新sizeCtl，协助扩容的线程数加1，成功才能参与扩容
                if (U.compareAndSwapInt(this, SIZECTL, sc, sc + 1)) {
                    // 帮忙扩容，传入的新数组不为空
                    transfer(tab, nextTab);
                    break;
                }
            }
            return nextTab;
        }

ConcurrentHashMap#transfer

    /**
     * 支持多线程并发复制原数组的数据到新数组
     * 假设原数组长度为n，每个线程每次可以负责一批任务，即步长，每完成一批任务再检测是否有其他没完成的，帮助搬运
     * transferIndex指向原数组后，从后往前的步长个元素分配为第一批任务，再将transferIndex指向新的位置，再往前的步长个元素属于第二批任务，依此类推
     */
    private final void transfer(ConcurrentHashMap.Node<K, V>[] tab, ConcurrentHashMap.Node<K, V>[] nextTab) {
        int n = tab.length, stride;
        // 复制的步长，最小不能小于16
        if ((stride = (NCPU > 1) ? (n >>> 3) / NCPU : n) < MIN_TRANSFER_STRIDE)
            stride = MIN_TRANSFER_STRIDE;
        // 如果扩容的新数组不存在，说明还没开始迁移，那就创建一个新数组
        // 上级调用方法保证第一次ConcurrentHashMap#transfer时，nextTab为null，而之后的线程调用时不为null
        if (nextTab == null) {
            try {
                // 扩容，新数组容量翻倍
                @SuppressWarnings("unchecked") ConcurrentHashMap.Node<K, V>[] nt = (ConcurrentHashMap.Node<K, V>[]) new ConcurrentHashMap.Node<?, ?>[n << 1];
                nextTab = nt;
            } catch (Throwable ex) {
                sizeCtl = Integer.MAX_VALUE;
                return;
            }
            // volatile更新，对后面的线程立即可见
            nextTable = nextTab;
            // 扩容用的索引，从后向前的
            transferIndex = n;
        }
        // 新数组的容量
        int nextn = nextTab.length;
        /*
         * 原数组中位置i处的节点全部搬到扩容的新数组后后，将该位置设为ForwardingNode，哈希值为MOVED，标志该位置已经处理了
         * 同时ForwardingNode也存放新数组，如果get的时候扩容没完成，哈希又定位到了原数组的该位置，那么从新数组查找
         */
        ConcurrentHashMap.ForwardingNode<K, V> fwd = new ConcurrentHashMap.ForwardingNode<K, V>(nextTab);
        boolean advance = true;
        boolean finishing = false;
        // 直到完成
        for (int i = 0, bound = 0; ; ) {
            ConcurrentHashMap.Node<K, V> f;
            int fh;
            // 当一个线程完成了一个位置的搬运后，继续向前推进
            while (advance) {
                int nextIndex, nextBound;
                // 从后向前，直到遇到本批任务的边界或者扩容已经完成，不过为什么是从后往前而不是从前往后？
                if (--i >= bound || finishing)
                    advance = false;
                    // 说明原数组所有位置都分配了线程处理了，不需要继续了
                else if ((nextIndex = transferIndex) <= 0) {
                    i = -1;
                    advance = false;
                }
                // CAS尝试更新transferIndex，代表当前数组已被分配给线程的最新索引
                else if (U.compareAndSwapInt(this, TRANSFERINDEX, nextIndex, nextBound = (nextIndex > stride ? nextIndex - stride : 0))) {
                    // 本批任务的边界
                    bound = nextBound;
                    // 本批任务的开始索引，即步长范围为(bound, i)
                    i = nextIndex - 1;
                    advance = false;
                }
            }
            // 已经全部搬运完成 || 已经搬运完成，transferIndex被别的线程CAS重置了？ || 没看懂
            if (i < 0 || i >= n || i + n >= nextn) {
                int sc;
                // 如果扩容已经全部完成
                if (finishing) {
                    nextTable = null;
                    // 替换数组
                    table = nextTab;
                    // 下次扩容阈值为原数组容量的1.5倍，即新数组的0.75倍
                    sizeCtl = (n << 1) - (n >>> 1);
                    return;
                }
                // 每个线程开始帮助搬运前将sizeCtl加1，完成后要减1
                if (U.compareAndSwapInt(this, SIZECTL, sc = sizeCtl, sc - 1)) {
                    // 第一个发起扩容的线程将sizeCtl修改为(resizeStamp(n) << RESIZE_STAMP_SHIFT) + 2)
                    // 注意下面的sc是上面CAS减1前的sizeCtl，当当前线程为唯一的扩容线程时，sc的低16位值应该为2
                    // 下面的条件成立时说明当前还有其他线程正在扩容
                    if ((sc - 2) != resizeStamp(n) << RESIZE_STAMP_SHIFT)
                        // 扩容还没完，只是当前线程完了，直接返回
                        return;
                    // 说明所有扩容的线程都已经完成搬运
                    finishing = advance = true;
                    /*
                     * i重置为n，即数组的长度，下次循环时最后退出的线程就会从原数组从后往前遍历一遍检查所有的节点是否都是ForwardingNode，然后才用扩容的新数组替代原数组并退出
                     * 因为在if (--i >= bound || finishing)这个判断后，i是肯定小于n的，不会满足if (i < 0 || i >= n || i + n >= nextn)这个退出条件
                     * mmp，Doug Lea太阴险了
                     */
                    i = n;
                }
            }
            // 如果原数组的该位置为空，说明没有写入数据
            else if ((f = tabAt(tab, i)) == null)
                // 直接用ForwardingNode标记已处理
                advance = casTabAt(tab, i, null, fwd);
                // ForwardingNode，说明该位置的节点已经被搬走了，可能是最后退出的线程在遍历检查
            else if ((fh = f.hash) == MOVED)
                advance = true;
            else {
                // 该节点加锁，所以put和transfer会有竞争，因为在transfer复制的过程中发生了put增加了新节点，但是复制后的扩容数组没有，这显然是不行的
                // 但是get没有，因为transfer是拷贝完才更新数组，操作完成前不会被get线程看到，如果不巧get的节点刚好被搬走了，还可以在新数组里边查找
                // get和put的话，如果追加到链表尾，那么是对get线程立即可见的，如果加到红黑树，是有竞争的，get线程不会被阻塞，但是put线程可能会挂起
                synchronized (f) {
                    // DCL，再次检查
                    if (tabAt(tab, i) == f) {
                        ConcurrentHashMap.Node<K, V> ln, hn;
                        // 说明是链表的节点
                        if (fh >= 0) {
                            // 0放在低位，即位置不变，否则放在高位，和HashMap一样，都是分成两条链表
                            int runBit = fh & n;
                            ConcurrentHashMap.Node<K, V> lastRun = f;
                            // 从头到尾遍历链表节点
                            for (ConcurrentHashMap.Node<K, V> p = f.next; p != null; p = p.next) {
                                int b = p.hash & n;
                                // 记录链表最后的runBit相同的连续节点的runBit，即分配到同一条链表的最后的连续几个节点
                                if (b != runBit) {
                                    // 决定该分配到哪条链表
                                    runBit = b;
                                    // 该段连续几个节点的起始节点
                                    lastRun = p;
                                }
                            }
                            // 该段的起始节点会被插到对应链表的尾部
                            if (runBit == 0) {
                                ln = lastRun;
                                hn = null;
                            } else {
                                hn = lastRun;
                                ln = null;
                            }
                            // 从头到尾遍历链表节点，直到遇到上面找到的连续相同的节点的起始节点，避免了对那段的操作，因为直接把该段的起始节点插到链表尾就行了
                            // 这个优化真是，嗯，有点细致
                            for (ConcurrentHashMap.Node<K, V> p = f; p != lastRun; p = p.next) {
                                int ph = p.hash;
                                K pk = p.key;
                                V pv = p.val;
                                if ((ph & n) == 0)
                                    // 这里链表方向反转了
                                    ln = new ConcurrentHashMap.Node<K, V>(ph, pk, pv, ln);
                                else
                                    hn = new ConcurrentHashMap.Node<K, V>(ph, pk, pv, hn);
                            }
                            // 高低位两条链表插入新数组
                            setTabAt(nextTab, i, ln);
                            setTabAt(nextTab, i + n, hn);
                            // 原数组该位置设为ForwardingNode，标记为已处理
                            setTabAt(tab, i, fwd);
                            advance = true;
                        }
                        // 说明是红黑树节点
                        else if (f instanceof ConcurrentHashMap.TreeBin) {
                            ConcurrentHashMap.TreeBin<K, V> t = (ConcurrentHashMap.TreeBin<K, V>) f;
                            ConcurrentHashMap.TreeNode<K, V> lo = null, loTail = null;
                            ConcurrentHashMap.TreeNode<K, V> hi = null, hiTail = null;
                            int lc = 0, hc = 0;
                            // 当成链表遍历
                            for (ConcurrentHashMap.Node<K, V> e = t.first; e != null; e = e.next) {
                                int h = e.hash;
                                // 创建新的红黑树节点
                                ConcurrentHashMap.TreeNode<K, V> p = new ConcurrentHashMap.TreeNode<K, V>(h, e.key, e.val, null, null);
                                // 分割为高低两个双向链表
                                if ((h & n) == 0) {
                                    // ...
                                } else {
                                    // ...
                                }
                            }
                            // 红黑树链表过短就转为普通链表，否则创建新的红黑树
                            ln = (lc <= UNTREEIFY_THRESHOLD) ? untreeify(lo) : (hc != 0) ? new ConcurrentHashMap.TreeBin<K, V>(lo) : t;
                            hn = (hc <= UNTREEIFY_THRESHOLD) ? untreeify(hi) : (lc != 0) ? new ConcurrentHashMap.TreeBin<K, V>(hi) : t;
                            // 插入扩容后的新数组
                            setTabAt(nextTab, i, ln);
                            setTabAt(nextTab, i + n, hn);
                            // 原数组该位置用ForwardingNode标记
                            setTabAt(tab, i, fwd);
                            advance = true;
                        }
                    }
                }
            }
        }
    }

ConcurrentHashMap#resizeStamp

    /**
     * 返回当前数组的长度的二进制值中开头0的数量
     */
    static final int resizeStamp(int n) {
        // Integer#numberOfLeadingZeros：计算一个Integer首部的0的数量
        // 0000 0000 0000 0000 1000 0000 0000 0000
        // 1是符号位，表示负数，用于后面的位移操作
        // 假设当前数组容量为16，那么方法返回 0000 0000 0000 0000 1000 0000 0001 0000
        return Integer.numberOfLeadingZeros(n) | (1 << (RESIZE_STAMP_BITS - 1));
    }

                                K ek;
                                if (e.hash == hash && ((ek = e.key) == key || (ek != null && key.equals(ek)))) {
                                    V ev = e.val;
                                    if (cv == null || cv == ev || (ev != null && cv.equals(ev))) {
                                        oldVal = ev;
                                        if (value != null)
                                            e.val = value;
                                        else if (pred != null)
                                            // 删除
                                            pred.next = e.next;
                                        else
                                            // 删除的是头节点的话要重新插入数组
                                            setTabAt(tab, i, e.next);
                                    }
                                    break;
                                }
                                pred = e;
                                if ((e = e.next) == null)
                                    break;
                            }
                        }
                        // 红黑树节点
                        else if (f instanceof ConcurrentHashMap.TreeBin) {
                            validated = true;
                            ConcurrentHashMap.TreeBin<K, V> t = (ConcurrentHashMap.TreeBin<K, V>) f;
                            ConcurrentHashMap.TreeNode<K, V> r, p;
                            // 查找也是完全无锁的
                            if ((r = t.root) != null && (p = r.findTreeNode(hash, key, null)) != null) {
                                V pv = p.val;
                                if (cv == null || cv == pv || (pv != null && cv.equals(pv))) {
                                    oldVal = pv;
                                    if (value != null)
                                        p.val = value;
                                    // 这里删除才会竞争红黑树的锁
                                    else if (t.removeTreeNode(p))
                                        setTabAt(tab, i, untreeify(t.first));
                                }
                            }
                        }
                    }
                }
                if (validated) {
                    if (oldVal != null) {
                        if (value == null)
                            addCount(-1L, -1);
                        return oldVal;
                    }
                    break;
                }
            }
        }
        return null;
    }

吭，简单说一下ConcurrentHashMap让我耳目一新的地方：

除了少部分地方使用了synchronized外，其它大部分地方都是用自旋+CAS+volatile+DCL模拟的轻量级锁；

synchronized只会锁住一个节点，缩小了锁的范围，而其它地方大量使用的CAS其实也是一个道理，各司其职，细化了范围，而不是直接加一个大锁；

在计算元素数量的时候，避免多个线程对baseCount的集中竞争修改，而是尝试分散到多个CounterCell中，最后计算的时候叠加起来就行了，虽然可能不准确；

还使用了@sun.misc.Contended注解避免伪共享，涉及操作系统底层的东西我也不了解，大概就是那个意思；

对于扩容，将原数组的元素划分为多批任务，分配给多个写线程并发进行，原本因为竞争而阻塞的线程还可以参与多线程扩容，这个想法，只能说，wow；

对于读，完全是无阻塞的，即使存在并发的修改或扩容，对于链表，因为使用了volatile修饰，修改是立即可见的，对于一棵红黑树，假如有修改的线程，那就退化为用链表的方式遍历查找，因为二叉树的搜索方式要求结构稳定，如果没有修改线程存在，那就使用更高效的红黑树查找。而假如搜索的位置刚好因为扩容被搬走了，则重定向到扩容后的新数组查找，完全无阻塞；

写和扩容是可能会有锁竞争的，不过范围缩小到了节点，为了避免扩容时复制遍历到新数组的过程中，删除或插入了新的节点，而扩容后的新数组中，已经删除的节点还存在，插入的节点却丢失这种情况；

另外写线程是可能被阻塞挂起的，不多说了，自己看代码吧。
