ReentrantReadWriteLock和ReentrantLock挺像的，但是多了一个读锁，即共享锁的处理，这里略过独占锁的AQS重复部分。

public class ReentrantReadWriteLock implements ReadWriteLock {

    public ReentrantReadWriteLock() {

    public ReentrantReadWriteLock(boolean fair) {

ReentrantReadWriteLock$Sync

    abstract static class Sync extends AbstractQueuedSynchronizer {
        /**
         * 读写锁，高16位表示读状态，低16位表示写状态
         * 写锁，独占锁，低16位表示写锁的重入次数
         * 读锁，共享锁，可以同时被多个线程持有，高16位表示读锁被获取的次数，包含了重入，而每个线程的重入次数保存在ThreadLocal
         */
        static final int SHARED_SHIFT = 16;

        /**
         * 读状态 = state >>> 16
         * 读状态 + 1 = state + (1 << 16)
         */
        static final int SHARED_UNIT = (1 << SHARED_SHIFT);

        /**
         * 最大重入次数，2^16 - 1
         */
        static final int MAX_COUNT = (1 << SHARED_SHIFT) - 1;

        /**
         * 写状态掩码
         * 写状态 = state & 0x0000FFFF
         * 写状态 + 1 = state + 1
         */
        static final int EXCLUSIVE_MASK = (1 << SHARED_SHIFT) - 1;

        static final class HoldCounter {
            /**
             * 当前线程读锁的重入次数
             */
            int count = 0;

            // Use id, not reference, to avoid garbage retention
            // 使用线程Id，而不是Thread的引用判断，避免HoldCounter和Thread相互引用
            final long tid = getThreadId(Thread.currentThread());
        }

        /**
         * ThreadLocal
         */
        static final class ThreadLocalHoldCounter extends ThreadLocal<HoldCounter> {
            @Override
            public HoldCounter initialValue() {
                return new HoldCounter();
            }
        }

        /**
         * ThreadLocal，每个线程一份，保存该线程读锁的重入次数
         */
        private transient ThreadLocalHoldCounter readHolds;

        /**
         * 缓存上一个获取读锁的线程重入次数，减少ThreadLocal#get
         * 比如某线程获取读锁，再释放读锁，如果此期间没有别的线程获取读锁，就可以使用缓存
         */
        private transient HoldCounter cachedHoldCounter;

        /**
         * 第一个获取读锁的线程
         * 即，获取读锁时读锁为空闲状态，还没被获取的那个线程
         */
        private transient Thread firstReader = null;

        /**
         * 第一个获取读锁的线程的读锁重入次数
         * 可以把firstReaderHoldCount当成一级缓存，cachedHoldCounter当成二级缓存，readHolds当成三级缓存
         */
        private transient int firstReaderHoldCount;

        Sync() {
            readHolds = new ThreadLocalHoldCounter();
            // 保证可见性，state是volatile变量，避免重排序
            setState(getState()); // ensures visibility of readHolds
        }
    }

ReentrantReadWriteLock$Sync#tryAcquireShared

        @Override
        protected final int tryAcquireShared(int unused) {
            Thread current = Thread.currentThread();
            int c = getState();
            // 写锁被持有 && 持有该锁的线程不是当前线程，那么获取读锁失败
            if (exclusiveCount(c) != 0 && getExclusiveOwnerThread() != current)
                return -1;
            // 也就是说，这里允许写锁不释放，直接降级为读锁

            // 读锁被获取的次数
            int r = sharedCount(c);
            /*
             * 这里读锁需不需要阻塞，是根据公平策略决定的
             * 对于非公平，只要AQS队列中第一个等待的节点不是在等待获取独占锁，那就不用阻塞，因为写锁是优先的，不能被读锁阻塞
             * 而公平的话，只要存在队列存在等待的节点，就要排队
             */
            // 不需要阻塞等待 && 读锁获取的线程数没超出最大限制 && CAS更新读锁的线程数成功
            if (!readerShouldBlock() && r < MAX_COUNT && compareAndSetState(c, c + SHARED_UNIT)) {
                // 如果读锁空闲
                if (r == 0) {
                    // 当前线程就是第一个获取读锁的线程
                    firstReader = current;
                    // 重入次数为1
                    firstReaderHoldCount = 1;
                }
                // 如果已经有线程获取读锁了，并且是第一个获取读锁的线程重入
                else if (firstReader == current) {
                    // 读锁重入次数自增
                    firstReaderHoldCount++;
                }
                // 不是第一个获取读锁的线程，可能是重入，也可能是第一次获取读锁
                else {
                    HoldCounter rh = cachedHoldCounter;
                    // 线程重入缓存还未初始化 || 缓存的不是当前线程，这里是根据线程Id判断的
                    if (rh == null || rh.tid != getThreadId(current))
                        // 搞一个
                        cachedHoldCounter = rh = readHolds.get();
                    // 说明是重入，并且缓存命中，但是重入次数为0，这个是怎么发生的？
                    else if (rh.count == 0)
                        readHolds.set(rh);
                    // 读锁重入次数自增
                    rh.count++;
                }
                // 获取读锁成功
                return 1;
            }
            // 这里是给不用阻塞等待读锁，但是因为别的线程竞争读锁导致CAS失败的情况用的，不要直接进入AQS队列阻塞，而是自旋重试
            return fullTryAcquireShared(current);
        }

ReentrantReadWriteLock$Sync#fullTryAcquireShared

        final int fullTryAcquireShared(Thread current) {
            HoldCounter rh = null;
            for (; ; ) {
                int c = getState();
                // 每次自旋先检查，如果写锁不空闲
                if (exclusiveCount(c) != 0) {
                    // 持有写锁的不是当前线程
                    if (getExclusiveOwnerThread() != current)
                        // 获取失败
                        return -1;
                }
                // 写锁空闲，但是根据公平策略，需要阻塞等待写锁
                else if (readerShouldBlock()) {
                    if (firstReader == current) {
                        // assert firstReaderHoldCount > 0;
                    }
                    // 如果当前线程不是最早获取读锁的线程
                    else {
                        if (rh == null) {
                            // 缓存的上一个获取读锁的线程的重入计数
                            rh = cachedHoldCounter;
                            // 线程重入缓存还未初始化 || 缓存的不是当前线程
                            if (rh == null || rh.tid != getThreadId(current)) {
                                // 从ThreadLocal中获取当前线程的读锁重入计数
                                rh = readHolds.get();
                                // 说明是刚才ThreadLocalHoldCounter#get的时候初始化的
                                if (rh.count == 0)
                                    // 删了
                                    readHolds.remove();
                            }
                        }
                        // 说明不是重入，获取锁失败，返回后进入AQS队列阻塞排队
                        if (rh.count == 0)
                            return -1;
                    }
                    // 所以这里是处理读锁重入的，虽然此时应该阻塞排队，但是已经获取读锁的线程是允许重入的
                }
                // 如果读锁的获取次数溢出，则抛出异常
                if (sharedCount(c) == MAX_COUNT)
                    throw new Error("Maximum lock count exceeded");
                // 如果CAS更新读锁的线程数成功
                if (compareAndSetState(c, c + SHARED_UNIT)) {
                    // 如果读锁空闲
                    if (sharedCount(c) == 0) {
                        // 当前线程就是第一个获取读锁的线程
                        firstReader = current;
                        // 重入次数为1
                        firstReaderHoldCount = 1;
                    }
                    // 如果已经有线程获取读锁了，并且是第一个获取读锁的线程重入
                    else if (firstReader == current) {
                        // 该线程的读锁重入次数自增
                        firstReaderHoldCount++;
                    }
                    // 不是第一个获取读锁的线程，可能是重入，也可能是第一次获取读锁
                    else {
                        if (rh == null)
                            rh = cachedHoldCounter;
                        if (rh == null || rh.tid != getThreadId(current))
                            rh = readHolds.get();
                        else if (rh.count == 0)
                            readHolds.set(rh);

                        // 该线程的读锁重入次数自增
                        rh.count++;
                        cachedHoldCounter = rh; // cache for release
                    }
                    // 获取读锁成功
                    return 1;
                }
            }
        }

AbstractQueuedSynchronizer#doAcquireShared

    private void doAcquireShared(int arg) {
        // 添加到AQS队列尾，节点类型为共享
        final Node node = addWaiter(Node.SHARED);
        boolean failed = true;
        try {
            boolean interrupted = false;
            // 自旋
            for (; ; ) {
                final Node p = node.predecessor();
                // 节点就是第一个排队等待的节点
                if (p == head) {
                    // 尝试获取共享锁
                    int r = tryAcquireShared(arg);
                    // 获取共享锁成功
                    if (r >= 0) {
                        // 更新当前节点为AQS队列的头节点，并向后传播唤醒等待共享锁的节点，直到遇到等待独占锁的节点
                        setHeadAndPropagate(node, r);
                        p.next = null; // help GC
                        if (interrupted)
                            selfInterrupt();
                        failed = false;
                        return;
                    }
                }
                // 前面还有节点排队，那就阻塞
                if (shouldParkAfterFailedAcquire(p, node) && parkAndCheckInterrupt())
                    interrupted = true;
            }
        } finally {
            // 子类实现的tryAcquire方法抛出异常，那么取消当前节点，并从AQS队列移除
            if (failed)
                cancelAcquire(node);
        }
    }

AbstractQueuedSynchronizer#setHeadAndPropagate

    private void setHeadAndPropagate(Node node, int propagate) {
        Node h = head; // Record old head for check below
        // 更新头节点
        setHead(node);
        // 共享锁获取成功 || 原本的头节点为空 || 原本头节点为SIGNAL、CONDITION或PROPAGATE状态 || 当前头节点为空 || 当前头节点状态为...
        if (propagate > 0 || h == null || h.waitStatus < 0 || (h = head) == null || h.waitStatus < 0) {
            Node s = node.next;
            // 如果第一个排队等待的节点在等待获取共享锁
            if (s == null || s.isShared())
                // 向后传播
                doReleaseShared();
        }
    }

AbstractQueuedSynchronizer#doReleaseShared

    private void doReleaseShared() {
        for (; ; ) {
            // AQS队列的头节点
            Node h = head;
            // 如果队列中存在等待锁的节点
            if (h != null && h != tail) {
                // 即当前拥有锁的节点的状态
                int ws = h.waitStatus;
                // 如果头节点是SIGNAL状态，说明后继节点需要被唤醒
                if (ws == Node.SIGNAL) {
                    // CAS尝试将头节点状态更新为0，失败则继续重试
                    if (!compareAndSetWaitStatus(h, Node.SIGNAL, 0))
                        // 可能该节点被取消，或者多个线程同时释放共享锁
                        continue;
                    /*
                     * 如果CAS成功，唤醒头节点的后继节点，即排队等待的第一个节点
                     * 该节点获取共享锁后成为新的头结点，然后会继续唤醒下一个，就是传播
                     * 独占锁的释放就简单多了，只用唤醒后面第一个非取消的节点就好了
                     */
                    unparkSuccessor(h);
                }
                // 上次自旋成功CAS更新头节点状态后，CAS尝试将头节点的状态设置成PROPAGATE传播状态，失败则继续重试
                else if (ws == 0 && !compareAndSetWaitStatus(h, 0, Node.PROPAGATE))
                    continue;
            }
            // 如果头节点没有变化，说明被唤醒的节点还没获取共享锁，退出循环，被唤醒的节点获取共享锁后还会进入这里，向后唤醒
            // 如果头节点改变了，说明被唤醒的节点获取了共享锁，继续唤醒下一个
            // 这里我没看懂为什么要这么做，因为被唤醒的节点也会唤醒后面的节点，有必要并发进行唤醒吗？还是说提高吞吐量？
            if (h == head)
                break;
        }
    }

ReentrantReadWriteLock$Sync#tryReleaseShared

        @Override
        protected final boolean tryReleaseShared(int unused) {
            Thread current = Thread.currentThread();
            // 当前线程是第一个获取读锁的线程
            if (firstReader == current) {
                // 没有重入，直接清空
                if (firstReaderHoldCount == 1)
                    firstReader = null;
                // 重入次数减少
                else
                    firstReaderHoldCount--;
            }
            // 如果不是
            else {
                // 上个获取读锁的线程的读锁重入次数缓存
                HoldCounter rh = cachedHoldCounter;
                // 缓存为空 || 缓存未命中
                if (rh == null || rh.tid != getThreadId(current))
                    // 从ThreadLocal获取
                    rh = readHolds.get();
                int count = rh.count;
                // 已经没有重入了
                if (count <= 1) {
                    // 移除ThreadLocal
                    readHolds.remove();
                    // 多次ReadLock#unlock
                    if (count <= 0)
                        throw unmatchedUnlockException();
                }
                // 重入次数减少
                --rh.count;
            }
            // 自旋至CAS成功
            for (; ; ) {
                int c = getState();
                // 读锁的获取次数减1
                int nextc = c - SHARED_UNIT;
                // CAS更新锁的状态
                if (compareAndSetState(c, nextc))
                    // 如果是0，表示读锁和写锁都空闲，可以唤醒后面的等待线程
                    return nextc == 0;
            }
        }

ReentrantReadWriteLock$Sync#tryAcquire

        @Override
        protected final boolean tryAcquire(int acquires) {
            Thread current = Thread.currentThread();
            int c = getState();
            // 写锁的重入次数
            int w = exclusiveCount(c);
            // 锁不空闲，读锁或写锁被持有
            if (c != 0) {
                // 写锁空闲，但读锁不空闲 || 写锁不空闲，但不是重入
                if (w == 0 || current != getExclusiveOwnerThread())
                    // 获取写锁失败
                    return false;
                // 写锁重入次数超过最大值
                if (w + exclusiveCount(acquires) > MAX_COUNT)
                    throw new Error("Maximum lock count exceeded");
                // 更新锁状态
                setState(c + acquires);
                // 重入获取写锁成功
                return true;
            }
            // 此时不存在读锁或写锁
            // 根据公平策略决定是否需要阻塞，非公平，完全不需要阻塞，而公平需要判断AQS同步队列中是否有节点等待，不能插队
            if (writerShouldBlock() || !compareAndSetState(c, c + acquires))
                return false;
            // 如果CAS更新成功，那么获取写锁成功，更新锁的持有线程
            setExclusiveOwnerThread(current);
            return true;
        }

        @Override
        protected final boolean tryRelease(int releases) {
            // 先校验是否持有写锁
            if (!isHeldExclusively())
                throw new IllegalMonitorStateException();
            // 写锁重入次数减少
            int nextc = getState() - releases;
            // 可以释放写锁
            boolean free = exclusiveCount(nextc) == 0;
            // 如果可以释放，置空锁的持有线程
            if (free)
                setExclusiveOwnerThread(null);
            // 更新锁状态
            setState(nextc);
            return free;
        }

总结：

  • Must release read lock before acquiring write lock.

必须先释放读锁，才能获取写锁，不允许读锁升级为写锁。

如果不释放读锁就直接获取写锁，会导致死锁。

因为读锁存在时，不能获取写锁，会直接进入AQS队列阻塞，等待前一个节点，即读锁释放后唤醒，但读锁不能被释放，因为线程已经阻塞了，这就是死锁。

  • Downgrade by acquiring read lock before releasing write lock.

在释放写锁前，可以获取读锁，允许写锁降级为读锁，此线程既持有写锁又持有读锁。

不过我没看出这个行为有什么意义。
