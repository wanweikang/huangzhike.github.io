SocketWrapperBase#write

    public final void write(boolean block, ByteBuffer from) throws IOException {
        if (from == null || from.remaining() == 0) {
            return;
        }
        if (block) {
            // 阻塞写
            writeBlocking(from);
        } else {
            // 非阻塞写
            writeNonBlocking(from);
        }
    }

SocketWrapperBase#writeNonBlocking

    protected void writeNonBlocking(ByteBuffer from) throws IOException {
        // 非阻塞写缓冲为空 && 写缓冲可写
        if (nonBlockingWriteBuffer.isEmpty() && socketBufferHandler.isWriteBufferWritable()) {
            // 非阻塞式将源缓冲数据写到SocketChannel
            writeNonBlockingInternal(from);
        }
        // 源缓冲还有剩余可读（上面没读完）
        if (from.remaining() > 0) {
            // 加到非阻塞写缓冲，非阻塞写缓冲内部维护一个双向阻塞链表
            nonBlockingWriteBuffer.add(from);
        }
    }

SocketBufferHandler#isWriteBufferWritable

    public boolean isWriteBufferWritable() {
        // 如果写缓冲设被为写模式
        if (writeBufferConfiguredForWrite) {
            // 可写
            return writeBuffer.hasRemaining();
        }
        // 如果写缓冲设被为读模式
        else {
            // 已读完
            return writeBuffer.remaining() == 0;
        }
    }

SocketWrapperBase#writeNonBlockingInternal

    protected void writeNonBlockingInternal(ByteBuffer from) throws IOException {
        // 写缓冲为空
        if (socketBufferHandler.isWriteBufferEmpty()) {
            // 直接非阻塞式将源缓冲写到SocketChannel，剩余的写入写缓冲
            writeNonBlockingDirect(from);
        }
        // 写缓冲不为空
        else {
            // 配置写缓冲准备写
            socketBufferHandler.configureWriteBufferForWrite();
            // 尽可能把源缓冲添加到写缓冲
            transfer(from, socketBufferHandler.getWriteBuffer());
            // 如果写缓冲不可写（写满了）
            if (!socketBufferHandler.isWriteBufferWritable()) {
                // 非阻塞式，把写缓冲的数据写到SocketChannel
                doWrite(false);
                // 最后如果写缓冲可写（上一步已经将写缓冲切为读模式了，这里是写缓冲已清空）
                if (socketBufferHandler.isWriteBufferWritable()) {
                    // 直接非阻塞式将源缓冲写到SocketChannel，剩余的写入写缓冲
                    writeNonBlockingDirect(from);
                }
            }
        }
    }

SocketBufferHandler#isWriteBufferEmpty

    public boolean isWriteBufferEmpty() {
        // 如果写缓冲设被为写模式
        if (writeBufferConfiguredForWrite) {
            // 没有被写过
            return writeBuffer.position() == 0;
        }
        // 如果写缓冲设被为读模式
        else {
            // 已经被读完
            return writeBuffer.remaining() == 0;
        }
    }

SocketWrapperBase#writeNonBlockingDirect

    protected void writeNonBlockingDirect(ByteBuffer from) throws IOException {
        // The socket write buffer capacity is socket.appWriteBufSize
        // TODO This only matters when using TLS. For non-TLS connections it should be possible to write the ByteBuffer in a single write
        // 写缓冲的容量
        int limit = socketBufferHandler.getWriteBuffer().capacity();
        // 先保存源缓冲的limit
        int fromLimit = from.limit();
        // 循环处理，直到源缓冲可读大小小于写缓冲的容量
        while (from.remaining() >= limit) {
            int newLimit = from.position() + limit;
            // 设置源缓冲的读limit，避免写缓冲溢出
            from.limit(newLimit);
            // 直接非阻塞写到SocketChannel，不经过写缓冲
            doWrite(false, from);
            // 设回源缓冲的limit
            from.limit(fromLimit);
            // 这里就是非阻塞，如果没成功写到预期的字节数
            if (from.position() != newLimit) {
                // 直接退出
                return;
            }
        } // 写成功
        // 如果源缓冲还有未读字节
        if (from.remaining() > 0) {
            // 配置写缓冲准备写
            socketBufferHandler.configureWriteBufferForWrite();
            // 尽可能写入到Socket关联的写缓冲
            transfer(from, socketBufferHandler.getWriteBuffer());
        }
    }

SocketBufferHandler#configureWriteBufferForWrite

    public void configureWriteBufferForWrite() {
        setWriteBufferConfiguredForWrite(true);
    }

SocketBufferHandler#setWriteBufferConfiguredForWrite

    private void setWriteBufferConfiguredForWrite(boolean writeBufferConfiguredForWrite) {
        // 如果已经处于要配置的状态就什么也不干
        if (this.writeBufferConfiguredForWrite != writeBufferConfiguredForWrite) {
            // 配置准备要写
            if (writeBufferConfiguredForWrite) {
                // Switching to write
                int remaining = writeBuffer.remaining();
                if (remaining == 0) {
                    writeBuffer.clear();
                } else {
                    writeBuffer.compact();
                    writeBuffer.position(remaining);
                    writeBuffer.limit(writeBuffer.capacity());
                }
            }
            // 配置准备要读
            else {
                // Switching to read
                writeBuffer.flip();
            }
            // 设置状态
            this.writeBufferConfiguredForWrite = writeBufferConfiguredForWrite;
        }
    }

SocketWrapperBase#transfer

    protected static int transfer(ByteBuffer from, ByteBuffer to) {
        // 源缓冲可读大小和目的缓冲可写大小，取最小值，避免溢出
        int max = Math.min(from.remaining(), to.remaining());
        if (max > 0) {
            int fromLimit = from.limit();
            // 设置源缓冲读的limit，避免目的缓冲溢出
            from.limit(from.position() + max);
            // 写入目的缓冲
            to.put(from);
            // 恢复源缓冲原本limit
            from.limit(fromLimit);
        }
        // 返回写的字节数
        return max;
    }

SocketWrapperBase#doWrite

    protected void doWrite(boolean block) throws IOException {
        // 将Socket关联的写缓冲切换到读模式
        socketBufferHandler.configureWriteBufferForRead();
        // 把写缓冲的数据写到SocketChannel
        // NioEndpoint$NioSocketWrapper#doWrite
        doWrite(block, socketBufferHandler.getWriteBuffer());
    }

SocketBufferHandler#configureWriteBufferForRead

    public void configureWriteBufferForRead() {
        setWriteBufferConfiguredForWrite(false);
    }

WriteBuffer#add

    public void add(ByteBuffer from) {
        // 从链表获取包装，大小为剩余未读字节
        ByteBufferHolder holder = getByteBufferHolder(from.remaining());
        // 包装缓冲
        holder.getBuf().put(from);
    }

WriteBuffer#getByteBufferHolder

    private ByteBufferHolder getByteBufferHolder(int capacity) {
        // 队尾
        ByteBufferHolder holder = buffers.peekLast();
        if (holder == null || holder.isFlipped() || holder.getBuf().remaining() < capacity) {
            // 分配缓冲
            ByteBuffer buffer = ByteBuffer.allocate(Math.max(bufferSize, capacity));
            // 没被切换为读模式
            holder = new ByteBufferHolder(buffer, false);
            // 加入链表尾
            buffers.add(holder);
        }
        return holder;
    }

总结：

非阻塞写，如果非阻塞写缓冲为空，并且写缓冲可写，才尝试非阻塞地将数据直接写入SocketChannel（步骤A），如果数据还有剩余没读的，才将剩余数据追加到非阻塞写缓冲的链表尾（步骤C）。

如果非阻塞写缓冲有数据，则不能跳过，这样就保证了数据的连续和顺序。

非阻塞地将数据直接写入SocketChannel（步骤A）：

  • 如果写缓冲为空，则可以直接非阻塞地将数据写入SocketChannel，每次写的字节数为写缓冲的容量（步骤B）。

    • 如果没写到预期的字节数，则退出，后面再将剩余数据追加入非阻塞写缓冲（步骤C）；

    • 如果还有剩余字节，则尽可能地将数据写入写缓冲，如果还有剩余，后面再将剩余数据追加入非阻塞写缓冲（步骤C）。

  • 如果写缓冲不为空，则尽可能把数据写入写缓冲，如果写缓冲满了，再非阻塞地将写缓冲的数据写到SocketChannel，如果写缓冲已经清空，则执行上面的步骤B。

所以非阻塞写不保证数据都写到SocketChannel，写不完的数据暂存在写缓冲和非阻塞写缓冲，顺序为写缓冲-非阻塞写缓冲-源数据。

SocketWrapperBase#flush

    public boolean flush(boolean block) throws IOException {
        boolean result = false;
        if (block) {
            // A blocking flush will always empty the buffer.
            flushBlocking();
        } else {
            result = flushNonBlocking();
        }
        return result;
    }

SocketWrapperBase#flushNonBlocking

    protected boolean flushNonBlocking() throws IOException {
        // 写缓冲不为空
        boolean dataLeft = !socketBufferHandler.isWriteBufferEmpty();
        // 必须先清空写缓冲
        if (dataLeft) {
            // 非阻塞式把写缓冲写到SocketChannel
            doWrite(false);
            // 写缓冲还不为空
            dataLeft = !socketBufferHandler.isWriteBufferEmpty();
        }
        // 写缓冲为空 && 非阻塞写缓冲不为空
        // 这样才能清空非阻塞写缓冲的数据
        if (!dataLeft && !nonBlockingWriteBuffer.isEmpty()) {
            // 非阻塞式把非阻塞写缓冲写到SocketChannel，返回非阻塞写缓冲是否清空
            dataLeft = nonBlockingWriteBuffer.write(this, false);
            // 非阻塞写缓冲没有剩余 && 写缓冲不为空
            if (!dataLeft && !socketBufferHandler.isWriteBufferEmpty()) {
                // 非阻塞式把写缓冲写到SocketChannel
                doWrite(false);
                // 写缓冲是否为空
                dataLeft = !socketBufferHandler.isWriteBufferEmpty();
            }
        }
        // 返回两个写缓冲最终是否清空
        return dataLeft;
    }

WriteBuffer#write

    boolean write(SocketWrapperBase<?> socketWrapper, boolean blocking) throws IOException {
        Iterator<ByteBufferHolder> bufIter = buffers.iterator();
        boolean dataLeft = false;
        // 遍历链表
        while (!dataLeft && bufIter.hasNext()) {
            ByteBufferHolder buffer = bufIter.next();
            // 切换到读模式
            buffer.flip();
            // 阻塞
            if (blocking) {
                // 阻塞写到SocketChannel
                socketWrapper.writeBlocking(buffer.getBuf());
            }
            // 非阻塞
            else {
                // 非阻塞写到SocketChannel，或者写到写缓冲
                socketWrapper.writeNonBlockingInternal(buffer.getBuf());
            }
            // 全部读完
            if (buffer.getBuf().remaining() == 0) {
                bufIter.remove();
            }
            // 还有剩余
            else {
                dataLeft = true;
            }
        }
        // 返回是否清空了非阻塞写缓冲
        return dataLeft;
    }

总结：

非阻塞刷，必须先清空写缓冲，才能清空非阻塞写缓冲。

假如清空了写缓冲，则遍历非阻塞写缓冲，执行上面非阻塞写的步骤A，这个过程可能将非阻塞写缓冲的数据写入SocketChannel或者写缓冲，然后返回是否清空了非阻塞写缓冲。

  • 非阻塞写缓冲清空，但写缓冲未清空：非阻塞地将写缓冲写入SocketChannel，最后返回写缓冲是否被清空，即最终是否有清空两个写缓冲。

  • 非阻塞写缓冲清空，写缓冲也已清空：最终已经清空。

  • 非阻塞写缓冲没清空：直接返回没有清空。

所以非阻塞Flush不保证所有数据都写到SocketChannel，要由上级调用者保证。
