NioEndpoint$NioSocketWrapper#doWrite

        @Override
        protected void doWrite(boolean block, ByteBuffer from) throws IOException {
            NioChannel socket = getSocket();
            if (socket instanceof ClosedNioChannel) {
                throw new ClosedChannelException();
            }
            // 阻塞写
            if (block) {
                // 写超时
                long writeTimeout = getWriteTimeout();
                Selector selector = null;
                try {
                    // EndPoint绑定端口时打开的NioSelectorPool
                    selector = pool.get();
                } catch (IOException x) {
                    // Ignore
                }
                try {
                    // 将数据写到SocketChannel
                    pool.write(from, socket, selector, writeTimeout);
                    if (block) {
                        // Make sure we are flushed
                        do {
                            // 这里非SSL安全连接直接返回true
                            if (socket.flush(true, selector, writeTimeout)) {
                                break;
                            }
                        } while (true);
                    }
                } finally {
                    if (selector != null) {
                        pool.put(selector);
                    }
                }
                // If there is data left in the buffer the socket will be registered for write further up the stack.
                // This is to ensure the socket is only registered for write once as both container and user code can trigger write registration.
            }
            // 非阻塞写
            else {
                // 直接写到SocketChannel
                if (socket.write(from) == -1) {
                    throw new EOFException();
                }
            }
            // 更新Socket的上次写时间
            updateLastWrite();
        }

NioSelectorPool#write

    public int write(ByteBuffer buf, NioChannel socket, Selector selector, long writeTimeout) throws IOException {
        // 共享，默认
        if (shared) {
            return blockingSelector.write(buf, socket, writeTimeout);
        }
        // 非共享
        SelectionKey key = null;
        int written = 0;
        boolean timedout = false;
        int keycount = 1; //assume we can write
        long time = System.currentTimeMillis(); //start the timeout timer
        try {
            while ((!timedout) && buf.hasRemaining()) {
                int cnt = 0;
                if (keycount > 0) { //only write if we were registered for a write
                    cnt = socket.write(buf); //write the data
                    if (cnt == -1) {
                        throw new EOFException();
                    }
                    written += cnt;
                    if (cnt > 0) {
                        time = System.currentTimeMillis(); //reset our timeout timer
                        continue; //we successfully wrote, try again without a selector
                    }
                }
                if (selector != null) {
                    //register OP_WRITE to the selector
                    if (key == null) {
                        key = socket.getIOChannel().register(selector, SelectionKey.OP_WRITE);
                    } else {
                        key.interestOps(SelectionKey.OP_WRITE);
                    }
                    if (writeTimeout == 0) {
                        timedout = buf.hasRemaining();
                    }
                    //
                    else if (writeTimeout < 0) {
                        keycount = selector.select();
                    }
                    // 阻塞超时
                    else {
                        keycount = selector.select(writeTimeout);
                    }
                }
                if (writeTimeout > 0 && (selector == null || keycount == 0)) {
                    timedout = (System.currentTimeMillis() - time) >= writeTimeout;
                }
            }
            if (timedout) {
                throw new SocketTimeoutException();
            }
        } finally {
            if (key != null) {
                key.cancel();
                if (selector != null)
                    selector.selectNow();//removes the key from this selector
            }
        }
        return written;
    }

NioBlockingSelector#write

    public int write(ByteBuffer buf, NioChannel socket, long writeTimeout) throws IOException {
        // 获取该Socket注册的事件
        SelectionKey key = socket.getIOChannel().keyFor(socket.getSocketWrapper().getPoller().getSelector());
        if (key == null) {
            throw new IOException(sm.getString("nioBlockingSelector.keyNotRegistered"));
        }
        KeyReference reference = keyReferenceStack.pop();
        if (reference == null) {
            reference = new KeyReference();
        }
        // Attachment
        NioSocketWrapper att = (NioSocketWrapper) key.attachment();
        int written = 0;
        boolean timedout = false;
        // 假设可写
        int keycount = 1;
        // 开启超时
        long time = System.currentTimeMillis();
        try {
            // 没有超时 && 没写完
            while (!timedout && buf.hasRemaining()) {
                // 只有注册了可写事件时才写
                if (keycount > 0) {
                    // SocketChannel#write
                    int cnt = socket.write(buf);
                    if (cnt == -1) {
                        throw new EOFException();
                    }
                    // 已写数
                    written += cnt;
                    // 写成功
                    if (cnt > 0) {
                        // 重置超时
                        time = System.currentTimeMillis();
                        // 跳过下面的Selector
                        continue;
                    }
                }
                // 写失败
                try {
                    if (att.getWriteLatch() == null || att.getWriteLatch().getCount() == 0) {
                        // WriteLatch
                        att.startWriteLatch(1);
                    }
                    // 向BlockPoller提交任务，向BlockSelector注册可写事件
                    poller.add(att, SelectionKey.OP_WRITE, reference);
                    // 阻塞至可写就绪或超时
                    att.awaitWriteLatch(AbstractEndpoint.toTimeout(writeTimeout), TimeUnit.MILLISECONDS);
                } catch (InterruptedException ignore) {
                    // Ignore
                }
                // 超时，BlockPoller没有CountDown唤醒
                if (att.getWriteLatch() != null && att.getWriteLatch().getCount() > 0) {
                    // 标记为不可写，在下一轮循环继续注册可写事件
                    keycount = 0;
                }
                // 没超时就唤醒
                else {
                    // 标记可写，下一轮循环尝试继续写
                    keycount = 1;
                    att.resetWriteLatch();
                }
                // 设置了写超时 && SockChannel暂时不可写
                if (writeTimeout > 0 && (keycount == 0)) {
                    // 计算是否已超时
                    timedout = (System.currentTimeMillis() - time) >= writeTimeout;
                }
            } // 已退出循环
            if (timedout) {
                throw new SocketTimeoutException();
            }
        } finally {
            // 最后LatchCountDown唤醒阻塞的线程
            poller.remove(att, SelectionKey.OP_WRITE);
            if (timedout && reference.key != null) {
                // 向BlockPoller提交一个SelectionKey#cancel事件
                poller.cancelKey(reference.key);
            }
            reference.key = null;
            keyReferenceStack.push(reference);
        }
        return written;
    }

NioBlockingSelector$BlockPoller#add

        public void add(final NioSocketWrapper key, final int ops, final KeyReference ref) {
            if (key == null) {
                return;
            }
            NioChannel nch = key.getSocket();
            final SocketChannel ch = nch.getIOChannel();
            if (ch == null) {
                return;
            }
            Runnable r = new RunnableAdd(ch, key, ops, ref);
            // 入队列
            events.offer(r);
            wakeup();
        }

NioBlockingSelector$BlockPoller#wakeup

        public void wakeup() {
            // 先加再获取，如果原值为-1，则当前正在阻塞选择中
            if (wakeupCounter.addAndGet(1) == 0)
                // 唤醒
                selector.wakeup();
        }

NioBlockingSelector$BlockPoller#run

        @Override
        public void run() {
            while (run) {
                try {
                    // 遍历执行任务队列
                    events();
                    int keyCount = 0;
                    try {
                        // 初始为0，每次通过BlockPoller#add时加1
                        int i = wakeupCounter.get();
                        // 说明有新的注册事件加入
                        if (i > 0) {
                            // 立刻非阻塞选择一次
                            keyCount = selector.selectNow();
                        }
                        // 上次执行完没有新的注册事件加入
                        else {
                            // 标明正在阻塞选择
                            wakeupCounter.set(-1);
                            // 超时
                            keyCount = selector.select(1000);
                        }
                        // 执行完恢复为0
                        wakeupCounter.set(0);
                        if (!run) {
                            break;
                        }
                    } catch (NullPointerException x) {
                        // continue...
                    } catch (CancelledKeyException x) {
                        // continue...
                    } catch (Throwable x) {
                        // continue...
                    }

                    // 如果有就绪事件
                    Iterator<SelectionKey> iterator = keyCount > 0 ? selector.selectedKeys().iterator() : null;
                    // 遍历就绪事件
                    while (run && iterator != null && iterator.hasNext()) {
                        SelectionKey sk = iterator.next();
                        NioSocketWrapper socketWrapper = (NioSocketWrapper) sk.attachment();
                        try {
                            // 处理前先移除
                            iterator.remove();
                            // 修改关心的事件
                            sk.interestOps(sk.interestOps() & (~sk.readyOps()));
                            // 可读就绪
                            if (sk.isReadable()) {
                                // 唤醒阻塞线程
                                countDown(socketWrapper.getReadLatch());
                            }
                            // 可写就绪
                            if (sk.isWritable()) {
                                // 唤醒阻塞线程
                                countDown(socketWrapper.getWriteLatch());
                            }
                        } catch (CancelledKeyException ckx) {
                            // ...
                        }
                    }
                } catch (Throwable t) {
                    // ...
                }
            } // 退出循环，BlockPoller不再运行
            // 清空事件队列
            events.clear();
            // If using a shared selector, the NioSelectorPool will also try and close the selector.
            // Try and avoid the ClosedSelectorException although because multiple threads are involved there is always the possibility of an Exception here.
            if (selector.isOpen()) {
                try {
                    // 执行一次select，移除所有cancel的SelectionKey
                    selector.selectNow();
                } catch (Exception ignore) {
                    // ...
                }
            }
            try {
                // 关闭
                selector.close();
            } catch (Exception ignore) {
                // ...
            }
        }

NioBlockingSelector$BlockPoller#events

        public boolean events() {
            Runnable r = null;
            int size = events.size();
            for (int i = 0; i < size && (r = events.poll()) != null; i++) {
                r.run();
            }
            return (size > 0);
        }

NioBlockingSelector$BlockPolle$RunnableAdd#run

            @Override
            public void run() {
                // 获取该SocketChannel注册到该Selector的SelectionKey
                SelectionKey sk = ch.keyFor(selector);
                try {
                    // 没有
                    if (sk == null) {
                        // 注册对应事件
                        sk = ch.register(selector, ops, key);
                        ref.key = sk;
                    }
                    // 无效
                    else if (!sk.isValid()) {
                        // 取消并CountDownLatch
                        cancel(sk, key, ops);
                    }
                    // 之前注册过
                    else {
                        // 修改注册事件
                        sk.interestOps(sk.interestOps() | ops);
                    }
                } catch (CancelledKeyException cx) {
                    cancel(sk, key, ops);
                } catch (ClosedChannelException cx) {
                    cancel(null, key, ops);
                }
            }

总结：

阻塞写就是先尝试直接向SocketChannel写数据，写不成功就向Selector注册该SocketChannel的可写事件并阻塞，由一个Poller线程轮询就绪事件并唤醒阻塞的工作线程，之后继续写，直到全部写完。

Selector默认是共享的，所有SocketChannel共享，也可以从对象池获取，这样可以分担单个Selector的压力。

Tomcat接收请求和回复响应是由两个Poller和Selector完成的，大概也是出于这个原因。

非阻塞写就是直接写，不成功就直接返回，由上级调用者决定后续操作。
