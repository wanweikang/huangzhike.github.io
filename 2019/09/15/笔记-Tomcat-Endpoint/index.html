NioEndpoint#startInternal

    @Override
    public void startInternal() throws Exception {
        if (!running) {
            running = true;
            paused = false;
            // 三个同步栈SynchronizedStack，对象池
            if (socketProperties.getProcessorCache() != 0) {
                processorCache = new SynchronizedStack<>(SynchronizedStack.DEFAULT_SIZE, socketProperties.getProcessorCache());
            }
            if (socketProperties.getEventCache() != 0) {
                eventCache = new SynchronizedStack<>(SynchronizedStack.DEFAULT_SIZE, socketProperties.getEventCache());
            }
            if (socketProperties.getBufferPool() != 0) {
                nioChannels = new SynchronizedStack<>(SynchronizedStack.DEFAULT_SIZE, socketProperties.getBufferPool());
            }

            /*
             * Worker线程池、Poller线程、Acceptor线程
             */

            // server.xml是否配置公共Executor
            if (getExecutor() == null) {
                // 没有就创建自己的Worker线程池，执行请求
                createExecutor();
            }
            // 初始化最大请求数LimitLatch，类似信号量，server.xml配置，接受TCP连接时自增，关闭TCP连接时自减
            initializeConnectionLatch();
            // 创建Poller，Poller是Runnable类型，内部打开了一个Selector，轮询事件交给Worker线程池处理
            poller = new Poller();
            Thread pollerThread = new Thread(poller, getName() + "-ClientPoller");
            pollerThread.setPriority(threadPriority);
            pollerThread.setDaemon(true);
            // 启动Poller线程
            pollerThread.start();
            // 实例化Acceptor对象，Acceptor是Runnable类型，将Endpoint对象传入
            // 启动Acceptor线程，Acceptor将请求Socket注册到Poller，由Poller轮询后续事件
            startAcceptorThread();
        }
    }

Acceptor#run

    @Override
    public void run() {
        // 用于发生异常时挂起线程
        int errorDelay = 0;
        // 循环直到接收到关闭命令
        while (endpoint.isRunning()) {
            // 暂停状态
            while (endpoint.isPaused() && endpoint.isRunning()) {
                state = AcceptorState.PAUSED;
                try {
                    // 阻塞挂起，轮询是否从暂停状态恢复
                    Thread.sleep(50);
                } catch (InterruptedException e) {
                    // Ignore
                }
            }
            // 非运行状态，退出
            if (!endpoint.isRunning()) {
                break;
            }
            // 设置状态为Running
            state = AcceptorState.RUNNING;
            try {
                // Latch，如果已经到达Connection连接最大值，等待
                endpoint.countUpOrAwaitConnection();
                // 如果等待期间被暂停了
                if (endpoint.isPaused()) {
                    continue;
                }
                U socket = null;
                try {
                    // 阻塞调用ServerSocketChannel#accept，接受请求，返回SocketChannel
                    socket = endpoint.serverSocketAccept();
                } catch (Exception ioe) {
                    // We didn't get a socket
                    endpoint.countDownConnection();
                    if (endpoint.isRunning()) {
                        // 遇到异常阻塞挂起自己errorDelay毫秒
                        errorDelay = handleExceptionWithDelay(errorDelay);
                        // re-throw
                        throw ioe;
                    } else {
                        break;
                    }
                }
                // 接受成功，重置标识
                errorDelay = 0;

                if (endpoint.isRunning() && !endpoint.isPaused()) {
                    // 配置SocketChannel，并将SocketChannel交给Processor
                    if (!endpoint.setSocketOptions(socket)) {
                        // 失败了就关闭SocketChannel
                        endpoint.closeSocket(socket);
                    }
                } else {
                    // 直接关闭SocketChannel
                    endpoint.destroySocket(socket);
                }
            } catch (Throwable t) {
                // ...
            }
        }
        // Acceptor线程结束
        state = AcceptorState.ENDED;
    }

NioEndpoint#setSocketOptions

    @Override
    protected boolean setSocketOptions(SocketChannel socket) {
        // Process the connection
        try {
            // 设置SocketChannel为非阻塞
            socket.configureBlocking(false);
            // 返回Socket
            Socket sock = socket.socket();
            // 设置Socket的属性，如缓冲区大小
            socketProperties.setProperties(sock);
            // Tomcat的Channel，非JDK
            NioChannel channel = null;
            if (nioChannels != null) {
                // 先从同步栈对象池中获取Tomcat的通道对象
                channel = nioChannels.pop();
            }
            // 没有就创建通道
            if (channel == null) {
                // 创建一个SocketBufferHandler，内部维护两个读写的缓冲
                SocketBufferHandler bufhandler = new SocketBufferHandler(socketProperties.getAppReadBufSize(), socketProperties.getAppWriteBufSize(), socketProperties.getDirectBuffer());

                // 配置了SSL
                if (isSSLEnabled()) {
                    // SSL的安全通道，selectorPool是之前NioEndpoint#bind打开的SelectorPool
                    channel = new SecureNioChannel(socket, bufhandler, selectorPool, this);
                } else {
                    // 普通通道
                    channel = new NioChannel(socket, bufhandler);
                }
            }
            // 如果已存在
            else {
                // 设置IOChannel，包含Socket
                channel.setIOChannel(socket);
                // 重置缓冲
                channel.reset();
            }
            // 把NioChannel和NioEndpoint包装成NioSocketWrapper
            // 将之前NioEndpoint#bind时打开的SelectorPool传入
            // SelectorPool用于给Socket提供Selector，有共享和非共享两种
            // 默认是所有SocketChannel共享一个Selector，并且是阻塞模式，由NioBlockingSelector完成读写
            // NioBlockingSelector维护一个轮询线程BlockPoller，第一次读写不成功时，注册到BlockPoller
            // 当配置为非共享时，从对象池获取，这样可以将每个SocketChannel的事件分散到不同Selector中，降低了单个的压力
            NioSocketWrapper socketWrapper = new NioSocketWrapper(channel, this);
            // 把NioSocketWrapper添加到NioChannel
            channel.setSocketWrapper(socketWrapper);
            socketWrapper.setReadTimeout(getConnectionTimeout());
            socketWrapper.setWriteTimeout(getConnectionTimeout());
            socketWrapper.setKeepAliveLeft(NioEndpoint.this.getMaxKeepAliveRequests());
            socketWrapper.setSecure(isSSLEnabled());
            // TCP握手完成，将该NioChannel注册到Poller，轮询后续事件
            poller.register(channel, socketWrapper);
            return true;
        } catch (Throwable t) {
            // ...
        }
        // Tell to close the socket
        return false;
    }

NioEndpoint$Poller#register

        public void register(final NioChannel socket, final NioSocketWrapper socketWrapper) {
            // 关心可读事件
            socketWrapper.interestOps(SelectionKey.OP_READ);//this is what OP_REGISTER turns into.
            PollerEvent r = null;
            if (eventCache != null) {
                // 先从对象池拿事件对象
                r = eventCache.pop();
            }
            if (r == null) {
                // 没有就新建一个，并设置为OP_REGISTER事件
                r = new PollerEvent(socket, OP_REGISTER);
            } else {
                // 重置为OP_REGISTER事件
                r.reset(socket, OP_REGISTER);
            }
            // 加入Poller事件队列，由Poller线程处理
            addEvent(r);
        }

NioEndpoint$Poller#addEvent

        private void addEvent(PollerEvent event) {
            // 入队列
            events.offer(event);
            // 如果wakeupCounter=-1，说明正在select中，还没结束select
            if (wakeupCounter.incrementAndGet() == 0) {
                // 唤醒Selector
                selector.wakeup();
            }
        }

NioEndpoint$Poller#run

        @Override
        public void run() {
            // Loop until destroy() is called
            while (true) {
                boolean hasEvents = false;
                try {
                    // 如果未关闭Poller
                    if (!close) {
                        // 遍历执行所有已注册的事件
                        hasEvents = events();
                        // 有事件加入
                        // CAS，先get原值，再set为-1，如果原值大于0
                        if (wakeupCounter.getAndSet(-1) > 0) {
                            // 非阻塞查询就绪事件数量
                            keyCount = selector.selectNow();
                        } else {
                            // 阻塞查询就绪事件数量，默认1s
                            keyCount = selector.select(selectorTimeout);
                        }
                        // select完成后恢复为0
                        wakeupCounter.set(0);
                    }
                    // 如果关闭Poller
                    if (close) {
                        // 关闭前要全部执行完
                        events();
                        timeout(0, false);
                        try {
                            selector.close();
                        } catch (IOException ioe) {
                            // ...
                        }
                        break;
                    }
                } catch (Throwable x) {
                    // ...
                }
                // 超时或唤醒，先处理events
                if (keyCount == 0) {
                    hasEvents = (hasEvents | events());
                }
                // 获取就绪事件
                Iterator<SelectionKey> iterator = keyCount > 0 ? selector.selectedKeys().iterator() : null;
                // 遍历就绪的key，比如Socket可读等
                while (iterator != null && iterator.hasNext()) {
                    SelectionKey sk = iterator.next();
                    NioSocketWrapper socketWrapper = (NioSocketWrapper) sk.attachment();
                    // 如果别的线程调用了Poller#cancelledKey关闭连接，Attachment可能是null
                    if (socketWrapper == null) {
                        iterator.remove();
                    } else {
                        // 处理前要先移除
                        iterator.remove();
                        // 处理就绪的事件，生成SocketProcessor交给Worker线程池执行
                        processKey(sk, socketWrapper);
                    }
                }
                // 回收废弃的KeepAlive长连接，每执行一遍完整的轮询，就遍历所有关心的选择键，查看通道的上次读写时间
                // 超时就移除之前注册的事件，并向Worker线程池提交SocketEvent.ERROR事件，最终关闭连接
                timeout(keyCount, hasEvents);
            } // while循环结束
            // 不再执行，释放Poller
            getStopLatch().countDown();
        }

NioEndpoint$Poller#events

        public boolean events() {
            boolean result = false;
            PollerEvent pe = null;
            // 遍历事件，出队列
            for (int i = 0, size = events.size(); i < size && (pe = events.poll()) != null; i++) {
                result = true;
                try {
                    // 依次执行，PollerEvent#run
                    pe.run();
                    // 执行完后重置
                    pe.reset();
                    if (running && !paused && eventCache != null) {
                        // 放回对象池
                        eventCache.push(pe);
                    }
                } catch (Throwable x) {
                    // ...
                }
            }
            return result;
        }

NioEndpoint$PollerEvent#run

        @Override
        public void run() {
            // 注册事件
            if (interestOps == OP_REGISTER) {
                try {
                    // 把客户端的新连接SocketChannel注册到Poller的Selector
                    socket.getIOChannel().register(socket.getSocketWrapper().getPoller().getSelector(),
                            // 关心可读事件
                            SelectionKey.OP_READ,
                            // 设置Attachment
                            socket.getSocketWrapper());
                } catch (Exception x) {
                    // ...
                }
            }
            // 其它事件
            else {
                // 获取Socket注册到Selector的SelectionKey
                final SelectionKey key = socket.getIOChannel().keyFor(socket.getSocketWrapper().getPoller().getSelector());
                try {
                    // 选择键被取消和从Selector移除，比如由于Socket关闭
                    if (key == null) {
                        try {
                            // 将ConnectionLatch倒数，因为Socket关闭的时候没有倒数
                            socket.socketWrapper.close();
                        } catch (Exception ignore) {
                        }
                    }
                    // key不为null
                    else {
                        final NioSocketWrapper socketWrapper = (NioSocketWrapper) key.attachment();
                        if (socketWrapper != null) {
                            // We are registering the key to start with, reset the fairness counter.
                            int ops = key.interestOps() | interestOps;
                            // 保存一份
                            socketWrapper.interestOps(ops);
                            // 修改关心的事件
                            key.interestOps(ops);
                        } else {
                            socket.getSocketWrapper().getPoller().cancelledKey(key, socket.getSocketWrapper());
                        }
                    }
                } catch (CancelledKeyException ckx) {
                    try {
                        socket.getSocketWrapper().getPoller().cancelledKey(key, socket.getSocketWrapper());
                    } catch (Exception ignore) {
                    }
                }
            }
        }

NioEndpoint$Poller#processKey

        protected void processKey(SelectionKey sk, NioSocketWrapper socketWrapper) {
            try {
                // Poller关闭
                if (close) {
                    // 取消
                    cancelledKey(sk, socketWrapper);
                }
                // 没关闭，有效
                // SelectionKey#cancel方法取消键时不会立即从Selector移除，而是添加到cancelledKeys中，下一次select时移除，所以要用isValid校验
                else if (sk.isValid() && socketWrapper != null) {
                    // 可读或可写
                    if (sk.isReadable() || sk.isWritable()) {
                        // 文件传输，compression节省带宽，sendfile节省CPU
                        if (socketWrapper.getSendfileData() != null) {
                            // 使用NIO的MappedByteBuffer
                            processSendfile(sk, socketWrapper, false);
                        }
                        // 其它
                        else {
                            // 处理之前，取消注册的事件，防止多线程同时读一个Socket
                            // 假如一个HTTP请求只收到一部分，SocketProcessor在等待另一部分，当数据到来时会触发可读事件，如果又重新创建了一个SocketProcess，会有并发处理问题
                            unreg(sk, socketWrapper, sk.readyOps());
                            boolean closeSocket = false;
                            // 先处理可读，处理请求
                            if (sk.isReadable()) {
                                if (socketWrapper.readOperation != null) {
                                    if (!socketWrapper.readOperation.process()) {
                                        closeSocket = true;
                                    }
                                }
                                // 处理可读事件，通过SocketProcessor分发到Worker线程池，返回处理结果标志
                                else if (!processSocket(socketWrapper, SocketEvent.OPEN_READ, true)) {
                                    closeSocket = true;
                                }
                            }
                            // 再处理可写，响应回复
                            if (!closeSocket && sk.isWritable()) {
                                if (socketWrapper.writeOperation != null) {
                                    if (!socketWrapper.writeOperation.process()) {
                                        closeSocket = true;
                                    }
                                } else if (!processSocket(socketWrapper, SocketEvent.OPEN_WRITE, true)) {
                                    closeSocket = true;
                                }
                            }
                            if (closeSocket) {
                                cancelledKey(sk, socketWrapper);
                            }
                        }
                    }
                }
                // 其它，取消无效的key
                else {
                    cancelledKey(sk, socketWrapper);
                }
            } catch (CancelledKeyException ckx) {
                // ...
            } catch (Throwable t) {
                // ...
            }
        }

AbstractEndpoint#processSocket

    public boolean processSocket(SocketWrapperBase<S> socketWrapper, SocketEvent event, boolean dispatch) {
        try {
            if (socketWrapper == null) {
                return false;
            }
            SocketProcessorBase<S> sc = null;
            if (processorCache != null) {
                // 先从对象池获取SocketProcessor
                sc = processorCache.pop();
            }
            if (sc == null) {
                // 没有再新建，包含了SocketWrapper和SocketEvent
                sc = createSocketProcessor(socketWrapper, event);
            } else {
                // 有就重置后再用
                sc.reset(socketWrapper, event);
            }
            // 获取Worker线程池
            Executor executor = getExecutor();
            // 分发到线程池执行
            if (dispatch && executor != null) {
                // SocketProcessor#doRun
                executor.execute(sc);
            }
            // 直接执行
            else {
                sc.run();
            }
        } catch (RejectedExecutionException ree) {
            // ...
            return false;
        } catch (Throwable t) {
            // ...
            return false;
        }
        return true;
    }

NioEndpoint#stopInternal

    @Override
    public void stopInternal() {
        if (!paused) {
            pause();
        }
        if (running) {
            // 设置标识，通知Acceptor线程跳出run循环
            running = false;
            if (poller != null) {
                // 通知Poller线程关闭
                poller.destroy();
                poller = null;
            }
            try {
                // 等待Poller线程结束，Poller#run -> Latch#countDown
                // 有阻塞超时
                if (!getStopLatch().await(selectorTimeout + 100, TimeUnit.MILLISECONDS)) {
                    // ...
                }
            } catch (InterruptedException e) {
                // ...
            }
            // 关闭Worker线程池
            shutdownExecutor();
            if (eventCache != null) {
                eventCache.clear();
                eventCache = null;
            }
            if (nioChannels != null) {
                nioChannels.clear();
                nioChannels = null;
            }
            if (processorCache != null) {
                processorCache.clear();
                processorCache = null;
            }
        }
    }

总结：

Connector内部有一个ProtocolHandler，用来处理请求，ProtocolHandler内部又包含了：Endpoint、Adapter。

  • Endpoint处理底层Socket的网络连接（TCP/IP协议相关）

    • Endpoint内部有个Acceptor，Acceptor有一个线程，用于监听端口，阻塞地接收Socket连接请求，并交给Poller处理后续的事件。

    • Endpoint内部有个Poller，Poller有一个Selector和线程，轮询就绪的事件，并封装Socket和事件提交给工作线程池处理。

    • Endpoint内部有个Handler，为Socket关联对应的Processor，由Processor处理对应的事件。

      • Processor将缓冲的字节解析为HTTP协议，生成Request和Response，交给Adapter处理。

  • Adapter是个转接，将Request和Response转换为HttpServletRequest和HttpServletResponse，决定请求的Pipeline，交给Container进行具体的处理。

关于Poller，Poller在将SocketProcessor提交到工作线程池前，会向Selector取消之前SocketChannel关注的事件，避免多线程重复处理。

后续该Socket的读写事件是由NioSelectorPool里的Selector维护的，默认共享，所有SocketChannel共用一个。
