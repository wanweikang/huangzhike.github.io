TreeMap#put

    public V put(K key, V value) {
        Entry<K, V> t = root;
        // 原本没有根节点，第一次插入
        if (t == null) {
            // 自己比较自己，大概是先检测一下比较方法会不会抛异常？
            compare(key, key); // type (and possibly null) check
            // 新建节点，并设自己为根节点
            root = new Entry<>(key, value, null);
            size = 1;
            modCount++;
            return null;
        }
        // 从根节点开始插入
        int cmp;
        Entry<K, V> parent;
        // split comparator and comparable paths
        Comparator<? super K> cpr = comparator;
        // 如果TreeMap设置了Comparator，优先使用Comparator比较
        if (cpr != null) {
            // 遍历树
            do {
                parent = t;
                // 比较要插入的节点和当前节点的大小
                cmp = cpr.compare(key, t.key);
                if (cmp < 0)
                    // 比当前节点小，从当前节点的左孩子继续
                    t = t.left;
                else if (cmp > 0)
                    // 比当前节点大，从当前节点的右孩子继续
                    t = t.right;
                else
                    // 相等，说明节点已存在，直接更新值
                    return t.setValue(value);
            } while (t != null);
        } else {
            if (key == null)
                throw new NullPointerException();
            @SuppressWarnings("unchecked") Comparable<? super K> k = (Comparable<? super K>) key;
            do {
                parent = t;
                cmp = k.compareTo(t.key);
                if (cmp < 0)
                    t = t.left;
                else if (cmp > 0)
                    t = t.right;
                else
                    return t.setValue(value);
            } while (t != null);
        }
        // 如果遍历树直到叶子节点，还没有找到相等的节点，说明是新增的节点，要插入新的节点到该叶子节点下
        Entry<K, V> e = new Entry<>(key, value, parent);
        if (cmp < 0)
            // 插左边
            parent.left = e;
        else
            // 插右边
            parent.right = e;
        // 插入后，从插入的节点开始平衡
        fixAfterInsertion(e);
        size++;
        modCount++;
        return null;
    }

TreeMap#compare

    @SuppressWarnings("unchecked")
    final int compare(Object k1, Object k2) {
        // 优先使用Comparator，其次才是元素的Comparable#compareTo
        // 因为Comparator比较灵活，而Comparable需要继承Comparable接口，对于某些final类如String，不方便自定义，当然String是继承了Comparable的
        return comparator == null ? ((Comparable<? super K>) k1).compareTo((K) k2) : comparator.compare((K) k1, (K) k2);
    }

TreeMap#fixAfterInsertion

    private void fixAfterInsertion(Entry<K, V> x) {
        // 新插入的节点为红色
        x.color = RED;
        // 从新插入的叶子节点开始，向上遍历，直到根节点，如果当前遍历的节点的父节点是红色，说明需要平衡（两个连续的红色）
        while (x != null && x != root && x.parent.color == RED) {
            // 当前节点的父亲 == 当前节点的爷爷的左孩子
            // 就是说当前节点的父节点是不是左节点，如果是
            if (parentOf(x) == leftOf(parentOf(parentOf(x)))) {
                // 获取当前节点的父亲的右兄弟，即叔叔
                Entry<K, V> y = rightOf(parentOf(parentOf(x)));
                // 如果自己，爸爸，叔叔都是红色
                if (colorOf(y) == RED) {
                    // 将爸爸设为黑色
                    setColor(parentOf(x), BLACK);
                    // 将叔叔也设为黑色，因为两条路的黑色数量必须一样
                    setColor(y, BLACK);
                    // 把爷爷设成红色，爷爷下面的节点都是已经处理好的了，可以不管
                    setColor(parentOf(parentOf(x)), RED);
                    // 下次循环从爷爷开始处理红色冲突
                    x = parentOf(parentOf(x));
                }
                // 如果自己和爸爸是红色，但是叔叔是黑色，这种情况因为不对称，不能直接设成黑色，只能改变路线
                else {
                    // 如果自己是爸爸的右孩子
                    if (x == rightOf(parentOf(x))) {
                        // 下次处理的开始位置不变，但是人变了
                        x = parentOf(x);
                        // 把爸爸左旋，即自己顶上爸爸的位置，爸爸下降为自己的左孩子
                        rotateLeft(x);
                    }
                    // 把新爸爸设为黑色
                    setColor(parentOf(x), BLACK);
                    // 先把爷爷设为红色
                    setColor(parentOf(parentOf(x)), RED);
                    // 再把爷爷右旋，即新爸爸顶上爷爷的位置，爷爷下降为新爸爸的右孩子
                    rotateRight(parentOf(parentOf(x)));
                }
            }
            // 如果当前节点的父节点是右节点，对称，略
            else {
                // ...
            }
        }
        // 根节点是黑色
        root.color = BLACK;
    }

TreeMap#getEntry

    final Entry<K, V> getEntry(Object key) {
        // Offload comparator-based version for sake of performance
        // 有先使用Comparator比较获取
        if (comparator != null)
            return getEntryUsingComparator(key);
        if (key == null)
            throw new NullPointerException();
        @SuppressWarnings("unchecked") Comparable<? super K> k = (Comparable<? super K>) key;
        Entry<K, V> p = root;
        while (p != null) {
            int cmp = k.compareTo(p.key);
            if (cmp < 0)
                p = p.left;
            else if (cmp > 0)
                p = p.right;
            else
                return p;
        }
        return null;
    }

TreeMap#remove

    /**
     * 找到删除的节点，找到替补节点，替补顶上，重新平衡
     */
    public V remove(Object key) {
        // 先遍历获取要删除的节点
        Entry<K, V> p = getEntry(key);
        if (p == null)
            return null;
        V oldValue = p.value;
        // 再从该节点开始删除
        deleteEntry(p);
        return oldValue;
    }

TreeMap#deleteEntry

    private void deleteEntry(Entry<K, V> p) {
        modCount++;
        size--;
        // If strictly internal, copy successor's element to p and then make p point to successor.
        // 如果要删除的节点有左右两个子节点
        if (p.left != null && p.right != null) {
            // 找到比该节点大的最小节点，就是继承者，顶替要删除的节点的位置
            Entry<K, V> s = successor(p);
            // 用继承者的值替代要删除的节点，那么现在树里边就存在两个继承者的值
            p.key = s.key;
            p.value = s.value;
            // 继承者节点就变成了要删除的节点，并且这个继承者一定没有左孩子
            // 转化成了要删除的节点有一个或没有子节点的情况
            p = s;
        } // p has 2 children
        // Start fixup at replacement node, if it exists.
        // 如果要删除的节点没有子节点，不需要替带节点
        // 如果要删除的节点只有一个子节点，该子节点就是替代节点
        Entry<K, V> replacement = (p.left != null ? p.left : p.right);
        // 要删除的节点只有一个子节点
        if (replacement != null) {
            // Link replacement to parent
            // 将替代节点的父指针指向要删除节点的父亲
            replacement.parent = p.parent;
            // 如果要删除的节点没有父亲，说明是根节点，那么替代节点就要成为新的根节点
            if (p.parent == null)
                root = replacement;
                // 如果要删除的节点是左孩子，那么将父亲的左指针指向替代节点
            else if (p == p.parent.left)
                p.parent.left = replacement;
                // 如果要删除的节点是右孩子，那么将父亲的右指针指向替代节点
            else
                p.parent.right = replacement;
            // Null out links so they are OK to use by fixAfterDeletion.
            // 将要删除的节点的所有指针置空，从树中移除
            p.left = p.right = p.parent = null;
            // Fix replacement
            // 如果删除的节点是黑色的，那么从替代节点开始平衡
            if (p.color == BLACK)
                fixAfterDeletion(replacement);
        }
        // 要删除的节点没有子节点，并且是根节点，说明这棵树只有一个节点
        else if (p.parent == null) { // return if we are the only node.
            root = null;
        }
        // 要删除的节点没有子节点
        else { //  No children. Use self as phantom replacement and unlink.
            // 如果要删除的节点是黑色的，那么先把要删除的节点当成替代节点，假设树已经少了一个黑色，开始平衡，平衡完成后就会多出一个黑色
            if (p.color == BLACK)
                fixAfterDeletion(p);
            // 然后再删除黑色，这样多出来的黑色就被移除，平衡才真正完成
            // 如果要删除的节点有父亲，把父亲对它的引用置空
            if (p.parent != null) {
                if (p == p.parent.left)
                    p.parent.left = null;
                else if (p == p.parent.right)
                    p.parent.right = null;
                p.parent = null;
            }
        }
    }

TreeMap#successor

    static <K, V> Entry<K, V> successor(Entry<K, V> t) {
        if (t == null)
            return null;
            // 如果右侧有子树，从右子树开始向左找到最左的节点，即大于该节点的最小节点，并返回
        else if (t.right != null) {
            Entry<K, V> p = t.right;
            while (p.left != null)
                p = p.left;
            return p;
        }
        // 如果右侧没有子树，那就从父节点向上回溯，第一个是左孩子的节点，就是大于要找节点的最小节点，并返回
        else {
            Entry<K, V> p = t.parent;
            Entry<K, V> ch = t;
            while (p != null && ch == p.right) {
                ch = p;
                p = p.parent;
            }
            return p;
        }
    }

TreeMap#fixAfterDeletion

    private void fixAfterDeletion(Entry<K, V> x) {
        // 从要调整的节点开始向上遍历修复，当前节点是黑色是才处理
        // 因为这个方法是处理缺失一个黑色节点的情况，如果当前节点是红色的，那直接把当前节点设为黑色就可以补上了，不需要循环
        while (x != root && colorOf(x) == BLACK) {
            // 当前的节点是左孩子
            if (x == leftOf(parentOf(x))) {
                // 获取当前节点的兄弟节点
                Entry<K, V> sib = rightOf(parentOf(x));
                // 如果兄弟节点是红色
                if (colorOf(sib) == RED) {
                    // 将兄弟节点设为黑色
                    setColor(sib, BLACK);
                    // 将父节点设为红色
                    setColor(parentOf(x), RED);
                    // 将父节点左旋
                    rotateLeft(parentOf(x));
                    // 父节点左旋后，当前节点的兄弟节点也变了
                    sib = rightOf(parentOf(x));
                }
                // 如果兄弟节点的两个孩子都是黑色
                if (colorOf(leftOf(sib)) == BLACK && colorOf(rightOf(sib)) == BLACK) {
                    // 将兄弟设为红色
                    setColor(sib, RED);
                    // 下个循环从父节点继续，如果父节点是黑色，继续处理，如果是红色，改为黑色，不再处理
                    x = parentOf(x);
                }
                // 如果兄弟节点有红色的孩子存在
                else {
                    // 如果兄弟节点的右孩子是黑色
                    if (colorOf(rightOf(sib)) == BLACK) {
                        // 将兄弟节点的左孩子设为黑色
                        setColor(leftOf(sib), BLACK);
                        // 将兄弟节点设为红色
                        setColor(sib, RED);
                        // 将兄弟节点右旋
                        rotateRight(sib);
                        // 兄弟节点左旋后，当前节点的兄弟节点也变了
                        sib = rightOf(parentOf(x));
                    }
                    // 设置兄弟节点为父节点的颜色
                    setColor(sib, colorOf(parentOf(x)));
                    // 设置父亲为黑色
                    setColor(parentOf(x), BLACK);
                    // 设置兄弟节点的右孩子为黑色
                    setColor(rightOf(sib), BLACK);
                    // 将父节点左旋
                    rotateLeft(parentOf(x));
                    // 平衡结束，不需要继续了
                    x = root;
                }
            }
            // 要删除的节点是右孩子，对称操作，略
            else { // symmetric
                // ...
            }
        } // 处理结束
        // 把最后处理的节点设为黑色
        setColor(x, BLACK);
    }

TreeMap的迭代器很多，以常用的为例（返回集合，再获取迭代器）：

  • TreeMap#keySet

  • TreeMap#values

  • TreeMap#entrySet

TreeMap的特点是有序，遍历是有序的，由小到大，类似二叉树的中序遍历，但是这种方式明显效率不好的，因为首先从根节点向左下遍历到最小的节点，再继续遍历到次小的节点，这期间不知要经过多少个节点。

代码就不贴了，这三个迭代器遍历的单位都是Entry，即节点。

但是KeyIterator只暴露了Key，ValueIterator只暴露了Value，EntryIterator暴露了Entry。

如果通过KeyIterator遍历Key，然后再通过Key获取Value，这又得重新遍历查询一遍，所以一般都用EntryIterator，方便。
