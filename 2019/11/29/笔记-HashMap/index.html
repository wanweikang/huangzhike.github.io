HashMap

public class HashMap<K, V> extends AbstractMap<K, V> implements Map<K, V>, Cloneable, Serializable {

    /**
     * 默认初始容量，指的是数组的长度，不是哈希表中的节点数量，1 * 2^4 = 16，必须是2的幂
     * 如果capacity为2的指数幂，在寻址时，hash & (capacity - 1) == hash % capacity，相当于取模，提高计算速度
     * 而且因为数组的容量一般都不会太大，也就是说(capacity - 1)的高位基本都是0，那么参与&哈希的有效数字基本来自低位
     * 如果是2的幂，如16，二进制表示为10000，(capacity - 1) = 15，二进制为1111，使得&哈希值的二进制位全为1，散列范围更广，减少哈希冲突
     * 而且这样方便扩容，相同的哈希重新定位时，要么在原位置，要么后移原容量
     */
    static final int DEFAULT_INITIAL_CAPACITY = 1 << 4; // aka 16

    /**
     * 最大容量，指的是数组的最大长度，1 * 2^30
     */
    static final int MAXIMUM_CAPACITY = 1 << 30;

    /**
     * 默认的加载因子
     */
    static final float DEFAULT_LOAD_FACTOR = 0.75f;

    /**
     * 树化阈值，链表长度到达时需要转化为红黑树
     */
    static final int TREEIFY_THRESHOLD = 8;

    /**
     * 非树化阈值，扩容时，尝试将树切分为两条链表，当链表长度小于此阈值时，将链表转化为普通链表
     */
    static final int UNTREEIFY_THRESHOLD = 6;

    /**
     * 最小树化容量，当数组长度未达到时不会进行红黑树转化，只是扩容和拆分链表
     */
    static final int MIN_TREEIFY_CAPACITY = 64;

    /**
     * 根节点数组集合，用transient说明不能序列化，百度了一下，找到一个解释：
     * HashMap的哈希值是依赖Object#hashCode方法的，这个方法是native方法，不同的虚拟机实现可能不一样，所以是为了避免跨虚拟机时的问题
     * 序列化时是通过writeObject方法的，遍历元素序列化，这样可以根据元素的数量而不是数组的长度控制消耗的空间
     */
    transient Node<K, V>[] table;

    /**
     * 键值对集合
     */
    transient Set<Entry<K, V>> entrySet;

    /**
     * 哈希表的大小，节点的数量
     */
    transient int size;

    /**
     * 哈希表的修改次数，FailFast
     */
    transient int modCount;

    /**
     * 哈希表的阈值，节点数量达到后就会扩容，等于容量乘以加载因子
     * 不过为什么是数组的容量*加载因子呢？这个加载因子是怎么算的？
     * 一般情况下都是直接原阈值翻倍，数组扩容也是直接翻倍
     */
    int threshold;

    /**
     * 加载因子，开始扩容时的元素比例
     * 太大会导致冲突变多，查找效率下降，太小会频繁扩容，影响性能
     * 所以初始化时应该设置合理的容量和加载因子
     */
    final float loadFactor;

}

HashMap#tableSizeFor

    /**
     * 获取大于等于指定容量的最小2次幂
     */
    static final int tableSizeFor(int cap) {
        // 避免传入2次幂，如8时，返回16
        int n = cap - 1;
        n |= n >>> 1;
        n |= n >>> 2;
        n |= n >>> 4;
        n |= n >>> 8;
        n |= n >>> 16;
        return (n < 0) ? 1 : (n >= MAXIMUM_CAPACITY) ? MAXIMUM_CAPACITY : n + 1;
    }

HashMap#hash

    static final int hash(Object key) {
        int h;
        /*
         * 因为Object#hashCode低位相同的可能性比较大
         * 把一个32位的Int类型无符号右移16位，相当于把高位移到低位，再和原值做异或运算，这样可以降低哈希冲突的几率
         * 对于Key为null的情况，放在数组的第一个位置
         */
        return (key == null) ? 0 : (h = key.hashCode()) ^ (h >>> 16);
    }

HashMap#putVal

    /**
     * 线程不安全，假如两个线程同时Put数据，并且此时两个Key发生了冲突，则可能有一个线程的值会被覆盖
     */
    final V putVal(int hash, K key, V value, boolean onlyIfAbsent, boolean evict) {
        Node<K, V>[] tab;
        Node<K, V> p;
        int n, i;
        // 如果数组为空，还没添加数据，即第一次Put
        if ((tab = table) == null || (n = tab.length) == 0)
            // 扩容，初始化数组，返回初始化后的数组长度
            n = (tab = resize()).length;
        // 返回数组中该哈希对应的节点，如果还没被占位
        if ((p = tab[i = (n - 1) & hash]) == null)
            // 直接创建一个新节点插入数组
            tab[i] = newNode(hash, key, value, null);
        else {
            // 说明发生了哈希冲突
            Node<K, V> e;
            K k;
            // 如果已经存在的Prev节点的哈希和新节点的哈希值相等 && 两个节点的Key也一样，==或equals，所以Key需要重写equals方法
            // 找到了Key对应的节点
            if (p.hash == hash && ((k = p.key) == key || (key != null && key.equals(k))))
                e = p;
            else if (p instanceof TreeNode)
                // 如果Key不相等，并且已存在的节点是红黑树的节点，那么遍历树，如果找到Key对应的节点就返回该节点，否则插入新的节点
                e = ((TreeNode<K, V>) p).putTreeVal(this, tab, hash, key, value);
            else {
                // 如果Key不相等，并且之前已存在的节点是链表的节点，那么从头遍历链表，同时累计链表长度
                for (int binCount = 0; ; ++binCount) {
                    // 下一个节点不存在，说明到了链表尾
                    if ((e = p.next) == null) {
                        // 创建新的链表节点，将新节点插入链表尾
                        p.next = newNode(hash, key, value, null);
                        // 如果加入新节点后，链表的长度到达了阈值，那么把链表转化为红黑树
                        if (binCount >= TREEIFY_THRESHOLD - 1) // -1 for 1st
                            treeifyBin(tab, hash);
                        break;
                    }
                    // 如果在链表中找到了哈希值相等的节点 && 两个节点的Key也一样，==或equals
                    // 找到了Key对应的节点
                    if (e.hash == hash && ((k = e.key) == key || (key != null && key.equals(k))))
                        break;
                    // 下一个节点继续遍历
                    p = e;
                }
            }
            // 说明找到了Key对应的节点
            if (e != null) { // existing mapping for key
                // 原本的值
                V oldValue = e.value;
                // 是否覆盖旧值
                if (!onlyIfAbsent || oldValue == null)
                    e.value = value;
                // 这个回调也是空方法，LinkedHashMap实现
                afterNodeAccess(e);
                // 返回旧的值
                return oldValue;
            }
        }
        // 修改次数自增
        ++modCount;
        // 节点的数量自增，如果HashMap中的元素数量大于阈值，那么扩容
        if (++size > threshold)
            resize();
        // 空方法，插入节点后的回调，LinkedHashMap实现
        afterNodeInsertion(evict);
        // 说明不存在相等的Key
        return null;
    }

HashMap#resize

    final Node<K, V>[] resize() {
        Node<K, V>[] oldTab = table;
        // 扩容前的数组容量
        int oldCap = (oldTab == null) ? 0 : oldTab.length;
        // 扩容前的阈值
        int oldThr = threshold;
        int newCap, newThr = 0;
        // 如果数组已经初始化
        if (oldCap > 0) {
            // 如果数组容量超出了最大容量，设置新阈值为最大，并且不再扩容
            if (oldCap >= MAXIMUM_CAPACITY) {
                threshold = Integer.MAX_VALUE;
                return oldTab;
            }
            // 否则扩容，数组容量翻倍
            // 如果扩容后的数组容量小于最大容量限制，并且扩容前的数组容量大于等于默认容量，那么阈值翻倍
            else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY && oldCap >= DEFAULT_INITIAL_CAPACITY)
                newThr = oldThr << 1; // double threshold
        }
        // 如果数组还没初始化，并且扩容前的阈值大于零，即初始化HashMap时设置了阈值，第一次Put的情况
        else if (oldThr > 0) // initial capacity was placed in threshold
            // 初始化数组的长度为阈值的大小
            newCap = oldThr;
            // 数组还没初始化，并且原本的阈值等于零，即初始化HashMap时不传参数，第一次Put的情况
        else {               // zero initial threshold signifies using defaults
            // 初始化数组容量为默认的初始容量
            newCap = DEFAULT_INITIAL_CAPACITY;
            // 初始化阈值为默认加载因子与默认初始容量的乘积
            newThr = (int) (DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);
        }
        // 如果新的阈值为零，那么把新的阈值改为数组容量与加载因子的乘积，但是不能超过最大容量限制
        if (newThr == 0) {
            float ft = (float) newCap * loadFactor;
            newThr = (newCap < MAXIMUM_CAPACITY && ft < (float) MAXIMUM_CAPACITY ? (int) ft : Integer.MAX_VALUE);
        }
        threshold = newThr;
        // 根据新的数组容量初始化新的数组
        @SuppressWarnings({"rawtypes", "unchecked"}) Node<K, V>[] newTab = (Node<K, V>[]) new Node[newCap];
        table = newTab;
        // 如果原本数组已经初始化，则需要扩容
        if (oldTab != null) {
            // 遍历原本数组的节点
            for (int j = 0; j < oldCap; ++j) {
                Node<K, V> e;
                // 获取数组对应位置的节点，如果存在
                if ((e = oldTab[j]) != null) {
                    // 先置空引用
                    oldTab[j] = null;
                    // 如果该节点没有后续节点
                    if (e.next == null)
                        // 根据哈希直接添加到新数组对应位置就行了
                        newTab[e.hash & (newCap - 1)] = e;
                        // 如果该节点是个红黑树节点
                    else if (e instanceof TreeNode)
                        // 从该根节点开始，把树拆分成两条链表或树，并添加到新数组
                        ((TreeNode<K, V>) e).split(this, newTab, j, oldCap);
                        // 如果该节点是个链表节点
                    else { // preserve order
                        Node<K, V> loHead = null, loTail = null;
                        Node<K, V> hiHead = null, hiTail = null;
                        Node<K, V> next;
                        // 开始遍历链表
                        do {
                            next = e.next;
                            // 将该链表拆分为高低两条链表，同时保持顺序
                            /*
                             * (n - 1) & hash)
                             * 假设数组长度为16，即10000，减1就是15，即01111，扩容后，数组长度由16变成32，即10000
                             * 情况一：假设哈希值为1353，即010101001001，这里右起第五位是0
                             * 16 & 1353 = 0, 15 & 1353 = 9, 31 & 1353 = 9
                             * 所以扩容后下标也是不变的
                             * 情况二：假设哈希值为1361，即010101010001，这里右起第五位是1
                             * 16 & 1361 = 16 !=0, 15 & 1361 = 1, 31 & 1361 = 17
                             * 所以扩容后下标后移了16，刚好是扩容的长度
                             */
                            if ((e.hash & oldCap) == 0) {
                                if (loTail == null)
                                    loHead = e;
                                else
                                    loTail.next = e;
                                loTail = e;
                            } else {
                                // ...
                            }
                        } while ((e = next) != null);
                        if (loTail != null) {
                            loTail.next = null;
                            // 低链表插入到新数组的位置和旧数组的位置一样
                            newTab[j] = loHead;
                        }
                        if (hiTail != null) {
                            hiTail.next = null;
                            // 高链表的插入位置后移扩容的长度
                            // 所以迁移的时候，节点要么在原位置，要么在原位置后移原容量长度
                            newTab[j + oldCap] = hiHead;
                        }
                    }
                }
            }
        }
        return newTab;
    }

HashMap$TreeNode#split

        final void split(HashMap<K, V> map, Node<K, V>[] tab, int index, int bit) {
            TreeNode<K, V> b = this;
            // Relink into lo and hi lists, preserving order
            // 切分成两条链表，保留顺序
            TreeNode<K, V> loHead = null, loTail = null;
            TreeNode<K, V> hiHead = null, hiTail = null;
            int lc = 0, hc = 0;
            // 红黑树节点是由链表节点转化而来的，保留了原本单向链表的后继节点的指针的同时，还加了一个前驱节点的指针，所以还是一个双向链表
            // 这里是从当前节点开始向后遍历，当成链表来使用，简化了遍历，不用像遍历二叉树那样繁琐低效
            for (TreeNode<K, V> e = b, next; e != null; e = next) {
                // 先获取当前节点的下一个节点
                next = (TreeNode<K, V>) e.next;
                // 再断开当前节点对它的引用
                e.next = null;
                // 如果把当前节点分到低链表
                if ((e.hash & bit) == 0) {
                    // 将当前节点追加到低链表尾，如果低链表还没有节点
                    if ((e.prev = loTail) == null)
                        // 那就是低链表的头节点
                        loHead = e;
                        // 已经存在尾节点
                    else
                        // 双向链表，还有一个指针
                        loTail.next = e;
                    // 更新低链表尾的节点
                    loTail = e;
                    // 统计新的低链表的节点数量
                    ++lc;
                }
                // 如果把当前节点分到高链表
                else {
                    // ...
                }
            }
            // 如果有节点分到低链表
            if (loHead != null) {
                // 如果低链表的长度小于阈值
                if (lc <= UNTREEIFY_THRESHOLD)
                    // 遍历将红黑树节点转为普通的链表节点，插入回数组的原位置
                    tab[index] = loHead.untreeify(map);
                else {
                    // 先把低链表的头节点插回原位
                    tab[index] = loHead;
                    // 如果有高链表，才把低链表转为新的红黑树
                    // 因为如果高链表不存在，说明原来所有的树节点仍在低链表上，树结构不变，不需要转化
                    if (hiHead != null) // (else is already treeified)
                        loHead.treeify(tab);
                }
            }
            if (hiHead != null) {
                if (hc <= UNTREEIFY_THRESHOLD)
                    // 分割树前是要扩容的，高链表就后移扩容的长度插入
                    tab[index + bit] = hiHead.untreeify(map);
                else {
                    tab[index + bit] = hiHead;
                    if (loHead != null)
                        hiHead.treeify(tab);
                }
            }
        }

HashMap$TreeNode#putTreeVal

        final TreeNode<K, V> putTreeVal(HashMap<K, V> map, Node<K, V>[] tab, int h, K k, V v) {
            Class<?> kc = null;
            boolean searched = false;
            // 红黑树的根节点
            TreeNode<K, V> root = (parent != null) ? root() : this;
            // 从根节点开始向下遍历
            for (TreeNode<K, V> p = root; ; ) {
                int dir, ph;
                K pk;
                // 先比较哈希值，如果要插入的节点的哈希值小于当前节点，那么要插到左边
                if ((ph = p.hash) > h)
                    dir = -1;
                    // 如果哈希值大于当前节点，那么要插到右边
                else if (ph < h)
                    dir = 1;
                    // 如果哈希值相等，并且Key也相等，== || equals，说明找到了节点，直接返回
                else if ((pk = p.key) == k || (k != null && k.equals(pk)))
                    return p;
                    // 如果哈希值相等，但是Key不相等
                    // 如果Key实现了Comparable接口，调用Comparable#compareTo比较，如果比较还相等
                    // 另外这里comparableClassFor方法会保存当前节点的类型，然后compareComparables方法比较时，如果和插入的节点不是同一类型，则认为相等
                    // 比如HashMap的Key是Object类型，当前节点的Key可能是String，新插入的节点的Key可能是Integer
                else if ((kc == null && (kc = comparableClassFor(k)) == null) || (dir = compareComparables(kc, k, pk)) == 0) {
                    // 只搜索一次，因为下面的方法会遍历子树比较，不需要重复查找
                    if (!searched) {
                        TreeNode<K, V> q, ch;
                        searched = true;
                        // 从该节点开始，先后遍历左右子树，如果找到了就直接返回找到的节点
                        if (((ch = p.left) != null && (q = ch.find(h, k, kc)) != null) || ((ch = p.right) != null && (q = ch.find(h, k, kc)) != null))
                            return q;
                    }
                    // 如果两者哈希值相等，但是子树无法找到对应的节点时，使用System#identityHashCode做最后的比较
                    // System#identityHashCode是根据内存地址计算哈希值的
                    dir = tieBreakOrder(k, pk);
                }
                TreeNode<K, V> xp = p;
                // 根据上面计算的值，判断插入到左还是右子节点
                if ((p = (dir <= 0) ? p.left : p.right) == null) {
                    Node<K, V> xpn = xp.next;
                    // 新建一个红黑树节点
                    TreeNode<K, V> x = map.newTreeNode(h, k, v, xpn);
                    // 插入
                    if (dir <= 0)
                        xp.left = x;
                    else
                        xp.right = x;
                    // 更新双向链表节点的指针
                    xp.next = x;
                    x.parent = x.prev = xp;
                    if (xpn != null)
                        ((TreeNode<K, V>) xpn).prev = x;
                    // 插入后，执行红黑树平衡，并用平衡后的根节点更新数组的位置，同时将新的根节点变成双向链表的头节点
                    moveRootToFront(tab, balanceInsertion(root, x));
                    return null;
                }
            }
        }

HashMap$TreeNode#find

        final TreeNode<K, V> find(int h, Object k, Class<?> kc) {
            // Present当前节点
            TreeNode<K, V> p = this;
            // 遍历二叉树
            do {
                int ph, dir;
                K pk;
                TreeNode<K, V> pl = p.left, pr = p.right, q;
                // 如果比当前节点的哈希小，从左孩子继续找
                if ((ph = p.hash) > h)
                    p = pl;
                    // 如果比当前节点的哈希大，从右孩子继续找
                else if (ph < h)
                    p = pr;
                    // 如果刚好找到，直接返回
                else if ((pk = p.key) == k || (k != null && k.equals(pk)))
                    return p;

                else if ((kc != null || (kc = comparableClassFor(k)) != null) && (dir = compareComparables(kc, k, pk)) != 0)
                    p = (dir < 0) ? pl : pr;

                else if ((q = pr.find(h, k, kc)) != null)
                    return q;
                // 如果右边子树找不到，下次再从左孩子继续
                else
                    p = pl;

        final void treeify(Node<K, V>[] tab) {
            TreeNode<K, V> root = null;
            // 从此头节点开始向后遍历
            for (TreeNode<K, V> x = this, next; x != null; x = next) {
                // 当前节点的后一个节点
                next = (TreeNode<K, V>) x.next;
                // 先把左右子节点置空
                x.left = x.right = null;
                // 第一次，还没设置树的根节点
                if (root == null) {
                    x.parent = null;
                    // 根节点为黑色
                    x.red = false;
                    // 设置头节点为根节点
                    root = x;
                }
                // 树已经有了根节点
                else {
                    K k = x.key;
                    int h = x.hash;
                    Class<?> kc = null;
                    // 从根节点开始向下遍历，和当前要插入的节点比较，两层循环
                    for (TreeNode<K, V> p = root; ; ) {
                        int dir, ph;
                        K pk = p.key;
                        // 先比较哈希值，如果要插入的节点比较小，那么准备插入到左边
                        if ((ph = p.hash) > h)
                            dir = -1;
                            // 如果要插入的节点比较大，那么准备插入到右边
                        else if (ph < h)
                            dir = 1;
                            // 如果哈希值相等，Comparable#compareTo也相等
                        else if ((kc == null && (kc = comparableClassFor(k)) == null) || (dir = compareComparables(kc, k, pk)) == 0)
                            // System#identityHashCode最后比较一次
                            dir = tieBreakOrder(k, pk);
                        TreeNode<K, V> xp = p;
                        // 更新下个遍历的节点，如果已经没有叶子节点了，说明到了底，准备插入
                        if ((p = (dir <= 0) ? p.left : p.right) == null) {
                            x.parent = xp;
                            if (dir <= 0)
                                xp.left = x;
                            else
                                xp.right = x;
                            // 插入后平衡红黑树，返回平衡后的根节点
                            root = balanceInsertion(root, x);
                            break;
                        }
                    }
                }
            }
            // 将平衡后的根节点，设为双向链表的头结点，并更新到数组对应的位置
            moveRootToFront(tab, root);
        }

                node = p;

            else if ((e = p.next) != null) {

                if (p instanceof TreeNode)
                    node = ((TreeNode<K, V>) p).getTreeNode(hash, key);
                    // 如果是个普通的链表节点，向后遍历链表找
                else {
                    do {
                        if (e.hash == hash && ((k = e.key) == key || (key != null && key.equals(k)))) {
                            node = e;
                            break;

                        p = e;
                    } while ((e = e.next) != null);
                }
