[笔记]-ArrayList

2019-11-29

jdk1.8.0_211

ArrayList

/**
 * 继承了AbstractList，是一个数组队列
 * 实现了RandomAccess标记接口，表示提供了快速随机访问功能
 * 实现了Cloneable标记接口，表示能被克隆
 * 实现Serializable标记接口，表示支持序列化
 */
public class ArrayList<E> extends AbstractList<E> implements List<E>, RandomAccess, Cloneable, java.io.Serializable {
    private static final long serialVersionUID = 8683452581122892189L;

    /**
     * 初始容量为10
     */
    private static final int DEFAULT_CAPACITY = 10;

    /*
     * 这两个都是空数组，不同的构造器初始化时使用不同的对象，用来区分性地扩容
     */

    /**
     * 当初始化时指定初始容量为零时使用
     */
    private static final Object[] EMPTY_ELEMENTDATA = {};

    /**
     * 当初始化时不指定初始容量时使用
     */
    private static final Object[] DEFAULTCAPACITY_EMPTY_ELEMENTDATA = {};

    /**
     * 底层使用数组保存元素，加了transient关键字表示该属性不能序列化
     * elementData是通过反射调用writeObject方法序列化的，一个个遍历元素
     * 百度原因：可以根据size序列化元素，而不是根据数组的长度序列化，节约空间，好吧
     */
    transient Object[] elementData; // non-private to simplify nested class access

    /**
     * 数组当前的容量
     */
    private int size;

    /**
     * 2^31 - 1 - 8
     * 关于减8，根据JDK的原注释，大概是虚拟机的数组实现问题
     */
    private static final int MAX_ARRAY_SIZE = Integer.MAX_VALUE - 8;

}

AbstractList

public abstract class AbstractList<E> extends AbstractCollection<E> implements List<E> {
    /**
     * 即ModifyCount，表示列表结构修改的次数
     */
    protected transient int modCount = 0;

}

构造方法

ArrayList

    /**
     * 根据指定容量构造一个空列表
     */
    public ArrayList(int initialCapacity) {
        // 指定了一个非空的容量
        if (initialCapacity > 0) {
            this.elementData = new Object[initialCapacity];
        }
        // 指定一个为零的容量
        else if (initialCapacity == 0) {
            this.elementData = EMPTY_ELEMENTDATA;
        } else {
            throw new IllegalArgumentException("Illegal Capacity: " + initialCapacity);
        }
    }

    /**
     * 根据默认容量构造一个空列表
     */
    public ArrayList() {
        this.elementData = DEFAULTCAPACITY_EMPTY_ELEMENTDATA;
    }

    /**
     * 从指定集合构造
     */
    public ArrayList(Collection<? extends E> c) {
        // 直接调用集合的Collection#toArray更新赋值
        elementData = c.toArray();
        // 更新列表元素的数量，如果不为空
        if ((size = elementData.length) != 0) {
            // Collection#toArray的返回类型可能不是Object[]，处理这个Bug
            if (elementData.getClass() != Object[].class)
                elementData = Arrays.copyOf(elementData, size, Object[].class);
        }
        // 空的就用空数组替代
        else {
            this.elementData = EMPTY_ELEMENTDATA;
        }
    }

添加元素时如果不添加到末尾，那么后面的元素全部要后移一位。

ArrayList#add

    /**
     * 添加元素到指定位置
     */
    public void add(int index, E element) {
        // 先检查索引是否在数组的范围内，否则抛出IndexOutOfBoundsException
        rangeCheckForAdd(index);
        // 先确保容量够，不够就要扩容，同时会增加ModCount
        // 传入添加该元素所需的最小容量
        ensureCapacityInternal(size + 1);  // Increments modCount!!
        // 后面的全部后移一位
        System.arraycopy(elementData, index, elementData, index + 1, size - index);
        // 将元素加入数组
        elementData[index] = element;
        // 增加列表元素的数量
        size++;
    }

ArrayList#ensureCapacityInternal

    private void ensureCapacityInternal(int minCapacity) {
        // 计算需要的容量，并自动扩容，如果需要的话
        ensureExplicitCapacity(calculateCapacity(elementData, minCapacity));
    }

ArrayList#calculateCapacity

    private static int calculateCapacity(Object[] elementData, int minCapacity) {
        // 默认的无参构造初始化的空列表，此时还没有添加元素
        if (elementData == DEFAULTCAPACITY_EMPTY_ELEMENTDATA) {
            // 返回两者的最大值，新的列表第一次添加数据会进来，默认容量10
            // 后面会自动扩容，也就是说，新的列表是空数组，第一次添加数据时才自动扩容
            return Math.max(DEFAULT_CAPACITY, minCapacity);
        }
        return minCapacity;
    }

ArrayList#ensureExplicitCapacity

    private void ensureExplicitCapacity(int minCapacity) {
        // 修改次数增加
        modCount++;
        // 需要扩容
        if (minCapacity - elementData.length > 0)
            grow(minCapacity);
    }

ArrayList#grow

    private void grow(int minCapacity) {
        // 原本数组的容量
        int oldCapacity = elementData.length;
        // 扩容后的新容量 = 原本容量 + 原本容量 / 2
        // 就是乘1.5
        int newCapacity = oldCapacity + (oldCapacity >> 1);
        // 如果小于需要的最小容量
        if (newCapacity - minCapacity < 0)
            // 更新
            newCapacity = minCapacity;
        // 处理极大值的问题，这操作没看懂
        if (newCapacity - MAX_ARRAY_SIZE > 0)
            newCapacity = hugeCapacity(minCapacity);
        // 创建新的数组并拷贝元素，也就是说每次扩容都会创建新的数组对象，再复制过去
        elementData = Arrays.copyOf(elementData, newCapacity);
    }

ArrayList#get

    public E get(int index) {
        // 越界检查，不能超过列表已有元素数量的大小，否则抛IndexOutOfBoundsException
        rangeCheck(index);
        // 直接根据索引从数组拿
        return elementData(index);
    }

ArrayList#elementData

    @SuppressWarnings("unchecked")
    E elementData(int index) {
        return (E) elementData[index];
    }

ArrayList#remove

    /**
     * 删除指定索引的元素
     */
    public E remove(int index) {
        // 越界检查，不能超过列表已有元素数量的大小，否则抛IndexOutOfBoundsException
        rangeCheck(index);
        // 修改次数增加，因为移除了元素
        modCount++;
        // 返回要删除位置的元素
        E oldValue = elementData(index);
        int numMoved = size - index - 1;
        if (numMoved > 0)
            // 后面的元素都需要前移一个位置
            System.arraycopy(elementData, index + 1, elementData, index, numMoved);
        elementData[--size] = null; // clear to let GC do its work
        return oldValue;
    }

ArrayList#remove

    /**
     * 删除指定元素
     */
    public boolean remove(Object o) {
        if (o == null) {
            for (int index = 0; index < size; index++)
                if (elementData[index] == null) {
                    // 快速删除，大概指的是省略了越界检查和返回值，其实也没快多少
                    fastRemove(index);
                    return true;
                }
        } else {
            for (int index = 0; index < size; index++)
                if (o.equals(elementData[index])) {
                    fastRemove(index);
                    return true;
                }
        }
        return false;
    }

ArrayList#size

    public int size() {
        // 直接返回数量，不用遍历计算
        return size;
    }

ArrayList#clone

    public Object clone() {
        try {
            ArrayList<?> v = (ArrayList<?>) super.clone();
            // 新的数组，但是元素还是原本的元素哦
            v.elementData = Arrays.copyOf(elementData, size);
            v.modCount = 0;
            return v;
        } catch (CloneNotSupportedException e) {
            // this shouldn't happen, since we are Cloneable
            throw new InternalError(e);
        }
    }

ArrayList$SubList

三种迭代器

  • ArrayList#iterator

  • ArrayList#listIterator

  • ArrayList#spliterator

关于Spliterator，这是JDK8加的，就是将元素分割为多份，多线程并行处理，对于ArrayList这种随机存取的还好，LinkedList这种的话，分割操作本身的性能应该就不太好吧。

ArrayList#iterator

    /**
     * 这里重写了父类AbstractList的方法，但是没加@Override
     */
    public Iterator<E> iterator() {
        return new Itr();
    }

ArrayList$Itr

    private class Itr implements Iterator<E> {
        /**
         * 迭代器有个游标，初始为0，虽然没有显式初始化，但是它是个原始类型，你懂的
         * 之所以叫游标而不叫索引是因为它不指向元素，而是指向元素之间，表示下一个迭代的元素位置，当调用Iterator#next时后移
         * 所以初始时在第一个元素前面，迭代完后在最后一个元素后面，即长度为N的集合有N+1个游标的位置
         */
        int cursor;       // index of next element to return

        /**
         * 上次迭代返回元素的索引
         */
        int lastRet = -1; // index of last element returned; -1 if no such

        int expectedModCount = modCount;
    }

ArrayList$Itr#hasNext

        /**
         * 游标后是否还有元素
         */
        public boolean hasNext() {
            return cursor != size;
        }

ArrayList$Itr#next

        @SuppressWarnings("unchecked")
        public E next() {
            // 返回下一个元素之前先检查ModCount和期待的ExpectedModCount是否一样
            // 不一样说明迭代过程中列表结构发生了修改，抛出ConcurrentModificationException，称为Fail-Fast
            checkForComodification();
            // 要迭代的下一个元素位置
            int i = cursor;
            // 越界检查，一般不会发生
            if (i >= size)
                throw new NoSuchElementException();
            Object[] elementData = ArrayList.this.elementData;
            // 正常调用Iterator#hasNext后再Iterator#next，游标是小于数组的大小的
            // 这里可能是多线程操作同一个迭代器Iterator，抛ConcurrentModificationException
            if (i >= elementData.length)
                throw new ConcurrentModificationException();
            // 游标后移
            cursor = i + 1;
            // 更新上次返回位置LastRet，同时也是当前迭代元素的索引，返回对应的数组元素
            return (E) elementData[lastRet = i];
        }

ArrayList$Itr#remove

        public void remove() {
            // 说明还没有调用Iterator#next，当然不能移除当前元素
            if (lastRet < 0)
                throw new IllegalStateException();
            // 检查ModCount，如果迭代过程中发生了修改，抛ConcurrentModificationException
            checkForComodification();
            try {
                // 调用列表的删除方法删除当前迭代的元素
                ArrayList.this.remove(lastRet);
                // 更新游标位置，因为删除了一个元素，所以其实是回退了一步
                cursor = lastRet;
                // 删除完恢复初始值
                lastRet = -1;
                // 更新修改后的期待值
                expectedModCount = modCount;
            } catch (IndexOutOfBoundsException ex) {
                throw new ConcurrentModificationException();
            }
        }

所以遍历时如果要移除List的元素，不能直接调用List#remove方法，而是要通过Iterator移除，否则会抛异常。

类似for...each这种方式遍历其实是个语法糖，反编译类文件后就会发现其实还是调用了迭代器。

可以看出迭代器还是多了挺多步骤的，所以只是简单的迭代的话，用普通的for (; ; ) { }会更加快些。
