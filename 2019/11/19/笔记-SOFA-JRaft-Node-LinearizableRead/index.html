[笔记]-SOFA-JRaft-Node-LinearizableRead

2019-11-19

关于一致性，这篇文章讲得挺好的。

线性一致读，不能直接从当前Leader节点读，因为无法确定当前Leader仍是Leader，比如网络分区。

Raft Log

同写请求一样，读请求也走一遍Raft协议，但开销较大。

ReadIndex Read

Leader将自己当前Log的CommitIndex记录到ReadIndex。

Leader向Follower发起一轮Heartbeat，如果半数以上节点返回了对应响应，那么就能确定自己仍是Leader。

Leader等待自己的状态机执行，直到ApplyIndex超过了ReadIndex，这样就能安全的提供线性一致读，不管读的时候是否仍是Leader。

Leader执行读请求，将结果返回给Client。

其实也可以从Follower读，但是当Follower进度落后太多的情况下，响应时间会变长。

Lease Read

类似ReadIndex，但省去了Log和网络交互。

Leader取一个比ElectionTimeout小一个数量级的租期（Lease），在租约期内不会发生选举，确保了Leader不会变，所以可以略过心跳。

Wait Free

节点有两个状态，日志状态和状态机状态，Leader的日志状态表示当前集群最新的数据状态，因为所有的写请求一定都先记录在日志中。

只要保证当前时刻，当前节点状态机是最新即可，但是Leader节点的状态机不一定是最新的，比如新当选的Leader。

当Leader已经成功应用了自己任期的第一条日志后，状态机状态一定是满足线性一致读的。

ReadOnlyServiceImpl#addRequest

    @Override
    public void addRequest(final byte[] reqCtx, final ReadIndexClosure closure) {
        // 服务正在关闭
        if (this.shutdownLatch != null) {
            Utils.runClosureInThread(closure, new Status(RaftError.EHOSTDOWN, "Was stopped"));
            throw new IllegalStateException("Service already shutdown.");
        }
        try {
            EventTranslator<ReadIndexEvent> translator = (event, sequence) -> {
                event.done = closure;
                event.requestContext = new Bytes(reqCtx);
                event.startTime = Utils.monotonicMs();
            };
            int retryTimes = 0;
            // 重试发布
            while (true) {
                // 如果发布到RingBuffer成功
                if (this.readIndexQueue.tryPublishEvent(translator)) {
                    break;
                } else {
                    retryTimes++;
                    // 超过发布重试次数
                    if (retryTimes > MAX_ADD_REQUEST_RETRY_TIMES) {
                        // 提交到线程池执行回调
                        Utils.runClosureInThread(closure, new Status(RaftError.EBUSY, "Node is busy, has too many read-only requests."));
                        // ...
                        return;
                    }
                    // 让一下线程
                    ThreadHelper.onSpinWait();
                }
            }
        } catch (final Exception e) {
            // 提交到线程池执行回调
            Utils.runClosureInThread(closure, new Status(RaftError.EPERM, "Node is down."));
        }
    }

ReadOnlyServiceImpl$ReadIndexEventHandler#onEvent

        @Override
        public void onEvent(final ReadIndexEvent newEvent, final long sequence, final boolean endOfBatch) throws Exception {
            // 要关闭
            if (newEvent.shutdownLatch != null) {
                // 先执行
                executeReadIndexEvents(this.events);
                this.events.clear();
                newEvent.shutdownLatch.countDown();
                return;
            }
            // 加入列表
            this.events.add(newEvent);
            // 到达批量大小
            if (this.events.size() >= ReadOnlyServiceImpl.this.raftOptions.getApplyBatch() || endOfBatch) {
                // 批量执行
                executeReadIndexEvents(this.events);
                this.events.clear();
            }
        }

ReadOnlyServiceImpl#executeReadIndexEvents

    private void executeReadIndexEvents(final List<ReadIndexEvent> events) {
        // ...
        final ReadIndexRequest.Builder rb = ReadIndexRequest.newBuilder() //
                .setGroupId(this.node.getGroupId()) //
                .setServerId(this.node.getServerId().toString());
        final List<ReadIndexState> states = new ArrayList<>(events.size());
        // 遍历读事件
        for (final ReadIndexEvent event : events) {
            rb.addEntries(ZeroByteStringHelper.wrap(event.requestContext.get()));
            states.add(new ReadIndexState(event.requestContext, event.done, event.startTime));
        }
        // 创建读请求
        final ReadIndexRequest request = rb.build();
        // 批处理，传入回调
        this.node.handleReadIndexRequest(request, new ReadIndexResponseClosure(states, request));
    }

NodeImpl#readLeader

    private void readLeader(final ReadIndexRequest request, final ReadIndexResponse.Builder respBuilder, final RpcResponseClosure<ReadIndexResponse> closure) {
        // 合法人数，即满足大多数节点的数量
        final int quorum = getQuorum();
        // 就一个节点，直接读
        if (quorum <= 1) {
            // Only one peer, fast path.
            respBuilder.setSuccess(true) //
                    .setIndex(this.ballotBox.getLastCommittedIndex());
            closure.setResponse(respBuilder.build());
            closure.run(Status.OK());
            return;
        }
        // 获取当前已日志提交的位置，由Follower的响应决定
        final long lastCommittedIndex = this.ballotBox.getLastCommittedIndex();
        // 如果Leader上台后还没提交过日志，拒绝读请求
        if (this.logManager.getTerm(lastCommittedIndex) != this.currTerm) {
            // Reject read only request when this leader has not committed any log entry at its term
            closure.run(new Status(RaftError.EAGAIN, "ReadIndex request rejected because leader has not committed any log entry at its term, logIndex=%d, currTerm=%d.", lastCommittedIndex, this.currTerm));
            return;
        }
        // 当前提交的位置
        respBuilder.setIndex(lastCommittedIndex);
        // 说明是来自Follower的读请求
        if (request.getPeerId() != null) {
            // request from follower, check if the follower is in current conf.
            final PeerId peer = new PeerId();
            peer.parse(request.getServerId());
            // 检查该Follower是否在配置中，不在就拒绝
            if (!this.conf.contains(peer)) {
                closure.run(new Status(RaftError.EPERM, "Peer %s is not in current configuration: {}.", peer, this.conf));
                return;
            }
        }
        // 默认ReadOnlySafe
        ReadOnlyOption readOnlyOpt = this.raftOptions.getReadOnlyOptions();
        // 如果选择了ReadOnlyLeaseBased但是Leader的租约超时了，需要切换为ReadOnlySafe
        if (readOnlyOpt == ReadOnlyOption.ReadOnlyLeaseBased && !isLeaderLeaseValid()) {
            // If leader lease timeout, we must change option to ReadOnlySafe
            readOnlyOpt = ReadOnlyOption.ReadOnlySafe;
        }
        switch (readOnlyOpt) {
            // 需要先向其它节点确认自己仍是Leader
            case ReadOnlySafe:
                final List<PeerId> peers = this.conf.getConf().getPeers();
                // ...
                // 收到足够心跳响应后执行回调
                final ReadIndexHeartbeatResponseClosure heartbeatDone = new ReadIndexHeartbeatResponseClosure(closure, respBuilder, quorum, peers.size());
                // 向其它Follower节点发送心跳
                for (final PeerId peer : peers) {
                    // 跳过自身
                    if (peer.equals(this.serverId)) {
                        continue;
                    }
                    this.replicatorGroup.sendHeartbeat(peer, heartbeatDone);
                }
                break;
            // 要求时钟不能漂移，在Leader当选后的一段时间内，如900ms，可以认为还是Leader，不需要心跳确认
            case ReadOnlyLeaseBased:
                // Responses to followers and local node.
                respBuilder.setSuccess(true);
                // 直接响应
                closure.setResponse(respBuilder.build());
                // 执行回调
                closure.run(Status.OK());
                break;
        }
    }

心跳回调

NodeImpl$ReadIndexHeartbeatResponseClosure#run

        @Override
        public synchronized void run(final Status status) {
            if (this.isDone) {
                return;
            }
            if (status.isOk() && getResponse().getSuccess()) {
                this.ackSuccess++;
            } else {
                this.ackFailures++;
            }
            // 加一是包含Leader自己的投票，如果得到了过半节点的承认，说明自己仍是Leader
            if (this.ackSuccess + 1 >= this.quorum) {
                // 成功
                this.respBuilder.setSuccess(true);
                // 设置响应
                this.closure.setResponse(this.respBuilder.build());
                // 执行回调
                this.closure.run(Status.OK());
                this.isDone = true;
            }
            // 过半节点都没有承认你，你还是自己下台吧
            else if (this.ackFailures >= this.failPeersThreshold) {
                // 失败
                this.respBuilder.setSuccess(false);
                this.closure.setResponse(this.respBuilder.build());
                this.closure.run(Status.OK());
                this.isDone = true;
            }
        }

响应回调

ReadOnlyServiceImpl$ReadIndexResponseClosure#run

        @Override
        public void run(final Status status) {
            if (!status.isOk()) {
                notifyFail(status);
                return;
            }
            // 之前生成的响应
            final ReadIndexResponse readIndexResponse = getResponse();
            if (!readIndexResponse.getSuccess()) {
                notifyFail(new Status(-1, "Fail to run ReadIndex task, maybe the leader stepped down."));
                return;
            }
            // Success
            final ReadIndexStatus readIndexStatus = new ReadIndexStatus(this.states, this.request, readIndexResponse.getIndex());
            for (final ReadIndexState state : this.states) {
                // 记录当前提交的Log的索引
                state.setIndex(readIndexResponse.getIndex());
            }
            boolean doUnlock = true;
            ReadOnlyServiceImpl.this.lock.lock();
            try {
                // 本地状态机已应用到该位置
                if (readIndexStatus.isApplied(ReadOnlyServiceImpl.this.fsmCaller.getLastAppliedIndex())) {
                    ReadOnlyServiceImpl.this.lock.unlock();
                    doUnlock = false;
                    // 执行成功回调
                    notifySuccess(readIndexStatus);
                }
                // 状态机还没应用到该位置
                else {
                    // 添加到Pending缓存
                    ReadOnlyServiceImpl.this.pendingNotifyStatus.computeIfAbsent(readIndexStatus.getIndex(), k -> new ArrayList<>(10)) //
                            .add(readIndexStatus);
                }
            } finally {
                if (doUnlock) {
                    ReadOnlyServiceImpl.this.lock.unlock();
                }
            }
        }

ReadOnlyServiceImpl#onApplied

    @Override
    public void onApplied(final long appliedIndex) {
        // TODO reuse pendingStatuses list?
        List<ReadIndexStatus> pendingStatuses = null;
        this.lock.lock();
        try {
            if (this.pendingNotifyStatus.isEmpty()) {
                return;
            }
            // 返回所有小于已应用位置的
            final Map<Long, List<ReadIndexStatus>> statuses = this.pendingNotifyStatus.headMap(appliedIndex, true);
            if (statuses != null) {
                pendingStatuses = new ArrayList<>(statuses.size() << 1);
                final Iterator<Map.Entry<Long, List<ReadIndexStatus>>> it = statuses.entrySet().iterator();
                while (it.hasNext()) {
                    final Map.Entry<Long, List<ReadIndexStatus>> entry = it.next();
                    pendingStatuses.addAll(entry.getValue());
                    // 移除
                    it.remove();
                }

            }
        } finally {
            this.lock.unlock();
            if (pendingStatuses != null && !pendingStatuses.isEmpty()) {
                for (final ReadIndexStatus status : pendingStatuses) {
                    // 通知
                    notifySuccess(status);
                }
            }
        }
    }

ReadOnlyServiceImpl#notifySuccess

    private void notifySuccess(final ReadIndexStatus status) {
        // ...
        final List<ReadIndexState> states = status.getStates();
        final int taskCount = states.size();
        for (int i = 0; i < taskCount; i++) {
            final ReadIndexState task = states.get(i);
            final ReadIndexClosure done = task.getDone(); // stack copy
            if (done != null) {
                // ...
                done.setResult(task.getIndex(), task.getRequestContext().get());
                done.run(Status.OK());
            }
        }
    }
