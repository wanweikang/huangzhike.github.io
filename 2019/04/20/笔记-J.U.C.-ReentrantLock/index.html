AbstractQueuedSynchronizer#acquire

    /**
     * 尝试获取独占锁，如果获取不到，加入等待队列等待被唤醒，如果被中断则中止
     */
    public final void acquire(int arg) {
        // 首先尝试获取锁，获取成功直接返回，锁空闲并且没有竞争的话一般都会成功，不需要入队列
        // 假如获取失败，则将当前线程加入到AQS队列尾，然后在AQS队列中阻塞等待获取锁，直到获取锁了才返回
        if (!tryAcquire(arg) && acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
            // 如果线程在阻塞过程中被中断过，则再中断一次（之前的被清除了）
            selfInterrupt();
    }

ReentrantLock$FairSync#tryAcquire

        @Override
        protected final boolean tryAcquire(int acquires) {
            final Thread current = Thread.currentThread();
            int c = getState();
            // 如果锁空闲
            if (c == 0) {
                // 如果没有前驱节点排队（公平，FCFS） && CAS更新state成功
                if (!hasQueuedPredecessors() && compareAndSetState(0, acquires)) {
                    // 设置当前线程为锁的持有线程
                    setExclusiveOwnerThread(current);
                    // 获取锁成功
                    return true;
                }
            }
            // 如果锁不空闲，但当前线程就是锁的持有线程，说明是重入
            else if (current == getExclusiveOwnerThread()) {
                // 重入次数增加
                int nextc = c + acquires;
                // 位数溢出了
                if (nextc < 0)
                    throw new Error("Maximum lock count exceeded");
                // 不用CAS，因为独占锁是当前线程持有，没有竞争
                setState(nextc);
                // 获取锁成功
                return true;
            }
            return false;
        }

AbstractQueuedSynchronizer#addWaiter

    /**
     * 将当前线程添加到AQS队尾，等待获取锁
     */
    private Node addWaiter(Node mode) {
        // 将线程包装成节点
        Node node = new Node(Thread.currentThread(), mode);
        Node pred = tail;
        // 如果队列不为空，尝试直接挂到AQS同步队列队尾
        if (pred != null) {
            node.prev = pred;
            // CAS更新尾节点的引用
            if (compareAndSetTail(pred, node)) {
                pred.next = node;
                return node;
            }
        }
        // 如果队列为空，或者同时有多个线程都把自己挂到队尾，CAS就会失败，那么循环至成功
        enq(node);
        return node;
    }

AbstractQueuedSynchronizer#enq

    /**
     * 自旋直到成功加到AQS队尾，返回前继节点，即原本的尾节点
     * 关于AQS队列的头节点，它是一个空节点，代表当前持有锁的线程，可以防止重复释放锁
     * 每个节点在阻塞前，都要将前驱节点设成SIGNAL，表示，当该前驱节点释放锁的时候，需要唤醒后继节点，否则会导致死锁
     * 但是第一个节点，没有前驱节点，所以会创建一个空节点
     */
    private Node enq(final Node node) {
        // 自旋
        for (; ; ) {
            Node t = tail;
            // 原本的尾节点为null，说明还没初始化，那么就创建一个空节点，CAS设为头节点，成功就将尾节点也指向它
            if (t == null) { // Must initialize
                if (compareAndSetHead(new Node()))
                    tail = head;
            }
            // 否则，将当前节点作为新的尾节点加入到AQS队尾
            else {
                node.prev = t;
                // CAS将该节点设置为新的尾节点
                if (compareAndSetTail(t, node)) {
                    t.next = node;
                    return t;
                }
                /*
                 * 注意这个入队尾的操作顺序，因为这些操作不是原子的，在别的线程并发遍历的情况下，可能会读到被破坏的队列结构
                 * 所以这里是先把新入队的节点的前指针指向原队尾，再CAS更新为新的尾节点，而遍历时则是从后向前，这样就没问题了
                 * 具体可以看AbstractQueuedSynchronizer#unparkSuccessor的逻辑
                 */
            }
        }
    }

AbstractQueuedSynchronizer#acquireQueued

    /**
     * 从队列中阻塞至获取锁，并返回当前线程在等待过程中有没有中断过
     * 如果获取到了锁，返回，否则，公平阻塞等待，直到被唤醒并获取锁了才返回
     */
    final boolean acquireQueued(final Node node, int arg) {
        boolean failed = true;
        try {
            // 表示在线程阻塞时，有没有被中断过
            boolean interrupted = false;
            // 自旋
            for (; ; ) {
                // 当前节点是通过它的前驱节点唤醒的
                final Node p = node.predecessor();
                // 公平锁，如果节点的前驱是头节点（空节点），就可以尝试获取锁
                if (p == head && tryAcquire(arg)) {
                    // 如果获取到锁，则将自己设置为新的头节点，并清除无用的引用
                    setHead(node);
                    // 清除原本的头节点对自己的引用
                    p.next = null; // help GC
                    failed = false;
                    // 返回有没有中断
                    return interrupted;
                }
                // 到这里说明当前节点的前面还有节点排队，或者是获取锁失败

                // 检查是否可以阻塞当前线程，还是继续自旋，如果阻塞，那么唤醒后再继续，还得获取锁
                if (shouldParkAfterFailedAcquire(p, node) && parkAndCheckInterrupt())
                    // 当前节点是因为中断而唤醒，而不是被前驱节点唤醒
                    interrupted = true;
                /*
                 * 如果设置好前继SIGNAL了，那么可以阻塞，唤醒后返回线程的中断状态并复位中断状态，继续自旋
                 * 之所以恢复中断状态，是因为假如当前线程是因为中断而被唤醒，如果不清除中断状态，下次自旋时LockSupport#park不会阻塞，那样就可能会一直自旋空转
                 */
            }
        } finally {
            // 抛出异常，那么就是子类实现的tryAcquire方法抛出的，那么取消当前节点
            if (failed)
                cancelAcquire(node);
        }
    }

AbstractQueuedSynchronizer#shouldParkAfterFailedAcquire

    /**
     * 返回当前线程是否应该阻塞
     * 挂起自己之前，需要将前驱节点设成SIGNAL，让前驱释放锁的时候唤醒自己
     */
    private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
        int ws = pred.waitStatus;
        // 如果前驱节点已经是SIGNAL，那么当前线程可以被阻塞
        if (ws == Node.SIGNAL)
            return true;
        // 如果前驱节点是CANCELLED，那么是个无效的节点，跳过并重试
        if (ws > 0) {
            do {
                // 前驱节点已取消，把前驱节点从AQS队列移除
                node.prev = pred = pred.prev;
            } while (pred.waitStatus > 0);
            // 重新连接链表
            pred.next = node;
        }
        // 如果前驱节点状态为 0 | CONDITION | PROPAGATE
        else {
            // CAS尝试更新前驱节点为SIGNAL
            compareAndSetWaitStatus(pred, ws, Node.SIGNAL);
        }
        // 意思是下次自旋继续
        return false;
    }

AbstractQueuedSynchronizer#parkAndCheckInterrupt

    /**
     * 阻塞当前线程，并返回线程被唤醒之后的中断状态
     * 线程被阻塞后唤醒一般有两种：
     * 前驱节点释放锁后，通过LockSupport#unpark唤醒当前线程
     * 线程中断，其它线程通过Thread#interrupt中断当前线程
     */
    private final boolean parkAndCheckInterrupt() {
        LockSupport.park(this); // 可响应中断，有中断标记，但是不会抛出InterruptedException
        return Thread.interrupted(); // 返回线程的中断状态，同时复位中断状态
    }

AbstractQueuedSynchronizer#cancelAcquire

    private void cancelAcquire(Node node) {
        if (node == null)
            return;
        node.thread = null;
        Node pred = node.prev;
        // 如果要取消的节点的前驱节点也是一个取消的节点，向前遍历找到一个有效的节点，同时更新节点向前的指针，也就是说此时从后向前的遍历中，会跳过取消的节点
        while (pred.waitStatus > 0)
            node.prev = pred = pred.prev;

        // 有效的前驱节点的后面的第一个被取消的无效节点
        Node predNext = pred.next;
        // 更新节点的状态为取消
        node.waitStatus = Node.CANCELLED;

        // 如果要取消的节点是尾节点 && CAS更新尾节点为前一个有效节点成功
        if (node == tail && compareAndSetTail(node, pred)) {
            // CAS置空尾节点的后指针，真正断开链表
            compareAndSetNext(pred, predNext, null);
        }
        // 否则如果要取消的节点不是尾节点，或者上面的CAS失败
        else {
            int ws;
            // 总之就是要保证有效的前驱节点的状态为SIGNAL，才能继续
            if (pred != head && ((ws = pred.waitStatus) == Node.SIGNAL || (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL))) && pred.thread != null) {
                Node next = node.next;
                // 当前取消的节点的后继节点处于有效状态，即没取消
                if (next != null && next.waitStatus <= 0)
                    // 连接链表
                    compareAndSetNext(pred, predNext, next);
            }
            // 其它情况
            else {
                // 唤醒后继节点，意义就是让后继节点唤醒后继续自旋，保证将它的前驱节点设为SIGNAL后再阻塞，避免出现无法唤醒的情况
                unparkSuccessor(node);
            }
            node.next = node; // help GC
        }
    }

        final boolean nonfairTryAcquire(int acquires) {
            final Thread current = Thread.currentThread();
            int c = getState();
            // 如果锁空闲
            if (c == 0) {
                // 相比公平锁的FairSync#tryAcquire方法，少了AbstractQueuedSynchronizer#hasQueuedPredecessors，不管队列还有没人排队，直接CAS
                // CAS更新成功，失败则可能是因为队列真正的头节点线程唤醒后，竞争失败
                if (compareAndSetState(0, acquires)) {
                    // 设置当前线程占有锁
                    setExclusiveOwnerThread(current);
                    return true;
                }

            }
            // 重入
            else if (current == getExclusiveOwnerThread()) {
                int nextc = c + acquires;
                if (nextc < 0)
                    throw new Error("Maximum lock count exceeded");
                setState(nextc);
                return true;
            }
            return false;
        }

AbstractQueuedSynchronizer#release

    /**
     * 释放锁
     */
    public final boolean release(int arg) {
        // 尝试释放锁，如果成功
        if (tryRelease(arg)) {
            Node h = head;
            // 释放成功后，唤醒头节点的后继节点，0说明已经释放过了，不需要重复释放
            if (h != null && h.waitStatus != 0)
                unparkSuccessor(h);
            return true;
        }
        // 释放失败
        return false;
    }

ReentrantLock$Sync#tryRelease

        @Override
        protected final boolean tryRelease(int releases) {
            // 对于独占锁，releases是1
            int c = getState() - releases; // 可重入，要释放重入获取的锁
            // 如果当前线程不是该锁的持有者，抛异常
            if (Thread.currentThread() != getExclusiveOwnerThread())
                throw new IllegalMonitorStateException();
            boolean free = false;
            // 重入锁全部释放才行
            if (c == 0) {
                // 只有全部释放才返回true
                free = true;
                // 设置锁的持有者为null，同步队列的节点可以获取了
                setExclusiveOwnerThread(null);
            }
            // 更新锁状态
            setState(c);
            return free;
        }

AbstractQueuedSynchronizer#unparkSuccessor

    /**
     * 头节点的后继节点被唤醒，后继节点就会尝试去获取锁
     */
    private void unparkSuccessor(Node node) {
        // 头节点，代表要释放锁的节点
        int ws = node.waitStatus;
        // 如果头节点状态不为CANCELLED，那么CAS更新为0，表示已释放，不能重复释放
        if (ws < 0)
            compareAndSetWaitStatus(node, ws, 0);
        // 从头结点的下一个节点开始寻找后继节点
        Node s = node.next;
        // 说明是个无效的节点，CANCELLED
        if (s == null || s.waitStatus > 0) {
            // 这个变量是用来判断有没有找到有效节点的
            s = null;
            // 那么从队列尾开始，向前寻找有效的节点，直到头节点
            for (Node t = tail; t != null && t != node; t = t.prev)
                if (t.waitStatus <= 0)
                    s = t;
            /*
             * 关于从尾部查找的逻辑
             * 可以参考AbstractQueuedSynchronizer#enq的逻辑，从后向前遍历可以避免读到被破坏的队列结构
             * 另外，Lock#lockInterruptibly的线程被中断，将CANCELLED的节点移出队列也是类似
             */
        }
        // 到这里说明找到了有效的后继节点，即waitStatus<=0，一般是头节点的后继节点

        // 如果找到一个有效的后继节点，就唤醒此节点
        if (s != null)
            LockSupport.unpark(s.thread);
    }
