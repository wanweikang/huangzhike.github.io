ThreadPoolExecutor

public class ThreadPoolExecutor extends AbstractExecutorService {
    /**
     * 高3位表示线程池状态，低29位表示线程池线程个数
     * workerCount：当前有效的线程数，最大线程数是2^29-1=536870911
     * runState：线程池的五种状态，Running、Shutdown、Stop、Tidying、Terminate
     * 初始化为RUNNING状态，并且任务数量初始化为0
     */
    private final AtomicInteger ctl = new AtomicInteger(ctlOf(RUNNING, 0));

    /**
     * 正在运行，能够接收新任务，处理已添加的任务
     */
    private static final int RUNNING = -1 << COUNT_BITS;

    /**
     * shutdown，不接收新任务，但能处理已添加的任务
     */
    private static final int SHUTDOWN = 0 << COUNT_BITS;

    /**
     * shutdownNow，不接收新任务，不处理已添加的任务，中断正在处理的任务
     */
    private static final int STOP = 1 << COUNT_BITS;

    /**
     * 所有任务已停止，准备清理
     * 变为TIDYING状态时，执行ThreadPoolExecutor#terminated
     */
    private static final int TIDYING = 2 << COUNT_BITS;

    /**
     * 已终止，TIDYING状态时，执行完ThreadPoolExecutor#terminated后
     */
    private static final int TERMINATED = 3 << COUNT_BITS;

    /**
     * 阻塞工作队列，一般用ArrayBlockingQueue或LinkedBlockingQueue
     */
    private final BlockingQueue<Runnable> workQueue;

    /**
     * 互斥锁
     */
    private final ReentrantLock mainLock = new ReentrantLock();

    /**
     * 线程集合，一个Worker对应一个线程
     */
    private final HashSet<Worker> workers = new HashSet<Worker>();

    /**
     * 终止条件
     */
    private final Condition termination = mainLock.newCondition();

    /**
     * 线程池中线程数量曾经达到过的最大值
     */
    private int largestPoolSize;

    /**
     * ThreadFactory对象，用于创建线程
     */
    private volatile ThreadFactory threadFactory;

    /**
     * 拒绝策略
     */
    private volatile RejectedExecutionHandler handler;

    /**
     * 线程存活时间
     */
    private volatile long keepAliveTime;

    private volatile boolean allowCoreThreadTimeOut;

    /**
     * 已完成任务数量
     */
    private long completedTaskCount;

    /**
     * 核心池大小
     */
    private volatile int corePoolSize;

    /**
     * 最大池大小
     */
    private volatile int maximumPoolSize;

    /**
     * 默认拒绝策略
     */
    private static final RejectedExecutionHandler defaultHandler = new AbortPolicy();

}

AbstractExecutorService#submit

    /**
     * 向线程池提交任务的时候，会创建一个FutureTask返回
     */
    @Override
    public Future<?> submit(Runnable task) {
        if (task == null)
            throw new NullPointerException();
        RunnableFuture<Void> ftask = newTaskFor(task, null);
        execute(ftask);
        return ftask;
    }

ThreadPoolExecutor#execute

    /**
     * 优先级为核心线程 -> 工作队列 -> 非核心线程
     */
    @Override
    public void execute(Runnable command) {
        if (command == null)
            throw new NullPointerException();
        int c = ctl.get();
        // 如果线程数量小于核心线程数，则创建线程
        if (workerCountOf(c) < corePoolSize) {
            // 添加成功则直接返回
            if (addWorker(command, true))
                return;
            // 否则再次获取线程池状态
            c = ctl.get();
        }
        // 否则说明核心线程数已满
        // 线程池正在运行 && 任务队列未满，添加进队列成功
        if (isRunning(c) && workQueue.offer(command)) {
            // 再次检查线程池状态，因为上面添加了任务到工作队列，可能要创建新的线程
            int recheck = ctl.get();
            // 如果线程池不是运行状态，那么从队列删除该任务，并尝试停止线程池，拒绝任务
            if (!isRunning(recheck) && remove(command))
                reject(command);
                // 否则如果是运行状态，但是当前工作线程数量为0，那么添加一个空任务，即创建新的线程
            else if (workerCountOf(recheck) == 0)
                addWorker(null, false);
        }
        // 否则如果核心线程数已满，并且队列已满，则创建一个任务线程，如果失败，说明线程数已经达到最大，那么拒绝任务
        else if (!addWorker(command, false))
            reject(command);
    }

ThreadPoolExecutor#addWorker

    /**
     * 创建线程
     * 第一个参数firstTask，如果不为空那就是包含新任务，否则只是创建新线程，从工作队列里拿任务执行
     * 如果core为true，要判断线程数是否达到核心线程数
     * 如果core为false，要判断线程数是否达到最大线程数
     */
    private boolean addWorker(Runnable firstTask, boolean core) {
        retry:
        // 状态的检测和更新
        for (; ; ) {
            int c = ctl.get();
            int rs = runStateOf(c);
            // 如果线程池为STOP，TIDYING，TERMINATED状态，直接拒绝
            // 如果线程池为SHUTDOWN状态，虽然不允许提交新任务，但是已提交的任务要完成，所以是允许创建线程的
            // 当为SHUTDOWN时，firstTask必须为空，因为不允许提交新任务，另外workQueue如果为空的话，那就没事可干了，不允许创建新线程
            if (rs >= SHUTDOWN && !(rs == SHUTDOWN && firstTask == null && !workQueue.isEmpty()))
                return false;
            // 自旋重试
            for (; ; ) {
                // 线程的数量
                int wc = workerCountOf(c);
                // 如果线程数量已经溢出那就直接返回
                if (wc >= CAPACITY || wc >= (core ? corePoolSize : maximumPoolSize))
                    return false;
                // 如果CAS将线程数自增成功，则退出外循环，否则重试
                if (compareAndIncrementWorkerCount(c))
                    break retry;
                // CAS不成功，说明存在竞争，重新获取线程状态
                c = ctl.get();  // Re-read ctl
                // 检查线程池运行状态，如果与之前的状态不同，则从外循环重新开始
                if (runStateOf(c) != rs)
                    continue retry;
                // else CAS failed due to workerCount change; retry inner loop
            }
        }
        // 正在运行的线程数自增成功后则准备创建线程，包装成Worker
        boolean workerStarted = false;
        boolean workerAdded = false;
        ThreadPoolExecutor.Worker w = null;
        // 添加任务到线程池，并启动任务所在的线程
        try {
            // 创建Worker，同时也创建了线程，并且指定firstTask为Worker的第一个任务
            w = new ThreadPoolExecutor.Worker(firstTask);
            // 获取Worker对应的线程
            final Thread t = w.thread;
            if (t != null) {
                final ReentrantLock mainLock = this.mainLock;
                // 获取锁
                mainLock.lock();
                try {
                    int rs = runStateOf(ctl.get());
                    // RUNNING || SHUTDOWN状态，但是需要完成工作队列里的任务
                    if (rs < SHUTDOWN || (rs == SHUTDOWN && firstTask == null)) {
                        if (t.isAlive())
                            throw new IllegalThreadStateException();
                        // 将Worker对象添加到线程池的Worker集合中
                        workers.add(w);
                        int s = workers.size();
                        // 记录线程池的线程数最大值
                        if (s > largestPoolSize)
                            largestPoolSize = s;
                        workerAdded = true;
                    }
                } finally {
                    // 释放锁
                    mainLock.unlock();
                }
                // 如果成功将任务添加到线程池中，则启动线程
                if (workerAdded) {
                    t.start();
                    workerStarted = true;
                }
            }
        } finally {
            // 在启动工作线程失败后，将工作线程从集合中移除，并将线程数递减，当然是有锁的
            if (!workerStarted)
                addWorkerFailed(w);
        }
        // 返回任务是否启动
        return workerStarted;
    }

ThreadPoolExecutor$Worker

    /**
     * 继承了AQS，实现了Runnable，所以是线程，也是锁，但是不可重入
     */
    private final class Worker extends AbstractQueuedSynchronizer implements Runnable {

        /**
         * 工作的线程
         */
        final Thread thread;

        /**
         * 创建Worker时指定的任务，如果有的话
         */
        Runnable firstTask;

        /**
         * 完成的任务数
         */
        volatile long completedTasks;

        Worker(Runnable firstTask) {
            // 设置AQS的同步状态为-1，防止runWorker前，线程没启动就被interrupt中断，直到调用runWorker，才设为0
            // 另外如果大于0，说明锁被获取
            setState(-1); // inhibit interrupts until runWorker
            this.firstTask = firstTask;
            // 通过线程工厂来创建线程
            this.thread = getThreadFactory().newThread(this);
        }
    }

ThreadPoolExecutor$Worker#runWorker

    final void runWorker(ThreadPoolExecutor.Worker w) {
        Thread wt = Thread.currentThread();
        Runnable task = w.firstTask;
        w.firstTask = null;
        // 允许中断
        w.unlock(); // allow interrupts
        boolean completedAbruptly = true;
        try {
            // 不断从工作队列里取出任务执行
            while (task != null || (task = getTask()) != null) {
                // 获取Worker的锁，不是因为并发，而是避免运行中被shutdown
                w.lock();
                // 如果线程池状态为STOP，TIDYING，TERMINATED，那么中断当前线程
                // 否则，判断当前线程是否被标记为中断，同时会清除中断标记，如果被标记为中断，那么再次判断状态，如果大于等于STOP，并且没有中断标记，那么就中断当前线程
                if ((runStateAtLeast(ctl.get(), STOP) || (Thread.interrupted() && runStateAtLeast(ctl.get(), STOP))) && !wt.isInterrupted())
                    wt.interrupt();
                try {
                    // 任何上抛的异常都会导致当前线程退出死亡
                    beforeExecute(wt, task);
                    Throwable thrown = null;
                    try {
                        // 执行任务
                        task.run();
                    } catch (RuntimeException x) {
                        thrown = x;
                        throw x;
                    } catch (Error x) {
                        thrown = x;
                        throw x;
                    } catch (Throwable x) {
                        thrown = x;
                        throw new Error(x);
                    } finally {
                        //
                        afterExecute(task, thrown);
                    }
                } finally {
                    // 准备下一个
                    task = null;
                    w.completedTasks++;
                    w.unlock();
                }
            } // while循环结束
            // 是正常完成，而不是因为线程运行时抛出异常而突然跳出
            completedAbruptly = false;
        } finally {
            // 可能是队列任务全部完成或者线程运行中抛出了异常，那么退出当前Worker，当前线程就没了
            processWorkerExit(w, completedAbruptly);
        }
    }

ThreadPoolExecutor#getTask

    private Runnable getTask() {
        boolean timedOut = false; // Did the last poll() time out?
        for (; ; ) {
            int c = ctl.get();
            int rs = runStateOf(c);

            // 满足条件才能取出任务执行
            // 如果线程池状态为SHUTDOWN，当工作队列为空时才不继续执行
            // 如果线程池状态为STOP，那么不管工作队列里还有没有任务，都不执行
            if (rs >= SHUTDOWN && (rs >= STOP || workQueue.isEmpty())) {
                decrementWorkerCount();
                return null;
            }
            int wc = workerCountOf(c);
            // 如果允许核心线程超时或者当前线程数已经超过核心线程数，那么有超时要求
            boolean timed = allowCoreThreadTimeOut || wc > corePoolSize;

            // 如果线程数量超出最大限制，那么CAS尝试减少线程数量，返回后runWorker的processWorkerExit中Worker线程就会退出
            // 如果数量没有超出限制，但是超出了核心线程数，上次从工作队列获取任务超时的话，也是一样的
            // 另外上面两个还有附加条件，即，当前存在工作线程，或者不存在工作线程，但是工作队列为空
            // 这个条件短路看得我头大，一般这里是不会超出maximumPoolSize的，除非后来又手动调整了maximumPoolSize大小
            if ((wc > maximumPoolSize || (timed && timedOut)) && (wc > 1 || workQueue.isEmpty())) {
                if (compareAndDecrementWorkerCount(c))
                    return null;
                continue;
            }
            try {
                // 根据有无超时限制，从工作队列阻塞获取任务
                Runnable r = timed ? workQueue.poll(keepAliveTime, TimeUnit.NANOSECONDS) : workQueue.take();
                // 有就返回
                if (r != null)
                    return r;
                // 超时还没有
                timedOut = true;
            } catch (InterruptedException retry) {
                // 捕捉Worker线程的中断异常，不算超时，继续
                timedOut = false;
            }
        }
    }

ThreadPoolExecutor#processWorkerExit

    private void processWorkerExit(ThreadPoolExecutor.Worker w, boolean completedAbruptly) {
        // 如果是线程运行中抛出异常而退出，那么要把线程数量减一，而如果是因为没任务而退出的话，在getTask里边就已经减了
        if (completedAbruptly)
            decrementWorkerCount();
        final ReentrantLock mainLock = this.mainLock;
        // 获取锁
        mainLock.lock();
        try {
            completedTaskCount += w.completedTasks;
            // 移除当前Worker
            workers.remove(w);
        } finally {
            mainLock.unlock();
        }
        // 每个Worker退出时都会尝试终止线程池，当然不一定会终止
        tryTerminate();
        int c = ctl.get();
        // 如果当前线程池状态是Running或Shutdown，所以Shutdown也可能会创建新的线程
        if (runStateLessThan(c, STOP)) {
            // 如果是因为没有任务而退出的，还要计算要不要增再加一个新的Worker线程
            if (!completedAbruptly) {
                int min = allowCoreThreadTimeOut ? 0 : corePoolSize;
                // 如果不需要维持核心线程，但是工作线程还有任务，那么至少得留一个线程
                if (min == 0 && !workQueue.isEmpty())
                    min = 1;
                // 线程数足够，不用增加
                if (workerCountOf(c) >= min)
                    return;
            }
            // 创建一个新的Worker线程
            addWorker(null, false);
        }
    }

ThreadPoolExecutor#shutdown

    /**
     * 不接受新任务，但是会把任务队列的任务全部执行完
     */
    @Override
    public void shutdown() {
        final ReentrantLock mainLock = this.mainLock;
        mainLock.lock();
        try {
            checkShutdownAccess(); // 检测权限
            advanceRunState(SHUTDOWN); // 自旋CAS修改运行状态为SHUTDOWN
            interruptIdleWorkers(); // 遍历停止未开启的线程
            onShutdown(); // hook for ScheduledThreadPoolExecutor
        } finally {
            mainLock.unlock();
        }
        tryTerminate();
    }

ThreadPoolExecutor#interruptIdleWorkers

    private void interruptIdleWorkers(boolean onlyOne) {
        final ReentrantLock mainLock = this.mainLock;
        mainLock.lock();
        try {
            // 遍历所有线程
            for (ThreadPoolExecutor.Worker w : workers) {
                Thread t = w.thread;
                // 只有空闲的Worker才能获取锁，获取不到就跳过
                if (!t.isInterrupted() && w.tryLock()) {
                    try {
                        t.interrupt();
                    } catch (SecurityException ignore) {
                    } finally {
                        w.unlock();
                    }
                }
                if (onlyOne)
                    break;
            }
        } finally {
            mainLock.unlock();
        }
    }

ThreadPoolExecutor#tryTerminate

    final void tryTerminate() {
        for (; ; ) {
            int c = ctl.get();
            // Running状态不终止
            // TIDYING或TERMINATED，已经终止过了不再终止
            // SHUTDOWN且工作队列还有任务也不能终止
            if (isRunning(c) || runStateAtLeast(c, TIDYING) || (runStateOf(c) == SHUTDOWN && !workQueue.isEmpty()))
                return;

            // 进行到这里，说明状态为STOP，或者SHUTDOWN并且工作队列为空

            // 还有正在工作的线程
            if (workerCountOf(c) != 0) {
                // 尝试中断一个空闲的线程，因为每个Worker线程退出时都会调用ThreadPoolExecutor#tryTerminate方法，所以不用担心
                interruptIdleWorkers(ONLY_ONE);
                return;
            }

            // 到这里说明线程池已经没有线程了

            final ReentrantLock mainLock = this.mainLock;
            mainLock.lock();
            try {
                // CAS修改状态为TIDYING成功
                if (ctl.compareAndSet(c, ctlOf(TIDYING, 0))) {
                    try {
                        // 执行终止回调，默认为空方法
                        terminated();
                    } finally {
                        // 修改状态为TERMINATED
                        ctl.set(ctlOf(TERMINATED, 0));
                        termination.signalAll();
                    }
                    return;
                }
            } finally {
                mainLock.unlock();
            }
            // else retry on failed CAS
        }
    }

线程池的作用：

  • 减少了创建和销毁线程的次数，每个工作线程都可以被重复利用

  • 调整线程池中工作线程的数量，限流作用

建议：

  • CPU密集型：避免线程争抢资源，无用的切换线程上下文，配置尽量少的线程，比如CPU个数；

  • IO密集型：需要长时间等待，可以配置多一些线程，比如CPU数的两倍。

获取CPU核心数：Runtime.getRuntime().availableProcessors()。
