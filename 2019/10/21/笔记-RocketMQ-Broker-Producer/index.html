SendMessageProcessor#processRequest

    @Override
    public RemotingCommand processRequest(ChannelHandlerContext ctx, RemotingCommand request) throws RemotingCommandException {
        SendMessageContext mqtraceContext;
        switch (request.getCode()) {
            // Consumer消费失败，将消息发回，并要求Broker重发
            // 一种是消息处理失败，一种是消息在Consumer的缓存ProcessQueue中超时被移除，退回Broker重发
            case RequestCode.CONSUMER_SEND_MSG_BACK:
                return this.consumerSendMsgBack(ctx, request);
            // Producer发送的消息
            default:
                // 解码请求头
                SendMessageRequestHeader requestHeader = parseRequestHeader(request);
                if (requestHeader == null) {
                    return null;
                }
                mqtraceContext = buildMsgContext(ctx, requestHeader);
                // 执行前置Hook
                this.executeSendMessageHookBefore(ctx, request, mqtraceContext);
                RemotingCommand response;
                // 批量消息
                if (requestHeader.isBatch()) {
                    response = this.sendBatchMessage(ctx, request, mqtraceContext, requestHeader);
                }
                // 单个消息
                else {
                    response = this.sendMessage(ctx, request, mqtraceContext, requestHeader);
                }
                // 执行后置Hook
                this.executeSendMessageHookAfter(response, mqtraceContext);
                return response;
        }
    }

SendMessageProcessor#consumerSendMsgBack

    private RemotingCommand consumerSendMsgBack(final ChannelHandlerContext ctx, final RemotingCommand request) throws RemotingCommandException {
        final RemotingCommand response = RemotingCommand.createResponseCommand(null);
        final ConsumerSendMsgBackRequestHeader requestHeader = (ConsumerSendMsgBackRequestHeader) request.decodeCommandCustomHeader(ConsumerSendMsgBackRequestHeader.class);
        // ...
        // Hook
        if (this.hasConsumeMessageHook() && !UtilAll.isBlank(requestHeader.getOriginMsgId())) {
            // ...
        }
        // 根据ConsumerGroup从本地缓存获取SubscriptionGroupConfig，没有找到并且允许自动创建则会自动创建，否则返回空
        SubscriptionGroupConfig subscriptionGroupConfig = this.brokerController.getSubscriptionGroupManager().findSubscriptionGroupConfig(requestHeader.getGroup());
        // 没找到ConsumerGroup对应的SubscriptionGroupConfig
        if (null == subscriptionGroupConfig) {
            // SUBSCRIPTION_GROUP_NOT_EXIST
            return response;
        }
        // Broker不可写
        if (!PermName.isWriteable(this.brokerController.getBrokerConfig().getBrokerPermission())) {
            // NO_PERMISSION
            return response;
        }
        // Topic不允许重试
        if (subscriptionGroupConfig.getRetryQueueNums() <= 0) {
            // SUCCESS
            return response;
        }
        // %RETRY% + ConsumerGroup
        String newTopic = MixAll.getRetryTopic(requestHeader.getGroup());
        // 随机选择一个QueueId，不过默认只有一个
        int queueIdInt = Math.abs(this.random.nextInt() % 99999999) % subscriptionGroupConfig.getRetryQueueNums();
        // ...
        // 根据Topic从本地缓存获取TopicConfig，没有就自动创建返回并向NameServer广播REGISTER_BROKER请求更新路由信息
        TopicConfig topicConfig = this.brokerController.getTopicConfigManager().createTopicInSendMessageBackMethod(newTopic, subscriptionGroupConfig.getRetryQueueNums(), PermName.PERM_WRITE | PermName.PERM_READ, topicSysFlag);
        // 基本不会发生
        if (null == topicConfig) {
            // SYSTEM_ERROR
            return response;
        }
        // 如果Topic不可写
        if (!PermName.isWriteable(topicConfig.getPerm())) {
            // NO_PERMISSION
            return response;
        }
        // 根据请求头携带的偏移从CommitLog获取原消息
        MessageExt msgExt = this.brokerController.getMessageStore().lookMessageByOffset(requestHeader.getOffset());
        // 消息不存在
        if (null == msgExt) {
            // SYSTEM_ERROR
            return response;
        }
        // ...
        msgExt.setWaitStoreMsgOK(false);
        // 延迟级别
        int delayLevel = requestHeader.getDelayLevel();
        int maxReconsumeTimes = subscriptionGroupConfig.getRetryMaxTimes();
        if (request.getVersion() >= MQVersion.Version.V3_4_9.ordinal()) {
            maxReconsumeTimes = requestHeader.getMaxReconsumeTimes();
        }
        // 已经超出消息的最大重消费次数
        if (msgExt.getReconsumeTimes() >= maxReconsumeTimes || delayLevel < 0) {
            // 准备将消息放入死信队列
            // %DLQ% + ConsumerGroup
            newTopic = MixAll.getDLQTopic(requestHeader.getGroup());
            // 只有一个Queue
            queueIdInt = Math.abs(this.random.nextInt() % 99999999) % DLQ_NUMS_PER_GROUP;
            // 根据Topic从本地缓存获取TopicConfig，没有就自动创建返回并向NameServer广播REGISTER_BROKER请求更新路由信息
            topicConfig = this.brokerController.getTopicConfigManager().createTopicInSendMessageBackMethod(newTopic, DLQ_NUMS_PER_GROUP, PermName.PERM_WRITE, 0);
            // 不会发生，除非是获取锁时被中断了
            if (null == topicConfig) {
                // SYSTEM_ERROR
                return response;
            }
        }
        // 还可以再消费
        else {
            if (0 == delayLevel) {
                // 重试次数越多，延时越久
                delayLevel = 3 + msgExt.getReconsumeTimes();
            }
            // 所以Consumer消费消息失败，消息重试间隔通过设置延迟实现
            msgExt.setDelayTimeLevel(delayLevel);
        }
        MessageExtBrokerInner msgInner = new MessageExtBrokerInner();
        msgInner.setTopic(newTopic);
        // ...
        msgInner.setQueueId(queueIdInt);
        // ...
        // 重试次数增加
        msgInner.setReconsumeTimes(msgExt.getReconsumeTimes() + 1);
        // ...
        // 重发的消息会再次加入CommitLog持久化
        PutMessageResult putMessageResult = this.brokerController.getMessageStore().putMessage(msgInner);
        // 消息持久化结果
        if (putMessageResult != null) {
            switch (putMessageResult.getPutMessageStatus()) {
                case PUT_OK:
                    // SUCCESS
                    return response;
                default:
                    break;
            }
            // SYSTEM_ERROR
            return response;
        }
        // SYSTEM_ERROR
        return response;
    }

SendMessageProcessor#sendMessage

    private RemotingCommand sendMessage(final ChannelHandlerContext ctx, final RemotingCommand request, final SendMessageContext sendMessageContext, final SendMessageRequestHeader requestHeader) throws RemotingCommandException {
        // 响应头
        final RemotingCommand response = RemotingCommand.createResponseCommand(SendMessageResponseHeader.class);
        final SendMessageResponseHeader responseHeader = (SendMessageResponseHeader) response.readCustomHeader();
        // RequestId
        response.setOpaque(request.getOpaque());
        // ...
        final long startTimstamp = this.brokerController.getBrokerConfig().getStartAcceptSendRequestTimeStamp();
        // Broker还没到可接受请求的时间，Broker：996是福报，我太难了
        if (this.brokerController.getMessageStore().now() < startTimstamp) {
            // SYSTEM_ERROR
            return response;
        }
        response.setCode(-1);
        // 检查Topic和Queue，如果不存在且允许Broker自动创建，则自动创建Topic
        super.msgCheck(ctx, requestHeader, response);
        if (response.getCode() != -1) {
            return response;
        }
        final byte[] body = request.getBody();
        int queueIdInt = requestHeader.getQueueId();
        // 根据Topic从本地缓存获取TopicConfig
        TopicConfig topicConfig = this.brokerController.getTopicConfigManager().selectTopicConfig(requestHeader.getTopic());
        if (queueIdInt < 0) {
            // 随机选一个
            queueIdInt = Math.abs(this.random.nextInt() % 99999999) % topicConfig.getWriteQueueNums();
        }
        // 包装消息为MessageExtBrokerInner
        MessageExtBrokerInner msgInner = new MessageExtBrokerInner();
        msgInner.setTopic(requestHeader.getTopic());
        msgInner.setQueueId(queueIdInt);
        // 处理RETRY消息和死信队列，如果RETRY消息超过最大重发次数，则把消息的Topic设置为DeadQueue，后续会加入死信队列
        if (!handleRetryAndDLQ(requestHeader, response, request, msgInner, topicConfig)) {
            return response;
        }
        msgInner.setBody(body);
        // ...
        PutMessageResult putMessageResult = null;
        Map<String, String> oriProps = MessageDecoder.string2messageProperties(requestHeader.getProperties());
        String traFlag = oriProps.get(MessageConst.PROPERTY_TRANSACTION_PREPARED);
        // 事务消息
        if (traFlag != null && Boolean.parseBoolean(traFlag)) {
            // 如果Broker拒绝事务消息，默认不拒绝
            if (this.brokerController.getBrokerConfig().isRejectTransactionMessage()) {
                // NO_PERMISSION
                return response;
            }
            // 半消息，持久化PREPARED消息进CommitLog
            putMessageResult = this.brokerController.getTransactionalMessageService().prepareMessage(msgInner);
        } else {
            // 持久化普通消息进CommitLog
            putMessageResult = this.brokerController.getMessageStore().putMessage(msgInner);
        }
        // 设置响应码，执行Hook等
        return handlePutMessageResult(putMessageResult, response, request, msgInner, responseHeader, sendMessageContext, ctx, queueIdInt);

    }

AbstractSendMessageProcessor#msgCheck

    protected RemotingCommand msgCheck(final ChannelHandlerContext ctx, final SendMessageRequestHeader requestHeader, final RemotingCommand response) {
        // Broker不可写 && 该Topic是顺序的
        if (!PermName.isWriteable(this.brokerController.getBrokerConfig().getBrokerPermission()) && this.brokerController.getTopicConfigManager().isOrderTopic(requestHeader.getTopic())) {
            // NO_PERMISSION
            return response;
        }
        // Topic是系统内置的Topic，TBW102
        if (!this.brokerController.getTopicConfigManager().isTopicCanSendMessage(requestHeader.getTopic())) {
            // SYSTEM_ERROR
            return response;
        }
        // 从本地缓存获取该Topic的TopicConfig
        TopicConfig topicConfig = this.brokerController.getTopicConfigManager().selectTopicConfig(requestHeader.getTopic());
        // 没有对应配置
        if (null == topicConfig) {
            int topicSysFlag = 0;
            // ...
            // 如果Broker启用了自动创建Topic，则创建对应Topic的TopicConfig并返回，同时向NameServer广播新的Topic的路由信息
            topicConfig = this.brokerController.getTopicConfigManager().createTopicInSendMessageMethod(requestHeader.getTopic(), requestHeader.getDefaultTopic(), RemotingHelper.parseChannelRemoteAddr(ctx.channel()), requestHeader.getDefaultTopicQueueNums(), topicSysFlag);
            // 没开启自动创建Topic，并且本地缓存没有该Topic的TopicConfig
            if (null == topicConfig) {
                // 如果以%RETRY%开头
                if (requestHeader.getTopic().startsWith(MixAll.RETRY_GROUP_TOPIC_PREFIX)) {
                    // 自动创建TopicConfig并向NameServer广播
                    topicConfig = this.brokerController.getTopicConfigManager().createTopicInSendMessageBackMethod(requestHeader.getTopic(), 1, PermName.PERM_WRITE | PermName.PERM_READ, topicSysFlag);
                }
            }
            if (null == topicConfig) {
                // TOPIC_NOT_EXIST
                return response;
            }
        }
        int queueIdInt = requestHeader.getQueueId();
        int idValid = Math.max(topicConfig.getWriteQueueNums(), topicConfig.getReadQueueNums());
        // 检查队列Id是否超出合法值
        if (queueIdInt >= idValid) {
            // SYSTEM_ERROR
            return response;
        }
        return response;
    }

SendMessageProcessor#handleRetryAndDLQ

    private boolean handleRetryAndDLQ(SendMessageRequestHeader requestHeader, RemotingCommand response, RemotingCommand request, MessageExt msg, TopicConfig topicConfig) {
        String newTopic = requestHeader.getTopic();
        // 如果Topic以%RETRY%开头
        if (null != newTopic && newTopic.startsWith(MixAll.RETRY_GROUP_TOPIC_PREFIX)) {
            // ConsumerGroup
            String groupName = newTopic.substring(MixAll.RETRY_GROUP_TOPIC_PREFIX.length());
            // 从本地缓存获取该ConsumerGroup的SubscriptionGroupConfig，如果没有并且开启了自动创建，则自动创建SubscriptionGroupConfig并加入缓存返回
            SubscriptionGroupConfig subscriptionGroupConfig = this.brokerController.getSubscriptionGroupManager().findSubscriptionGroupConfig(groupName);
            if (null == subscriptionGroupConfig) {
                // SUBSCRIPTION_GROUP_NOT_EXIST
                return false;
            }
            // 最大重新消费次数
            int maxReconsumeTimes = subscriptionGroupConfig.getRetryMaxTimes();
            // 兼容高版本
            if (request.getVersion() >= MQVersion.Version.V3_4_9.ordinal()) {
                maxReconsumeTimes = requestHeader.getMaxReconsumeTimes();
            }
            // 已消费次数
            int reconsumeTimes = requestHeader.getReconsumeTimes() == null ? 0 : requestHeader.getReconsumeTimes();
            // 消费次数大于等于可消费次数，说明要加入死信队列
            if (reconsumeTimes >= maxReconsumeTimes) {
                // 该ConsumerGroup的死信队列的Topic名
                // %DLQ% + ConsumerGroup
                newTopic = MixAll.getDLQTopic(groupName);
                // 随机生成QueueId，一个ConsumerGroup只有一个死信队列，这里永远返回0
                int queueIdInt = Math.abs(this.random.nextInt() % 99999999) % DLQ_NUMS_PER_GROUP;
                // 从本地缓存获取该死信队列Topic的TopicConfig，没有就自动创建加入并返回，同时向NameServer广播新配置
                topicConfig = this.brokerController.getTopicConfigManager().createTopicInSendMessageBackMethod(newTopic, DLQ_NUMS_PER_GROUP, PermName.PERM_WRITE, 0);
                // 修改消息的Topic为死信队列的Topic
                msg.setTopic(newTopic);
                msg.setQueueId(queueIdInt);
                if (null == topicConfig) {
                    // SYSTEM_ERROR
                    return false;
                }
            }
        }
        // ...
        return true;
    }
