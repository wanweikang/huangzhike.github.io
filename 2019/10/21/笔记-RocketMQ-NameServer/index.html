RouteInfoManager#registerBroker

    public RegisterBrokerResult registerBroker(final String clusterName, final String brokerAddr, final String brokerName, final long brokerId, final String haServerAddr, final TopicConfigSerializeWrapper topicConfigWrapper, final List<String> filterServerList, final Channel channel) {
        RegisterBrokerResult result = new RegisterBrokerResult();
        try {
            try {
                // 写锁，独占，不允许别的线程读写
                this.lock.writeLock().lockInterruptibly();
                // 根据ClusterName获取Broker集群下所有BrokerName列表
                Set<String> brokerNames = this.clusterAddrTable.get(clusterName);
                // 该集群下不存在BrokerName
                if (null == brokerNames) {
                    brokerNames = new HashSet<String>();
                    // 加入
                    this.clusterAddrTable.put(clusterName, brokerNames);
                }
                // 加入
                brokerNames.add(brokerName);
                boolean registerFirst = false;
                // 获取该BrokerName下的Broker列表，Master和Slave
                BrokerData brokerData = this.brokerAddrTable.get(brokerName);
                // 该BrokerName首次注册
                if (null == brokerData) {
                    registerFirst = true;
                    brokerData = new BrokerData(clusterName, brokerName, new HashMap<Long, String>());
                    // 加入
                    this.brokerAddrTable.put(brokerName, brokerData);
                }
                // 该BrokerName下的Broker地址列表，包括主从
                Map<Long, String> brokerAddrsMap = brokerData.getBrokerAddrs();
                Iterator<Entry<Long, String>> it = brokerAddrsMap.entrySet().iterator();
                // 遍历该BrokerName下的Broker地址列表
                while (it.hasNext()) {
                    Entry<Long, String> item = it.next();
                    // 不同的Broker，IP地址不允许重复，所以一个主机上不能同时部署主从？
                    if (null != brokerAddr && brokerAddr.equals(item.getValue()) && brokerId != item.getKey()) {
                        // 移除旧的
                        it.remove();
                    }
                }
                // 该Broker是否已注册
                String oldAddr = brokerData.getBrokerAddrs().put(brokerId, brokerAddr);
                registerFirst = registerFirst || (null == oldAddr);
                // TopicConfigWrapper不为空 && 当前Broker是Master节点
                if (null != topicConfigWrapper && MixAll.MASTER_ID == brokerId) {
                    // 根据BrokerLiveInfo的DataVersion判断是否有变化
                    if (this.isBrokerTopicConfigChanged(brokerAddr, topicConfigWrapper.getDataVersion()) || registerFirst) {
                        // Broker发过来的Topic信息
                        ConcurrentMap<String, TopicConfig> tcTable = topicConfigWrapper.getTopicConfigTable();
                        if (tcTable != null) {
                            // 遍历Topic列表
                            for (Map.Entry<String, TopicConfig> entry : tcTable.entrySet()) {
                                // 
                                this.createAndUpdateQueueData(brokerName, entry.getValue());
                            }
                        }
                    }
                }
                // 更新Broker生存信息
                BrokerLiveInfo prevBrokerLiveInfo = this.brokerLiveTable.put(brokerAddr, new BrokerLiveInfo(System.currentTimeMillis(), topicConfigWrapper.getDataVersion(), channel, haServerAddr));
                // ...
                if (filterServerList != null) {
                    if (filterServerList.isEmpty()) {
                        this.filterServerTable.remove(brokerAddr);
                    } else {
                        this.filterServerTable.put(brokerAddr, filterServerList);
                    }
                }
                // 如果当前Broker是Slave
                if (MixAll.MASTER_ID != brokerId) {
                    String masterAddr = brokerData.getBrokerAddrs().get(MixAll.MASTER_ID);
                    if (masterAddr != null) {
                        // 获取Master的信息
                        BrokerLiveInfo brokerLiveInfo = this.brokerLiveTable.get(masterAddr);
                        if (brokerLiveInfo != null) {
                            // 向Slave返回Master地址
                            result.setHaServerAddr(brokerLiveInfo.getHaServerAddr());
                            result.setMasterAddr(masterAddr);
                        }
                    }
                }
            } finally {
                // 释放写锁
                this.lock.writeLock().unlock();
            }
        } catch (Exception e) {
            // ...
        }
        return result;
    }

RouteInfoManager#createAndUpdateQueueData

    private void createAndUpdateQueueData(final String brokerName, final TopicConfig topicConfig) {
        // 队列
        QueueData queueData = new QueueData();
        queueData.setBrokerName(brokerName);
        queueData.setWriteQueueNums(topicConfig.getWriteQueueNums());
        queueData.setReadQueueNums(topicConfig.getReadQueueNums());
        queueData.setPerm(topicConfig.getPerm());
        queueData.setTopicSynFlag(topicConfig.getTopicSysFlag());
        // 获取该Topic下的队列列表
        List<QueueData> queueDataList = this.topicQueueTable.get(topicConfig.getTopicName());
        // 没有就加入
        if (null == queueDataList) {
            queueDataList = new LinkedList<QueueData>();
            queueDataList.add(queueData);
            this.topicQueueTable.put(topicConfig.getTopicName(), queueDataList);
        }
        // 已有
        else {
            boolean addNewOne = true;
            Iterator<QueueData> it = queueDataList.iterator();
            while (it.hasNext()) {
                QueueData qd = it.next();
                if (qd.getBrokerName().equals(brokerName)) {
                    // 没变化
                    if (qd.equals(queueData)) {
                        addNewOne = false;
                    } else {
                        // 移除旧的
                        it.remove();
                    }
                }
            }
            if (addNewOne) {
                // 加入新的
                queueDataList.add(queueData);
            }
        }
    }

DefaultRequestProcessor#getRouteInfoByTopic

    public RemotingCommand getRouteInfoByTopic(ChannelHandlerContext ctx, RemotingCommand request) throws RemotingCommandException {
        final RemotingCommand response = RemotingCommand.createResponseCommand(null);
        final GetRouteInfoRequestHeader requestHeader = (GetRouteInfoRequestHeader) request.decodeCommandCustomHeader(GetRouteInfoRequestHeader.class);
        // 从RouteInfoManager获取Topic的路由信息
        TopicRouteData topicRouteData = this.namesrvController.getRouteInfoManager().pickupTopicRouteData(requestHeader.getTopic());
        if (topicRouteData != null) {
            // 如果NameServer启用顺序消息，默认不启用
            if (this.namesrvController.getNamesrvConfig().isOrderMessageEnable()) {
                String orderTopicConf = this.namesrvController.getKvConfigManager().getKVConfig(NamesrvUtil.NAMESPACE_ORDER_TOPIC_CONFIG, requestHeader.getTopic());
                topicRouteData.setOrderTopicConf(orderTopicConf);
            }
            byte[] content = topicRouteData.encode();
            response.setBody(content);
            response.setCode(ResponseCode.SUCCESS);
            response.setRemark(null);
            return response;
        }
        response.setCode(ResponseCode.TOPIC_NOT_EXIST);
        // ...
        return response;
    }
    public TopicRouteData pickupTopicRouteData(final String topic) {
        TopicRouteData topicRouteData = new TopicRouteData();
        // ...
        try {
            try {
                // 获取读锁
                this.lock.readLock().lockInterruptibly();
                // 获取Topic的Queue列表
                List<QueueData> queueDataList = this.topicQueueTable.get(topic);
                if (queueDataList != null) {
                    // 加入
                    topicRouteData.setQueueDatas(queueDataList);
                    foundQueueData = true;
                    Iterator<QueueData> it = queueDataList.iterator();
                    // 遍历Topic的Queue列表
                    while (it.hasNext()) {
                        QueueData qd = it.next();
                        // Queue所属的BrokerName
                        brokerNameSet.add(qd.getBrokerName());
                    }
                    // 遍历BrokerName列表
                    for (String brokerName : brokerNameSet) {
                        // 根据BrokerName获取Broker列表，Master和Slave
                        BrokerData brokerData = this.brokerAddrTable.get(brokerName);
                        if (null != brokerData) {
                            // 复制一份
                            BrokerData brokerDataClone = new BrokerData(brokerData.getCluster(), brokerData.getBrokerName(), (HashMap<Long, String>) brokerData.getBrokerAddrs().clone());
                            // 加入
                            brokerDataList.add(brokerDataClone);
                            foundBrokerData = true;
                            // 遍历该BrokerName下的主从Broker地址
                            for (final String brokerAddr : brokerDataClone.getBrokerAddrs().values()) {
                                List<String> filterServerList = this.filterServerTable.get(brokerAddr);
                                filterServerMap.put(brokerAddr, filterServerList);
                            }
                        }
                    }
                }
            } finally {
                this.lock.readLock().unlock();
            }
        } catch (Exception e) {
            // ...
        }
        if (foundBrokerData && foundQueueData) {
            return topicRouteData;
        }
        return null;
    }
