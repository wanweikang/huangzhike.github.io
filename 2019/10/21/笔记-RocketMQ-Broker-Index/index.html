IndexService#buildIndex

    public void buildIndex(DispatchRequest req) {
        // 获取IndexFile，没有就创建
        IndexFile indexFile = retryGetAndCreateIndexFile();
        if (indexFile != null) {
            // 当前IndexFile的最后一条消息在CommitLog的物理偏移
            long endPhyOffset = indexFile.getEndPhyOffset();
            DispatchRequest msg = req;
            String topic = msg.getTopic();
            String keys = msg.getKeys();
            // 说明已创建对应索引
            if (msg.getCommitLogOffset() < endPhyOffset) {
                return;
            }
            final int tranType = MessageSysFlag.getTransactionValue(msg.getSysFlag());
            // 消息的事务类型
            switch (tranType) {
                case MessageSysFlag.TRANSACTION_NOT_TYPE:
                case MessageSysFlag.TRANSACTION_PREPARED_TYPE:
                case MessageSysFlag.TRANSACTION_COMMIT_TYPE:
                    break;
                // Rollback消息不创建索引
                case MessageSysFlag.TRANSACTION_ROLLBACK_TYPE:
                    return;
            }
            /*
             * 如果消息设置了UNIQ_KEY，就用Topic + “#” + UNIQ_KEY
             * 如果消息设置了KEYS（空格分隔），就用Topic + “#” + KEY
             */
            // 设置了UniqKey
            if (req.getUniqKey() != null) {
                // 根据Topic和UniqKey生成Key，并生成Index
                // UniqKey是唯一的，而MessageId在CommitLog不唯一，比如重试消息是会多次放入CommitLog的
                indexFile = putKey(indexFile, msg, buildKey(topic, req.getUniqKey()));
                if (indexFile == null) {
                    return;
                }
            }
            // 设置了KEYS
            if (keys != null && keys.length() > 0) {
                // 空格分隔
                String[] keyset = keys.split(MessageConst.KEY_SEPARATOR);
                // 遍历KEYS
                for (int i = 0; i < keyset.length; i++) {
                    String key = keyset[i];
                    if (key.length() > 0) {
                        // 为该Key生成索引
                        indexFile = putKey(indexFile, msg, buildKey(topic, key));
                        if (indexFile == null) {
                            return;
                        }
                    }
                }
            }
        } else {
        }
    }

IndexService#retryGetAndCreateIndexFile

    public IndexFile retryGetAndCreateIndexFile() {
        IndexFile indexFile = null;
        // 最多重试3次，直到成功获取或创建IndexFile
        for (int times = 0; null == indexFile && times < MAX_TRY_IDX_CREATE; times++) {
            indexFile = this.getAndCreateLastIndexFile();
            // 获取或创建成功就退出
            if (null != indexFile)
                break;
            // 创建失败就挂起1秒再重试
            try {
                Thread.sleep(1000);
            } catch (InterruptedException e) {
            }
        }
        // 重试创建失败
        if (null == indexFile) {
            this.defaultMessageStore.getAccessRights().makeIndexFileError();
        }
        return indexFile;
    }

IndexService#getAndCreateLastIndexFile

    public IndexFile getAndCreateLastIndexFile() {
        IndexFile indexFile = null;
        IndexFile prevIndexFile = null;
        long lastUpdateEndPhyOffset = 0;
        long lastUpdateIndexTimestamp = 0;
        // 先尝试获取IndexFile
        {
            // 获取读锁
            this.readWriteLock.readLock().lock();
            // 已存在IndexFile文件列表
            if (!this.indexFileList.isEmpty()) {
                // 获取最后一个
                IndexFile tmp = this.indexFileList.get(this.indexFileList.size() - 1);
                // IndexFile没写满
                if (!tmp.isWriteFull()) {
                    // 就决定是你了
                    indexFile = tmp;
                }
                // 写满了
                else {
                    lastUpdateEndPhyOffset = tmp.getEndPhyOffset();
                    lastUpdateIndexTimestamp = tmp.getEndTimestamp();
                    prevIndexFile = tmp;
                }
            }
            this.readWriteLock.readLock().unlock();
        }
        // 没有IndexFile就尝试创建
        if (indexFile == null) {
            try {
                // 文件名是时间戳
                String fileName = this.storePath + File.separator + UtilAll.timeMillisToHumanString(System.currentTimeMillis());
                // 创建
                indexFile = new IndexFile(fileName, this.hashSlotNum, this.indexNum, lastUpdateEndPhyOffset, lastUpdateIndexTimestamp);
                // 获取写锁
                this.readWriteLock.writeLock().lock();
                // 加入文件列表
                this.indexFileList.add(indexFile);
            } catch (Exception e) {
                // ...
            } finally {
                this.readWriteLock.writeLock().unlock();
            }
            if (indexFile != null) {
                // 上一个写满了的IndexFile
                final IndexFile flushThisFile = prevIndexFile;
                // 开一个线程刷上一个IndexFile
                Thread flushThread = new Thread(new Runnable() {
                    @Override
                    public void run() {
                        IndexService.this.flush(flushThisFile);
                    }
                }, "FlushIndexFileThread");
                flushThread.setDaemon(true);
                flushThread.start();
            }
        }
        return indexFile;
    }

IndexService#putKey

    private IndexFile putKey(IndexFile indexFile, DispatchRequest msg, String idxKey) {
        // 不断重试PutKey
        for (boolean ok = indexFile.putKey(idxKey, msg.getCommitLogOffset(), msg.getStoreTimestamp()); !ok; ) {
            // 可能写的过程中IndexFile满了，创建新的
            indexFile = retryGetAndCreateIndexFile();
            if (null == indexFile) {
                return null;
            }
            // 如果满了，就写新的IndexFile
            ok = indexFile.putKey(idxKey, msg.getCommitLogOffset(), msg.getStoreTimestamp());
        }
        return indexFile;
    }

IndexService#putKey

    public boolean putKey(final String key, final long phyOffset, final long storeTimestamp) {
        // IndexFile的Index未满
        if (this.indexHeader.getIndexCount() < this.indexNum) {
            // 对键做哈希运算
            int keyHash = indexKeyHashMethod(key);
            // 计算键对应的Slot位置
            int slotPos = keyHash % this.hashSlotNum;
            // 计算对应的Slot在IndexFile的位置
            // IndexHeader大小 + Slot的索引 * Slot的大小
            int absSlotPos = IndexHeader.INDEX_HEADER_SIZE + slotPos * hashSlotSize;
            FileLock fileLock = null;
            try {
                // fileLock = this.fileChannel.lock(absSlotPos, hashSlotSize, false);
                // 获取该位置的Slot的值，如果该位置的Slot已存在值，则哈希冲突
                int slotValue = this.mappedByteBuffer.getInt(absSlotPos);
                // 说明该Slot还没设置值
                if (slotValue <= invalidIndex || slotValue > this.indexHeader.getIndexCount()) {
                    // 后面用到
                    slotValue = invalidIndex;
                }
                // 当前消息的存储时间戳 - 当前IndexFile的第一条消息的时间戳
                long timeDiff = storeTimestamp - this.indexHeader.getBeginTimestamp();
                // 转为秒
                timeDiff = timeDiff / 1000;
                // 当前消息就是IndexFile的第一条消息
                if (this.indexHeader.getBeginTimestamp() <= 0) {
                    timeDiff = 0;
                } else if (timeDiff > Integer.MAX_VALUE) {
                    timeDiff = Integer.MAX_VALUE;
                } else if (timeDiff < 0) {
                    timeDiff = 0;
                }
                // 计算当前Index在IndexFile的存储位置偏移
                // IndexHeader大小 + Slot数量 * Slot大小 + Index数量 * Index大小
                int absIndexPos = IndexHeader.INDEX_HEADER_SIZE + this.hashSlotNum * hashSlotSize + this.indexHeader.getIndexCount() * indexSize;
                // 添加Index
                // 消息的MessageKey的哈希值，4字节
                this.mappedByteBuffer.putInt(absIndexPos, keyHash);
                // 消息在CommitLog的物理偏移，8字节
                this.mappedByteBuffer.putLong(absIndexPos + 4, phyOffset);
                // 时间差，4字节
                this.mappedByteBuffer.putInt(absIndexPos + 4 + 8, (int) timeDiff);
                // 冲突的同Slot的上一条消息在IndexFile的位置偏移，4字节
                this.mappedByteBuffer.putInt(absIndexPos + 4 + 8 + 4, slotValue);
                // 更新Slot
                // 当前Index的序号
                this.mappedByteBuffer.putInt(absSlotPos, this.indexHeader.getIndexCount());
                // 当前Index是第一条Index
                if (this.indexHeader.getIndexCount() <= 1) {
                    // 设置IndexHeader的第一条消息在CommitLog的物理偏移
                    this.indexHeader.setBeginPhyOffset(phyOffset);
                    // 设置IndexHeader的第一条消息的存储时间戳
                    this.indexHeader.setBeginTimestamp(storeTimestamp);
                }
                // 更新IndexHeader的Slot计数
                this.indexHeader.incHashSlotCount();
                // 更新IndexHeader的Index计数
                this.indexHeader.incIndexCount();
                // 设置IndexHeader的最后一条消息在CommitLog的物理偏移
                this.indexHeader.setEndPhyOffset(phyOffset);
                // 设置IndexHeader的最后一条消息的存储时间戳
                this.indexHeader.setEndTimestamp(storeTimestamp);
                return true;
            } catch (Exception e) {
                // ...
            } finally {
                // ...
            }
        } else {
            // ...
        }
        return false;
    }

IndexService#queryOffset

    public QueryOffsetResult queryOffset(String topic, String key, int maxNum, long begin, long end) {
        // 存放查询到的消息在CommitLog的偏移
        List<Long> phyOffsets = new ArrayList<Long>(maxNum);
        long indexLastUpdateTimestamp = 0;
        long indexLastUpdatePhyoffset = 0;
        // 最多返回的消息数，多个消息是因为哈希冲突，Broker只是根据哈希过滤，Consumer需要根据字符串再次过滤
        maxNum = Math.min(maxNum, this.defaultMessageStore.getMessageStoreConfig().getMaxMsgsNumBatch());
        try {
            // 获取读锁
            this.readWriteLock.readLock().lock();
            // 存在IndexFile
            if (!this.indexFileList.isEmpty()) {
                // 从后往前遍历IndexFile
                for (int i = this.indexFileList.size(); i > 0; i--) {
                    IndexFile f = this.indexFileList.get(i - 1);
                    boolean lastFile = i == this.indexFileList.size();
                    if (lastFile) {
                        indexLastUpdateTimestamp = f.getEndTimestamp();
                        indexLastUpdatePhyoffset = f.getEndPhyOffset();
                    }
                    // IndexFile保存了第一和最后一条消息的时间
                    if (f.isTimeMatched(begin, end)) {
                        // 从该IndexFile查询物理偏移
                        f.selectPhyOffset(phyOffsets, buildKey(topic, key), maxNum, begin, end, lastFile);
                    }
                    // 该IndexFile的起始时间比查询的起始时间小
                    if (f.getBeginTimestamp() < begin) {
                        // 跳过后面的IndexFile，因为是倒序遍历，后面的时间都比它小
                        break;
                    }
                    // 查询到的消息数量超过了限制
                    if (phyOffsets.size() >= maxNum) {
                        // 跳
                        break;
                    }
                }
            }
        } catch (Exception e) {
            // ...
        } finally {
            this.readWriteLock.readLock().unlock();
        }
        return new QueryOffsetResult(phyOffsets, indexLastUpdateTimestamp, indexLastUpdatePhyoffset);
    }

IndexService#selectPhyOffset

    public void selectPhyOffset(final List<Long> phyOffsets, final String key, final int maxNum, final long begin, final long end, boolean lock) {
        // 获取MappedFile引用计数
        if (this.mappedFile.hold()) {
            // 获得哈希值
            int keyHash = indexKeyHashMethod(key);
            // 哈希对应的Slot的位置
            int slotPos = keyHash % this.hashSlotNum;
            // 对应的Slot在IndexFile的物理偏移
            int absSlotPos = IndexHeader.INDEX_HEADER_SIZE + slotPos * hashSlotSize;
            FileLock fileLock = null;
            try {
                // 获取该Slot最新一条的Index的序号
                int slotValue = this.mappedByteBuffer.getInt(absSlotPos);
                if (slotValue <= invalidIndex || slotValue > this.indexHeader.getIndexCount() || this.indexHeader.getIndexCount() <= 1) {
                } else {
                    // 遍历Index链表
                    for (int nextIndexToRead = slotValue; ; ) {
                        // 查询到消息位置的数量超过限制
                        if (phyOffsets.size() >= maxNum) {
                            // 再见
                            break;
                        }
                        // Index在IndexFile的物理偏移
                        int absIndexPos = IndexHeader.INDEX_HEADER_SIZE + this.hashSlotNum * hashSlotSize + nextIndexToRead * indexSize;
                        // 键的哈希，4字节
                        int keyHashRead = this.mappedByteBuffer.getInt(absIndexPos);
                        // 消息在CommitLog的物理偏移，8字节
                        long phyOffsetRead = this.mappedByteBuffer.getLong(absIndexPos + 4);
                        // 时间差，8字节
                        long timeDiff = (long) this.mappedByteBuffer.getInt(absIndexPos + 4 + 8);
                        // 同Slot的上一个Index的序号，4字节
                        int prevIndexRead = this.mappedByteBuffer.getInt(absIndexPos + 4 + 8 + 4);
                        if (timeDiff < 0) {
                            break;
                        }
                        // 转为毫秒
                        timeDiff *= 1000L;
                        // IndexFile的第一条消息的时间 + 时间差
                        long timeRead = this.indexHeader.getBeginTimestamp() + timeDiff;
                        // 匹配时间范围
                        boolean timeMatched = (timeRead >= begin) && (timeRead <= end);
                        // 哈希匹配 && 时间匹配
                        if (keyHash == keyHashRead && timeMatched) {
                            // 加入结果
                            phyOffsets.add(phyOffsetRead);
                        }
                        // 下个要读的Index无效
                        if (prevIndexRead <= invalidIndex || prevIndexRead > this.indexHeader.getIndexCount() || prevIndexRead == nextIndexToRead || timeRead < begin) {
                            // 再见
                            break;
                        }
                        // 链表的下个节点的序号
                        nextIndexToRead = prevIndexRead;
                    }
                }
            } catch (Exception e) {
                // ...
            } finally {
                // ...
            }
        }
    }
