RemotingCommand#encodeHeader

    public ByteBuffer encodeHeader(final int bodyLength) {
        // 消息总长度，包括序列化类型与消息头长度的四字节
        int length = 4;

        byte[] headerData;
        // 编码消息头
        headerData = this.headerEncode();
        // 消息总长度加上消息头长度
        length += headerData.length;
        // 消息总长度加上消息体长度
        length += bodyLength;
        // 分配缓冲，包括消息总长度自身占的四字节
        ByteBuffer result = ByteBuffer.allocate(4 + length - bodyLength);
        /*
         * 协议分为四部分
         */
        // 消息总长度，四字节
        result.putInt(length);
        // 序列化类型与消息头长度，共四字节
        result.put(markProtocolType(headerData.length, serializeTypeCurrentRPC));
        // 消息头数据
        result.put(headerData);
        // 切换为读模式
        result.flip();
        return result;
    }

RemotingCommand#markProtocolType

    /**
     * 高低位转换，将小端字节序转为大端字节序
     */
    public static byte[] markProtocolType(int source, SerializeType type) {
        byte[] result = new byte[4];
        // 高位在前
        result[0] = type.getCode();
        result[1] = (byte) ((source >> 16) & 0xFF);
        result[2] = (byte) ((source >> 8) & 0xFF);
        // 低位最后
        result[3] = (byte) (source & 0xFF);
        return result;
    }

RemotingCommand#decode

    public static RemotingCommand decode(final ByteBuffer byteBuffer) {
        // 消息长度，四字节
        int length = byteBuffer.limit();
        // 序列化类型和消息头长度，四字节
        int oriHeaderLen = byteBuffer.getInt();
        // 后三个字节，消息头长度
        int headerLength = getHeaderLength(oriHeaderLen);
        byte[] headerData = new byte[headerLength];
        // 填充消息头
        byteBuffer.get(headerData);
        // 根据序列化类型解码消息头
        RemotingCommand cmd = headerDecode(headerData, getProtocolType(oriHeaderLen));
        // 消息体长度
        int bodyLength = length - 4 - headerLength;
        byte[] bodyData = null;
        if (bodyLength > 0) {
            bodyData = new byte[bodyLength];
            // 填充消息体
            byteBuffer.get(bodyData);
        }
        cmd.body = bodyData;
        return cmd;
    }

RemotingCommand#getHeaderLength

    public static int getHeaderLength(int length) {
        // 截断
        return length & 0xFFFFFF;
    }

RemotingCommand#getProtocolType

    public static SerializeType getProtocolType(int source) {
        // 右移三字节并截断
        return SerializeType.valueOf((byte) ((source >> 24) & 0xFF));
    }

NettyRemotingAbstract#invokeSyncImpl

    public RemotingCommand invokeSyncImpl(final Channel channel, final RemotingCommand request, final long timeoutMillis) throws InterruptedException, RemotingSendRequestException, RemotingTimeoutException {

        // 请求ID
        final int opaque = request.getOpaque();
        try {
            // 根据请求ID创建ResponseFuture
            final ResponseFuture responseFuture = new ResponseFuture(channel, opaque, timeoutMillis, null, null);
            // 加入ResponseTable，等待收到响应后处理
            this.responseTable.put(opaque, responseFuture);
            final SocketAddress addr = channel.remoteAddress();
            // 发送请求
            channel.writeAndFlush(request).addListener(new ChannelFutureListener() {
                @Override
                public void operationComplete(ChannelFuture f) throws Exception {
                    // 发送成功
                    if (f.isSuccess()) {
                        responseFuture.setSendRequestOK(true);
                        return;
                    } else {
                        responseFuture.setSendRequestOK(false);
                    }
                    // 发送失败，从ResponseTable移除
                    responseTable.remove(opaque);
                    responseFuture.setCause(f.cause());
                    // 唤醒线程，解除阻塞
                    responseFuture.putResponse(null);
                    // ...
                }
            });
            // 阻塞等待收到响应后唤醒或超时
            RemotingCommand responseCommand = responseFuture.waitResponse(timeoutMillis);
            if (null == responseCommand) {
                // ...
            }
            return responseCommand;
        } finally {
            // 最后从ResponseTable移除
            this.responseTable.remove(opaque);
        }
    }

    public void processMessageReceived(ChannelHandlerContext ctx, RemotingCommand msg) throws Exception {
        final RemotingCommand cmd = msg;
        if (cmd != null) {
            switch (cmd.getType()) {
                case REQUEST_COMMAND:
                    // 处理请求
                    processRequestCommand(ctx, cmd);
                    break;
                case RESPONSE_COMMAND:
                    // 处理响应
                    processResponseCommand(ctx, cmd);
                    break;
                default:
                    break;
            }
        }
    }

NettyRemotingAbstract#processResponseCommand

    public void processResponseCommand(ChannelHandlerContext ctx, RemotingCommand cmd) {
        // 请求Id
        final int opaque = cmd.getOpaque();
        // 从根据请求Id从ResponseTable获取ResponseFuture
        final ResponseFuture responseFuture = responseTable.get(opaque);
        if (responseFuture != null) {
            responseFuture.setResponseCommand(cmd);
            // 从ResponseTable移除
            responseTable.remove(opaque);
            // 提交给线程池执行回调
            if (responseFuture.getInvokeCallback() != null) {
                executeInvokeCallback(responseFuture);
            }
            // 说明是同步请求
            else {
                // 解除阻塞
                responseFuture.putResponse(cmd);
                responseFuture.release();
            }
        } else {
            // ...
        }
    }

    public void processRequestCommand(final ChannelHandlerContext ctx, final RemotingCommand cmd) {
        // 根据Code获取匹配的处理器及对应的线程池
        final Pair<NettyRequestProcessor, ExecutorService> matched = this.processorTable.get(cmd.getCode());
        // 没有就使用默认的
        final Pair<NettyRequestProcessor, ExecutorService> pair = null == matched ? this.defaultRequestProcessor : matched;
        // 请求Id
        final int opaque = cmd.getOpaque();
        if (pair != null) {
            // 创建任务
            Runnable run = () -> {
                try {
                    doBeforeRpcHooks(RemotingHelper.parseChannelRemoteAddr(ctx.channel()), cmd);
                    // 处理请求，生成响应
                    final RemotingCommand response = pair.getObject1().processRequest(ctx, cmd);
                    doAfterRpcHooks(RemotingHelper.parseChannelRemoteAddr(ctx.channel()), cmd, response);
                    // 非单向
                    if (!cmd.isOnewayRPC()) {
                        if (response != null) {
                            // 请求Id
                            response.setOpaque(opaque);
                            response.markResponseType();
                            try {
                                // 回复请求
                                ctx.writeAndFlush(response);
                            } catch (Throwable e) {
                                // ...
                            }
                        } else {
                        }
                    }
                } catch (Throwable e) {
                    // ...
                }
            };
            // 处理器是否拒绝请求
            if (pair.getObject1().rejectRequest()) {
                // ...
            }
            try {
                // 封装任务
                final RequestTask requestTask = new RequestTask(run, ctx.channel(), cmd);
                // 提交到对应线程池
                pair.getObject2().submit(requestTask);
            } catch (RejectedExecutionException e) {
                // ...
            }
        } else {
            // ...
        }
    }

NettyRemotingAbstract#invokeAsyncImpl

    public void invokeAsyncImpl(final Channel channel, final RemotingCommand request, final long timeoutMillis, final InvokeCallback invokeCallback) throws InterruptedException, RemotingTooMuchRequestException, RemotingTimeoutException, RemotingSendRequestException {
        long beginStartTime = System.currentTimeMillis();
        // 请求Id
        final int opaque = request.getOpaque();
        // 阻塞超时方式获取信号量，用于限流，异步请求完成后才释放
        boolean acquired = this.semaphoreAsync.tryAcquire(timeoutMillis, TimeUnit.MILLISECONDS);
        // 获得了信号量
        if (acquired) {
            // 包装信号量，只允许释放一次，不然会增加许可
            final SemaphoreReleaseOnlyOnce once = new SemaphoreReleaseOnlyOnce(this.semaphoreAsync);
            long costTime = System.currentTimeMillis() - beginStartTime;
            // 获取信号量后超时了？
            if (timeoutMillis < costTime) {
                // 释放信号量
                once.release();
            }
            final ResponseFuture responseFuture = new ResponseFuture(channel, opaque, timeoutMillis - costTime, invokeCallback, once);
            // 加入ResponseTable，等待响应处理，对于因为网络原因没有收到Response的请求，后台有个计划任务每隔一秒扫描一次，处理超时
            this.responseTable.put(opaque, responseFuture);
            try {
                // 发送
                channel.writeAndFlush(request).addListener(new ChannelFutureListener() {
                    @Override
                    public void operationComplete(ChannelFuture f) throws Exception {
                        if (f.isSuccess()) {
                            responseFuture.setSendRequestOK(true);
                            return;
                        }
                        // 发送失败，执行失败回调，并释放信号量
                        requestFail(opaque);
                    }
                });
            } catch (Exception e) {
                // 发生异常也要释放信号量
                responseFuture.release();
                // ...
            }
        } else {
            // ...
        }
    }

NettyRemotingAbstract#scanResponseTable

    public void scanResponseTable() {
        final List<ResponseFuture> rfList = new LinkedList<ResponseFuture>();
        Iterator<Entry<Integer, ResponseFuture>> it = this.responseTable.entrySet().iterator();
        // 遍历
        while (it.hasNext()) {
            Entry<Integer, ResponseFuture> next = it.next();
            ResponseFuture rep = next.getValue();
            // 超时了
            if ((rep.getBeginTimestamp() + rep.getTimeoutMillis() + 1000) <= System.currentTimeMillis()) {
                // 释放信号量
                rep.release();
                // 超时需要移除
                it.remove();
                rfList.add(rep);
            }
        }
        // 超时的响应
        for (ResponseFuture rf : rfList) {
            try {
                // 执行回调
                executeInvokeCallback(rf);
            } catch (Throwable e) {
                // ...
            }
        }
    }
