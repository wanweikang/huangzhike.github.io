    public PutMessageResult putMessage(final MessageExtBrokerInner msg) {
        // ...
        final int tranType = MessageSysFlag.getTransactionValue(msg.getSysFlag());
        // 非事务消息 || Commit型事务消息
        if (tranType == MessageSysFlag.TRANSACTION_NOT_TYPE || tranType == MessageSysFlag.TRANSACTION_COMMIT_TYPE) {
            // 如果设置了延迟投递
            if (msg.getDelayTimeLevel() > 0) {
                if (msg.getDelayTimeLevel() > this.defaultMessageStore.getScheduleMessageService().getMaxDelayLevel()) {
                    msg.setDelayTimeLevel(this.defaultMessageStore.getScheduleMessageService().getMaxDelayLevel());
                }
                // 修改Topic为SCHEDULE_TOPIC_XXXX，不会被Consumer消费，因为没订阅这个Topic，后续由后台线程扫描恢复原本主题
                topic = ScheduleMessageService.SCHEDULE_TOPIC;
                // 根据延时级别计算QueueId，其实就是DelayLevel-1，后面会把消息放到SCHEDULE_TOPIC_XXXX的队列里
                queueId = ScheduleMessageService.delayLevel2QueueId(msg.getDelayTimeLevel());
                // 备份原本的Topic和QueueId
                MessageAccessor.putProperty(msg, MessageConst.PROPERTY_REAL_TOPIC, msg.getTopic());
                MessageAccessor.putProperty(msg, MessageConst.PROPERTY_REAL_QUEUE_ID, String.valueOf(msg.getQueueId()));
                msg.setPropertiesString(MessageDecoder.messageProperties2String(msg.getProperties()));
                // 修改Topic
                msg.setTopic(topic);
                // 修改QueueId
                msg.setQueueId(queueId);
                /*
                 * 简单说一下延时消息的逻辑：
                 * 延时消息到达Broker后，修改Topic为SCHEDULE_TOPIC_XXXX，QueueId为DelayLevel-1
                 * ReputMessageService将消息从CommitLog中更新ConsumeQueue时，将TagsCode换成时间戳，因为延时消息不需要Tag过滤，没有Consumer订阅
                 * DeliverDelayedMessageTimerTask任务线程遍历ConsumeQueue，根据TagsCode取出到时的消息，恢复原本的Topic和QueueId，再次写入CommitLog
                 * 然后再次更新ConsumeQueue，这样就可以被消费了
                 */
            }
        }
        // ...
        MappedFile unlockMappedFile = null;
        // 获取最后一个MappedFile，因为是顺序写入
        MappedFile mappedFile = this.mappedFileQueue.getLastMappedFile();
        // 获取锁
        putMessageLock.lock(); //spin or ReentrantLock ,depending on store config
        try {
            // ...
            // 在回调中将消息写入MappedFile，返回结果
            result = mappedFile.appendMessage(msg, this.appendMessageCallback);
            // ...
        } finally {
            // ...
        }
        // ...
        return putMessageResult;
    }

    public DispatchRequest checkMessageAndReturnSize(java.nio.ByteBuffer byteBuffer, final boolean checkCRC, final boolean readBody) {
        try {
            // ...
            // Properties长度，2字节
            short propertiesLength = byteBuffer.getShort();
            Map<String, String> propertiesMap = null;
            if (propertiesLength > 0) {
                // ...
                String tags = propertiesMap.get(MessageConst.PROPERTY_TAGS);
                if (tags != null && tags.length() > 0) {
                    // TagsCode
                    tagsCode = MessageExtBrokerInner.tagsString2tagsCode(MessageExt.parseTopicFilterType(sysFlag), tags);
                }
                // 处理延时消息
                {
                    // 延时级别
                    String t = propertiesMap.get(MessageConst.PROPERTY_DELAY_TIME_LEVEL);
                    // 如果是延时消息
                    if (ScheduleMessageService.SCHEDULE_TOPIC.equals(topic) && t != null) {
                        int delayLevel = Integer.parseInt(t);
                        // 检查，如果大于最大的延时级别
                        if (delayLevel > this.defaultMessageStore.getScheduleMessageService().getMaxDelayLevel()) {
                            delayLevel = this.defaultMessageStore.getScheduleMessageService().getMaxDelayLevel();
                        }
                        if (delayLevel > 0) {
                            // 计算延迟消息应该被投递的时间，覆盖原有的TagsCode
                            tagsCode = this.defaultMessageStore.getScheduleMessageService().computeDeliverTimestamp(delayLevel, storeTimestamp);
                        }
                    }
                }
            }
            // ...
            // 封装消息，返回DispatchRequest
            return new DispatchRequest(topic, queueId, physicOffset, totalSize, tagsCode, storeTimestamp, queueOffset, keys, uniqKey, sysFlag, preparedTransactionOffset, propertiesMap);
        } catch (Exception e) {
        }
        return new DispatchRequest(-1, false /* success */);
    }

ScheduleMessageService#start

    public void start() {
        if (started.compareAndSet(false, true)) {
            // 守护线程
            this.timer = new Timer("ScheduleMessageTimerThread", true);
            // 遍历所有延迟级别
            for (Map.Entry<Integer, Long> entry : this.delayLevelTable.entrySet()) {
                // 延迟级别
                Integer level = entry.getKey();
                // 延迟毫秒
                Long timeDelay = entry.getValue();
                // 延迟级别对应的消费进度偏移
                Long offset = this.offsetTable.get(level);
                if (null == offset) {
                    offset = 0L;
                }
                if (timeDelay != null) {
                    // 每个延迟级别都有一个TimerTask负责扫描，不支持任意精度的延迟
                    this.timer.schedule(new DeliverDelayedMessageTimerTask(level, offset), FIRST_DELAY_TIME);
                }

            }
            // 定时任务，每隔十秒将消费进度持久化到${user.home}/store/config/delayOffset.json
            this.timer.scheduleAtFixedRate(new TimerTask() {

                @Override
                public void run() {
                    try {
                        if (started.get())
                            ScheduleMessageService.this.persist();
                    } catch (Throwable e) {
                        // ...
                    }
                }
            }, 10000, this.defaultMessageStore.getMessageStoreConfig().getFlushDelayOffsetInterval());
        }
    }

        public void executeOnTimeup() {
            // 根据Topic和延迟级别找到ConsumeQueue，之前加入时设置QueueId = DelayLevel - 1
            // 所有的延迟消息都放在SCHEDULE_TOPIC主题下，根据延迟级别不同，ConsumeQueue队列不同
            ConsumeQueue cq = ScheduleMessageService.this.defaultMessageStore.findConsumeQueue(SCHEDULE_TOPIC, delayLevel2QueueId(delayLevel));
            // 初始化时传入的该延迟级别的ConsumeQueue的消息消费进度
            long failScheduleOffset = offset;
            if (cq != null) {
                // 根据该延迟级别的消费进度偏移获取所有的未消费消息
                SelectMappedBufferResult bufferCQ = cq.getIndexBuffer(this.offset);
                // 获取到消息缓冲
                if (bufferCQ != null) {
                    try {
                        long nextOffset = offset;
                        int i = 0;
                        ConsumeQueueExt.CqExtUnit cqExtUnit = new ConsumeQueueExt.CqExtUnit();
                        // 遍历
                        for (; i < bufferCQ.getSize(); i += ConsumeQueue.CQ_STORE_UNIT_SIZE) {
                            long offsetPy = bufferCQ.getByteBuffer().getLong();
                            int sizePy = bufferCQ.getByteBuffer().getInt();
                            // TagsCode
                            long tagsCode = bufferCQ.getByteBuffer().getLong();
                            if (cq.isExtAddr(tagsCode)) {
                                // ...
                            }
                            long now = System.currentTimeMillis();
                            // 根据TagsCode计算延时到期时间，更新ConsumeQueue时，如果是延时消息，会将TagsCode换成时间戳，因为不需要用Tag过滤
                            long deliverTimestamp = this.correctDeliverTimestamp(now, tagsCode);
                            nextOffset = offset + (i / ConsumeQueue.CQ_STORE_UNIT_SIZE);
                            long countdown = deliverTimestamp - now;
                            // 消息到期，可以被消费
                            if (countdown <= 0) {
                                // 根据消息在CommitLog的偏移及消息大小从CommitLog查询消息
                                MessageExt msgExt = ScheduleMessageService.this.defaultMessageStore.lookMessageByOffset(offsetPy, sizePy);
                                if (msgExt != null) {
                                    try {
                                        // 先恢复消息原先的QueueId及Topic
                                        MessageExtBrokerInner msgInner = this.messageTimeup(msgExt);
                                        // 重新写入CommitLog
                                        PutMessageResult putMessageResult = ScheduleMessageService.this.writeMessageStore.putMessage(msgInner);
                                        // 写入成功
                                        if (putMessageResult != null && putMessageResult.getPutMessageStatus() == PutMessageStatus.PUT_OK) {
                                            // 跳过，继续下一个消息
                                            continue;
                                        }
                                        // 写入失败
                                        else {
                                            // 十秒后继续下一次调度任务
                                            ScheduleMessageService.this.timer.schedule(new DeliverDelayedMessageTimerTask(this.delayLevel, nextOffset), DELAY_FOR_A_PERIOD);
                                            // 更新当前延迟级别的消费进度偏移
                                            ScheduleMessageService.this.updateOffset(this.delayLevel, nextOffset);
                                            return;
                                        }
                                    } catch (Exception e) {
                                        // ...
                                    }
                                }
                            }
                            // 时间还没到
                            else {
                                // 根据剩余时间，启动定时任务
                                ScheduleMessageService.this.timer.schedule(new DeliverDelayedMessageTimerTask(this.delayLevel, nextOffset), countdown);
                                // 更新当前延迟级别的消费进度偏移
                                ScheduleMessageService.this.updateOffset(this.delayLevel, nextOffset);
                                // 返回
                                return;
                            }
                        } // 消息遍历结束
                        //
                        nextOffset = offset + (i / ConsumeQueue.CQ_STORE_UNIT_SIZE);
                        ScheduleMessageService.this.timer.schedule(new DeliverDelayedMessageTimerTask(this.delayLevel, nextOffset), DELAY_FOR_A_WHILE);
                        ScheduleMessageService.this.updateOffset(this.delayLevel, nextOffset);
                        return;
                    } finally {
                        // ...
                    }
                }
                // ConsumeQueue的消费偏移后面没有数据
                else {
                    // ...
                }
            }
            // 计划下一次任务
            ScheduleMessageService.this.timer.schedule(new DeliverDelayedMessageTimerTask(this.delayLevel, failScheduleOffset), DELAY_FOR_A_WHILE);
        }

2019年11月19日更新：

RocketMQ是不支持任意精度的延时消息的，而是一个延时级别的消息放在一个队列里边，按收到消息的顺序持久化，也按照此顺序发送消息，巧妙地避免了消息排序的开销，同时有一个独立的定时器，开销也不大。

如果想基于RocketMQ实现任意精度的延时消息，我觉得首先得解决消息排序的问题，今天偶然看到一位大佬的文章后豁然开朗，以下是我对他的思路的理解：

  • 消息写入CommitLog；

  • Dispatcher处理延迟消息；

  • 假设MQ每次只处理将来30分钟的延迟消息，假设第一次处理的开始时间为整点，如2019年11月19日00点00分，下一次处理的开始时间就是2019年11月19日00点30分，以此类推；

  • 系统有一个TimeWheel，每次Tick的精度为1s，完成一圈需要30*60s，每个格子的链表则存放该秒需要发送的消息索引（内存的消耗可以接受）；

  • 系统有一个DelayMsgFile文件队列，专门存放延时消息，按顺序只放该区间的消息，比如第一个文件存放2019年11月19日00点00分2019年11月19日00点30分的消息，第二个文件存放2019年11月19日30点00分2019年11月19日01点00分的消息，依次类推，存储时每个消息单元包含了消息的相对时间索引，即该消息的投递时间相对于该时间区间左端点的秒数差，比如，如果一个消息计划在2019年11月19日00点03分投递，那么它的时间索引就是3*60，对应的就是TimeWheel的某个格子；

  • TimeWheel启动时，从对应的区间的DelayMsgFile加载消息到TimeWheel，走到对应格子后就将链表里的消息投递到ConsumeQueue，这样就对消费者可见。这段时间内，如果有新的、在区间内的延时消息，直接放入TimeWheel，否则持久化入对应的DelayMsgFile。当TimeWheel走完一圈后，删除对应的DelayMsgFile。

这个做法也是巧妙地避开了延时消息的排序问题，大概这就是大佬吧，可以提出创造性的解决方案，像我，就想不出来😬。
