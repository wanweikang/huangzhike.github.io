DefaultMQPushConsumerImpl#pullMessage

    public void pullMessage(final PullRequest pullRequest) {
        // 获取处理队列
        final ProcessQueue processQueue = pullRequest.getProcessQueue();
        // 如果ProcessQueue被标记为废弃
        if (processQueue.isDropped()) {
            // 直接返回
        }
        pullRequest.getProcessQueue().setLastPullTimestamp(System.currentTimeMillis());
        try {
            // 确保消费者处于运行状态
            this.makeSureStateOK();
        } catch (MQClientException e) {
            // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
            this.executePullRequestLater(pullRequest, PULL_TIME_DELAY_MILLS_WHEN_EXCEPTION);
            return;
        }
        // 如果消费者被暂停
        if (this.isPause()) {
            // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
        }
        long cachedMessageCount = processQueue.getMsgCount().get();
        long cachedMessageSizeInMiB = processQueue.getMsgSize().get() / (1024 * 1024);
        // 如果Consumer本地堆积未处理的消息过多，超过消息拉取阈值，则需要执行流量控制
        if (cachedMessageCount > this.defaultMQPushConsumer.getPullThresholdForQueue()) {
            // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
        }
        // 如果Consumer本地堆积消息过大，超过消息拉取队列大小，则需要执行流量控制
        if (cachedMessageSizeInMiB > this.defaultMQPushConsumer.getPullThresholdSizeForQueue()) {
            // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
        }
        // 不需要顺序消费消息
        if (!this.consumeOrderly) {
            // 消息偏移跨度过大，则需要执行流量控制
            if (processQueue.getMaxSpan() > this.defaultMQPushConsumer.getConsumeConcurrentlyMaxSpan()) {
                // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
            }
        }
        // 需要顺序消费消息
        else {
            // 已经向Broker获取了该MessageQueue的分布式锁
            if (processQueue.isLocked()) {
                // PullRequest对象是重复使用的，第一次拉取时没标记锁定
                if (!pullRequest.isLockedFirst()) {
                    // 向Broker发送QUERY_CONSUMER_OFFSET请求，获取MessageQueue的消费进度
                    final long offset = this.rebalanceImpl.computePullFromWhere(pullRequest.getMessageQueue());
                    // 标记锁定
                    pullRequest.setLockedFirst(true);
                    // 设置拉取偏移
                    pullRequest.setNextOffset(offset);
                }
            }
            // 没有获取锁
            else {
                // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
            }
        }
        // 检查订阅关系，可能延时后，Topic或Consumer的配置发生了变化
        final SubscriptionData subscriptionData = this.rebalanceImpl.getSubscriptionInner().get(pullRequest.getMessageQueue().getTopic());
        // 没找到该Topic的订阅数据
        if (null == subscriptionData) {
            // 提交延时任务，将PullRequest放回队列，延时拉取，然后返回
        }
        final long beginTimestamp = System.currentTimeMillis();
        // 创建拉取回调
        PullCallback pullCallback = new PullCallback() {
            @Override
            public void onSuccess(PullResult pullResult) {
                if (pullResult != null) {
                    // 处理拉取结果，客户端根据Tag的字符串再次过滤
                    // 百度了一下，RocketMQ之所以设计为Broker根据哈希过滤而不是直接使用字符串，是因为ConsumeQueue为了性能设计为定长单元，显然不能使用字符串
                    pullResult = DefaultMQPushConsumerImpl.this.pullAPIWrapper.processPullResult(pullRequest.getMessageQueue(), pullResult, subscriptionData);
                    // 拉取状态
                    switch (pullResult.getPullStatus()) {
                        // 找到了消息
                        case FOUND:
                            long prevRequestOffset = pullRequest.getNextOffset();
                            // 根据Broker返回的下次拉取的开始偏移更新PullRequest
                            pullRequest.setNextOffset(pullResult.getNextBeginOffset());
                            // RoundTime
                            long pullRT = System.currentTimeMillis() - beginTimestamp;
                            // 统计
                            DefaultMQPushConsumerImpl.this.getConsumerStatsManager().incPullRT(pullRequest.getConsumerGroup(), pullRequest.getMessageQueue().getTopic(), pullRT);
                            long firstMsgOffset = Long.MAX_VALUE;
                            // 如果没有消息，可能是在上面被Consumer过滤了Tag
                            if (pullResult.getMsgFoundList() == null || pullResult.getMsgFoundList().isEmpty()) {
                                // 直接提交PullRequest到队列，执行下一次拉取
                                DefaultMQPushConsumerImpl.this.executePullRequestImmediately(pullRequest);
                            }
                            // 有消息
                            else {
                                firstMsgOffset = pullResult.getMsgFoundList().get(0).getQueueOffset();
                                // 统计
                                DefaultMQPushConsumerImpl.this.getConsumerStatsManager().incPullTPS(pullRequest.getConsumerGroup(), pullRequest.getMessageQueue().getTopic(), pullResult.getMsgFoundList().size());
                                // 把消息放入ProcessQueue，返回是否分发给消费者消费
                                boolean dispatchToConsume = processQueue.putMessage(pullResult.getMsgFoundList());
                                // 创建ConsumeRequest对象，提交到线程池，调用MessageListener消费消息，处理完成会通知ProcessQueue
                                DefaultMQPushConsumerImpl.this.consumeMessageService.submitConsumeRequest(pullResult.getMsgFoundList(), processQueue, pullRequest.getMessageQueue(), dispatchToConsume);

                                if (DefaultMQPushConsumerImpl.this.defaultMQPushConsumer.getPullInterval() > 0) {
                                    // 延迟间隔时间后再次提交PullRequest到队列
                                    DefaultMQPushConsumerImpl.this.executePullRequestLater(pullRequest, DefaultMQPushConsumerImpl.this.defaultMQPushConsumer.getPullInterval());
                                }
                                // PushConsumer
                                else {
                                    // 直接提交PullRequest到队列，执行下一次拉取
                                    DefaultMQPushConsumerImpl.this.executePullRequestImmediately(pullRequest);
                                }
                            }
                            break;
                        // 没有找到新的消息
                        case NO_NEW_MSG:
                            // 使用Broker校准的偏移
                            pullRequest.setNextOffset(pullResult.getNextBeginOffset());
                            DefaultMQPushConsumerImpl.this.correctTagsOffset(pullRequest);
                            // 直接提交PullRequest到队列，执行下一次拉取
                            DefaultMQPushConsumerImpl.this.executePullRequestImmediately(pullRequest);
                            break;
                        case NO_MATCHED_MSG:
                            pullRequest.setNextOffset(pullResult.getNextBeginOffset());
                            DefaultMQPushConsumerImpl.this.correctTagsOffset(pullRequest);
                            DefaultMQPushConsumerImpl.this.executePullRequestImmediately(pullRequest);
                            break;
                        // 偏移非法
                        case OFFSET_ILLEGAL:
                            // 使用Broker校准的偏移
                            pullRequest.setNextOffset(pullResult.getNextBeginOffset());
                            // 将该ProcessQueue标记为丢弃
                            pullRequest.getProcessQueue().setDropped(true);
                            // 提交延时任务，将PullRequest放回队列，延时拉取
                            DefaultMQPushConsumerImpl.this.executeTaskLater(new Runnable() {

                                @Override
                                public void run() {
                                    try {
                                        // 更新Consumer本地该MessageQueue的消费进度缓存
                                        DefaultMQPushConsumerImpl.this.offsetStore.updateOffset(pullRequest.getMessageQueue(), pullRequest.getNextOffset(), false);
                                        // 持久化当前消息队列的消费进度，集群模式下是向Broker发送UPDATE_CONSUMER_OFFSET的单向请求
                                        DefaultMQPushConsumerImpl.this.offsetStore.persist(pullRequest.getMessageQueue());
                                        // 移除当前MessageQueue的ProcessQueue，等待下次执行负载均衡
                                        DefaultMQPushConsumerImpl.this.rebalanceImpl.removeProcessQueue(pullRequest.getMessageQueue());
                                    } catch (Throwable e) {
                                        // ...
                                    }
                                }
                            }, 10000);
                            break;
                        default:
                            break;
                    }
                }
            }

            @Override
            public void onException(Throwable e) {
                // 提交延时任务，将PullRequest放回队列，延时拉取
            }
        }; // 回调代码结束
        // 是否同步消费进度到Broker，除了定时任务向Broker同步消费进度，拉取消息时也可能会顺便同步
        boolean commitOffsetEnable = false;
        long commitOffsetValue = 0L;
        // 如果是集群消费模式
        if (MessageModel.CLUSTERING == this.defaultMQPushConsumer.getMessageModel()) {
            // 从本地缓存获取该MessageQueue的消费进度
            commitOffsetValue = this.offsetStore.readOffset(pullRequest.getMessageQueue(), ReadOffsetType.READ_FROM_MEMORY);
            //
            if (commitOffsetValue > 0) {
                // 同步
                commitOffsetEnable = true;
            }
        }
        String subExpression = null;
        boolean classFilter = false;
        SubscriptionData sd = this.rebalanceImpl.getSubscriptionInner().get(pullRequest.getMessageQueue().getTopic());
        if (sd != null) {
            // ...
        }
        // 系统标记
        int sysFlag = PullSysFlag.buildSysFlag(commitOffsetEnable, // commitOffset
                true, // suspend
                subExpression != null, // subscription
                classFilter // class filter
        );
        try {
            // 执行拉取
            this.pullAPIWrapper.pullKernelImpl(pullRequest.getMessageQueue(), //
                    subExpression, //
                    subscriptionData.getExpressionType(), //
                    subscriptionData.getSubVersion(), //
                    pullRequest.getNextOffset(), //
                    this.defaultMQPushConsumer.getPullBatchSize(), // 批量拉取大小
                    sysFlag, //
                    commitOffsetValue, //
                    BROKER_SUSPEND_MAX_TIME_MILLIS, //
                    CONSUMER_TIMEOUT_MILLIS_WHEN_SUSPEND, //
                    CommunicationMode.ASYNC, // 异步
                    pullCallback);
        } catch (Exception e) {
            // 提交延时任务，将PullRequest放回队列，延时拉取
            this.executePullRequestLater(pullRequest, PULL_TIME_DELAY_MILLS_WHEN_EXCEPTION);
        }
    }

PullAPIWrapper#pullKernelImpl

    public PullResult pullKernelImpl(final MessageQueue mq, final String subExpression, final String expressionType, final long subVersion, final long offset, final int maxNums, final int sysFlag, final long commitOffset, final long brokerSuspendMaxTimeMillis, final long timeoutMillis, final CommunicationMode communicationMode, final PullCallback pullCallback) throws MQClientException, RemotingException, MQBrokerException, InterruptedException {

        // 根据MessageQueue的BrokerName及上次从Broker拉取返回的建议BrokerId，决定Broker的地址
        FindBrokerResult findBrokerResult = this.mQClientFactory.findBrokerAddressInSubscribe(mq.getBrokerName(), this.recalculatePullFromWhichNode(mq), false);
        // 没有就从NameServer更新后再获取
        if (null == findBrokerResult) {
            // ...
        }
        if (findBrokerResult != null) {
            // 检查版本...
            int sysFlagInner = sysFlag;
            // 如果是向Slave拉取消息
            if (findBrokerResult.isSlave()) {
                // 清除同步消费偏移的标记
                sysFlagInner = PullSysFlag.clearCommitOffsetFlag(sysFlagInner);
            }
            PullMessageRequestHeader requestHeader = new PullMessageRequestHeader();
            // ConsumerGroup
            requestHeader.setConsumerGroup(this.consumerGroup);
            // 拉取的Topic
            requestHeader.setTopic(mq.getTopic());
            // 拉取的MessageQueueId
            requestHeader.setQueueId(mq.getQueueId());
            // 本次拉取的起始偏移
            requestHeader.setQueueOffset(offset);
            // 一次拉取的最大消息数量
            requestHeader.setMaxMsgNums(maxNums);
            requestHeader.setSysFlag(sysFlagInner);
            // 向Broker同步Consumer对该MessageQueue的消费进度
            requestHeader.setCommitOffset(commitOffset);
            requestHeader.setSuspendTimeoutMillis(brokerSuspendMaxTimeMillis);
            requestHeader.setSubscription(subExpression);
            requestHeader.setSubVersion(subVersion);
            requestHeader.setExpressionType(expressionType);
            String brokerAddr = findBrokerResult.getBrokerAddr();
            // 有过滤标记
            if (PullSysFlag.hasClassFilterFlag(sysFlagInner)) {
                // 随机选择一个FilterServer的地址，之后就从FilterServer拉取消息
                brokerAddr = computPullFromWhichFilterServer(mq.getTopic(), brokerAddr);
            }
            // 发送PULL_MESSAGE请求，从Broker拉取消息
            PullResult pullResult = this.mQClientFactory.getMQClientAPIImpl().pullMessage(brokerAddr, requestHeader, timeoutMillis, communicationMode, pullCallback);
            return pullResult;
        }
        // 抛异常
    }

    public PullResult processPullResult(final MessageQueue mq, final PullResult pullResult, final SubscriptionData subscriptionData) {
        PullResultExt pullResultExt = (PullResultExt) pullResult;
        // 根据Broker的返回建议，更新下次建议拉取的Broker
        this.updatePullFromWhichNode(mq, pullResultExt.getSuggestWhichBrokerId());
        // 拉取到了消息
        if (PullStatus.FOUND == pullResult.getPullStatus()) {
            ByteBuffer byteBuffer = ByteBuffer.wrap(pullResultExt.getMessageBinary());
            // 解码拉取的消息列表
            List<MessageExt> msgList = MessageDecoder.decodes(byteBuffer);
            List<MessageExt> msgListFilterAgain = msgList;
            if (!subscriptionData.getTagsSet().isEmpty() && !subscriptionData.isClassFilterMode()) {
                msgListFilterAgain = new ArrayList<MessageExt>(msgList.size());
                for (MessageExt msg : msgList) {
                    if (msg.getTags() != null) {
                        // Broker只是根据Tag的哈希值过滤，因为可能存在哈希冲突，Consumer需要根据字符串再次过滤
                        if (subscriptionData.getTagsSet().contains(msg.getTags())) {
                            msgListFilterAgain.add(msg);
                        }
                    }
                }
            }
            // 执行钩子函数，用于过滤消息
            if (this.hasHook()) {
                // ...
            }
            for (MessageExt msg : msgListFilterAgain) {
                // 一些属性处理
            }
            pullResultExt.setMsgFoundList(msgListFilterAgain);
        }
        pullResultExt.setMessageBinary(null);
        return pullResult;
    }

    public boolean putMessage(final List<MessageExt> msgs) {
        // 是否分发给消费者消费
        boolean dispatchToConsume = false;
        try {
            // 获取写锁
            this.lockTreeMap.writeLock().lockInterruptibly();
            try {
                int validMsgCnt = 0;
                for (MessageExt msg : msgs) {
                    // 将消息加入TreeMap，等待被消费
                    // 注意这里TreeMap会自动排序，键为消息在队列的偏移（索引）
                    MessageExt old = msgTreeMap.put(msg.getQueueOffset(), msg);
                    // 消息未重复
                    if (null == old) {
                        validMsgCnt++;
                        // 更新队列最大偏移
                        this.queueOffsetMax = msg.getQueueOffset();
                        msgSize.addAndGet(msg.getBody().length);
                    }
                }
                msgCount.addAndGet(validMsgCnt);
                // Map有消息被消费 && 当前此ProcessQueue没被消费
                if (!msgTreeMap.isEmpty() && !this.consuming) {
                    // 分发给消费者消费
                    dispatchToConsume = true;
                    // 标记为正在被消费
                    this.consuming = true;
                }
                if (!msgs.isEmpty()) {
                    // ...
                }
            } finally {
                this.lockTreeMap.writeLock().unlock();
            }
        } catch (InterruptedException e) {
            // ...
        }
        return dispatchToConsume;
    }

       @Override
        public void run() {
            // ProcessQueue被标记为废弃就不再消费
            if (this.processQueue.isDropped()) {
                return;
            }
            // 获取该MessageQueue的锁对象
            final Object objLock = messageQueueLock.fetchLockObject(this.messageQueue);
            // 获取同步锁
            synchronized (objLock) {
                // 广播模式 || 集群消费模式，ProcessQueue被锁定并且没失效
                if (MessageModel.BROADCASTING.equals(ConsumeMessageOrderlyService.this.defaultMQPushConsumerImpl.messageModel()) || (this.processQueue.isLocked() && !this.processQueue.isLockExpired())) {
                    final long beginTime = System.currentTimeMillis();
                    //
                    for (boolean continueConsume = true; continueConsume; ) {
                        // 每次消费前检查，ProcessQueue被标记为废弃就不再消费
                        if (this.processQueue.isDropped()) {
                            break;
                        }
                        // 集群消费模式 && ProcessQueue没被锁定
                        if (MessageModel.CLUSTERING.equals(ConsumeMessageOrderlyService.this.defaultMQPushConsumerImpl.messageModel()) && !this.processQueue.isLocked()) {
                            // 稍后向Broker发送LOCK_BATCH_MQ请求，获取该MessageQueue的分布式锁，然后重新提交ConsumeRequest请求
                            ConsumeMessageOrderlyService.this.tryLockLaterAndReconsume(this.messageQueue, this.processQueue, 10);
                            break;
                        }
                        // 集群消费模式 && ProcessQueue锁失效
                        if (MessageModel.CLUSTERING.equals(ConsumeMessageOrderlyService.this.defaultMQPushConsumerImpl.messageModel()) && this.processQueue.isLockExpired()) {
                            // 稍后向Broker发送LOCK_BATCH_MQ请求，获取该MessageQueue的分布式锁，然后重新提交ConsumeRequest请求
                            ConsumeMessageOrderlyService.this.tryLockLaterAndReconsume(this.messageQueue, this.processQueue, 10);
                            break;
                        }
                        // 提交分布式锁请求耗时
                        long interval = System.currentTimeMillis() - beginTime;
                        // 时间过长，超过60秒
                        if (interval > MAX_TIME_CONSUME_CONTINUOUSLY) {
                            // 重新提交ConsumeRequest请求
                            ConsumeMessageOrderlyService.this.submitConsumeRequestLater(processQueue, messageQueue, 10);
                            break;
                        }
                        // 本次消费的消息数量
                        final int consumeBatchSize = ConsumeMessageOrderlyService.this.defaultMQPushConsumer.getConsumeMessageBatchMaxSize();
                        // 从处理队列拿出消息，顺序消费会把待消费的消息取出，并放入临时的TreeMap中，按队列偏移排序，所以是有序消费
                        // 另外，takeMessags拼写错误
                        List<MessageExt> msgs = this.processQueue.takeMessags(consumeBatchSize);
                        defaultMQPushConsumerImpl.resetRetryAndNamespace(msgs, defaultMQPushConsumer.getConsumerGroup());
                        if (!msgs.isEmpty()) {
                            final ConsumeOrderlyContext context = new ConsumeOrderlyContext(this.messageQueue);
                            ConsumeOrderlyStatus status = null;
                            ConsumeMessageContext consumeMessageContext = null;
                            // 执行前置Hook
                            if (ConsumeMessageOrderlyService.this.defaultMQPushConsumerImpl.hasHook()) {
                                // ...
                            }
                            long beginTimestamp = System.currentTimeMillis();
                            ConsumeReturnType returnType = ConsumeReturnType.SUCCESS;
                            boolean hasException = false;
                            try {
                                // 先获取ProcessQueue锁
                                this.processQueue.getLockConsume().lock();
                                // 再次检查
                                if (this.processQueue.isDropped()) {
                                    break;
                                }
                                // 再使用监听器消费消息，返回消费状态，注意消费都是批量的
                                status = messageListener.consumeMessage(Collections.unmodifiableList(msgs), context);
                            } catch (Throwable e) {
                                // ...
                            } finally {
                                // 最后释放ProcessQueue锁
                                this.processQueue.getLockConsume().unlock();
                            }
                            // ...
                            long consumeRT = System.currentTimeMillis() - beginTimestamp;
                            if (null == status) {
                                // ...
                            } else if (consumeRT >= defaultMQPushConsumer.getConsumeTimeout() * 60 * 1000) {
                                // ...
                            } else if (ConsumeOrderlyStatus.SUSPEND_CURRENT_QUEUE_A_MOMENT == status) {
                                // ...
                            }
                            // 消费成功
                            else if (ConsumeOrderlyStatus.SUCCESS == status) {
                                returnType = ConsumeReturnType.SUCCESS;
                            }
                            // ...
                            // 执行后置Hook
                            if (ConsumeMessageOrderlyService.this.defaultMQPushConsumerImpl.hasHook()) {
                                // ...
                            }
                            ConsumeMessageOrderlyService.this.getConsumerStatsManager().incConsumeRT(ConsumeMessageOrderlyService.this.consumerGroup, messageQueue.getTopic(), consumeRT);
                            // 根据消费结果决定是否继续消费
                            continueConsume = ConsumeMessageOrderlyService.this.processConsumeResult(msgs, status, context, this);
                        } else {
                            continueConsume = false;
                        }
                    }
                }
                // 集群消费模式，队列没被锁定或者锁失效
                else {
                    if (this.processQueue.isDropped()) {
                        return;
                    }
                    // 稍后向Broker发送LOCK_BATCH_MQ请求，获取该MessageQueue的分布式锁，然后重新提交ConsumeRequest请求
                    // 集群模式下，队列必须被锁定才能被消费
                    ConsumeMessageOrderlyService.this.tryLockLaterAndReconsume(this.messageQueue, this.processQueue, 100);
                }
            }

ConsumeMessageOrderlyService#processConsumeResult

    public boolean processConsumeResult(final List<MessageExt> msgs, final ConsumeOrderlyStatus status, final ConsumeOrderlyContext context, final ConsumeRequest consumeRequest) {
        boolean continueConsume = true;
        long commitOffset = -1L;
        // 默认，自动提交消费进度
        if (context.isAutoCommit()) {
            switch (status) {
                case COMMIT:
                case ROLLBACK:
                case SUCCESS:
                    // 返回本批消息的队列消费偏移
                    commitOffset = consumeRequest.getProcessQueue().commit();
                    // 统计
                    this.getConsumerStatsManager().incConsumeOKTPS(consumerGroup, consumeRequest.getMessageQueue().getTopic(), msgs.size());
                    break;
                case SUSPEND_CURRENT_QUEUE_A_MOMENT:
                    this.getConsumerStatsManager().incConsumeFailedTPS(consumerGroup, consumeRequest.getMessageQueue().getTopic(), msgs.size());
                    // 消息未超过最大重消费次数，或者超出了并且把消息发回Broker的死信队列失败
                    if (checkReconsumeTimes(msgs)) {
                        // 设置可重新消费，将本批消息放回ProcessQueue
                        consumeRequest.getProcessQueue().makeMessageToCosumeAgain(msgs);
                        // 提交延迟消费请求，延迟消费该队列
                        this.submitConsumeRequestLater(consumeRequest.getProcessQueue(), consumeRequest.getMessageQueue(), context.getSuspendCurrentQueueTimeMillis());
                        // 不能继续消费，顺序消息，不能跳过
                        continueConsume = false;
                    }
                    // 消息超过最大重消费次数，并且发送回死信队列成功
                    else {
                        // 返回本批消息的队列消费偏移
                        commitOffset = consumeRequest.getProcessQueue().commit();
                    }
                    break;
                default:
                    break;
            }
        } else {
            // ...
        }
        // 队列的消费偏移 >= 0 && ProcessQueue未被标志为废弃
        if (commitOffset >= 0 && !consumeRequest.getProcessQueue().isDropped()) {
            // 更新本地缓存中的该MessageQueue的消费进度
            this.defaultMQPushConsumerImpl.getOffsetStore().updateOffset(consumeRequest.getMessageQueue(), commitOffset, false);
        }
        return continueConsume;
    }

ProcessQueue#commit

    public long commit() {
        try {
            // 获取写锁
            this.lockTreeMap.writeLock().lockInterruptibly();
            try {
                // 从本批消费的消息中获取最大消息队列偏移
                Long offset = this.consumingMsgOrderlyTreeMap.lastKey();
                msgCount.addAndGet(0 - this.consumingMsgOrderlyTreeMap.size());
                for (MessageExt msg : this.consumingMsgOrderlyTreeMap.values()) {
                    msgSize.addAndGet(0 - msg.getBody().length);
                }
                // 消费完就清除
                this.consumingMsgOrderlyTreeMap.clear();
                if (offset != null) {
                    // 返回消费进度
                    return offset + 1;
                }
            } finally {
                this.lockTreeMap.writeLock().unlock();
            }
        } catch (InterruptedException e) {
            // ...
        }
        return -1;
    }

ConsumeMessageConcurrentlyService#submitConsumeRequest

    @Override
    public void submitConsumeRequest(final List<MessageExt> msgs, final ProcessQueue processQueue, final MessageQueue messageQueue, final boolean dispatchToConsume) {
        final int consumeBatchSize = this.defaultMQPushConsumer.getConsumeMessageBatchMaxSize();
        // 提交消费的数量小于等于批量消费的数量
        if (msgs.size() <= consumeBatchSize) {
            ConsumeRequest consumeRequest = new ConsumeRequest(msgs, processQueue, messageQueue);
            try {
                // 提交给线程池，所以是多线程并发消费
                this.consumeExecutor.submit(consumeRequest);
            } catch (RejectedExecutionException e) {
                // 被拒绝就稍后再提交
                this.submitConsumeRequestLater(consumeRequest);
            }
        }
        // 大于
        else {
            // 两层循环拆分消息
            for (int total = 0; total < msgs.size(); ) {
                List<MessageExt> msgThis = new ArrayList<MessageExt>(consumeBatchSize);
                for (int i = 0; i < consumeBatchSize; i++, total++) {
                    if (total < msgs.size()) {
                        msgThis.add(msgs.get(total));
                    } else {
                        break;
                    }
                }
                ConsumeRequest consumeRequest = new ConsumeRequest(msgThis, processQueue, messageQueue);
                try {
                    // 拆分后分批提交
                    this.consumeExecutor.submit(consumeRequest);
                } catch (RejectedExecutionException e) {
                    for (; total < msgs.size(); total++) {
                        msgThis.add(msgs.get(total));
                    }
                    this.submitConsumeRequestLater(consumeRequest);
                }
            }
        }
    }

ConsumeMessageConcurrentlyService#processConsumeResult

    public void processConsumeResult(final ConsumeConcurrentlyStatus status, final ConsumeConcurrentlyContext context, final ConsumeRequest consumeRequest) {
        // 初始值为Integer.MAX_VALUE
        int ackIndex = context.getAckIndex();
        if (consumeRequest.getMsgs().isEmpty())
            return;
        // 消费状态
        switch (status) {
            // 成功
            case CONSUME_SUCCESS:
                if (ackIndex >= consumeRequest.getMsgs().size()) {
                    // 本批消息中最后一个消息的索引
                    ackIndex = consumeRequest.getMsgs().size() - 1;
                }
                // 本批消息成功消费的数量
                int ok = ackIndex + 1;
                int failed = consumeRequest.getMsgs().size() - ok;
                this.getConsumerStatsManager().incConsumeOKTPS(consumerGroup, consumeRequest.getMessageQueue().getTopic(), ok);
                this.getConsumerStatsManager().incConsumeFailedTPS(consumerGroup, consumeRequest.getMessageQueue().getTopic(), failed);
                break;
            // 稍后再消费
            case RECONSUME_LATER:
                // 这批消息都要发回Broker
                ackIndex = -1;
                this.getConsumerStatsManager().incConsumeFailedTPS(consumerGroup, consumeRequest.getMessageQueue().getTopic(), consumeRequest.getMsgs().size());
                break;
            default:
                break;
        }
        // 消费模式
        switch (this.defaultMQPushConsumer.getMessageModel()) {
            // 广播模式
            case BROADCASTING:
                // 从最后一个消费成功的消息后的消息开始
                for (int i = ackIndex + 1; i < consumeRequest.getMsgs().size(); i++) {
                    MessageExt msg = consumeRequest.getMsgs().get(i);
                    // 消费失败，忽略并丢弃
                }
                break;
            // 集群模式
            case CLUSTERING:
                List<MessageExt> msgBackFailed = new ArrayList<MessageExt>(consumeRequest.getMsgs().size());
                // 遍历消费失败的消息
                for (int i = ackIndex + 1; i < consumeRequest.getMsgs().size(); i++) {
                    MessageExt msg = consumeRequest.getMsgs().get(i);
                    // 直接将消息发回Broker，期待重新发送，这里也会导致乱序
                    boolean result = this.sendMessageBack(msg, context);
                    // 发回失败
                    if (!result) {
                        msg.setReconsumeTimes(msg.getReconsumeTimes() + 1);
                        msgBackFailed.add(msg);
                    }
                }
                if (!msgBackFailed.isEmpty()) {
                    // 移除消费失败的消息
                    consumeRequest.getMsgs().removeAll(msgBackFailed);
                    // 发回Broker失败，则再次提交消费请求，所以跳过了消费失败的消息，不是有序的
                    this.submitConsumeRequestLater(msgBackFailed, consumeRequest.getProcessQueue(), consumeRequest.getMessageQueue());
                }
                break;
            default:
                break;
        }
        // 消费完毕，将消费成功的消息从ProcessQueue移除，返回下条待消费的消息的偏移
        long offset = consumeRequest.getProcessQueue().removeMessage(consumeRequest.getMsgs());
        if (offset >= 0 && !consumeRequest.getProcessQueue().isDropped()) {
            // 更新消费偏移
            this.defaultMQPushConsumerImpl.getOffsetStore().updateOffset(consumeRequest.getMessageQueue(), offset, true);
        }
    }

ProcessQueue#removeMessage

    public long removeMessage(final List<MessageExt> msgs) {
        long result = -1;
        final long now = System.currentTimeMillis();
        try {
            // 获取写锁
            this.lockTreeMap.writeLock().lockInterruptibly();
            this.lastConsumeTimestamp = now;
            try {
                if (!msgTreeMap.isEmpty()) {
                    // 消息的最大偏移 + 1
                    result = this.queueOffsetMax + 1;
                    int removedCnt = 0;
                    // 遍历消息
                    for (MessageExt msg : msgs) {
                        // 删除消费成功的消息
                        MessageExt prev = msgTreeMap.remove(msg.getQueueOffset());
                        if (prev != null) {
                            removedCnt--;
                            msgSize.addAndGet(0 - msg.getBody().length);
                        }
                    }
                    msgCount.addAndGet(removedCnt);
                    // 删除后还有消息，说明剩下消费失败的消息
                    if (!msgTreeMap.isEmpty()) {
                        // TreeMap，按偏移升序，第一个是最小的偏移，这里就是消费失败的第一条消息的队列偏移
                        // 假设多线程并发消费队列里的消息1，2，3，4，5，可能消费的顺序是1，4，2，5，3
                        // 返回最小偏移可以避免消费未被消费，但是可能会导致重复消费
                        result = msgTreeMap.firstKey();
                    }
                }
            } finally {
                this.lockTreeMap.writeLock().unlock();
            }
        } catch (Throwable t) {
            // ...
        }
        return result;
    }

顺序消费下，一个Queue同时只允许一个Consumer消费，Consumer只有先向Broker获取到Queue的锁才能消费消息，同时消息消费失败时会重试，直到超过最大次数，则将消息发回Broker的死信队列，不会再次消费。

并发消费下，Consumer拉取到消息后回被放入ProcessQueue，有多个线程同时处理队列中的消息，同时消息消费失败时会跳过该消息，并将消息发回Broker，下次继续消费。

另外，顺序消费需要注意消息堆积的问题。

顺序消息也不是绝对的，两条顺序消息MsgA和MsgB，MsgA发到BrokerA的QueueA，之后BrokerA关闭，MsgB发到BrokerB的QueueB，这样就不能保证顺序消费了。

补充一下Pull，Push之类的模式的区别：

Push

  • 优点：实时；

  • 缺点：Push的速率可能会超过Consumer的消费速率；Broker需要维护Consumer的状态，特别是Consumer数量比较多时；

Pull

  • 优点：Consumer可调节消费速度；状态由Consumer维护，每次Pull请求都会携带偏移；方便批量消费消息；

  • 缺点：可能不及时，不断轮询消耗比较大；

Long Polling

  • 优点：基本上结合了Push和Pull的优点

  • 缺点：假如Broker一直有消息，实际上长轮询就等于无间隔的Pull，一直重复请求-响应，网络交互开销较大

长轮询的超时：Consumer和Broker之间网络交互需要时间，本地时钟速率也不同（虽然可以忽略），假如一个Pull请求的超时时间是5秒，在Broker端阻塞了5秒才返回，那么Consumer在收到Broker响应前就会判定超时。
