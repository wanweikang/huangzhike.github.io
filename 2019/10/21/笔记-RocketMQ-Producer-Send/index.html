DefaultMQProducerImpl#sendDefaultImpl

    private SendResult sendDefaultImpl(Message msg, final CommunicationMode communicationMode, final SendCallback sendCallback, final long timeout) throws MQClientException, RemotingException, MQBrokerException, InterruptedException {
        this.makeSureStateOK();
        Validators.checkMessage(msg, this.defaultMQProducer);
        final long invokeID = random.nextLong();
        long beginTimestampFirst = System.currentTimeMillis();
        long beginTimestampPrev = beginTimestampFirst;
        long endTimestamp = beginTimestampFirst;
        // 先从this.topicPublishInfoTable查找Topic的发布路由信息，没有则从NameServer获取，并更新Broker地址信息
        TopicPublishInfo topicPublishInfo = this.tryToFindTopicPublishInfo(msg.getTopic());
        if (topicPublishInfo != null && topicPublishInfo.ok()) {
            boolean callTimeout = false;
            MessageQueue mq = null;
            Exception exception = null;
            SendResult sendResult = null;
            // 最多重试次数，只有同步发送这里才有重试
            // 因为NameServer的Broker检测是有延迟的，为了保证高可用，需要失败重试和规避状态不佳的Broker
            int timesTotal = communicationMode == CommunicationMode.SYNC ? 1 + this.defaultMQProducer.getRetryTimesWhenSendFailed() : 1;
            int times = 0;
            String[] brokersSent = new String[timesTotal];
            // 重试
            for (; times < timesTotal; times++) {
                // 上次使用的BrokerName，第一次为null
                String lastBrokerName = null == mq ? null : mq.getBrokerName();
                // 选择一个MessageQueue，有启用和关闭延时容忍两种配置，默认关闭
                // 如果关闭延时容忍，会选择下一个MessageQueue发送，同时避开上次失败的BrokerName下的MessageQueue
                // 如果开启延时容忍，会选择下一个MessageQueue发送，同时优先上次失败的BrokerName下的MessageQueue，直到该Broker下的队列全部失败，才会尝试其他Broker的可用的队列
                MessageQueue mqSelected = this.selectOneMessageQueue(topicPublishInfo, lastBrokerName);
                if (mqSelected != null) {
                    mq = mqSelected;
                    brokersSent[times] = mq.getBrokerName();
                    try {
                        beginTimestampPrev = System.currentTimeMillis();
                        // ...
                        long costTime = beginTimestampPrev - beginTimestampFirst;
                        // 多次重发的总耗时已经超时
                        if (timeout < costTime) {
                            callTimeout = true;
                            // 不再重试
                            break;
                        }
                        // 发送消息
                        sendResult = this.sendKernelImpl(msg, mq, communicationMode, sendCallback, topicPublishInfo, timeout - costTime);
                        endTimestamp = System.currentTimeMillis();
                        // 更新该BrokerName的发送延迟
                        // MQFaultStrategy有个latencyMax和notAvailableDuration数组，分别代表发送最大延迟和不可用时间，当Broker发送延迟在某个区间时，有个对应的不可用时间
                        this.updateFaultItem(mq.getBrokerName(), endTimestamp - beginTimestampPrev, false);
                        switch (communicationMode) {
                            case ASYNC:
                                return null;
                            case ONEWAY:
                                return null;
                            // 同步发送
                            case SYNC:
                                // 发送结果有问题
                                if (sendResult.getSendStatus() != SendStatus.SEND_OK) {
                                    // 默认关闭
                                    if (this.defaultMQProducer.isRetryAnotherBrokerWhenNotStoreOK()) {
                                        // 下个循环重试
                                        continue;
                                    }
                                }
                                return sendResult;
                            default:
                                break;
                        }
                    } catch (RemotingException e) {
                        // 跳过并重试
                    } catch (MQClientException e) {
                        // 跳过并重试
                    } catch (MQBrokerException e) {
                        // 根据异常跳过并重试
                    } catch (InterruptedException e) {
                        // 抛异常
                    }
                } else {
                    break;
                }
            }
            if (sendResult != null) {
                // 返回结果
                return sendResult;
            }
            // 抛异常
        }
        // 抛异常
    }

DefaultMQProducerImpl#tryToFindTopicPublishInfo

    private TopicPublishInfo tryToFindTopicPublishInfo(final String topic) {
        // 先从本地缓存获取Topic的路由信息
        TopicPublishInfo topicPublishInfo = this.topicPublishInfoTable.get(topic);
        // 本地没有
        if (null == topicPublishInfo || !topicPublishInfo.ok()) {
            // 生成并加入
            this.topicPublishInfoTable.putIfAbsent(topic, new TopicPublishInfo());
            // 从NameServer获取Topic的路由信息并更新，如果没有提前向Broker手动注册对应Topic，这里获取不到路由信息
            this.mQClientFactory.updateTopicRouteInfoFromNameServer(topic);
            topicPublishInfo = this.topicPublishInfoTable.get(topic);
        }
        // 获取路由信息成功
        if (topicPublishInfo.isHaveTopicRouterInfo() || topicPublishInfo.ok()) {
            return topicPublishInfo;
        }
        // 获取不成功
        else {
            /*
             * 通过内置的Topic从NameServer获取并更新Topic路由信息，如果Broker开启了自动创建Topic，则NameServer会有对应的TBW102路由
             * 该Topic的消息会发到对应Broker的默认TBW102主题下，Broker收到消息后才自动创建对应Topic及向NameServer注册路由，就是所谓的自动创建Topic
             * 官方的建议是生产环境关闭自动创建Topic的功能，这种方式，在这里虽然会获取所有的支持自动创建的Broker列表，但是在后面发送时会选择一个Broker发送消息
             * 对应的Broker才会自动创建和注册Topic，结果就是只有一个Broker向NameServer注册了该Topic
             * Producer每隔30秒从NameServer更新路由信息，如果这段时间内没有再发消息，那么别的Broker就不会注册该Topic，之后则从NameServer获取路由发送，这样无法完成负载均衡
             * 假设有两个Broker，每个Broker有4个MessageQueue，则只要连续发超过8条消息，则两个Broker都会创建Topic
             */
            this.mQClientFactory.updateTopicRouteInfoFromNameServer(topic, true, this.defaultMQProducer);
            topicPublishInfo = this.topicPublishInfoTable.get(topic);
            return topicPublishInfo;
        }
    }

MQFaultStrategy#selectOneMessageQueue

    public MessageQueue selectOneMessageQueue(final TopicPublishInfo tpInfo, final String lastBrokerName) {
        // 开启了发送延时容错，默认关闭
        if (this.sendLatencyFaultEnable) {
            try {
                // 线程本地变量，获取上次使用的Queue索引，并自增
                int index = tpInfo.getSendWhichQueue().getAndIncrement();
                // 轮询MessageQueue列表重试
                for (int i = 0; i < tpInfo.getMessageQueueList().size(); i++) {
                    // 获取对应的队列位置并自增
                    int pos = Math.abs(index++) % tpInfo.getMessageQueueList().size();
                    if (pos < 0)
                        pos = 0;
                    // 获取对应的MessageQueue
                    MessageQueue mq = tpInfo.getMessageQueueList().get(pos);
                    // MessageQueue对应的BrokerName可用，不处于不可用时间
                    if (latencyFaultTolerance.isAvailable(mq.getBrokerName())) {
                        // 第一次发送 || 重发，选择上次发送失败对应的Broker
                        if (null == lastBrokerName || mq.getBrokerName().equals(lastBrokerName))
                            return mq;
                    }
                }
                // 没找到最好的Broker，在之前发送失败的Broker中选择一个可用的
                final String notBestBroker = latencyFaultTolerance.pickOneAtLeast();
                int writeQueueNums = tpInfo.getQueueIdByBroker(notBestBroker);
                if (writeQueueNums > 0) {
                    final MessageQueue mq = tpInfo.selectOneMessageQueue();
                    if (notBestBroker != null) {
                        mq.setBrokerName(notBestBroker);
                        mq.setQueueId(tpInfo.getSendWhichQueue().getAndIncrement() % writeQueueNums);
                    }
                    return mq;
                } else {
                    latencyFaultTolerance.remove(notBestBroker);
                }
            } catch (Exception e) {
                // ...
            }
            // 都没有，按顺序选择下一个
            return tpInfo.selectOneMessageQueue();
        }
        // 选择下一个MessageQueue，避开上次失败的Broker的MessageQueue
        // 每个TopicPublishInfo有一个索引，选择前自增，用于轮询该Topic下的MessageQueue列表
        return tpInfo.selectOneMessageQueue(lastBrokerName);
    }

TopicPublishInfo#selectOneMessageQueue

    public MessageQueue selectOneMessageQueue() {
        // ThreadLocal变量，初始为随机数，获取并自增
        int index = this.sendWhichQueue.getAndIncrement();
        // 取余
        int pos = Math.abs(index) % this.messageQueueList.size();
        if (pos < 0)
            pos = 0;
        return this.messageQueueList.get(pos);
    }

TopicPublishInfo#selectOneMessageQueue

    public MessageQueue selectOneMessageQueue(final String lastBrokerName) {
        // 第一次
        if (lastBrokerName == null) {
            // 选择一个MessageQueue
            return selectOneMessageQueue();
        }
        // 上次Broker失败，重试
        else {
            // 获取并自增
            int index = this.sendWhichQueue.getAndIncrement();
            for (int i = 0; i < this.messageQueueList.size(); i++) {
                // 获取对应的队列位置并自增，其实就是轮询MessageQueue列表
                int pos = Math.abs(index++) % this.messageQueueList.size();
                if (pos < 0)
                    pos = 0;
                MessageQueue mq = this.messageQueueList.get(pos);
                // 不能是上次失败的Broker
                if (!mq.getBrokerName().equals(lastBrokerName)) {
                    return mq;
                }
            }
            // 最后还没有，跳过一个MessageQueue
            return selectOneMessageQueue();
        }
    }

DefaultMQProducerImpl#sendKernelImpl

    private SendResult sendKernelImpl(final Message msg, final MessageQueue mq, final CommunicationMode communicationMode, final SendCallback sendCallback, final TopicPublishInfo topicPublishInfo, final long timeout) throws MQClientException, RemotingException, MQBrokerException, InterruptedException {
        long beginStartTime = System.currentTimeMillis();
        // 根据BrokerName从本地缓存获取MasterBroker的地址
        String brokerAddr = this.mQClientFactory.findBrokerAddressInPublish(mq.getBrokerName());
        // 地址为空
        if (null == brokerAddr) {
            // 先从this.topicPublishInfoTable查找Topic的发布路由信息，没有则从NameServer获取，并更新Broker地址信息
            tryToFindTopicPublishInfo(mq.getTopic());
            // 再次获取地址
            brokerAddr = this.mQClientFactory.findBrokerAddressInPublish(mq.getBrokerName());
        }
        SendMessageContext context = null;
        if (brokerAddr != null) {
            // Broker监听两个端口接受消息，其中VIPChannel不接受Consumer消费消息
            // 如果开启了使用VIPChannel发送消息，则使用VIPChannel，默认关闭
            brokerAddr = MixAll.brokerVIPChannel(this.defaultMQProducer.isSendMessageWithVIPChannel(), brokerAddr);
            byte[] prevBody = msg.getBody();
            try {
                // ...
                // 如果是PREPARED事务消息
                final String tranMsg = msg.getProperty(MessageConst.PROPERTY_TRANSACTION_PREPARED);
                if (tranMsg != null && Boolean.parseBoolean(tranMsg)) {
                    // 设置事务消息标记位
                    sysFlag |= MessageSysFlag.TRANSACTION_PREPARED_TYPE;
                }
                if (hasCheckForbiddenHook()) {
                    // 执行钩子
                }
                // 有设置发送消息的钩子Hook，一般用于记录消息轨迹
                if (this.hasSendMessageHook()) {
                    // 执行钩子
                }
                // 生成消息头
                SendMessageRequestHeader requestHeader = new SendMessageRequestHeader();
                requestHeader.setProducerGroup(this.defaultMQProducer.getProducerGroup());
                // 真实Topic
                requestHeader.setTopic(msg.getTopic());
                // 默认Topic，TBW102
                requestHeader.setDefaultTopic(this.defaultMQProducer.getCreateTopicKey());
                requestHeader.setDefaultTopicQueueNums(this.defaultMQProducer.getDefaultTopicQueueNums());
                // 消息发送到的队列Id
                requestHeader.setQueueId(mq.getQueueId());
                requestHeader.setSysFlag(sysFlag);
                requestHeader.setBornTimestamp(System.currentTimeMillis());
                requestHeader.setFlag(msg.getFlag());
                requestHeader.setProperties(MessageDecoder.messageProperties2String(msg.getProperties()));
                // 没被消费过
                requestHeader.setReconsumeTimes(0);
                requestHeader.setUnitMode(this.isUnitMode());
                requestHeader.setBatch(msg instanceof MessageBatch);
                // 仅Consumer使用，如果Topic以%RETRY%开头，把消息重发回Broker，以后再次消费
                if (requestHeader.getTopic().startsWith(MixAll.RETRY_GROUP_TOPIC_PREFIX)) {
                    String reconsumeTimes = MessageAccessor.getReconsumeTime(msg);
                    if (reconsumeTimes != null) {
                        // 请求头设置已经重消费的次数
                        requestHeader.setReconsumeTimes(Integer.valueOf(reconsumeTimes));
                        // 重发的消息要清除消费的次数
                        MessageAccessor.clearProperty(msg, MessageConst.PROPERTY_RECONSUME_TIME);
                    }

                    String maxReconsumeTimes = MessageAccessor.getMaxReconsumeTimes(msg);
                    if (maxReconsumeTimes != null) {
                        // 请求头设置最大重新消费次数
                        requestHeader.setMaxReconsumeTimes(Integer.valueOf(maxReconsumeTimes));
                        // 重发的消息要清除最大重新消费的次数
                        MessageAccessor.clearProperty(msg, MessageConst.PROPERTY_MAX_RECONSUME_TIMES);
                    }
                }
                SendResult sendResult = null;
                switch (communicationMode) {
                    // 异步发送
                    case ASYNC:
                        // ...
                    case ONEWAY:
                        // 同步发送
                    case SYNC:
                        long costTimeSync = System.currentTimeMillis() - beginStartTime;
                        // 发送前检查是否已经超时
                        if (timeout < costTimeSync) {
                            // 抛异常
                        }
                        // 执行发送
                        sendResult = this.mQClientFactory.getMQClientAPIImpl().sendMessage(brokerAddr, mq.getBrokerName(), msg, requestHeader, timeout - costTimeSync, communicationMode, context, this);
                        break;
                    default:
                        assert false;
                        break;
                }
                // 回调Hook
                if (this.hasSendMessageHook()) {
                    // 执行钩子
                }
                return sendResult;
            } catch (RemotingException e) {
                // 执行钩子，抛异常
            } catch (MQBrokerException e) {
                // 执行钩子，抛异常
            } catch (InterruptedException e) {
                // 执行钩子，抛异常
            } finally {
                msg.setBody(prevBody);
                msg.setTopic(NamespaceUtil.withoutNamespace(msg.getTopic(), this.defaultMQProducer.getNamespace()));
            }
        }
        // 抛异常
    }
