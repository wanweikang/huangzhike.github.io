            regFuture.addListener(new ChannelFutureListener() {
                @Override
                public void operationComplete(ChannelFuture future) throws Exception {
                    // ...
                }
            });
            return promise;
        }
    }

AbstractUnsafe#register

    @Override
    public final void register(EventLoop eventLoop, final ChannelPromise promise) {
        // ...
        // 已经注册则失败
        if (isRegistered()) {
            promise.setFailure(new IllegalStateException("registered to an event loop already"));
            return;
        }
        // EventLoop不兼容当前Channel
        if (!isCompatible(eventLoop)) {
            promise.setFailure(new IllegalStateException("incompatible event loop type: " + eventLoop.getClass().getName()));
            return;
        }
        // 注册Channel到EventLoop
        AbstractChannel.this.eventLoop = eventLoop;
        // 当前线程为EventLoop线程直接执行，比如unregister再register
        if (eventLoop.inEventLoop()) {
            register0(promise);
        }
        // 否则提交任务给EventLoop线程执行（异步）
        else {
            try {
                // 由Channel绑定的EventLoop执行，这里会启动EventLoop线程
                eventLoop.execute(new Runnable() {
                    @Override
                    public void run() {
                        register0(promise);
                    }
                });
            } catch (Throwable t) {
                // ...
            }
        }
    }

AbstractUnsafe#register0

    private void register0(ChannelPromise promise) {
        try {
            // 确保Channel没有关闭
            if (!promise.setUncancellable() || !ensureOpen(promise)) {
                return;
            }
            boolean firstRegistration = neverRegistered;
            // JDK的Channel注册到Selector，模板方法
            // AbstractNioChannel#doRegister
            doRegister();
            neverRegistered = false;
            registered = true;
            // 完成ChannelInitializer的回调，执行initChannel，将处理业务逻辑的用户Handler添加到ChannelPipeline
            pipeline.invokeHandlerAddedIfNeeded();
            // 触发Channel异步注册成功回调，开始bind或者connect
            safeSetSuccess(promise);
            // 传播Channel注册事件，head开始
            pipeline.fireChannelRegistered();
            // 服务端ServerSocketChannel接受的客户端Channel此时已被激活
            if (isActive()) {
                if (firstRegistration) {
                    // 首次注册且激活触发Channel激活事件
                    pipeline.fireChannelActive();
                }
                // 已设置autoRead
                else if (config().isAutoRead()) {
                    // 读取数据，模板方法
                    beginRead();
                }
            }
        } catch (Throwable t) {
            // ...
        }
    }

AbstractNioChannel#doRegister

    @Override
    protected void doRegister() throws Exception {
        boolean selected = false;
        for (; ; ) {
            try {
                // 将Channel注册到EventLoop的Selector上，0表示注册时不关心任何事件，attachment为Netty的Channel对象本身
                selectionKey = javaChannel().register(eventLoop().unwrappedSelector(), 0, this);
                return;
            } catch (CancelledKeyException e) {
                if (!selected) {
                    // 重新selectNow，清除因取消操作而缓存的选择键
                    eventLoop().selectNow();
                    selected = true;
                } else {
                    throw e;
                }
            }
        }
    }

DefaultChannelPipeline#invokeHandlerAddedIfNeeded

    /**
     * 两种情况（但是只执行一次）：
     * AbstractUnsafe#register进入，在DefaultChannelPipeline#fireChannelRegistered前被调用，在异步注册完成将所有Handler全部加入
     * DefaultChannelPipeline#fireChannelRegistered调用HeadContext#channelRegistered方法
     */
    final void invokeHandlerAddedIfNeeded() {
        assert channel.eventLoop().inEventLoop();
        // 只执行一次
        if (firstRegistration) {
            firstRegistration = false;
            // 可能是异步操作，不过基本不是
            callHandlerAddedForAllHandlers();
        }
    }

DefaultChannelPipeline#callHandlerAddedForAllHandlers

    private void callHandlerAddedForAllHandlers() {
        final PendingHandlerCallback pendingHandlerCallbackHead;
        synchronized (this) {
            // Pipeline，不是Channel
            assert !registered;
            // Pipeline已完成注册
            registered = true;
            pendingHandlerCallbackHead = this.pendingHandlerCallbackHead;
            // 用完就删
            this.pendingHandlerCallbackHead = null;
        }

        // 不在synchronized块内
        PendingHandlerCallback task = pendingHandlerCallbackHead;
        while (task != null) {
            // 遍历链表依次执行
            task.execute();
            task = task.next;
        }
    }

    private ChannelFuture doResolveAndConnect0(final Channel channel, SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise promise) {
        try {
            // 获取绑定的EventLoop
            final EventLoop eventLoop = channel.eventLoop();
            // 地址解析
            final AddressResolver<SocketAddress> resolver = this.resolver.getResolver(eventLoop);

            if (!resolver.isSupported(remoteAddress) || resolver.isResolved(remoteAddress)) {
                // 向Channel注册的EventLoop提交异步连接任务
                doConnect(remoteAddress, localAddress, promise);
                return promise;
            }

            final Future<SocketAddress> resolveFuture = resolver.resolve(remoteAddress);

            if (resolveFuture.isDone()) {
                final Throwable resolveFailureCause = resolveFuture.cause();
                if (resolveFailureCause != null) {
                    // ...
                } else {
                    doConnect(resolveFuture.getNow(), localAddress, promise);
                }
                return promise;
            }
            // 回调
            resolveFuture.addListener(new FutureListener<SocketAddress>() {
                @Override
                public void operationComplete(Future<SocketAddress> future) throws Exception {
                    if (future.cause() != null) {
                        // ...
                    } else {
                        doConnect(future.getNow(), localAddress, promise);
                    }
                }
            });
        } catch (Throwable cause) {
            // ...
        }
        return promise;
    }

Bootstrap#doConnect

    private static void doConnect(final SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise connectPromise) {

        final Channel channel = connectPromise.channel();
        // 由Channel绑定的EventLoop执行
        channel.eventLoop().execute(new Runnable() {
            @Override
            public void run() {
                if (localAddress == null) {
                    // AbstractChannel#connect
                    // Pipeline#connect
                    // Tail#connect(AbstractChannelHandlerContext#connect)
                    // -->
                    // Head#connect(AbstractNioUnsafe#connect)
                    channel.connect(remoteAddress, connectPromise);
                } else {
                    channel.connect(remoteAddress, localAddress, connectPromise);
                }
                connectPromise.addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
            }
        });
    }

AbstractNioUnsafe#connect

        @Override
        public final void connect(final SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise promise) {
            // Channel已被关闭
            if (!promise.setUncancellable() || !ensureOpen(promise)) {
                return;
            }
            try {
                // 已经在连接
                if (connectPromise != null) {
                    // ...
                }
                boolean wasActive = isActive();
                // 模板方法
                // NioSocketChannel#doConnect
                if (doConnect(remoteAddress, localAddress)) {
                    // 连接已完成，触发Channel的Active事件
                    fulfillConnectPromise(promise, wasActive);
                }
                // 连接可能未完成
                else {
                    connectPromise = promise;
                    requestedRemoteAddress = remoteAddress;
                    // Netty的连接超时
                    int connectTimeoutMillis = config().getConnectTimeoutMillis();
                    if (connectTimeoutMillis > 0) {
                        // 向EventLoop提交一个调度任务，如果超时时间已到还没被取消或者完成，则设置异步结果失败然后关闭连接
                        connectTimeoutFuture = eventLoop().schedule(new Runnable() {
                            @Override
                            public void run() {
                                // ...
                            }
                        }, connectTimeoutMillis, TimeUnit.MILLISECONDS);
                    }
                    promise.addListener(new ChannelFutureListener() {
                        @Override
                        public void operationComplete(ChannelFuture future) throws Exception {
                            // 如果连接操作取消，则取消连接超时检测任务
                            if (future.isCancelled()) {
                                // ...
                            }
                        }
                    });
                }
            } catch (Throwable t) {
                // ...
            }
        }

NioSocketChannel#doConnect

    @Override
    protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
        if (localAddress != null) {
            doBind0(localAddress);
        }
        boolean success = false;
        try {
            // JDK中的Channel在非阻塞模式下调用connect方法时，会立即返回
            boolean connected = SocketUtils.connect(javaChannel(), remoteAddress);
            // 连接还没成功
            if (!connected) {
                // 关心OP_CONNECT事件，EventLoop负责轮询，连接就绪时会调用AbstractNioUnsafe#finishConnect
                selectionKey().interestOps(SelectionKey.OP_CONNECT);
            }
            success = true;
            // 返回是否连接成功
            return connected;
        } finally {
            if (!success) {
                doClose();
            }
        }
    }

AbstractNioUnsafe#finishConnect

        @Override
        public final void finishConnect() {
            // 只由EventLoop处理就绪SelectionKey的OP_CONNECT事件时调用，完成连接，连接被取消或超时不会调用
            assert eventLoop().inEventLoop();
            try {
                // Channel是否已经连接
                boolean wasActive = isActive();
                // 模板方法
                doFinishConnect();
                // 首次Active触发Active事件
                fulfillConnectPromise(connectPromise, wasActive);
            } catch (Throwable t) {
                fulfillConnectPromise(connectPromise, annotateConnectException(t, requestedRemoteAddress));
            } finally {
                if (connectTimeoutFuture != null) {
                    // 连接完成，取消超时检测任务
                    connectTimeoutFuture.cancel(false);
                }
                connectPromise = null;
            }
        }

AbstractNioUnsafe#fulfillConnectPromise

        private void fulfillConnectPromise(ChannelPromise promise, boolean wasActive) {
            if (promise == null) {
                return;
            }
            // Channel已连接
            boolean active = isActive();
            // false表示取消操作
            boolean promiseSet = promise.trySuccess();
            // 调用前没激活 && 现在已激活
            if (!wasActive && active) {
                // 触发Active事件
                pipeline().fireChannelActive();
            }
            if (!promiseSet) {
                // 操作已被取消，关闭Channel
                close(voidPromise());
            }
        }

AbstractNioChannel#doBeginRead

    @Override
    protected void doBeginRead() throws Exception {
        final SelectionKey selectionKey = this.selectionKey;
        // 选择键被取消
        if (!selectionKey.isValid()) {
            return;
        }
        // 设置NIO底层读事件正在进行
        readPending = true;
        final int interestOps = selectionKey.interestOps();
        // 没关心则要关心Read事件
        if ((interestOps & readInterestOp) == 0) {
            selectionKey.interestOps(interestOps | readInterestOp);
        }
    }

NioByteUnsafe#read

        @Override
        public final void read() {
            final ChannelConfig config = config();
            if (shouldBreakReadReady(config)) {
                clearReadPending();
                return;
            }
            final ChannelPipeline pipeline = pipeline();
            final ByteBufAllocator allocator = config.getAllocator();
            final RecvByteBufAllocator.Handle allocHandle = recvBufAllocHandle();
            allocHandle.reset(config);
            ByteBuf byteBuf = null;
            boolean close = false;
            try {
                do {
                    // 分配一个ByteBuf缓冲
                    byteBuf = allocHandle.allocate(allocator);
                    // 模板方法，读取数据到缓冲
                    allocHandle.lastBytesRead(doReadBytes(byteBuf));
                    // 没有数据可读
                    if (allocHandle.lastBytesRead() <= 0) {
                        byteBuf.release();
                        byteBuf = null;
                        // 读取数据量为负，说明对端已关闭
                        close = allocHandle.lastBytesRead() < 0;
                        if (close) {
                            readPending = false;
                        }
                        break;
                    }
                    allocHandle.incMessagesRead(1);
                    // 读完成
                    readPending = false;
                    // 触发ChannelRead事件，用户处理
                    pipeline.fireChannelRead(byteBuf);
                    byteBuf = null;
                } while (allocHandle.continueReading());

                allocHandle.readComplete();
                // 触发ReadComplete事件
                // 在HeadContext#channelReadComplete，如果开启autoRead则会继续调用AbstractNioChannel#doBeginRead
                pipeline.fireChannelReadComplete();
                if (close) {
                    closeOnRead(pipeline);
                }
            } catch (Throwable t) {
                handleReadException(pipeline, byteBuf, t, close, allocHandle);
            } finally {
                // 读事件没在进行 && 没有配置autoRead
                if (!readPending && !config.isAutoRead()) {
                    // 不再关注读事件
                    removeReadOp();
                }
            }
        }

AbstractChannelHandlerContext#write

    private void write(Object msg, boolean flush, ChannelPromise promise) {
        // ...
        try {
            if (isNotValidPromise(promise, true)) {
                // ...
            }
        } catch (RuntimeException e) {
            // ...
        }
        final AbstractChannelHandlerContext next = findContextOutbound(flush ? (MASK_WRITE | MASK_FLUSH) : MASK_WRITE);
        final Object m = pipeline.touch(msg, next);
        EventExecutor executor = next.executor();
        if (executor.inEventLoop()) {
            if (flush) {
                next.invokeWriteAndFlush(m, promise);
            } else {
                next.invokeWrite(m, promise);
            }
        }
        // 不在EventLoop线程执行的就提交执行
        else {
            final AbstractWriteTask task;
            if (flush) {
                task = WriteAndFlushTask.newInstance(next, m, promise);
            } else {
                task = WriteTask.newInstance(next, m, promise);
            }
            if (!safeExecute(executor, task, promise, m)) {
                task.cancel();
            }
        }
    }

AbstractNioByteChannel#doWrite

    @Override
    protected void doWrite(ChannelOutboundBuffer in) throws Exception {
        // 自旋写入次数
        int writeSpinCount = config().getWriteSpinCount();
        // 自旋
        do {
            Object msg = in.current();
            // 数据已全部写完
            if (msg == null) {
                // 清除OP_WRITE事件
                clearOpWrite();
                return;
            }
            // 返回该次是否写入
            writeSpinCount -= doWriteInternal(in, msg);
        } while (writeSpinCount > 0);
        // 是否写完成
        incompleteWrite(writeSpinCount < 0);

AbstractNioByteChannel#doWriteInternal

    private int doWriteInternal(ChannelOutboundBuffer in, Object msg) throws Exception {
        if (msg instanceof ByteBuf) {
            ByteBuf buf = (ByteBuf) msg;
            // ByteBuf不可读，此时数据已写完
            if (!buf.isReadable()) {
                // 完成时，清理缓冲区
                in.remove();
                return 0;
            }
            // 模板方法，返回已写入的数据量，在非阻塞模式下写操作可能返回0表示未写入数据
            final int localFlushedAmount = doWriteBytes(buf);
            // 成功写入了数据
            if (localFlushedAmount > 0) {
                in.progress(localFlushedAmount);
                if (!buf.isReadable()) {
                    in.remove();
                }
                return 1;
            }
        }
        // FileRegion是Netty对NIO底层的FileChannel的封装，负责将File中的数据写入到WritableChannel中
        else if (msg instanceof FileRegion) {
            // ...
        } else {
            // 其他类型不支持
            throw new Error();
        }
        // 写入不成功，返回Integer.MAX_VALUE
        return WRITE_STATUS_SNDBUF_FULL;
    }

AbstractNioByteChannel#incompleteWrite

    protected final void incompleteWrite(boolean setOpWrite) {
        // 写入数据不成功
        if (setOpWrite) {
            // 设置继续关心OP_WRITE事件
            setOpWrite();
        }
        // 写操作次数达到writeSpinCount但没写完
        else {
            // 不再关心写事件
            clearOpWrite();
            // 向EventLoop提交一个新的flush任务，此时可以响应其他请求，不会占用全部资源使其他请求得不到响应
            eventLoop().execute(flushTask);
        }
    }
