DefaultChannelPipeline

public class DefaultChannelPipeline implements ChannelPipeline {
    /**
     * 双向链表头
     */
    final AbstractChannelHandlerContext head;

    /**
     * 双向链表尾
     */
    final AbstractChannelHandlerContext tail;

    /**
     * 对应Channel
     */
    private final Channel channel;

    /**
     * ChannelHandler添加任务队列链表头
     */
    private PendingHandlerCallback pendingHandlerCallbackHead;

    protected DefaultChannelPipeline(Channel channel) {
        this.channel = ObjectUtil.checkNotNull(channel, "channel");
        succeededFuture = new SucceededChannelFuture(channel, null);
        voidPromise = new VoidChannelPromise(channel, true);
        tail = new TailContext(this);
        head = new HeadContext(this);
        head.next = tail;
        tail.prev = head;
    }
}

DefaultChannelPipeline#addLast

    @Override
    public final ChannelPipeline addLast(EventExecutorGroup group, String name, ChannelHandler handler) {
        final AbstractChannelHandlerContext newCtx;
        synchronized (this) {
            // 检查Handler是否重复添加
            checkMultiplicity(handler);
            // 新建一个Context
            newCtx = newContext(group, filterName(name, handler), handler);
            // 双向链表插入操作
            addLast0(newCtx);
            // 比如添加ChannelInitializer时
            if (!registered) {
                // Pipeline还没有完成注册，现在不能执行Handler，设置Handler状态为ADD_PENDING
                newCtx.setAddPending();
                // 新增一个任务用于注册后触发handlerAdded事件
                callHandlerCallbackLater(newCtx, true);
                return this;
            }

            EventExecutor executor = newCtx.executor();
            // 当前线程不是EventLoop线程，那么提交给EventLoop线程完成
            if (!executor.inEventLoop()) {
                callHandlerAddedInEventLoop(newCtx, executor);
                return this;
            }
        }
        // 比如ChannelInitializer添加的用户Handler
        // 当前线程为EventLoop线程且已注册则直接触发HandlerAdd事件
        callHandlerAdded0(newCtx);
        return this;
    }

DefaultChannelPipeline#newContext

    private AbstractChannelHandlerContext newContext(EventExecutorGroup group, String name, ChannelHandler handler) {
        // 返回一个默认Context，传入EventExecutor用于执行Handler中事件处理代码，group为null返回null
        return new DefaultChannelHandlerContext(this, childExecutor(group), name, handler);
    }

DefaultChannelPipeline#childExecutor

    private EventExecutor childExecutor(EventExecutorGroup group) {
        if (group == null) {
            return null;
        }
        Boolean pinEventExecutor = channel.config().getOption(ChannelOption.SINGLE_EVENTEXECUTOR_PER_GROUP);
        // 如果Channel参数配置非同一个线程处理
        if (pinEventExecutor != null && !pinEventExecutor) {
            return group.next();
        }
        Map<EventExecutorGroup, EventExecutor> childExecutors = this.childExecutors;
        if (childExecutors == null) {
            childExecutors = this.childExecutors = new IdentityHashMap<EventExecutorGroup, EventExecutor>(4);
        }
        // 保证Pipeline中的事件为同一个EventExecutor处理
        EventExecutor childExecutor = childExecutors.get(group);
        if (childExecutor == null) {
            childExecutor = group.next();
            childExecutors.put(group, childExecutor);
        }
        return childExecutor;
    }

DefaultChannelPipeline#callHandlerCallbackLater

    private void callHandlerCallbackLater(AbstractChannelHandlerContext ctx, boolean added) {
        // 必须没注册
        assert !registered;
        // added为true表示HandlerAdd任务，false表示HandlerRemove任务
        PendingHandlerCallback task = added ? new PendingHandlerAddedTask(ctx) : new PendingHandlerRemovedTask(ctx);
        PendingHandlerCallback pending = pendingHandlerCallbackHead;
        if (pending == null) {
            // 链表头部
            pendingHandlerCallbackHead = task;
        } else {
            // 插入到链表尾部
            while (pending.next != null) {
                pending = pending.next;
            }
            pending.next = task;
        }
    }

ChannelInitializer#handlerAdded

    @Override
    public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
        // Channel已注册
        if (ctx.channel().isRegistered()) {
            // 避免并发操作，操作前加个标记，完成后移除
            if (initChannel(ctx)) {
                removeState(ctx);
            }
        }
    }

ChannelInitializer#handlerAdded

    @SuppressWarnings("unchecked")
    private boolean initChannel(ChannelHandlerContext ctx) throws Exception {
        if (initMap.add(ctx)) {
            try {
                // 将用户自定义Handler加入Pipeline
                initChannel((C) ctx.channel());
            } catch (Throwable cause) {
                exceptionCaught(ctx, cause);
            } finally {
                ChannelPipeline pipeline = ctx.pipeline();
                if (pipeline.context(this) != null) {
                    // 将自己从Pipeline移除
                    pipeline.remove(this);
                }
            }
            return true;
        }
        return false;
    }
