NioEventLoop

public final class NioEventLoop extends SingleThreadEventLoop {

    /**
     * SelectorProvider
     */
    private final SelectorProvider provider;

    /**
     * 唤醒标记，select()方法会阻塞
     */
    private final AtomicBoolean wakenUp = new AtomicBoolean();

    /**
     * 选择策略
     */
    private final SelectStrategy selectStrategy;

    /**
     * Selector
     */
    private Selector selector;

    private final IntSupplier selectNowSupplier = new IntSupplier() {
        @Override
        public int get() throws Exception {
            return selectNow();
        }
    };

    private Selector unwrappedSelector;

    /**
     * 就绪事件的集合，优化时使用
     */
    private SelectedSelectionKeySet selectedKeys;

    /**
     * IO任务占总任务比例
     */
    private volatile int ioRatio = 50;

    /**
     * 取消的键数目
     */
    private int cancelledKeys;

    private boolean needsToSelectAgain;

    NioEventLoop(NioEventLoopGroup parent, Executor executor, SelectorProvider selectorProvider, SelectStrategy strategy, RejectedExecutionHandler rejectedExecutionHandler) {
        super(parent, executor, false, DEFAULT_MAX_PENDING_TASKS, rejectedExecutionHandler);
        // ...
        provider = selectorProvider;
        // 可能优化处理
        final SelectorTuple selectorTuple = openSelector();
        selector = selectorTuple.selector;
        unwrappedSelector = selectorTuple.unwrappedSelector;
        selectStrategy = strategy;
    }

}

    @Override
    protected void run() {
        for (; ; ) {
            try {
                try {
                    // 如果有任务待执行，返回selectNow()；否则返回SelectStrategy.SELECT
                    switch (selectStrategy.calculateStrategy(selectNowSupplier, hasTasks())) {
                        case SelectStrategy.CONTINUE:
                            continue;
                        case SelectStrategy.BUSY_WAIT:
                        case SelectStrategy.SELECT:
                            // select查询是否有就绪的IO事件，进入前设置唤醒标记wakenUp为false，标明线程将要被阻塞
                            select(wakenUp.getAndSet(false));
                            // 唤醒标记wakenUp为true，说明当前任务队列有任务
                            if (wakenUp.get()) {
                                // 唤醒一个阻塞在select上的线程，不过只有NioEventLoop一个线程，所以当前应该没有线程被阻塞？
                                // 这里wakeUp，下个select会立即返回，所以目的是？
                                selector.wakeup();
                            }
                        default:
                            // selectNow后进来
                    }
                } catch (IOException e) {
                    rebuildSelector0();
                    handleLoopException(e);
                    continue;
                }
                cancelledKeys = 0;
                needsToSelectAgain = false;
                // IO时间 / 任务时间
                final int ioRatio = this.ioRatio;
                if (ioRatio == 100) {
                    try {
                        // 先处理就绪的IO事件
                        processSelectedKeys();
                    } finally {
                        // 最后再执行任务队列
                        runAllTasks();
                    }
                } else {
                    final long ioStartTime = System.nanoTime();
                    try {
                        // 处理就绪的IO事件
                        processSelectedKeys();
                    } finally {
                        final long ioTime = System.nanoTime() - ioStartTime;
                        // 给定时间内执行任务
                        runAllTasks(ioTime * (100 - ioRatio) / ioRatio);
                    }
                }
            } catch (Throwable t) {
                // 防止连续异常消耗CPU
                handleLoopException(t);
            }

            try {
                if (isShuttingDown()) {
                    // ...
                }
            } catch (Throwable t) {
                handleLoopException(t);
            }
        }
    }

NioEventLoop#select

    private void select(boolean oldWakenUp) throws IOException {
        Selector selector = this.selector;
        try {
            int selectCnt = 0;
            long currentTimeNanos = System.nanoTime();
            // 每次select的截止时间戳，delayNanos返回最近的一个调度任务的到期时间，没有调度任务返回1秒
            long selectDeadLineNanos = currentTimeNanos + delayNanos(currentTimeNanos);
            for (; ; ) {
                // 将超时时间换算为毫秒
                long timeoutMillis = (selectDeadLineNanos - currentTimeNanos + 500000L) / 1000000L;
                // 不足1ms，不再select阻塞
                if (timeoutMillis <= 0) {
                    // 如果一次select操作都没有进行
                    if (selectCnt == 0) {
                        // selectNow后马上返回
                        selector.selectNow();
                        selectCnt = 1;
                    }
                    break;
                }
                // 如果任务队列非空，并且唤醒标志为false，那么需要CAS更新为true，如果成功
                if (hasTasks() && wakenUp.compareAndSet(false, true)) {
                    // selectNow返回，不耽误任务执行
                    selector.selectNow();
                    selectCnt = 1;
                    break;
                }
                // 线程将阻塞至超时或者事件就绪或者被唤醒
                // 阻塞期间如果使用外部线程提交一个任务，会调用NioEventLoop#wakeup，线程唤醒跳出，执行提交任务
                // 如果提交多个任务，使用wakenUp唤醒标记可以使selector.wakeup只执行一次，wakeUp消耗大，尽量少调用
                int selectedKeys = selector.select(timeoutMillis);
                selectCnt++;
                // 有就绪的IO事件 || select前就需要唤醒 || 外部设置wakenUp为真 || 有待执行任务 || 有待执行调度任务
                if (selectedKeys != 0 || oldWakenUp || wakenUp.get() || hasTasks() || hasScheduledTasks()) {
                    // 跳出循环
                    break;
                }
                // 如果线程被中断，那么退出
                if (Thread.interrupted()) {
                    selectCnt = 1;
                    break;
                }
                long time = System.nanoTime();
                if (time - TimeUnit.MILLISECONDS.toNanos(timeoutMillis) >= currentTimeNanos) {
                    selectCnt = 1;
                } else if (SELECTOR_AUTO_REBUILD_THRESHOLD > 0 && selectCnt >= SELECTOR_AUTO_REBUILD_THRESHOLD) {
                    // 处理JDK的BUG
                    selector = selectRebuildSelector(selectCnt);
                    selectCnt = 1;
                    break;
                }
                currentTimeNanos = time;
            }
            // ...
        } catch (CancelledKeyException e) {
            // ...
        }
    }

NioEventLoop#selectRebuildSelector

    private Selector selectRebuildSelector(int selectCnt) throws IOException {
        /*
         * EPOLL的BUG导致select方法不阻塞而直接返回且返回值为0，出现空轮询，解决办法：
         * 对select返回0的操作计数，如果次数大于阈值SELECTOR_AUTO_REBUILD_THRESHOLD就新建一个Selector
         * 将注册到旧的Selector上的Channel重新注册到新的Selector上
         */
        rebuildSelector();
        Selector selector = this.selector;
        // 重建Selector之后立即selectNow
        selector.selectNow();
        return selector;
    }

        if (selectedKeys != null) {

            processSelectedKeysPlain(selector.selectedKeys());

NioEventLoop#processSelectedKeysPlain

    private void processSelectedKeysPlain(Set<SelectionKey> selectedKeys) {
        // 选择键的集合为空直接返回
        if (selectedKeys.isEmpty()) {
            return;
        }
        Iterator<SelectionKey> i = selectedKeys.iterator();
        for (; ; ) {
            final SelectionKey k = i.next();
            final Object a = k.attachment();
            // 处理前先移除
            i.remove();
            // IO事件由Netty处理，默认
            if (a instanceof AbstractNioChannel) {
                processSelectedKey(k, (AbstractNioChannel) a);
            }
            // IO事件由自定义任务处理，在注册了自定义的SelectableChannel后使用
            else {
                @SuppressWarnings("unchecked") NioTask<SelectableChannel> task = (NioTask<SelectableChannel>) a;
                processSelectedKey(k, task);
            }
            if (!i.hasNext()) {
                break;
            }
            // 取消的选择键达到CLEANUP_INTERVAL时，selectAgain
            if (needsToSelectAgain) {
                // selectNow并将needsToSelectAgain标记设置为false
                selectAgain();
                selectedKeys = selector.selectedKeys();
                if (selectedKeys.isEmpty()) {
                    break;
                } else {
                    i = selectedKeys.iterator();
                }
            }
        }
    }

NioEventLoop#processSelectedKey

    private void processSelectedKey(SelectionKey k, AbstractNioChannel ch) {
        final AbstractNioChannel.NioUnsafe unsafe = ch.unsafe();
        // 选择键无效
        if (!k.isValid()) {
            final EventLoop eventLoop;
            try {
                eventLoop = ch.eventLoop();
            } catch (Throwable ignored) {
                return;
            }
            // Channel已不在该EventLoop，直接返回
            if (eventLoop != this || eventLoop == null) {
                return;
            }
            // Channel还在EventLoop，关闭Channel
            unsafe.close(unsafe.voidPromise());
            return;
        }
        try {
            int readyOps = k.readyOps();
            // 客户端连接就绪
            if ((readyOps & SelectionKey.OP_CONNECT) != 0) {
                int ops = k.interestOps();
                // 连接完成后客户端除了连接事件都感兴趣
                ops &= ~SelectionKey.OP_CONNECT;
                k.interestOps(ops);
                // 完成连接
                unsafe.finishConnect();
            }
            // 可写就绪
            if ((readyOps & SelectionKey.OP_WRITE) != 0) {
                ch.unsafe().forceFlush();
            }
            // readyOps == 0为对epoll的bug的处理，防止死循环
            if ((readyOps & (SelectionKey.OP_READ | SelectionKey.OP_ACCEPT)) != 0 || readyOps == 0) {
                // OP_READ及OP_ACCEPT都抽象为read事件
                // Server NioMessageUnsafe#read
                // Client NioByteUnsafe#read
                unsafe.read();
            }
        } catch (CancelledKeyException ignored) {
            unsafe.close(unsafe.voidPromise());
        }
    }

NioEventLoopGroup下包含多个NioEventLoop。

NioEventLoop包含一个Selector，一个任务队列，一个调度任务队列。

NioEventLoop的Selector上可以注册多个Channel。

对于Boss，轮询到的是是连接事件（OP_ACCEPT）。

对于Worker，轮询到的是IO读写事件。

每个Channel分配一个EventLoop，多个Channel可能共享一个EventLoop。

一个Channel上的所有读写事件都只会在一个EventLoop上处理。

在非IO线程调用Channel的方法时，将相关操作封装成一个任务并放入任务队列，保证相关操作在IO线程中串行执行。

    @Override
    public void execute(Runnable task) {
        // ...
        boolean inEventLoop = inEventLoop();
        // 直接添加到队列
        addTask(task);

        // 第一次是在主线程中运行的，会启动线程
        if (!inEventLoop) {
            // 延迟启动，只有当提交第一个任务时线程才启动
            startThread();
            if (isShutdown()) {
                // ...
            }
        }
        // addTaskWakesUp是否唤醒线程，构造方法配置，wakesUpForTask由子类覆盖，默认唤醒
        if (!addTaskWakesUp && wakesUpForTask(task)) {
            wakeup(inEventLoop);
        }
