ProtocolCodeBasedEncoder#encode

    @Override
    protected void encode(ChannelHandlerContext ctx, Serializable msg, ByteBuf out) throws Exception {
        // 初始化Connection时设置
        Attribute<ProtocolCode> att = ctx.channel().attr(Connection.PROTOCOL);
        ProtocolCode protocolCode;
        if (att == null || att.get() == null) {
            // 默认
            protocolCode = this.defaultProtocolCode;
        } else {
            // 使用自定义
            protocolCode = att.get();
        }
        Protocol protocol = ProtocolManager.getProtocol(protocolCode);
        // 处理
        protocol.getEncoder().encode(ctx, msg, out);
    }

RpcCommandEncoderV2#encode

    @Override
    public void encode(ChannelHandlerContext ctx, Serializable msg, ByteBuf out) throws Exception {
        try {
            if (msg instanceof RpcCommand) {
                // 获取当前写索引
                int index = out.writerIndex();
                RpcCommand cmd = (RpcCommand) msg;
                // 协议码ProtocolCode
                out.writeByte(RpcProtocolV2.PROTOCOL_CODE);
                Attribute<Byte> version = ctx.channel().attr(Connection.VERSION);
                byte ver = RpcProtocolV2.PROTOCOL_VERSION_1;
                if (version != null && version.get() != null) {
                    ver = version.get();
                }
                // 协议版本ProtocolVersion
                out.writeByte(ver);
                // 请求类型RequestType，request/response/oneway
                out.writeByte(cmd.getType());
                // 请求命令类型CommandCode，request/response/heartbeat
                out.writeShort(((RpcCommand) msg).getCmdCode().value());
                // 请求命令版本号CommandVersion
                out.writeByte(cmd.getVersion());
                // 请求RequestId
                out.writeInt(cmd.getId());
                // 序列化器Codec
                out.writeByte(cmd.getSerializer());
                // 协议开关Switch，协议级别的开关控制，比如CRC校验，安全校验等
                out.writeByte(cmd.getProtocolSwitch().toByte());
                if (cmd instanceof RequestCommand) {
                    // 超时Timeout
                    out.writeInt(((RequestCommand) cmd).getTimeout());
                }
                if (cmd instanceof ResponseCommand) {
                    ResponseCommand response = (ResponseCommand) cmd;
                    // 响应码ResponseStatus，方便客户端定位
                    out.writeShort(response.getResponseStatus().getValue());
                }
                // 业务请求类名长度ClassLen
                out.writeShort(cmd.getClazzLength());
                // 业务请求头长度HeaderLen
                out.writeShort(cmd.getHeaderLength());
                // 业务请求体长度ContentLen
                out.writeInt(cmd.getContentLength());
                if (cmd.getClazzLength() > 0) {
                    // 业务请求类名ClassName
                    out.writeBytes(cmd.getClazz());
                }
                if (cmd.getHeaderLength() > 0) {
                    // 业务请求头HeaderContent
                    out.writeBytes(cmd.getHeader());
                }
                if (cmd.getContentLength() > 0) {
                    // 业务请求体BodyContent
                    out.writeBytes(cmd.getContent());
                }
                if (ver == RpcProtocolV2.PROTOCOL_VERSION_2 && cmd.getProtocolSwitch().isOn(ProtocolSwitch.CRC_SWITCH_INDEX)) {
                    byte[] frame = new byte[out.readableBytes()];
                    // 整个包
                    out.getBytes(index, frame);
                    // CRC校验码
                    out.writeInt(CrcUtil.crc32(frame));
                }
            } else {
                String warnMsg = "msg type [" + msg.getClass() + "] is not subclass of RpcCommand";
                logger.warn(warnMsg);
            }
        } catch (Exception e) {
            logger.error("Exception caught!", e);
            throw e;
        }
    }

解码

ProtocolCodeBasedDecoder#decode

    @Override
    protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
        // 先标记读索引
        in.markReaderIndex();
        // 先解码协议码ProtocolCode
        ProtocolCode protocolCode = decodeProtocolCode(in);
        if (null != protocolCode) {
            /*
             * 再解码协议版本ProtocolVersion
             * 读之前先重置读索引，再读一次协议码ProtocolCode，然后再读协议版本ProtocolVersion
             */
            byte protocolVersion = decodeProtocolVersion(in);
            if (ctx.channel().attr(Connection.PROTOCOL).get() == null) {
                // 方便后续处理
                ctx.channel().attr(Connection.PROTOCOL).set(protocolCode);
                // 说明带有协议版本号
                if (DEFAULT_ILLEGAL_PROTOCOL_VERSION_LENGTH != protocolVersion) {
                    ctx.channel().attr(Connection.VERSION).set(protocolVersion);
                }
            }
            Protocol protocol = ProtocolManager.getProtocol(protocolCode);
            if (null != protocol) {
                // 再次重置读索引
                in.resetReaderIndex();
                // 解码
                protocol.getDecoder().decode(ctx, in, out);
            } else {
                throw new CodecException("Unknown protocol code: [" + protocolCode + "] while decode in ProtocolDecoder.");
            }
        }
    }

RpcProtocolDecoder#decodeProtocolVersion

    @Override
    protected byte decodeProtocolVersion(ByteBuf in) {
        // 先重置读索引
        in.resetReaderIndex();
        // 可读长度>=协议码长度+默认协议版本长度
        if (in.readableBytes() >= protocolCodeLength + DEFAULT_PROTOCOL_VERSION_LENGTH) {
            // 再次读协议码ProtocolCode
            byte rpcProtocolCodeByte = in.readByte();
            // 协议码表明带有协议版本号
            if (rpcProtocolCodeByte >= MIN_PROTOCOL_CODE_WITH_VERSION) {
                // 继续读协议版本号返回
                return in.readByte();
            }
        }
        // 返回默认非法协议版本号长度标识
        return DEFAULT_ILLEGAL_PROTOCOL_VERSION_LENGTH;
    }

RpcCommandDecoderV2#decode

    @Override
    public void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
        // 可读长度>=最小长度
        if (in.readableBytes() >= lessLen) {
            // 再次标记读索引
            in.markReaderIndex();
            // 再次读取协议码ProtocolCode
            byte protocol = in.readByte();
            // 再次重置读索引（反复读来读去。。。）
            in.resetReaderIndex();
            // 再次确认
            if (protocol == RpcProtocolV2.PROTOCOL_CODE) {
                // 可读长度>协议码长度+协议版本长度+请求类型长度
                if (in.readableBytes() > 2 + 1) {
                    // 获取当前读索引位置
                    int startIndex = in.readerIndex();
                    // 再次标记读索引（这是第几次了）
                    in.markReaderIndex();
                    // 跳过协议码ProtocolCode，读索引++
                    in.readByte();
                    // 再次读取协议版本ProtocolVersion，读索引++
                    byte version = in.readByte();
                    // 读取请求类型RequestType
                    byte type = in.readByte();
                    // REQUEST或ONEWAY
                    if (type == RpcCommandType.REQUEST || type == RpcCommandType.REQUEST_ONEWAY) {
                        // 可读长度>请求头长度-（协议码长度+协议版本长度+请求类型长度）
                        if (in.readableBytes() >= RpcProtocolV2.getRequestHeaderLength() - 3) {
                            // 请求命令类型CommandCode，request/response/heartbeat
                            short cmdCode = in.readShort();
                            // 请求命令版本号CommandVersion
                            byte ver2 = in.readByte();
                            // 请求RequestId
                            int requestId = in.readInt();
                            // 序列化器Codec
                            byte serializer = in.readByte();
                            // 协议开关Switch，协议级别的开关控制，比如CRC校验，安全校验等
                            byte protocolSwitchValue = in.readByte();
                            // 超时Timeout
                            int timeout = in.readInt();
                            // 业务请求类名长度ClassLen
                            short classLen = in.readShort();
                            // 业务请求头长度HeaderLen
                            short headerLen = in.readShort();
                            // 业务请求体长度ContentLen
                            int contentLen = in.readInt();
                            byte[] clazz = null;
                            byte[] header = null;
                            byte[] content = null;
                            int lengthAtLeastForV1 = classLen + headerLen + contentLen;
                            // CRC校验开启
                            boolean crcSwitchOn = ProtocolSwitch.isOn(ProtocolSwitch.CRC_SWITCH_INDEX, protocolSwitchValue);
                            // 最小长度
                            int lengthAtLeastForV2 = classLen + headerLen + contentLen;
                            if (crcSwitchOn) {
                                // 加上CRC校验的长度
                                lengthAtLeastForV2 += 4;
                            }
                            // 对协议版本的处理
                            if ((version == RpcProtocolV2.PROTOCOL_VERSION_1 && in.readableBytes() >= lengthAtLeastForV1) || (version == RpcProtocolV2.PROTOCOL_VERSION_2 && in.readableBytes() >= lengthAtLeastForV2)) {
                                if (classLen > 0) {
                                    clazz = new byte[classLen];
                                    // 业务请求类名ClassName
                                    in.readBytes(clazz);
                                }
                                if (headerLen > 0) {
                                    header = new byte[headerLen];
                                    // 业务请求头HeaderContent
                                    in.readBytes(header);
                                }
                                if (contentLen > 0) {
                                    content = new byte[contentLen];
                                    // 业务请求体BodyContent
                                    in.readBytes(content);
                                }
                                if (version == RpcProtocolV2.PROTOCOL_VERSION_2 && crcSwitchOn) {
                                    // CRC校验，失败就抛异常
                                    checkCRC(in, startIndex);
                                }
                            } else {
                                // 收到的数据包不够，再次重置读索引，下次继续读
                                in.resetReaderIndex();
                                return;
                            }
                            // 生成RequestCommand
                            RequestCommand command;
                            if (cmdCode == CommandCode.HEARTBEAT_VALUE) {
                                // 心跳
                                command = new HeartbeatCommand();
                            } else {
                                // 其它
                                command = createRequestCommand(cmdCode);
                            }
                            command.setType(type);
                            command.setVersion(ver2);
                            command.setId(requestId);
                            command.setSerializer(serializer);
                            command.setProtocolSwitch(ProtocolSwitch.create(protocolSwitchValue));
                            command.setTimeout(timeout);
                            command.setClazz(clazz);
                            command.setHeader(header);
                            command.setContent(content);
                            // 解码完成
                            out.add(command);
                        } else {
                            // 收到的数据包不够，再次重置读索引，下次继续读
                            in.resetReaderIndex();
                        }
                    } else if (type == RpcCommandType.RESPONSE) {
                        if (in.readableBytes() >= RpcProtocolV2.getResponseHeaderLength() - 3) {
                            short cmdCode = in.readShort();
                            byte ver2 = in.readByte();
                            int requestId = in.readInt();
                            byte serializer = in.readByte();
                            byte protocolSwitchValue = in.readByte();
                            short status = in.readShort();
                            short classLen = in.readShort();
                            short headerLen = in.readShort();
                            int contentLen = in.readInt();
                            byte[] clazz = null;
                            byte[] header = null;
                            byte[] content = null;
                            int lengthAtLeastForV1 = classLen + headerLen + contentLen;
                            boolean crcSwitchOn = ProtocolSwitch.isOn(ProtocolSwitch.CRC_SWITCH_INDEX, protocolSwitchValue);
                            int lengthAtLeastForV2 = classLen + headerLen + contentLen;
                            if (crcSwitchOn) {
                                lengthAtLeastForV2 += 4;
                            }
                            if ((version == RpcProtocolV2.PROTOCOL_VERSION_1 && in.readableBytes() >= lengthAtLeastForV1) || (version == RpcProtocolV2.PROTOCOL_VERSION_2 && in.readableBytes() >= lengthAtLeastForV2)) {
                                if (classLen > 0) {
                                    clazz = new byte[classLen];
                                    in.readBytes(clazz);
                                }
                                if (headerLen > 0) {
                                    header = new byte[headerLen];
                                    in.readBytes(header);
                                }
                                if (contentLen > 0) {
                                    content = new byte[contentLen];
                                    in.readBytes(content);
                                }
                                if (version == RpcProtocolV2.PROTOCOL_VERSION_2 && crcSwitchOn) {
                                    checkCRC(in, startIndex);
                                }
                            } else {
                                in.resetReaderIndex();
                                return;
                            }
                            ResponseCommand command;
                            if (cmdCode == CommandCode.HEARTBEAT_VALUE) {
                                command = new HeartbeatAckCommand();
                            } else {
                                command = createResponseCommand(cmdCode);
                            }
                            command.setType(type);
                            command.setVersion(ver2);
                            command.setId(requestId);
                            command.setSerializer(serializer);
                            command.setProtocolSwitch(ProtocolSwitch.create(protocolSwitchValue));
                            command.setResponseStatus(ResponseStatus.valueOf(status));
                            command.setClazz(clazz);
                            command.setHeader(header);
                            command.setContent(content);
                            command.setResponseTimeMillis(System.currentTimeMillis());
                            command.setResponseHost((InetSocketAddress) ctx.channel().remoteAddress());
                            out.add(command);
                        } else {
                            in.resetReaderIndex();
                        }
                    } else {
                        String emsg = "Unknown command type: " + type;
                        logger.error(emsg);
                        throw new RuntimeException(emsg);
                    }
                }

            } else {
                String emsg = "Unknown protocol: " + protocol;
                logger.error(emsg);
                throw new RuntimeException(emsg);
            }

        }
    }
