ES6的第一个版本：ECMAScript 2015

Babel将ES6代码转为ES5代码。

静态字符串一律使用单引号或反引号，不使用双引号。

动态字符串使用反引号。

// bad
const a = "foobar";
const b = 'foo' + a + 'bar';

// acceptable
const c = `foobar`;

// good
const a  = 'foobar';
const b = `foo${a}bar`;
const c = 'foobar';

let命令：声明的变量只在代码块内有效。

{
    let a = 10;
    var b = 1;
}
a // ReferenceError: a is not defined.
b // 1

for(let i = 0; i < 100; i++) {}
i // ReferenceError: i is not defined

var messages = ["1", "2", "3!"];
for (var i = 0; i < messages.length; i++) {
    setTimeout(function () {
        console.log(messages[i]);
    }, i * 1500);
}
// undefined

var a = [];
for (var i = 0; i < 10; i++) {
    a[i] = function () {
        console.log(i);
    };
}
a[6](); // 10

// let声明的变量仅在块级作用域内有效
for (let i = 0; i < 10; i++) {
    a[i] = function () {
        console.log(i);
    };
}
a[6](); // 6

let不像var变量提升，如果区块存在let和const命令，let之前，该变量不可用。

称为“暂时性死区”（temporal dead zone，TDZ）。

if (true) {
    // TDZ开始
    tmp = 'abc'; // ReferenceError
    console.log(tmp); // ReferenceError

    let tmp; // TDZ结束
    console.log(tmp); // undefined

    tmp = 123;
    console.log(tmp); // 123
}

function bar(x=y, y=2) {
    return [x, y];
}

bar(); // 报错，参数x默认值等于y，此时y还没声明，属于”死区“。

let不允许在相同作用域内，重复声明变量。

function () {
    let a = 10;
    var a = 1; // 报错
}

// 不能在函数内部重新声明参数。
function func(arg) {
    let arg; // 报错
}

function func(arg) {
    {let arg; // 不报错}
}

let新增了块级作用域。

function f1() {
    let n = 5;
    if (true) {
        let n = 10;
    }
    console.log(n); // 5
    // 如果使用var，最后输出10。
}

// 立即执行匿名函数（IIFE）不再必要
(function () {
    var tmp = 1;
}());

// 块级作用域写法
{
    let tmp = 1;
}

ES6函数本身的作用域，在其所在的块级作用域之内。

function f() { console.log('I am outside!'); }
(function () {
    if(false) {
        function f() { console.log('I am inside!'); } // 重复声明函数f
    }
    f();
}());
// ES5中，得到“I am inside!”，因为ES5函数提升，函数声明提升到当前作用域顶部
// ES6中，得到“I am outside!”，因为ES6支持块级作用域
  • const一旦声明，值就不能改变。
  • const的作用域与let命令相同：只在声明所在的块级作用域内有效。
  • const也不存在提升，只能在声明的后面使用。
  • const也与let一样不可重复声明。
  • const只是指向变量所在的地址，将一个对象声明为常量必须小心。
const foo = {}; // foo储存的是一个地址，指向一个对象。
foo.prop = 123; // 可以添加新属性
foo = {} // 不可变的是这个地址，不能把foo指向另一个地址

const arr = [];
arr.push("Hello"); // ok
arr.length = 0; // ok
arr = ["Dave"]; // 报错

将对象冻结，使用Object.freeze方法。

const foo = Object.freeze({}); // foo指向一个冻结的对象
foo.prop = 123; // 添加新属性不起作用

// 除了将对象本身冻结，对象的属性也冻结
var constantize = (obj) => {
    Object.freeze(obj);
    Object.keys(obj).forEach( (key, value) => {
        if ( typeof obj[key] === 'object' ) {
            constantize( obj[key] );
        }
    });
};

const声明只在当前代码块有效。如果想设置跨模块的常量，可采用下面写法。

// constants.js 模块
export const A = 1;

// test1.js 模块
import * as constants from './constants';
console.log(constants.A); // 1

// test2.js 模块
import {A} from './constants';
console.log(A); // 1

注：const定义是不可重新赋值，与不可变的值不同，const 定义的 Object，定义之后仍可以修改其属性。

所以使用场景很广，反之let的使用场景应该相对较少，只会在 loop(for，while 循环)及少量必须重定义的变量上用到。

let和const，优先用const，在全局环境，只应设置常量。

所有函数都应设为常量。

// bad
var a = 1, b = 2, c = 3;

// good
const a = 1;
const b = 2;
const c = 3;

// best
const [a, b, c] = [1, 2, 3];

const简写

const d = {
    // 直接指定原型对象
    __proto__: basicConfig,

    // 属性简写
    a,

    // 方法简写
    c() {
        return this.a;
    }
}

全局对象是最顶层的对象，浏览器环境是window对象，Node.js是global对象，所有全局变量都是全局对象的属性。

ES6规定，var命令和function命令声明的全局变量，属于全局对象的属性；

let命令、const命令、class命令声明的全局变量，不属于全局对象的属性。

var a = 1;
window.a // 1
// 在node环境，写成global.a
// 采用通用方法，写成this.a

let b = 1;
window.b // undefined

扩展运算符（spread）是三个点。

类似rest参数的逆运算，将数组转为逗号分隔的参数序列。

主要用于函数调用。

function push(array, ...items) {
    array.push(...items); // 函数调用，使用扩展运算符将数组变为参数序列
}

function add(x, y) {
    return x + y;
}

var numbers = [4, 38];
// 函数调用，使用扩展运算符将数组变为参数序列
add(...numbers) // 42 

const date = new Date(...[2015, 1, 1]);

// 扩展运算符与正常函数参数结合使用
function fun(a, b, c, d,) {}
var arr = [0, 1];
fun(...arr, 2, ...[3]);

扩展运算符不需要apply方法展开数组。

Math.max(14, 3, 77);
Math.max.apply(null, [14, 3, 77]) // ES5写法
Math.max(...[14, 3, 77]) // ES6写法

// push一个数组到另一个数组尾
var arr1 = [0, 1, 2];
var arr2 = [3, 4, 5];

// ES5写法
Array.prototype.push.apply(arr1, arr2); // push方法参数不能是数组，通过apply使用push
// ES6写法
arr1.push(...arr2); // 扩展运算符直接将数组传入push方法

扩展运算符用于数组赋值。

var a = [1];
var b = [2, 3, 4];
var c = [0, ...a, 5, ...b];

c; // [0, 1, 5, 2, 3, 4]

// 将一个数组拷贝进另一个数组
const arr2 = [...arr1];

// 扩展运算符与解构赋值结合，生成数组
const [first, ...rest] = [1, 2, 3, 4, 5];
first // 1
rest // [2, 3, 4, 5]

const [first, ...rest] = [];
first // undefined
rest // []:

扩展运算符用于数组赋值，只能放在参数的最后一位，否则报错。

const [...butLast, last] = [1, 2, 3, 4, 5]; // 报错

函数只能返回一个值，如果返回多个值，只能返回数组或对象。

var dateFields = readDateFields(database);
var d = new Date(...dateFields);
// 通过扩展运算符，直接传入构造函数Date

扩展运算符将字符串转为数组。

[..."hello"] // [ "h", "e", "l", "l", "o" ]

任何类似数组的对象，都可以用扩展运算符转为真正的数组。

var nodeList = document.querySelectorAll('div');
var array = [...nodeList];

Array.from方法，将类数组对象转为数组。

const foo = document.querySelectorAll('.foo');
const nodes = Array.from(foo);

ES6引入rest参数（“…变量名”），用于定义函数，不需要用arguments对象。

rest参数搭配的变量是一个数组。

function add(...values) {
     let sum = 0;
     for (var val of values) {
        sum += val;
     }
     return sum;
}
// 利用rest参数，可以向该函数传入任意数目的参数
add(2, 5, 3); // 10

// arguments写法
const sortNumbers = () =>
    Array.prototype.slice.call(arguments).sort();

// rest参数写法
const sortNumbers = (...numbers) => numbers.sort();
// rest参数中的变量代表一个数组，数组方法都可以用于这个变量

// 利用rest参数改写数组push方法
function push(arr, ...items) {
    items.forEach(function(item) {
        arr.push(item);
    });
}

var a = [];
push(a, 1, 2, 3);

rest参数之后不能有其他参数，否则报错。

函数的length属性，不包括rest参数。

(function(a) {}).length // 1
(function(...a) {}).length // 0
(function(a, ...b) {}).length // 1
