使用模板引擎：

// 在html定义一个html模板
<script id = "user_info_template" type = "text/x-jquery-tmpl">
    <div class = "user_info">
        <span class="user_name">#{name}</span>
        <span class="user_age">#{age}</span>
        <span class="user_gender">#{gender}</span>
    <div>
</script>

<script>
    var user = {name:'david supper',age:12, gender:'man'}

    $(document).ready(function() {
        var template = $("#user_info_template").html();
        var html = renderTemplate(template, user);
    });

    function renderTemplate(template, data) {
        var content = '';

        if(data instanceof Array) {
            for(var i = 0; i < data.length; i++) {
                content += renderTemplate(template, data[i]);
            }
        } else {
            var t=template;
            for(var attr in data) {
                var value = data[attr];
                t = t.replace(new RegExp("#{"+attr+"}", "gm"), value);
            }
            content += t;
        }
        return content;
    }
</script>

function TemplateEngine(html, options) {
    // 变量，占位符
    var re = /<%(.+?)%>/g,
        // 语句关键字
        reExp = /(^( )?(var|if|for|else|switch|case|break|{|}|;))(.*)?/g,
        // 存放代码字符串，最后new Function调用
        code = 'with(obj) { var r=[];\n',
        // 记录占位符位置
        cursor = 0,
        result, match;

    var add = function(line, js) {
            // 如果是js，是变量？还是if等语句？注意没有双引号
            js ? (code += line.match(reExp) ? line + '\n' : 'r.push(' + line + ');\n') : 
                // 是html片段，如果不是空字符串的话就push进，同时双引号要转义，因为外层可能也有双引号
                (code += line != '' ? 'r.push("' + line.replace(/"/g, '\\"') + '");\n' : '');

            return add;
        }

    while (match = re.exec(html)) {
        // 传入上一次结束到下一个占位符"");
    //     r.push(this.skills[index]);
    //     r.push("");
    // }
    // r.push("");

    code = (code + 'return r.join(""); }').replace(/[\r\t\n]/g, ' ');

    try {
        // eval字符串，apply固定this为JSON数据对象
        result = new Function('obj', code).apply(options, [options]);
    } catch (err) {
        console.error("'" + err.message + "'", " in \n\nCode:\n", code, "\n");
    }

    return result;
}
