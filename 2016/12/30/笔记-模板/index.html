    function renderTemplate(template, data) {
        var content = '';

        if(data instanceof Array) {
            for(var i = 0; i < data.length; i++) {
                content += renderTemplate(template, data[i]);
            }
        } else {
            var t=template;
            for(var attr in data) {
                var value = data[attr];
                t = t.replace(new RegExp("#{"+attr+"}", "gm"), value);
            }
            content += t;
        }
        return content;
    }
</script>
