[笔记]-闭包

2016-10-29

2016-11-17重新整理

  • 关于JavaScript词法作用域：
function foo() {
    console.log(a);
}
function bar() {
    var a = 3;
    foo();
}
var a = 2;
bar(); // 2

而动态作用域不关心何处声明，只关心何处调用，作用域链是基于调用栈的，而不是作用域嵌套。如果JavaScript具有动态作用域，上面输出3。

  • 闭包
function outer() {
    var a = 1;
    function inner() {
        console.log(a);
    }
    return inner;
}

var closure = outer(); // 返回inner赋给closure
// inner引用了外部函数的变量a，outer执行完后，a没有被销毁
closure(); // 执行返回的函数，依然能访问变量a
  • 函数inner()在定义时的词法作用域以外的地方被调用。
  • 由于inner()声明的位置，它可访问outer()内部作用域，outer()执行后内部作用域依然存在，这个引用叫闭包。
  • 因为函数是JavaScript中唯一拥有自身作用域的结构，因此闭包依赖于函数。
  • 闭包使函数可以继续访问定义时的词法作用域（JavaScript的函数运行在定义的作用域里，而不是执行的作用域里）。
  • 因此，在函数外可以读取函数内变量。
  • 闭包还可以使内部变量记住上一次调用的结果。
  • 每次重新引用函数的时候，闭包是不同的。
function outer() {
    var count = 0;
    function inner() {
        count++;
        console.log(count);
    }
    return inner;
}

var inn1 = outer();
var inn2 = outer();
inn1 === inn2; // false，每次调用都返回一个新函数
inn1(); //1
inn1(); //2
// inn1()和inn2()的互不影响
inn2(); //1
inn2(); //2

inn2 = null;
inn1 = null;
  • 无论函数在何处被调用，闭包总能访问定义它时所处作用域中的变量：
function fun(n,o) {
    console.log(o);
    return {
        fun: function(m){
            return fun(m,n);
        }
    };
}

var a = fun(0); // undefined
a.fun(1); // 0
a.fun(2); // 0

var b = fun(0).fun(1).fun(2).fun(3); // undefined, 0, 1, 2

定时器闭包

function delay(message) {
    setTimeout(function timer() {
        console.log(message);
    }, 1000);
}
delay("Hello, closure!");
  • 将内部函数timer传递给setTimeout，timer引用变量message，涵盖delay作用域的闭包。
  • delay执行1000毫秒后，内部作用域不会消失，timer依然有delay作用域。

循环的闭包

for(var i = 0; i < 10; i++) {
    setTimeout(function() {
        console.log(i);
    }, i*1000);
}
// 不会每秒一次输出0到9，而会每秒一次输出10十次
  • console.log被调用的时候，匿名函数保持对外部变量i的引用，此时for循环结束，i值10。
  • 每次循环中创建变量i的拷贝，使用自执行匿名函数：
for (var i=1; i<=5; i++) {
    (function() {
        var j = i;
        setTimeout(function() {
            console.log(j);
        }, j*1000);
    })();
}
  • 更好：
for (var i=1; i<=5; i++) {
    (function(j) {
        setTimeout(function() {
            console.log(j);
        }, j*1000);
    })(i);
}
  • 匿名函数立即执行，把i作为参数，此时函数内j变量就有了i的拷贝。
  • 另外的方法：从匿名自执行函数中返回一个函数：
for(var i = 0; i < 10; i++) {
    setTimeout((function(j) {
        return function() {
            console.log(j);
        }
    })(i), i*1000)
}
  • IIFE每次迭代时都创建一个新作用域。
  • let声明在这个块作用域中声明一个变量。
for (var i=1; i<=5; i++) {
    let j = i; // 闭包的块作用域
    setTimeout(function timer() {
        console.log(j);
    }, j*1000);
}
  • for循环头部的let声明使i在循环每次迭代都会声明。随后的每个迭代会使用上个迭代的值初始化变量。
for (let i=1; i<=5; i++) {
    setTimeout(function timer() {
        console.log(i);
    }, i*1000);
}
for (var i = 0; i < Things.length; i++) {
    Things[i].onclick=function() {
        console.log(i);
    }
}
// 保存变量
for (var i = 0; i < Things.length; i++) {
    Things[i].index=i;
    var that = this;
    Things[i].onclick=function() {
        console.log(Things[that.index].index);
    }
}
  • 或者使用闭包，把每次循环的i值都封闭起来。当在事件函数中顺着作用域链中从内到外查找变量i时，会先找到被封闭在闭包环境中的i。

数组与闭包

  • 返回的函数在其定义内部引用了局部变量arr，所以返回后，内部的变量还被引用。
var results = (function count() {
    var arr = [];
    for (var i=1; i<=3; i++) {
        // 每次循环都创建了新函数
        arr.push(function () {
            return i * i;
        });
    }
    // 把创建的3个函数都添加到arr中返回
    return arr;
})();

results[0]();
results[1]();
results[2](); 
// 结果不是1，4，9，而是16，16，16
  • 返回的函数引用了变量i，但并非立刻执行，等到最后3个函数都返回时，所引用的变量i已变成4。
  • 返回函数不要引用任何循环变量，或后续会变化的变量。
for (var i = 0; i < 5; i++) {
    var btn = document.createElement('button');
    btn.appendChild(document.createTextNode('Button ' + i));
    btn.addEventListener('click', function(){console.log(i);}); // 全是5
    document.body.appendChild(btn);
}
  • 如果要引用循环变量，方法是创建一个函数，用该函数的参数保存循环变量当前的值：
var results = (function count() {
    var arr = [];
    for (var i=1; i<=3; i++) {
        arr.push((function (n) {
            return function () {
                return n * n;
            }
        })(i));
    }
    return arr;
})();

results[0](); // 1
results[1](); // 4
results[2](); // 9

事件与闭包

(function(eles) { 
    for (var i = 0, len = eles.length; i<len; i++) {
        eles[i].onclick = function() { 
            alert(i);
        };
    }
})(eles);
  • for循环中每个ele点击事件绑定了一个匿名函数，当循环结束后，变量i的值变为5；
  • 此时再点击ele，执行相应的匿名函数，这时变量i已经是5了，所以点击弹出5，因为返回的匿名函数都引用了同一个变量i；
(function(eles) { 
    var what = function(i) { 
        return function() { 
            alert(i);
        }
    };
    for (var i = 0, len = eles.length; i<len; i ++) {
        eles[i].onclick =what(i);
    };
})(eles);
  • for循环时给点击事件绑定的what函数传递i后返回一个内部的匿名函数，因为参数是按值传递的，此时保存的就是当前i的值。

根据事件响应写一个计数器

var count = 0 // 容易被更改
var foo = document.querySelector(".foo");
foo.addEventListener("click", function() {
    count++;
    console.log(count);
}, false)
foo.addEventListener("click", (function() {
    var count = 0;
    return function() {
        count++;
        console.log(count);
    }
})(), false)
$(function() {
    var ele = $("div#panel");
    this.id = "content";
    ele.click(function() {
        console.log(this.id); // panel，两个this引用的并非同一个对象
    });
});
$(function() {
    var ele = $("div#panel");
    this.id = "content";
    var self = this;
    ele.click(function() {
        // 保存的是外部的局部变量self的引用
        console.log(self.id); // content
    });
});
