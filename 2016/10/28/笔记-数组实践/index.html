获取数组中最后一个元素：Array.prototype.slice(begin,end)获取begin和end之间的数组元素。

不设置end参数，将数组的默认长度当end值。

设置负值为begin值，获取最后一个元素：

var array = [1, 2, 3, 4, 5, 6];
array.slice(-1); // [6]
array.slice(-2); // [5, 6]
array.slice(-3); // [4, 5, 6]

数组截断。

var array = [1, 2, 3, 4, 5, 6];
array.length = 3;
array; // [1,2,3]

清空数组。

var array = [12 ,222 ,1000];  
array.length = 0;

替换所有：String.replace()使用字符串或正则替换字符串，只替换第一次出现的字符串，使用正则中的/g模拟 replaceAll。

var string = "john john";
string.replace(/hn/, "ana"); // "joana john"
string.replace(/hn/g, "ana"); // "joana joana"

合并数组：Array.prototype.concat()。

var array1 = [1, 2, 3];
var array2 = [4, 5, 6];
array1.concat(array2); // [1, 2, 3, 4, 5, 6];

不适合用合并两个大型数组，因为消耗内存存储新创建的数组。

用arr1.push.apply(arr1, arr2)不创建新数组，将第一个第二个数组合并，减少内存使用：

var array1 = [1, 2, 3];
var array2 = [4, 5, 6];
[].push.apply(array1, array2); // [1, 2, 3, 4, 5, 6];

数组间追加。

var array1 = [12, "foo", {name "Joe"}, -2458];
var array2 = ["Doe", 555, 100];
Array.prototype.push.apply(array1, array2);
array1; // [12 , "foo" , {name "Joe"} , -2458 , "Doe" , 555 , 100]

function shuffle(a) { 
    var b = []; 
    while (a.length) { 
        var index = ~~(Math.random() * a.length); 
        b.push(a[index]);
        a.splice(index, 1); 
    } 
    return b;
}
// 一：每random一个下标，如果被选过了，继续random，如果没有，标记，存入返回数组，直到所有都被标记
// 二：每次选中后，直接从数组中删除，无需标记
function shuffle(a) { 
    return a.concat().sort(function(a, b) { 
        return Math.random() - 0.5; 
    });
}

数组乱序，正解应该是Fisher–Yates Shuffle，复杂度O(n)。

遍历数组元素，将其与之前的任意元素交换。

从后往前：

function shuffle(array) {
    var _array = array.concat();

    for (var i = _array.length; i--; ) {
        var j = Math.floor(Math.random() * (i + 1));
        // 交换顺序
        var temp = _array[i];
        _array[i] = _array[j];
        _array[j] = temp;
    }

    return _array;
}

从前往后：

function shuffle(a) {
    var length = a.length;
    var shuffled = Array(length);

    for (var index = 0, rand; index < length; index++) {
        rand = ~~(Math.random() * (index + 1));
        if (rand !== index) {
            shuffled[index] = shuffled[rand];
        }
        shuffled[rand] = a[index];
    }

    return shuffled;
}

数组方法性能测试：push()与unshift()方法都能给数组添加元素，push()末尾添加，unshift()开头添加，unshift()的效率较低，每添加一个元素，都要把现有元素下移一个位置。

var arr = [];
// +new Date()相当于new Date().valueOf()，返回当前时间的毫秒数
var startTime = +new Date(); 
// push性能测试 
for (var i = 0; i < 100000; i++) { 
    arr.push(i); 
}
var endTime = +new Date();
console.log("push耗时"+(endTime-startTime)+"毫秒"); 
startTime = +new Date(); 
arr = []; 
// unshift性能测试 
for (var i = 0; i < 100000; i++) { 
    arr.unshift(i); 
}
endTime = +new Date();
console.log("unshift耗时"+(endTime-startTime)+"毫秒");

unshift()比push()慢差不多100倍。因此，慎用unshift，特别是大数组。

如果unshift，先把放进数组的元素用push添加，再reverse，reverse方法性能不错。

var arr = [], s = +new Date; 
for (var i = 0; i < 100000; i++) { 
    arr.push(i); 
}
arr.reverse(); 
console.log("reverse耗时："+(+new Date - s)+"毫秒");

数组去重

O(n^2) 复杂度

function unique(a) {
    var res = []; // 临时数组存储元素

    for (var i = 0, len = a.length; i < len; i++) {
        var item = a[i];
        // 该元素已在res中，则是重复的元素
        for (var j = 0, jLen = res.length; j < jLen; j++) {
            if (res[j] === item)
                break;
        }
        // 如果没有，放入res
        if (j === jLen)
            res.push(item);
    }

    return res;
}

将原数组中重复元素的最后一个元素放入结果数组中。

function unique(a) {
    var res = [];

    for (var i = 0, len = a.length; i < len; i++) {
        for (var j = i + 1; j < len; j++) {
            // 如果发现相同元素，则 i 自增进入下一个循环比较
            if (a[i] === a[j]) {
                j = ++i;
            }
        }
        res.push(a[i]);
    }
    return res;
}

var a = [1, 1, '1', '2', 1];
var ans = unique(a);
console.log(ans); // => ["1", "2", 1]

将数组用sort排序后，相同的元素放在相邻的位置，比较前后位置的元素就可以了。

function unique(a) {
    return a.concat().sort().filter(function(item, pos, ary) {
        return !pos || item != ary[pos - 1];
    });
}

var a = [1, 1, 3, 2, 1, 2, 4, '1'];
var ans = unique(a);
console.log(ans); // [1, 2, 3, 4]
// 1 和 "1" 被排在一起，因为toString() 结果相同
// 可以针对数组中可能出现的不同类型，来写这个比较函数
function unique(a) {
    var seen = {};

    return a.filter(function(item) {
        // 用 JavaScript 中的 Object 对象来当做哈希表
        return seen.hasOwnProperty(item) ? false : (seen[item] = true);
    });
}

var a = [1, 1, 3, 2, "1", 2, 4];
var ans = unique(a);
console.log(ans); // [1, 3, 2, 4]

Array.prototype.unique2 = function() {
    var arr = [this[0]]; // 结果数组
    // 从第二项开始遍历
    for(var i = 1; i < this.length; i++) {
        // 如果当前数组的第i项在当前数组中第一次出现位置不是i，那么第i项是重复的，忽略掉
        if (this.indexOf(this[i]) == i) {
            // 否则存入结果数组
            arr.push(this[i]);
        }
    }
    return arr;
};
Array.prototype.unique3 = function() {
    var hash = {},arr=[]; // hash为hash表，arr为临时数组
    for(var i = 0; i < this.length; i++) {
        var item = this[i];
        var key = typeof(item) + item;
        // 遍历当前数组，如果hash表中没有当前项
        if (!hash[key]) {
            // 存入hash表
            hash[key] = true; 
            // 把当前数组当前项push到临时数组里
            arr.push(key); 
        }
    }
    return arr;
};
// 核心是构建了一个 hash 对象来替代 indexOf. 
// JavaScript对象的键值只能是字符串，var key = typeof(item) + item 区分数值 1 和字符串 "1"
