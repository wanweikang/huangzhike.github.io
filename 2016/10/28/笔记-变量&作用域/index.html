[笔记]-变量&作用域

2016-10-28

2016-11-18重新整理

局部变量只可能通过两种方式声明，一个作为函数参数，另一个通过var关键字。

var a = 1; // 全局变量

(function test(i) {
    // 函数test内的局部作用域
    i = 5; // 局部变量
    var a = 3; // 局部变量
})(10);

不使用var声明变量会产生隐式全局变量。

// 全局作用域
for (var i = 0; i < 10; i++) {
    // 外部循环在第一次Loop后就终止，因为Loop覆盖了全局变量 i
    // 在Loop的for循环中使用var声明可以避免
    (function () {
        // 函数作用域
        // 没用var声明
        for(i = 0; i < 10; i++) { 
            console.log(i);
        }
    })();
}
(function() {
    var a = 1;
    a = 2;
})();
a; // Uncaught ReferenceError: a is not defined(…)
  • 隐式全局变量和明确定义的全局变量的差异，在于delete操作符能否删除。
  • 通过var创建的全局变量是不能被删除的。
  • 无var创建的隐式全局变量能被删除。
  • 隐式全局变量不是真正的全局变量，是全局对象的属性（属性可以通过delete操作符删除，变量不能）。
// 定义三个全局变量
var a = 1;
u = 2; 
(function () {
    fu = 3; 
}());

delete a; // false
a = 2;
delete a; // false

delete u; // true
delete fu; // true

JavaScript支持大括号创建的代码段，但没有块级作用域，除了全局作用域，只有函数可以创建作用域。

function test() { 
    // 一个作用域
    for(var i = 0; i < 10; i++) { 
    // 没有作用域
    }
    console.log(i); // 10，外部可以访问i
}
  • 函数的作用域
function p(a,b) {
    return  pow(a) + pow(b); 
    function pow(m) {
        return Math.pow(m,2); // 返回m的平方
    }
}

pow(4); // 报错，全局作用域下，没有pow

静态作用域（词法作用域）：函数在定义创建的时候（不是调用），已经确定了函数体内自由变量的作用域。

另外，使用new Function创建的函数指向全局，而不是定义的环境。

每引用一个变量，JavaScript会向上遍历整个作用域直到找到变量。如果到达全局作用域仍未找到，抛出ReferenceError。

function a() {
    console.log(x);
}

(function b() {
    var x = 2;
    // 函数a是在函数b外声明的，不会在b中查找
    a();  // Uncaught ReferenceError: x is not defined(…)

})()
var x = 1;
function a() {
    console.log(x);
}

(function show(f) {
    var x = 2;
    (function () {
        f(); // 1，传入的函数是在外部定义的
    })();
})(a);

闭包的两种情况：函数作为返回值；函数作为参数传递。

第一，函数作为返回值

function fn() {
    var b=1;
    return function bar(a) {
        console.log(a+b);
    };
}

var f = fn(); // bar函数作为返回值，赋给f
f(5); // 引用了fn作用域下的变量b的值

第二，函数作为参数被传递

var b = 1;
function fn(a) {
    console.log(a+b);
};

(function(f) {
    var b = 3;
    f(5); // b变量的取值是 1，而不是 3。

})(fn);  // 函数fn作为参数被传入另一个函数

自由变量跨作用域取值：去创建这个函数的作用域取值，而不是“父作用域”。

JavaScript的函数可以嵌套，内部函数可以访问外部函数定义的变量，反过来则不行：

function foo() {
    var a =1;
    function bar() {
        var b = a; // bar可以访问foo的变量a
    }
    var c = b+ 1; // ReferenceError  foo不可以访问bar的变量b
}

将参数从一个函数传递到另一个函数的推荐做法：

function foo(num1,num2) {
    return num1+num2;
}
function bar(num1,num2) {
    return foo.apply(null, arguments); // 将bar参数传到foo
}

调用栈和调用位置：

function baz() {
    // 当前调用栈是：baz
    console.log("baz");
    bar(); // <-- bar的调用位置（当前调用位置在baz中）
}

function bar() {
    // 当前调用栈是baz -> bar
    console.log("bar");
    foo(); // <-- foo的调用位置（当前调用位置在bar中）
}

function foo() {
    // 当前调用栈是baz -> bar -> foo
    console.log("foo");
}

baz(); // <-- baz的调用位置（当前调用位置是全局作用域）
